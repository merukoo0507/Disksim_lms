  30 // The g4 skew analysis works by generating a trace of requests to
  31 // issue to the disk, and then analyzing the result of that to derive
  32 // the skew values.
  33 
  34 // The trace follows the structure of the g4 layout search tree for
  35 // the disk's layout mapping.  We effectively do a depth-first search
  36 // from the top down with a recursive algorithm.  Operating at each
  37 // node, we take the lowest LBN in that node as the "zero point" and
  38 // measure the angular offset of the first lbn in each entry in the
  39 // node from the first entry of the whole node.  Then we recursively
  40 // expand the structure of the lbn range of each index entry.
  41 
  42 // The implementation does both passes in the same code path.
  43

 103 void cb(double t, struct disksim_request *r, void *ctx) {

 107   // This is a dumb hack to deal with ds iface being wrong -- the

 113 // Read lbns l1 and l2 back-to-back, timing the second request.  In
 114 // trace generation mode, we just print out the IOs we would do.  In
 115 // calibration mode, we inject both IOs into an instance of disksim.
 116 double
 117 time_second_request(int l1, int l2, struct dsstuff *ds, struct trace *t,
 118                     struct dm_disk_if *d) {
 119   struct disksim_request *r1, *r2;
 120 
 121   ddbg_assert(l1 < d->dm_sectors);
 122   ddbg_assert(l2 < d->dm_sectors);
 123 
 124   if(mode == GENTRACE) {
 125     fprintf(t->fp, "%f 0 %d 1 1\n", ds->now, l1);
 126     fprintf(t->fp, "%f 0 %d 1 1\n", ds->now + 0.1, l2);
 127     ds->now += 20.0;
 128   } else {
 129     r1 = calloc(1, sizeof(*r1));
 130     r2 = calloc(1, sizeof(*r2));
 131 
 132     r1->blkno = l1;
 133     r1->bytecount = 512;
 134     r1->flags |= DISKSIM_READ;
 135     r1->start = ds->now;
 136     //    r1->reqctx = &t1;
 137   
 138     r2->blkno = l2;
 139     r2->bytecount = 512;
 140     r2->flags |= DISKSIM_READ;
 141     r2->start = ds->now + 0.001;
 142     //    r2->reqctx = &t2;
 143   
 144     disksim_interface_request_arrive(ds->iface, ds->now, r1);
 145     disksim_interface_request_arrive(ds->iface, ds->now, r2);
 146   
 147     // pump disksim
 148 
 149     do {
 150       ds->now = ds->next;
 151       ds->next = -1;
 152       disksim_interface_internal_event(ds->iface, ds->now, 0);
 153     } while(ds->next != -1);
 154 
 155     // roll now forward slightly, some small ~1ms int-arr
 156     ds->now += 1.0;
 157 
 158     free(r1);
 159     free(r2);
 160 
 161     //    printf("%d -> %d : %f\n", l1, l2, ds->compl[1] - ds->compl[0]);
 162 
 163     ds->compl_next = 0;
 164   }
 165 
 166   return ds->compl[1] - ds->compl[0]; 
 167 }

 183 // Parse the next request in the input trace in disksim "validate"
 184 // format.
 185 int
 186 trace_get_next(struct trace *t, int *lbn, double *result) {
 187   int rv;
 188 
 189   rv = fscanf(t->fp, "%*s %*s %d %*d %lf %*f", lbn, result) == 2;
 190   *result /=  1000.0;
 191   
 192   return rv;

 195 // Get the timing for accessing lbn2 after lbn1 from the trace.
 196 double
 197 get_tracetime(struct trace *t, int lbn1, int lbn2) {
 198   double result, t1;
 199   int l1, l2;
 200   
 201   do { // try to resync
 202     ddbg_assert(trace_get_next(t, &l1, &t1));
 203   } while(l1 != lbn1);
 204   ddbg_assert(trace_get_next(t, &l2, &result));
 205 
 206   ddbg_assert(l1 == lbn1);
 207   ddbg_assert(l2 == lbn2);
 208   return result;
 209 }

 212 // We actually want the last lbn on the source track.  If you use the
 213 // first lbn on the source track, some disks will make it every time,
 214 // some will make it about half of the time and have a rotation miss
 215 // the other half.
 216 int
 217 adjust_lbns(struct dm_disk_if *d, int *l1, int *l2) {
 218   struct dm_pbn pbn;
 219   int l0, ln;
 220 
 221   if(l1) {
 222     d->layout->dm_translate_ltop(d, *l1, MAP_FULL, &pbn, 0);
 223     d->layout->dm_get_track_boundaries(d, &pbn, &l0, &ln, 0);
 224     *l1 = ln;
 225   }
 226 
 227   if(l2) {
 228     d->layout->dm_translate_ltop(d, *l2, MAP_FULL, &pbn, 0);
 229     d->layout->dm_get_track_boundaries(d, &pbn, &l0, &ln, 0);
 230     *l2 = l0;
 231   }
 232 
 233   return 0;
 234 }

 236 // The following functions are used to do a linear least-squares fit
 237 // following the article at mathworld.wolfram.com
 238 static double
 239 mean(double *x, int n) {
 240   int i;
 241   double sum = 0.0;
 242   double mean;
 243   for(i = 0; i < n; i++) {
 244     sum += x[i];
 245   }
 246   
 247   mean = sum / n;
 248   return mean;
 249 }
 250 
 251 // sum of squares
 252 static double
 253 ss(double *x, int n) {
 254   int i;
 255   double sum = 0.0;
 256   double res;
 257   for(i = 0; i < n; i++) {
 258     sum += (x[i] * x[i]);
 259   }
 260   res = sum - n * mean(x,n) * mean(x,n);
 261   return res;
 262 }
 263 
 264 static double
 265 ssxy(double *x, double *y, int n) {
 266   int i;
 267   double sum = 0.0;
 268   double res;
 269   for(i = 0; i < n; i++) {
 270     sum += (x[i] * y[i]);
 271   }
 272   res = sum - (n * mean(x,n) * mean(y,n));
 273   return res;
 274 }
 275 
 276 // variance
 277 double vari(double *x, int n) {
 278   double res = ss(x,n) / n;
 279   return res;
 280 }
 281 
 282 double covar(double *x, double *y, int n) {
 283   double res = ssxy(x,y,n) / n;
 284   return res;
 285 }

 287 // y = a + bx
 288 // The data is the collection of pairs (x[i], y[i]) for i in [0,n)
 289 // Returns the correlation coeffecient, r
 290 double
 291 linreg(double *x, double *y, int n, double *a, double *b) {
 292   double r;
 293   *b = ssxy(x,y,n) / ss(x,n);
 294   *a = mean(y,n) - *b * mean(x,n);
 295 
 296   r = sqrt(ssxy(x,y,n) * ssxy(x,y,n) / (ss(x,n) * ss(y,n)));
 297 
 298   return r;
 299 }

 301 // Compute the difference between the samples in d1 and d2, returning
 302 // the slope of a line through the points, (i, d1[i] - d2[i])
 303 // for i in [0,n)
 304 // Returns the y intercept in aa.
 305 // Return value is the slope of the fitted line.  If it decides we're
 306 // done, returns 0.0
 307 double
 308 find_slope(double *d1, double *d2, int n, double *aa,
 309            double period) {
 310   int i;
 311   double a,b,r;
 312   double yi;
 313   double *err = calloc(n, sizeof(*err));
 314   double *x = calloc(n, sizeof(*x));
 315   double fit_err, tot_err;
 316   double var;
 317   double thresh = period / 2.0;
 318 
 319   int addct = 0;
 320   int n2 = 0;
 321   
 322   if(n == 1) {
 323     *aa = d1[0] - d2[0];
 324     return 0.0;
 325   }
 326 
 327   if(n == 2) {
 328     *aa = 0.0;
 329     return d1[1] - d2[1];
 330   }
 331 
 332   for(i = 1; i < n; i++) {
 333     yi = d1[i] - d2[i];
 334 
 335     if(-thresh <= yi && yi <= thresh) {
 336       //      printf("%3d\t%f\n", i, yi);
 337       err[n2] = yi;
 338       x[n2] = i;
 339       n2++;
 340     }
 341   }
 342 
 343   r = linreg(x, err, n2, &a, &b);
 344 
 345   for(i = 0; i < n2; i++) {
 346     printf("%3d\t%f\n", (int)x[i], err[i]);
 347   }
 348 
 349   var = vari(err,n);
 350 
 351   printf("%s() a = %f, b = %f, r = %f, var = %f\n", __func__, a, b, r, var);
 352 
 353   *aa = a;
 354 
 355   free(err);
 356   free(x);
 357 
 358   return b;
 359 }

 361 // Make an angle be within [0,1)
 362 double fix_angle(double a) {
 363   if(a < 0) {
 364     return a - (int)a + 1;
 365   } else {
 366     return a - (int)a;
 367   }
 368 }

 371 // Time accessing lbn li immediately after accessing lbn l0 and convert the
 372 // time into an angle in [0,1) according to the rotational period of
 373 // the disk.
 374 double 
 375 measure_one_skew(struct dm_disk_if *d,
 376                  struct dsstuff *ds,
 377                  struct trace *trace,
 378                  int l0,
 379                  int li) {
 380   double t, tt, a = 0.0;
 381   double period = dm_time_itod(d->mech->dm_period(d));
 382 
 383   adjust_lbns(d, &l0, &li);
 384 
 385   t = time_second_request(l0, li, ds, trace, d);
 386   if(mode == CALIB) {
 387     tt = get_tracetime(trace, l0, li);
 388     a = fix_angle(-(t - tt) / period);
 389   }
 390 
 391   return a;
 392 }

 395 // do_idx_ent() does the hard work to measure the skews for a single
 396 // index entry.  We need to determine e->off, the angular offset of
 397 // the first lbn of this instance taking the first lbn of parent as
 398 // the 0 point and e->alen, the angular offset of the i+1st instance
 399 // of e from the ith.
 400 void
 401 do_idx_ent(struct dm_disk_if *d,        // diskmodel
 402            struct dm_layout_g4 *l,      // root of g4 layout
 403            struct idx_ent *e,           // the entry in question
 404            struct idx *parent,          // The index containing e
 405            struct idx_ent *parent_e,    // The entry for parent in its parent
 406            int parent_off,              // Which entry we are in parent
 407            struct dsstuff *ds,          // Disksim instance
 408            struct trace *t,             // io trace
 409            int lbn) {                   // The first lbn of parent
 410   int i;
 411   int l0;
 412   int dist;
 413 
 414   double yi;
 415 
 416   double *times;
 417   double *tracetimes;
 418 
 419   // for bootstrapping off and len
 420   double off0time, len0time;
 421   int off0lbn[2], len0lbn[2];
 422 
 423   // Work internally in floating-point, then convert back to the
 424   // integer representation at the end.
 425   double aoff = dm_angle_itod(e->off);
 426   double alen =  dm_angle_itod(e->alen);
 427   
 428   double period = dm_time_itod(d->mech->dm_period(d));
 429 
 430   // Number of instances of this entry.
 431   int quot = e->runlen / e->len;
 432 
 433   // times[i] is disksim's prediction of the amount of time to access the 
 434   // first lbn of the ith instance after accessing the first instance.
 435   times = calloc(quot, sizeof(double));
 436   // Actual times from the trace replay against the real disk.
 437   tracetimes = calloc(quot, sizeof(double));
 438   // li[i] contains the first lbn of the ith instance of e.
 439   int *li = calloc(quot, sizeof(*li));
 440 
 441   printf("%s() lbn %d e->lbn %d alen %f off %f quot %d\n", 
 442          __func__, lbn, e->lbn, alen, aoff, quot);
 443 
 444   if(e->alen != 0 || e->off != 0) {
 445     printf("%s() already done, apparently\n", __func__);
 446     return;
 447   }
 448 
 449   // Bootstrap offset by measuring the first instance.
 450   if(e->lbn != 0) {
 451     aoff = measure_one_skew(d, ds, t, lbn, lbn + e->lbn);
 452     e->off = dm_angle_dtoi(aoff);
 453   }
 454 
 455   // Check for the end of the lbn space.
 456   if(lbn + e->lbn + e->len >= d->dm_sectors) {
 457     e->alen = dm_angle_dtoi(0.0);
 458     return;
 459   }
 460 
 461   // Bootstrap alen by measuring the skew from the first to the second
 462   // instance.
 463   alen = measure_one_skew(d, ds, t, lbn + e->lbn, 
 464                           lbn + e->lbn + e->len);
 465   e->alen = dm_angle_dtoi(alen);    
 466 
 467   printf("first off %f len %f\n", aoff, alen);
 468 
 469   // Expand out the lbns of all of the instances.  For calibration
 470   // (second pass), read in the values from the trace.  To generate
 471   // the trace (first pass), time how long it takes to read the first
 472   // lbn of the ith instance after reading the last lbn on the first
 473   // track of the first instance.
 474   for(i = 1; i < quot; i++) {
 475     l0 = lbn;
 476     li[i] = l0 + e->lbn + e->len * i;
 477 
 478     adjust_lbns(d, &l0, &li[i]);
 479 
 480     if(mode == CALIB) {
 481       tracetimes[i] = get_tracetime(t, l0, li[i]);
 482 
 483       printf("%d (%d,%d) trace %f pred %f\n", i, 
 484              l0, li[i], 
 485              tracetimes[i], times[i]);
 486     }
 487     else {
 488       time_second_request(l0, li[i], ds, t, d);
 489     }
 490   }
 491 
 492   if(mode == GENTRACE) {
 493     return;
 494   }
 495 
 496   // Now calibrate the value.  We first look at the first 2 instances,
 497   // then 4, then 8, etc, refining our estimate at each step.
 498   for(dist = 2; dist < quot ; ) {
 499     // y = a + bx; r is the correlation coeffecient.
 500     double a, b;
 501     for(i = 1; i < dist; i++) {
 502       times[i] = time_second_request(l0, li[i], ds, t, d);
 503       printf("%d (%d,%d) trace %f pred %f\n", i, 
 504              l0, li[i], 
 505              tracetimes[i], times[i]);
 506     }
 507 
 508     // Do a linear least squares fit of the difference between the
 509     // predicted and actual service times.  The slope (b) of that line
 510     // corresponds to the error in our estimate of alen and the y
 511     // intercept corresponds to the error in our estimate of off.
 512     b = find_slope(times, tracetimes, dist, &a, period);
 513 
 514     // Adjust the instance-to-instance skew according to the fit.    
 515     alen = fix_angle(alen - b / period);
 516     e->alen = dm_angle_dtoi(alen);
 517     
 518     // If e is the first instance in its parent index, its offset is
 519     // defined to be 0.  Otherwise, correct according to the fit.
 520     if(e->lbn > lbn) {
 521       printf("fix aoff (%d, %d)\n", lbn, e->lbn);
 522       aoff = fix_angle(aoff - a / period);
 523       e->off = dm_angle_dtoi(aoff);
 524     }
 525 
 526     printf("alen -> %f, off -> %f (dist %d)\n", alen, aoff, dist);
 527 
 528     if(dist == quot) {
 529       break;
 530     } else if(dist * 2 >= quot) {
 531       dist = quot;
 532     } else {
 533       dist *= 2;
 534     }
 535   }
 536 
 537   free(li);
 538   free(tracetimes);
 539   free(times);
 540   return;
 541 }

 543 // do_idx() is the main recursive function which measures the skews
 544 // for all of the entries of a given index node.  Each entry has 2
 545 // parameters to measure: the offset of the first instance of that
 546 // entry from the beginning of the index containing it and the
 547 // "length" of each instance, i.e. the offset of the i+1st instance
 548 // relative to the ith.  For each index entry, it first measures the
 549 // offset of the entry relative to the start of the index
 550 // (do_idx_ent()) and then recurses into the structure of that entry.
 551 int do_idx(struct dm_disk_if *d,    // diskmodel
 552            struct idx_ent *ie,      // The entry for idx in its parent
 553            struct idx *idx,         // The index node in question
 554            struct dm_layout_g4 *l,  // g4 layout
 555            struct dsstuff *ds,      // disksim instance
 556            struct trace *t,         // io trace
 557            int lbn) {               // First lbn of idx
 558   int i;
 559   struct idx_ent *e;
 560 
 561   fprintf(stderr, "%s(lbn %d)\n", __func__, lbn);
 562 
 563   for(i = 0, e = &idx->ents[0]; i < idx->ents_len; i++, e++) {
 564     // Measure the angular offset of this entry from the beginning of
 565     // the index node.
 566     do_idx_ent(d, l, &idx->ents[i], idx, ie, i, ds, t, lbn);
 567 
 568     // Recursively expand the structure of this entry.
 569     if(e->childtype == IDX) {
 570       do_idx(d, e, e->child.i, l, ds, t, lbn + e->lbn);
 571     }
 572   }
 573 
 574   for(i = 0, e = &idx->ents[0]; i < idx->ents_len; i++, e++) {
 575     printf("%s() off %f alen %f\n", __func__, 
 576            dm_angle_itod(idx->ents[i].off),
 577            dm_angle_itod(idx->ents[i].alen));
 578   }
 579 
 580   return 0;
 581 }
