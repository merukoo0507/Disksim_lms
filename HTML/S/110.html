 194 int lp_register_module(struct lp_mod *m) {
 195   int c;
 196   for(c = 0; c < lp_max_mod; c++) {
 197     if(!strcmp(m->name, lp_modules[c]->name)) {
 198       return -1;
 199     }
 200   }
 201   if(c >= lp_mod_size) {
 202     lp_mod_size *= 2; lp_mod_size++;
 203     lp_modules = realloc(lp_modules, lp_mod_size * sizeof(int *));
 204   }
 205   if(c > 0) 
 206     lp_modules[c] = m;
 207   else 
 208     lp_modules[0] = m;
 209 
 210   lp_max_mod++;
 211 
 212   return c;
 213 }

 242 static void destroy_value(struct lp_value *v) {
 243   int d;
 244   switch(v->t) {
 245   case S:
 246     free(v->v.s); 
 247     break;
 248     
 249   case LIST:
 250     destroy_list(v->v.l);
 251     break;
 252     
 253   case I: 
 254   case D: break;
 255 
 256   case TOPOSPEC:
 257     for(d = 0; d < v->v.t.len; d++)
 258       destroy_topospec(&v->v.t.l[d]);
 259 
 260     free(v->v.t.l);
 261     break;
 262       
 263   default:
 264     destroy_block(v->v.b);
 265   }
 266 
 267   free(v);
 268 }

 378 void unparse_type(int t, FILE *outfile) {
 379   if(t < 0) {
 380     fprintf(outfile, "%s", lp_builtin_names[abs(t)]);
 381   }
 382   else {
 383     if(t > lp_max_mod) {
 384       fprintf(outfile, "<UNKNOWN TYPE>");
 385     }
 386     else {
 387       fprintf(outfile, "%s", lp_modules[t]->name);
 388     }
 389   }

 392 void unparse_param(struct lp_param *p, FILE *outfile) {
 393   fprintf(outfile, "%s = ", p->name);
 394   if(p->source_file && p->v->source_file) {
 395     if(strcmp(p->source_file, p->v->source_file)) {
 396       unparse_source(p->v->source_file, outfile);
 397       return;
 398     }
 399   }
 400 
 401   unparse_value(p->v, outfile);
 402 }

 404 void unparse_value(struct lp_value *v, FILE *outfile) {
 405   int c;
 406   switch(v->t) {
 407   case I:
 408     fprintf(outfile, "%d", v->v.i);
 409     break;
 410   case D:
 411     if(v->v.d == 0.0) {
 412       fprintf(outfile, "0.0");
 413     }
 414     else {
 415       fprintf(outfile, "%f", v->v.d);
 416     }
 417     break;
 418   case S:
 419     fprintf(outfile, "%s", v->v.s);
 420     break;
 421   case LIST:
 422     unparse_list(v->v.l, outfile);
 423     break;
 424   case TOPOSPEC:
 425     for(c = 0; c < v->v.t.len; c++)
 426       unparse_topospec(&v->v.t.l[c], outfile);
 427     break;
 428   default:
 429   case BLOCK:
 430     unparse_block(v->v.b, outfile);
 431     break;
 432   }
 433 }

 435 void unparse_list(struct lp_list *l, FILE *outfile) {
 436   int printed = 0;
 437   int c;
 438 
 439 /*    indent(outfile); */
 440 
 441   fprintf(outfile, "[ ");
 442   indent_level++;
 443   for(c = 0; c < l->values_len; c++) { 
 444     if(!l->values[c]) 
 445       continue;
 446 
 447     if(c) {
 448       fprintf(outfile, ", ");
 449       if(!(c % l->linelen)) {
 450         fprintf(outfile, "\n");
 451         indent(outfile);
 452       }
 453     }
 454     else {
 455       fprintf(outfile, "\n");
 456       indent(outfile);
 457     }
 458 
 459     printed++;
 460     unparse_value(l->values[c], outfile);
 461   }
 462   indent_level--;
 463 
 464   if((c > 0) && printed) {
 465     fprintf(outfile, "\n");
 466     indent(outfile);
 467   }
 468   fprintf(outfile, "]");
 469 }

 472 void unparse_block(struct lp_block *b, FILE *outfile) {
 473   int c;
 474 /*    indent(outfile); */
 475   if(b->name) {
 476     /* a block definition */
 477     fprintf(outfile, "%s %s {\n", lp_modules[b->type]->name, b->name);
 478   }
 479   else {
 480     /* a block value */
 481     fprintf(outfile, "%s {\n", lp_modules[b->type]->name);
 482 
 483   }
 484 
 485   indent_level++;
 486   for(c = 0; c < b->params_len; c++) {
 487     if(!b->params[c]) continue;
 488 
 489     if(c) fprintf(outfile, ",\n");
 490 
 491     unparse_param(b->params[c], outfile);
 492   }
 493   indent_level--;
 494   fprintf(outfile, "\n");
 495   indent(outfile);
 496   fprintf(outfile, "}");
 497   if(b->name) {
 498     fprintf(outfile, " # end of %s spec", b->name);
 499     fprintf(outfile, "\n\n");
 500   }
 501 }

 524 void unparse_tlt(struct lp_tlt *tlt, FILE *outfile, char *infile) {
 525 
 526   if(tlt->source_file && infile) {
 527     if(strcmp(tlt->source_file, infile)) {
 528       unparse_source(tlt->source_file, outfile);
 529       return;
 530     }
 531   }
 532   
 533   switch(tlt->what) {
 534   case TLT_BLOCK: unparse_block(tlt->it.block, outfile); break;
 535   case TLT_TOPO:  unparse_tl_topospec(tlt->it.topo, outfile); break;
 536   case TLT_INST:  unparse_inst(tlt->it.inst, outfile); break;
 537   default:        ddbg_assert(0); break;
 538   };
 539   
 540 }

 557 /* instantiate all the elements of <l> as <name> */
 558 int lp_inst_list(struct lp_inst *i)
 559 {
 560   int c;
 561   
 562   /*      unparse_block(spec, outputfile); */
 563 
 564   for(c = 0; c < i->l->values_len; c++) {
 565     if(!i->l->values[c]) continue;
 566     
 567     ddbg_assert3(i->l->values[c]->t == S, 
 568                ("bad type for component %s", i->l->values[c]->v.s));
 569     
 570     lp_instantiate(i->l->values[c]->v.s, i->name);
 571   }
 572   
 573   return 0;

 576 /* instantiate <targ> as <name> */
 577 int *lp_instantiate(char *targ, char *name) {
 578   char *nametmp;
 579   struct lp_block *spec;
 580   int *obj;
 581     
 582 /*    unparse_block(b, outputfile); */
 583 
 584   spec = lp_lookup_spec(name);
 585   ddbg_assert3(spec != 0, ("no such type %s.\n", name));
 586 
 587   // this is a bit of a hack; we swap the name of the component
 588   // being instantiated with the name of the module that's being
 589   // instantiated so that the loader function sees the target
 590   // name
 591   nametmp = spec->name;
 592   spec->name = targ;
 593   
 594   //  fprintf(stderr, "*** Instantiating %s as %s\n", targ, name);
 595 
 596   obj = lp_override_inst(spec, 
 597                          targ,
 598                          lp_modules[spec->type]->fn,
 599                          overrides, 
 600                          overrides_len);
 601 
 602   // swap the name back
 603   spec->name = nametmp;
 604 
 605 
 606   if(!obj) {
 607     return 0;
 608   }
 609   
 610   if(lp_modules[spec->type]->callback) {
 611     lp_modules[spec->type]->callback(lp_modules[spec->type]->ctx, obj);
 612   }
 613 
 614   return obj;
 615 }

 617 /* 0 on success, nonzero on error */
 618 int check_types(struct lp_block *b) {
 619   int c = 0;
 620 
 621   /* we'll do all of the type checking here so that 
 622    * the per-mod loaders only have to sanitize values */
 623 
 624   for(c = 0; c < b->params_len; c++) {
 625     if(b->params[c]) {
 626       if(lp_param_name(b->type, b->params[c]->name) == -1) {
 627 
 628         fprintf(stderr, "*** warning: parameter %s not valid in context %s\n",
 629                 b->params[c]->name, lp_modules[b->type]->name);
 630         
 631       }
 632 
 633       else if(lp_modules[b->type]->modvars[lp_param_name(b->type, b->params[c]->name)].type
 634          != PTYPE(b->params[c])) {
 635 
 636         /* implicitly convert ints to doubles */
 637         if((lp_modules[b->type]->modvars[lp_param_name(b->type, b->params[c]->name)].type
 638             == D)
 639            && (PTYPE(b->params[c]) == I)) {
 640           b->params[c]->v->t = D;
 641           DVAL(b->params[c]) = (double) IVAL(b->params[c]);
 642         }
 643           
 644         else { 
 645           fprintf(stderr, "*** error: type error: %s::\"%s\" cannot take type ",
 646                   lp_modules[b->type]->name,
 647                   b->params[c]->name);
 648           
 649 /*        unparse_type(PTYPE(b->params[c]), stderr); */
 650           fprintf(stderr, "\n");
 651 
 652           die("check_types() failed");
 653         }
 654       }
 655       
 656 
 657 
 658       if(PTYPE(b->params[c]) >= BLOCK) {
 659         check_types(BVAL(b->params[c]));
 660       }
 661       else if(PTYPE(b->params[c]) == LIST) {
 662         int d;
 663         struct lp_list *l = LVAL(b->params[c]);
 664         for(d = 0; d < l->values_len; d++) {
 665           if(l->values[d]) {
 666             if(l->values[d]->t >= BLOCK) {
 667               check_types(l->values[d]->v.b);
 668             }
 669           }
 670         }
 671       }
 672 
 673     }
 674   }
 675   return 0;
 676 }

 700   for(c = 0; c < lp_max_mod; c++) {

 920 int lp_add_param(struct lp_param ***b, int *plen,
 921                  struct lp_param *p)
 922 {
 923   int c;
 924 
 925   /* look for dupe params */
 926   for(c = 0; c < *plen; c++) {
 927     if(!(*b)[c]) continue;
 928     if(!strcmp((*b)[c]->name, p->name)) {
 929       fprintf(stderr, "*** error: redefined %s\n", p->name);
 930       return -1;
 931     }
 932   }
 933 
 934 
 935   for(c = 0; c < *plen; c++) {
 936     if(!(*b)[c]) {
 937       (*b)[c] = p;
 938       break;
 939     }
 940   }
 941   if(c == *plen) {
 942     /* didn't find a free slot -- double the array */
 943     int newlen = 2 * (*plen) + 1;
 944     (*b) = realloc((*b), newlen * sizeof(int *));
 945     bzero((int *)(*b) + *plen, ((*plen) + 1) * sizeof(int*));
 946     (*b)[(*plen)] = p;
 947     *plen = newlen;
 948   }
 949 
 950   return 0;
 951 }

1016 /* splits s into a trailing number (i) and the leading part */
1017 int dumb_split(char *s, char **t, int *i) {
1018   int c = strlen(s) - 1;
1019   
1020   while(c && isdigit(s[c])) c--;
1021   if(!c) return -1;
1022   c++;
1023   (*i) = atoi(&s[c]);
1024   *t = strdup(s);
1025   (*t)[c] = 0;
1026   return 0;
1027 }
1028 
1029 /* separate 'foo:bar' into 'foo' and 'bar' */
1030 int dumb_split2(char *s, char **s1, char **s2) {
1031   int c = 0;
1032   (*s1) = strdup(s);
1033   while(s[c] && (s[c] != ':')) c++;
1034   (*s1)[c] = 0;
1035   if(s[c]) (*s2) = strdup(s + c + 1);
1036   else return -1;
1037 
1038   return 0;
1039 }

1045 static int param_override(struct lp_block *b, 
1046                           char *pname,
1047                           char *pval) 
1048 {
1049   int c;
1050   
1051   for(c = 0; c < b->params_len; c++) {
1052     if(!b->params[c]) continue;
1053     if(!strcmp(b->params[c]->name, pname)) {
1054       switch(b->params[c]->v->t) {
1055       case I:
1056         b->params[c]->v->v.i = atoi(pval);
1057         break;
1058       case D:
1059         b->params[c]->v->v.d = atof(pval);
1060         break;
1061       case S:
1062         free(b->params[c]->v->v.s);
1063         b->params[c]->v->v.s = strdup(pval);
1064         return 0;
1065         break;
1066 
1067       default:
1068         ddbg_assert(0);
1069       }
1070 
1071       return -1;
1072     } 
1073   }
1074 
1075   
1076   return -1;
1077 }

1081 int range_match(char *range, char *name) {
1082   char *base1, *base2, *base3;
1083   char r1[128], r2[128];
1084   int i1, i2, i3;
1085 
1086   if(!strcmp(range, name)) { return 1; }
1087 
1088   if(!strcmp(range, "*")) return 1;
1089 
1090   if(sscanf(range, "%s .. %s", r1, r2) != 2) { 
1091     char *prefix; int junk;
1092 
1093     dumb_split(name, &prefix, &junk);
1094 
1095     /* i.e. driver* matches driver2 and driver and driver2344 but not
1096      * driverqux 
1097      */
1098     if((strlen(range) == ( strlen(prefix) + 1)) &&
1099        !strncmp(range,prefix,strlen(prefix)) &&
1100        (range[strlen(prefix)] == '*')) return 1;
1101 
1102     return 0; 
1103   }
1104 
1105   
1106   dumb_split(r1, &base1, &i1);
1107   dumb_split(r2, &base2, &i2);
1108   if(strcmp(base1, base2) || (i1 < 0) || (i2 < i1)) {
1109     fprintf(stderr, "*** error: bad range \"%s .. %s\"\n", r1, r2);
1110     return 0;
1111   }
1112 
1113   dumb_split(name, &base3, &i3);
1114   if(!strcmp(base3, base2) && (i3 >= i1) && (i3 <= i2)) {
1115     return 1;
1116   }
1117   else {
1118     return 0;
1119   }
1120 }

1123 /* instantiate a component with overrides 
1124  * tname type to instantiate,
1125  * cname name of instantiated component
1126  * loader is the module loader function for tname
1127  */
1128 int *lp_override_inst(struct lp_block *spec, 
1129                   char *cname, 
1130                   lp_modloader_t loader,
1131                   char **overrides,
1132                   int overrides_len)
1133 {
1134   int c, d;
1135   struct lp_block *spec_copy;
1136   char *p1, *p2; 
1137   int *result;
1138   
1139   spec_copy = copy_block(spec);
1140 
1141   for(c = 0; c < overrides_len; c += 3) {
1142     if(range_match(overrides[c], cname)) {
1143       
1144       /* overrides[c+2] could be an int, a real, a string or a list or
1145        * a block.  need parser to deal with lists and blocks
1146        * reasonably so we aren't going to deal with them here */
1147      
1148       if(!dumb_split2(overrides[c+1], &p1, &p2)) {
1149         /* descend hierarchy */
1150         for(d = 0; d < spec_copy->params_len; d++) {
1151           if(!spec_copy->params[d]) continue;
1152           if(!strcmp(p1, spec_copy->params[d]->name)) {
1153             if(spec_copy->params[d]->v->t != BLOCK) {
1154               fprintf(stderr, "*** error: tried to recurse through non-block parameter.\n");
1155               return 0;
1156             }
1157             else {
1158               param_override(spec_copy->params[d]->v->v.b, p2, 
1159                              overrides[c+2]);
1160             }
1161           }
1162         }
1163 
1164       }
1165       else {
1166         param_override(spec_copy, 
1167                        overrides[c+1], 
1168                        overrides[c+2]);
1169         
1170       }
1171     }
1172   }
1173 
1174 
1175 
1176   if(!check_types(spec_copy)) {
1177     result = loader(spec_copy, 0);
1178   }
1179 
1180   // XXX don't leak (segfaults)
1181   //  destroy_block(spec_copy);
1182   return result; 
1183 }

1273 int
1274 lp_loadparams(void *it, struct lp_block *b, struct lp_mod *m) {
1275   int c;
1276   int needed = 0;
1277   int param_stack[LP_STACK_MAX];
1278   int stack_ptr = 0;  // index of first free slot
1279   // XXX not static size
1280   char *paramvec = calloc(m->modvars_len, sizeof(char));
1281 
1282 
1283   // This is pretty gross; there should be a better solution.
1284   // dirname() munges its operand.  Its result may also be a static
1285   // buffer somewhere, hence the 2 copies.
1286   char *tmp = strdup(b->source_file);
1287   lp_cwd = strdup(dirname(tmp));
1288   //  free(tmp);
1289  
1290   for(c = 0; c < b->params_len; c++) {
1291     int pnum, deps;
1292     
1293     if(!b->params[c]) continue;
1294     
1295   TOP:
1296     pnum = lp_param_name(lp_mod_name(m->name), b->params[c]->name);
1297     
1298     // Don't initialize things more than once.
1299     // Should warn here, probably.
1300     if(BIT_TEST(paramvec, pnum)) continue;
1301     
1302     
1303     if(stack_ptr > 0) {
1304       for(c = 0; c < b->params_len; c++) {
1305         if(lp_param_name(lp_mod_name(m->name), b->params[c]->name) == needed)
1306           goto FOUND;
1307       }
1308       break;
1309     }
1310 
1311 
1312   FOUND:
1313   
1314     deps = m->param_deps[pnum](paramvec);
1315     if(deps > -1) {
1316       ddbg_assert(stack_ptr < LP_STACK_MAX);
1317       param_stack[stack_ptr++] = c;
1318       needed = deps;
1319       continue;
1320     }
1321     else {
1322       switch(PTYPE(b->params[c])) {
1323       case I:     
1324         ((lp_paramloader_int)m->param_loaders[pnum])(it, IVAL(b->params[c])); 
1325         break; 
1326       case D:     
1327         ((lp_paramloader_double)m->param_loaders[pnum])(it, DVAL(b->params[c])); 
1328         break; 
1329       case S:     
1330         ((lp_paramloader_string)m->param_loaders[pnum])(it, SVAL(b->params[c])); 
1331         break; 
1332       case LIST:  
1333         ((lp_paramloader_list)m->param_loaders[pnum])(it, LVAL(b->params[c])); 
1334          break; 
1335          
1336       case BLOCK:
1337       default:    
1338         ((lp_paramloader_block)m->param_loaders[pnum])(it, BVAL(b->params[c])); 
1339         break; 
1340       }
1341     }
1342 
1343     BIT_SET(paramvec, pnum);
1344     if(stack_ptr > 0) { 
1345       c = param_stack[--stack_ptr]; 
1346       goto TOP;  
1347     }
1348   }
1349 
1350   for(c = 0; c < m->modvars_len; c++) {
1351     if(m->modvars[c].req && !BIT_TEST(paramvec,c)) {
1352       fprintf(stderr, "*** error: in %s spec -- missing required parameter %s\n", m->name, m->modvars[c].name);
1353       break;
1354     }
1355   }
1356 
1357   free(paramvec);
1358 
1359   return 0; // ???
1360 }
