 292 // a module description
 293 struct lp_mod {
 294   // unique; used in input files
 295   char *name; 
 296 
 297   // the variable specs for this module
 298   struct lp_varspec *modvars;
 299   int modvars_len;
 300   
 301   // loader function
 302   lp_modloader_t fn;
 303 
 304   // when we successfully instantiate an object of this type, 
 305   // we will call <callback>(ctx, ptr) where ptr is a pointer to the object
 306   // why not do this in the loader?  because the loader might come from
 307   // libdiskmodel and as a user of it, you want to do something with
 308   // each dm_disk without modifying the dm code itself.
 309   lp_load_callback callback;
 310   void *ctx;
 311   
 312   // array of per-parameter loader functions
 313   // index with foo_mod_t
 314   void **param_loaders;
 315   lp_paramdep_t *param_deps;
 316 };
