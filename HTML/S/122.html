 103 // if we're using remotely recent gcc, we can use __PRETTY_FUNCTION__
 104 // if we're using C99, we can use __func__

 122 /* like assert() modulo above */
 123 #ifndef _DDBG_DISABLE_ASSERTIONS
 124 #define ddbg_assert(cond) do { int _ltassert_cond = (int)(cond); \
 125 if(!_ltassert_cond) { \
 126 ddbg_assert_msg(__FILE__, __LINE__, #cond, __func__, ""); \
 127 ddbg_assert_fail(__FILE__, __LINE__, #cond, __func__, ""); \
 128 } } while(0)
 129 #else
 130 #define ddbg_assert(cond) 
 131 #endif
 132 
 133 /* print a message */
 134 #ifndef _DDBG_DISABLE_ASSERTIONS
 135 #define ddbg_assert2(cond,fmt) do { int _ltassert_cond = (int)(cond); \
 136 if(!_ltassert_cond) { \
 137 ddbg_assert_msg(__FILE__, __LINE__, #cond, __func__, fmt); \
 138 ddbg_assert_fail(__FILE__, __LINE__, #cond, __func__, fmt); \
 139 } } while(0)
 140 #else
 141 #define ddbg_assert2(x,y)
 142 #endif
 143 
 144 
 145 /* invoke by e.g. 
 146  *   ddbg_assert3(cond, ("foo %d", 3));
 147  */
 148 
 149 #ifndef _DDBG_DISABLE_ASSERTIONS
 150 #define ddbg_assert3(cond,msg) do { int _ltassert_cond = (int)(cond);\
 151 if(!_ltassert_cond) { \
 152  ddbg_assert_msg(__FILE__, __LINE__, #cond, __func__, ""); \
 153  ddbg_assert_printmsg msg; \
 154  ddbg_assert_fail(__FILE__, __LINE__, #cond, __func__, ""); \
 155 } } while(0)
 156 #else
 157 #define ddbg_assert3(x,y)
 158 #endif
