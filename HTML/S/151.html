  26 /*

  27  * calculates the cost of reading and writing a block of data across planes.

  28  */
  29 static double ssd_crossover_cost
  30 (ssd_t *s, ssd_element_metadata *metadata, int from_blk, int to_blk)
  31 {
  32     if (ssd_same_plane_blocks(s, metadata, from_blk, to_blk)) {
  33         return 0;
  34     } else {
  35         double xfer_cost;
  36 
  37         // we need to read and write back across the pins

  38         xfer_cost = ssd_data_transfer_cost(s, s->params.page_size);
  39         return (2 * xfer_cost);
  40     }
  41 }

  43 /*

  44  * writes a page to the current active page. if there is no active page,

  45  * allocate one and then move.

  46  */
  47 static double ssd_move_page(int lpn, int from_blk, int plane_num, int elem_num, ssd_t *s)
  48 {
  49     double cost = 0;
  50     ssd_element_metadata *metadata = &s->elements[elem_num].metadata;
  51 
  52     switch(s->params.copy_back) {
  53         case SSD_COPY_BACK_DISABLE:
  54             if (ssd_last_page_in_block(metadata->active_page, s)) {
  55                 _ssd_alloc_active_block(-1, elem_num, s);
  56             }
  57             break;
  58 
  59         case SSD_COPY_BACK_ENABLE:
  60             ASSERT(metadata->plane_meta[plane_num].active_page == metadata->active_page);
  61             if (ssd_last_page_in_block(metadata->active_page, s)) {
  62                 _ssd_alloc_active_block(plane_num, elem_num, s);
  63             }
  64             break;
  65 
  66         default:
  67             fprintf(stderr, "Error: invalid copy back policy %d\n",
  68                 s->params.copy_back);
  69             exit(1);
  70     }
  71 
  72     cost += _ssd_write_page_osr(s, metadata, lpn);
  73 
  74     return cost;
  75 }

  77 /*

  78  * reads the data out of a page and writes it back the active page.

  79  */
  80 static double ssd_clean_one_page
  81 (int lp_num, int pp_index, int blk, int plane_num, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
  82 {
  83     double cost = 0;
  84     double xfer_cost = 0;
  85 
  86     cost += s->params.page_read_latency;
  87     cost += ssd_move_page(lp_num, blk, plane_num, elem_num, s);
  88 
  89     // if the write is within the same plane, then the data need

  90     // not cross the pins. but if not, add the cost of transferring

  91     // the bytes across the pins

  92     xfer_cost = ssd_crossover_cost(s, metadata, blk, SSD_PAGE_TO_BLOCK(metadata->active_page, s));
  93 
  94     cost += xfer_cost;
  95 
  96     ssd_assert_valid_pages(plane_num, metadata, s);
  97     ASSERT(metadata->block_usage[blk].page[pp_index] == -1);
  98 
  99     // stat -- we move 'valid_pages' out of this block

 100     s->elements[elem_num].stat.pages_moved ++;
 101     s->elements[elem_num].stat.tot_xfer_cost += xfer_cost;
 102 
 103     return cost;
 104 }

 106 /*

 107  * update the remaining life time and time of last erasure

 108  * for this block.

 109  * we did some operations since the last update to

 110  * simtime variable and the time took for these operations are

 111  * in the 'cost' variable. so the time of last erasure is

 112  * cost + the simtime

 113  */
 114 void ssd_update_block_lifetime(double time, int blk, ssd_element_metadata *metadata)
 115 {
 116     metadata->block_usage[blk].rem_lifetime --;
 117     metadata->block_usage[blk].time_of_last_erasure = time;
 118 
 119     if (metadata->block_usage[blk].rem_lifetime < 0) {
 120         fprintf(stderr, "Error: Negative lifetime %d (block is being erased after it's dead)\n",
 121             metadata->block_usage[blk].rem_lifetime);
 122         ASSERT(0);
 123         exit(1);
 124     }
 125 }

 128  * updates the status of erased blocks

 130 void ssd_update_free_block_status(int blk, int plane_num, ssd_element_metadata *metadata, ssd_t *s)
 131 {
 132     int bitpos;
 133 
 134     // clear the bit corresponding to this block in the

 135     // free blocks list for future use

 136     bitpos = ssd_block_to_bitpos(s, blk);
 137     ssd_clear_bit(metadata->free_blocks, bitpos);
 138     metadata->block_usage[blk].state = SSD_BLOCK_CLEAN;
 139     metadata->block_usage[blk].bsn = 0;
 140     metadata->tot_free_blocks ++;
 141     metadata->plane_meta[plane_num].free_blocks ++;
 142     ssd_assert_free_blocks(s, metadata);
 143 
 144     // there must be no valid pages in the erased block

 145     ASSERT(metadata->block_usage[blk].num_valid == 0);
 146     ssd_assert_valid_pages(plane_num, metadata, s);
 147 }

 192 /*

 193  * computes the average lifetime of all the blocks in a plane.

 194  */
 195 double ssd_compute_avg_lifetime_in_plane(int plane_num, int elem_num, ssd_t *s)
 196 {
 197     int i;
 198     int bitpos;
 199     double tot_lifetime = 0;
 200     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 201 
 202     bitpos = plane_num * s->params.blocks_per_plane;
 203     for (i = bitpos; i < bitpos + (int)s->params.blocks_per_plane; i ++) {
 204         int block = ssd_bitpos_to_block(i, s);
 205         ASSERT(metadata->block_usage[block].plane_num == plane_num);
 206         tot_lifetime += metadata->block_usage[block].rem_lifetime;
 207     }
 208 
 209     return (tot_lifetime / s->params.blocks_per_plane);
 210 }

 217 {
 218     double tot_lifetime = 0;

 220     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);

 304 static int _ssd_pick_block_to_clean
 305 (int blk, int plane_num, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
 306 {
 307     int block_life;
 308 
 309     if (plane_num != -1) {
 310         if (metadata->block_usage[blk].plane_num != plane_num) {
 311             return 0;
 312         }
 313     }
 314 
 315     block_life = metadata->block_usage[blk].rem_lifetime;
 316 
 317     // if the block is already dead, skip it

 318     if (block_life == 0) {
 319         return 0;
 320     }
 321 
 322     // clean only those blocks that are sealed.

 323     if (!ssd_can_clean_block(s, metadata, blk)) {
 324         return 0;
 325     }
 326 
 327     return 1;
 328 }

 330 /*

 331  * migrate data from a cold block to "to_blk"

 332  */
 333 int ssd_migrate_cold_data(int to_blk, double *mcost, int plane_num, int elem_num, ssd_t *s)
 334 {
 335     int i;
 336     int from_blk = -1;
 337     double oldest_erase_time = simtime;
 338     double cost = 0;
 339     int bitpos;
 340 
 341 #if SSD_ASSERT_ALL
 342     int f1;
 343     int f2;
 344 #endif
 345 
 346     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 347 
 348     // first select the coldest of all blocks.

 349     // one way to select is to find the one that has the oldest

 350     // erasure time.

 351     if (plane_num == -1) {
 352         for (i = 0; i < s->params.blocks_per_element; i ++) {
 353             if (metadata->block_usage[i].num_valid > 0) {
 354                 if (metadata->block_usage[i].time_of_last_erasure < oldest_erase_time) {
 355                     oldest_erase_time = metadata->block_usage[i].time_of_last_erasure;
 356                     from_blk = i;
 357                 }
 358             }
 359         }
 360     } else {
 361 
 362 #if SSD_ASSERT_ALL
 363         f1 = ssd_free_bits(plane_num, elem_num, metadata, s);
 364         ASSERT(f1 == metadata->plane_meta[metadata->block_usage[to_blk].plane_num].free_blocks);
 365 #endif
 366 
 367         bitpos = plane_num * s->params.blocks_per_plane;
 368         for (i = bitpos; i < bitpos + (int)s->params.blocks_per_plane; i ++) {
 369             int block = ssd_bitpos_to_block(i, s);
 370             ASSERT(metadata->block_usage[block].plane_num == plane_num);
 371 
 372             if (metadata->block_usage[block].num_valid > 0) {
 373                 if (metadata->block_usage[block].time_of_last_erasure < oldest_erase_time) {
 374                     oldest_erase_time = metadata->block_usage[block].time_of_last_erasure;
 375                     from_blk = block;
 376                 }
 377             }
 378         }
 379     }
 380 
 381     ASSERT(from_blk != -1);
 382     if (plane_num != -1) {
 383         ASSERT(metadata->block_usage[from_blk].plane_num == metadata->block_usage[to_blk].plane_num);
 384     }
 385 
 386     // next, clean the block to which we'll transfer the

 387     // cold data

 388     cost += _ssd_clean_block_fully(to_blk, metadata->block_usage[to_blk].plane_num, elem_num, metadata, s);
 389 
 390 #if SSD_ASSERT_ALL
 391     if (plane_num != -1) {
 392         f2 = ssd_free_bits(plane_num, elem_num, metadata, s);
 393         ASSERT(f2 == metadata->plane_meta[metadata->block_usage[to_blk].plane_num].free_blocks);
 394     }
 395 #endif
 396 
 397     // then, migrate the cold data to the worn out block.

 398     // for which, we first read all the valid data

 399     cost += metadata->block_usage[from_blk].num_valid * s->params.page_read_latency;
 400     // include the write cost

 401     cost += metadata->block_usage[from_blk].num_valid * s->params.page_write_latency;
 402     // if the src and dest blocks are on different planes

 403     // include the transfer cost also

 404     cost += ssd_crossover_cost(s, metadata, from_blk, to_blk);
 405 
 406     // the cost of erasing the cold block (represented by from_blk)

 407     // will be added later ...

 408 
 409     // finally, update the metadata

 410     metadata->block_usage[to_blk].bsn = metadata->block_usage[from_blk].bsn;
 411     metadata->block_usage[to_blk].num_valid = metadata->block_usage[from_blk].num_valid;
 412     metadata->block_usage[from_blk].num_valid = 0;
 413 
 414     for (i = 0; i < s->params.pages_per_block; i ++) {
 415         int lpn = metadata->block_usage[from_blk].page[i];
 416         if (lpn != -1) {
 417             ASSERT(metadata->lba_table[lpn] == (from_blk * s->params.pages_per_block + i));
 418             metadata->lba_table[lpn] = to_blk * s->params.pages_per_block + i;
 419         }
 420         metadata->block_usage[to_blk].page[i] = metadata->block_usage[from_blk].page[i];
 421     }
 422     metadata->block_usage[to_blk].state = metadata->block_usage[from_blk].state;
 423 
 424     bitpos = ssd_block_to_bitpos(s, to_blk);
 425     ssd_set_bit(metadata->free_blocks, bitpos);
 426     metadata->tot_free_blocks --;
 427     metadata->plane_meta[metadata->block_usage[to_blk].plane_num].free_blocks --;
 428 
 429 #if SSD_ASSERT_ALL
 430     if (plane_num != -1) {
 431         f2 = ssd_free_bits(plane_num, elem_num, metadata, s);
 432         ASSERT(f2 == metadata->plane_meta[metadata->block_usage[to_blk].plane_num].free_blocks);
 433     }
 434 #endif
 435 
 436     ssd_assert_free_blocks(s, metadata);
 437     ASSERT(metadata->block_usage[from_blk].num_valid == 0);
 438 
 439 #if ASSERT_FREEBITS
 440     if (plane_num != -1) {
 441         f2 = ssd_free_bits(plane_num, elem_num, metadata, s);
 442         ASSERT(f1 == f2);
 443     }
 444 #endif
 445 
 446     *mcost = cost;
 447 
 448     // stat

 449     metadata->tot_migrations ++;
 450     metadata->tot_pgs_migrated += metadata->block_usage[to_blk].num_valid;
 451     metadata->mig_cost += cost;
 452 
 453     return from_blk;
 454 }

 466

 476     }
 477

 481 /*

 482  * a greedy solution, where we find the block in a plane with the least

 483  * num of valid pages and return it.

 484  */
 485 static int ssd_pick_block_to_clean2(int plane_num, int elem_num, double *mcost, ssd_element_metadata *metadata, ssd_t *s)
 486 {
 487     double avg_lifetime = 1;
 488     int i;
 489     int size;
 490     int block = -1;
 491     int min_valid = s->params.pages_per_block - 1; // one page goes for the summary info

 492     listnode *greedy_list;
 493 
 494     *mcost = 0;
 495 
 496     // find the average life time of all the blocks in this element

 497     avg_lifetime = ssd_compute_avg_lifetime(plane_num, elem_num, s);
 498 
 499     // we create a list of greedily selected blocks

 500     ll_create(&greedy_list);
 501     for (i = 0; i < s->params.blocks_per_element; i ++) {
 502         if (_ssd_pick_block_to_clean(i, plane_num, elem_num, metadata, s)) {
 503 
 504             // greedily select the block

 505             if (metadata->block_usage[i].num_valid <= min_valid) {
 506                 ASSERT(i == metadata->block_usage[i].block_num);
 507                 ll_insert_at_head(greedy_list, (void*)&metadata->block_usage[i]);
 508                 min_valid = metadata->block_usage[i].num_valid;
 509                 block = i;
 510             }
 511         }
 512     }
 513 
 514     ASSERT(block != -1);
 515     block = -1;
 516 
 517     // from the greedily picked blocks, select one after rate

 518     // limiting the overly used blocks

 519     size = ll_get_size(greedy_list);
 520 
 521     //printf("plane %d elem %d size %d avg lifetime %f\n",

 522     //  plane_num, elem_num, size, avg_lifetime);

 523 
 524     for (i = 0; i < size; i ++) {
 525         block_metadata *bm;
 526         int mig_blk;
 527 
 528         listnode *n = ll_get_nth_node(greedy_list, i);
 529         bm = ((block_metadata *)n->data);
 530 
 531         if (i == 0) {
 532             ASSERT(min_valid == bm->num_valid);
 533         }
 534 
 535         // this is the last of the greedily picked blocks.

 536         if (i == size -1) {
 537             // select it!

 538             block = bm->block_num;
 539             break;
 540         }
 541 
 542         if (s->params.cleaning_policy == DISKSIM_SSD_CLEANING_POLICY_GREEDY_WEAR_AGNOSTIC) {
 543             block = bm->block_num;
 544             break;
 545         } else {
 546 #if MIGRATE
 547             // migration

 548             mig_blk = ssd_pick_wear_aware_with_migration(bm->block_num, bm->rem_lifetime, avg_lifetime, mcost, bm->plane_num, elem_num, s);
 549             if (mig_blk != bm->block_num) {
 550                 // data has been migrated and we have a new

 551                 // block to use

 552                 block = mig_blk;
 553                 break;
 554             }
 555 #endif
 556 
 557             // pick this block giving consideration to its life time

 558             if (ssd_pick_wear_aware(bm->block_num, bm->rem_lifetime, avg_lifetime, s)) {
 559                 block = bm->block_num;
 560                 break;
 561             }
 562         }
 563     }
 564 
 565     ll_release(greedy_list);
 566 
 567     ASSERT(block != -1);
 568     return block;
 569 }

 571 static int ssd_pick_block_to_clean1(int plane_num, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
 572 {
 573     int i;
 574     int block = -1;
 575     int min_valid = s->params.pages_per_block - 1; // one page goes for the summary info

 576 
 577     for (i = 0; i < s->params.blocks_per_element; i ++) {
 578         if (_ssd_pick_block_to_clean(i, plane_num, elem_num, metadata, s)) {
 579             if (metadata->block_usage[i].num_valid < min_valid) {
 580                 min_valid = metadata->block_usage[i].num_valid;
 581                 block = i;
 582             }
 583         }
 584     }
 585 
 586     if (block != -1) {
 587         return block;
 588     } else {
 589         fprintf(stderr, "Error: we cannot find a block to clean in plane %d\n", plane_num);
 590         ASSERT(0);
 591         exit(1);
 592     }
 593 }

 600 /*

 601  * this routine cleans one page at a time in a block. if all the

 602  * valid pages are moved, then the block is erased.

 603  */
 604 static double _ssd_clean_block_partially(int plane_num, int elem_num, ssd_t *s)
 605 {
 606     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 607     plane_metadata *pm = &metadata->plane_meta[plane_num];
 608     double cost = 0;
 609     int block;
 610     int i;
 611 
 612     ASSERT(pm->clean_in_progress);
 613     block = pm->clean_in_block;
 614 
 615     if (metadata->block_usage[block].num_valid > 0) {
 616         // pick a page that is not yet cleaned and move it

 617         for (i = 0; i < s->params.pages_per_block; i ++) {
 618             int lp_num = metadata->block_usage[block].page[i];
 619 
 620             if (lp_num != -1) {
 621                 cost += ssd_clean_one_page(lp_num, i, block, plane_num, elem_num, metadata, s);
 622                 break;
 623             }
 624         }
 625     }
 626 
 627     // if we've moved all the valid pages out of this block,

 628     // then we're done with it. so, erase it and update the

 629     // system state.

 630     if (metadata->block_usage[block].num_valid == 0) {
 631         // add the cost of erasing the block

 632         cost += s->params.block_erase_latency;
 633 
 634         ssd_update_free_block_status(block, plane_num, metadata, s);
 635         ssd_update_block_lifetime(simtime+cost, block, metadata);
 636         pm->clean_in_progress = 0;
 637         pm->clean_in_block = -1;
 638     }
 639 
 640     return cost;
 641 }

 643 double ssd_clean_block_partially(int plane_num, int elem_num, ssd_t *s)
 644 {
 645     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 646     plane_metadata *pm = &metadata->plane_meta[plane_num];
 647     double cost = 0;
 648     double mcost = 0;
 649 
 650     // see if we've already started the cleaning

 651     if (!pm->clean_in_progress) {
 652         // pick a block to be cleaned

 653         pm->clean_in_block = ssd_pick_block_to_clean(plane_num, elem_num, &mcost, metadata, s);
 654         pm->clean_in_progress = 1;
 655     }
 656 
 657     // yes, the cleaning on this plane is already initiated

 658     ASSERT(pm->clean_in_block != -1);
 659     cost = _ssd_clean_block_partially(plane_num, elem_num, s);
 660 
 661     return (cost+mcost);
 662 }

 665 /*

 666  * this routine cleans one block by reading all its valid

 667  * pages and writing them to the current active block. the

 668  * cleaned block is also erased.

 669  */
 670 double _ssd_clean_block_fully(int blk, int plane_num, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
 671 {
 672     double cost = 0;
 673     plane_metadata *pm = &metadata->plane_meta[plane_num];
 674 
 675     ASSERT((pm->clean_in_progress == 0) && (pm->clean_in_block = -1));
 676     pm->clean_in_block = blk;
 677     pm->clean_in_progress = 1;
 678 
 679     // stat

 680     pm->num_cleans ++;
 681     s->elements[elem_num].stat.num_clean ++;
 682 
 683     do {
 684         cost += _ssd_clean_block_partially(plane_num, elem_num, s);
 685     } while (metadata->block_usage[blk].num_valid > 0);
 686 
 687     return cost;
 688 }

 690 static double ssd_clean_block_fully(int plane_num, int elem_num, ssd_t *s)
 691 {
 692     int blk;
 693     double cost = 0;
 694     double mcost = 0;
 695     ssd_element_metadata *metadata = &s->elements[elem_num].metadata;
 696     plane_metadata *pm = &metadata->plane_meta[plane_num];
 697 
 698     ASSERT((pm->clean_in_progress == 0) && (pm->clean_in_block = -1));
 699 
 700     blk = ssd_pick_block_to_clean(plane_num, elem_num, &mcost, metadata, s);
 701     ASSERT(metadata->block_usage[blk].plane_num == plane_num);
 702 
 703     cost = _ssd_clean_block_fully(blk, plane_num, elem_num, metadata, s);
 704     return (cost+mcost);
 705 }

 708 static usage_table *ssd_build_usage_table(int elem_num, ssd_t *s)
 709 {
 710     int i;
 711     usage_table *table;
 712     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 713 
 714     //////////////////////////////////////////////////////////////////////////////

 715     // allocate the hash table. it has (pages_per_block + 1) entries,

 716     // one entry for values from 0 to pages_per_block.

 717     table = (usage_table *) malloc(sizeof(usage_table) * (s->params.pages_per_block + 1));
 718     memset(table, 0, sizeof(usage_table) * (s->params.pages_per_block + 1));
 719 
 720     // find out how many blocks have a particular no of valid pages

 721     for (i = 0; i < s->params.blocks_per_element; i ++) {
 722         int usage = metadata->block_usage[i].num_valid;
 723         table[usage].len ++;
 724     }
 725 
 726     // allocate space to store the block numbers

 727     for (i = 0; i <= s->params.pages_per_block; i ++) {
 728         table[i].block = (int*)malloc(sizeof(int)*table[i].len);
 729     }
 730 
 731     /////////////////////////////////////////////////////////////////////////////

 732     // fill in the block numbers in their appropriate 'usage' buckets

 733     for (i = 0; i < s->params.blocks_per_element; i ++) {
 734         usage_table *entry;
 735         int usage = metadata->block_usage[i].num_valid;
 736 
 737         entry = &(table[usage]);
 738         entry->block[entry->temp ++] = i;
 739     }
 740 
 741     return table;
 742 }
 743 
 744 static void ssd_release_usage_table(usage_table *table, ssd_t *s)
 745 {
 746     int i;
 747 
 748     for (i = 0; i <= s->params.pages_per_block; i ++) {
 749         free(table[i].block);
 750     }
 751 
 752     // release the table

 753     free(table);
 754 }

 756 /*

 757  * pick a random block with at least 1 empty page slot and clean it

 758  */
 759 static double ssd_clean_blocks_random(int plane_num, int elem_num, ssd_t *s)
 760 {
 761     double cost = 0;
 762 #if 1
 763     printf("ssd_clean_blocks_random: not yet fixed\n");
 764     exit(1);
 765 #else
 766     long blk = 0;
 767     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 768 
 769     do {
 770         // get a random number to select a block

 771         blk = DISKSIM_lrand48() % s->params.blocks_per_element;
 772 
 773         // if this is plane specific cleaning, then skip all the

 774         // blocks that don't belong to this plane.

 775         if ((plane_num != -1) && (metadata->block_usage[blk].plane_num != plane_num)) {
 776             continue;
 777         }
 778 
 779         // clean only those blocks that are used.

 780         if (ssd_can_clean_block(s, metadata, blk)) {
 781 
 782             int valid_pages = metadata->block_usage[blk].num_valid;
 783 
 784             // if all the pages in the block are valid, continue to

 785             // select another random block

 786             if (valid_pages == s->params.pages_per_block) {
 787                 continue;
 788             } else {
 789                 // invoke cleaning until we reach the high watermark

 790                 cost += _ssd_clean_block_fully(blk, elem_num, metadata, s);
 791 
 792                 if (ssd_stop_cleaning(plane_num, elem_num, s)) {
 793                     // we're done with creating enough free blocks. so quit.

 794                     break;
 795                 }
 796             }
 797         } else { // block is already free. so continue.

 798             continue;
 799         }
 800     } while (1);
 801 #endif
 802 
 803     return cost;
 804 }

 808 /*

 809  * first we create a hash table of blocks according to their

 810  * usage. then we select blocks with the least usage and clean

 811  * them.

 812  */
 813 static double ssd_clean_blocks_greedy(int plane_num, int elem_num, ssd_t *s)
 814 {
 815     double cost = 0;
 816     double avg_lifetime;
 817     int i;
 818     usage_table *table;
 819     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 820 
 821     /////////////////////////////////////////////////////////////////////////////

 822     // build the histogram

 823     table = ssd_build_usage_table(elem_num, s);
 824 
 825     //////////////////////////////////////////////////////////////////////////////

 826     // find the average life time of all the blocks in this element

 827     avg_lifetime = ssd_compute_avg_lifetime(plane_num, elem_num, s);
 828 
 829     /////////////////////////////////////////////////////////////////////////////

 830     // we now have a hash table of blocks, where the key of each

 831     // bucket is the usage count and each bucket has all the blocks with

 832     // the same usage count (i.e., the same num of valid pages).

 833     for (i = 0; i <= s->params.pages_per_block; i ++) {
 834         int j;
 835         usage_table *entry;
 836 
 837         // get the bucket of blocks with 'i' valid pages

 838         entry = &(table[i]);
 839 
 840         // free all the blocks with 'i' valid pages

 841         for (j = 0; j < entry->len; j ++) {
 842             int blk = entry->block[j];
 843             int block_life = metadata->block_usage[blk].rem_lifetime;
 844 
 845             // if this is plane specific cleaning, then skip all the

 846             // blocks that don't belong to this plane.

 847             if ((plane_num != -1) && (metadata->block_usage[blk].plane_num != plane_num)) {
 848                 continue;
 849             }
 850 
 851             // if the block is already dead, skip it

 852             if (block_life == 0) {
 853                 continue;
 854             }
 855 
 856             // clean only those blocks that are sealed.

 857             if (ssd_can_clean_block(s, metadata, blk)) {
 858 
 859                 // if we care about wear-leveling, then we must rate limit overly cleaned blocks

 860                 if (s->params.cleaning_policy == DISKSIM_SSD_CLEANING_POLICY_GREEDY_WEAR_AWARE) {
 861 
 862                     // see if this block's remaining lifetime is within

 863                     // a certain threshold of the average remaining lifetime

 864                     // of all blocks in this element

 865                     if (block_life < (SSD_LIFETIME_THRESHOLD_X * avg_lifetime)) {
 866                         // we have to rate limit this block as it has exceeded

 867                         // its cleaning limits

 868                         printf("Rate limiting block %d (block life %d avg life %f\n",
 869                             blk, block_life, avg_lifetime);
 870 
 871                         if (ssd_rate_limit(block_life, avg_lifetime)) {
 872                             // skip this block and go to the next one

 873                             continue;
 874                         }
 875                     }
 876                 }
 877 
 878                 // okies, finally here we're with the block to be cleaned.

 879                 // invoke cleaning until we reach the high watermark.

 880                 cost += _ssd_clean_block_fully(blk, metadata->block_usage[blk].plane_num, elem_num, metadata, s);
 881 
 882                 if (ssd_stop_cleaning(plane_num, elem_num, s)) {
 883                     // no more cleaning is required -- so quit.

 884                     break;
 885                 }
 886             }
 887         }
 888 
 889         if (ssd_stop_cleaning(plane_num, elem_num, s)) {
 890             // no more cleaning is required -- so quit.

 891             break;
 892         }
 893     }
 894 
 895     // release the table

 896     ssd_release_usage_table(table, s);
 897 
 898     // see if we were able to generate enough free blocks

 899     if (!ssd_stop_cleaning(plane_num, elem_num, s)) {
 900         printf("Yuck! we couldn't generate enough free pages in plane %d elem %d ssd %d\n",
 901             plane_num, elem_num, s->devno);
 902     }
 903 
 904     return cost;
 905 }

 907 double ssd_clean_element_no_copyback(int elem_num, ssd_t *s)
 908 {
 909     double cost = 0;
 910 
 911     if (!ssd_start_cleaning(-1, elem_num, s)) {
 912         return cost;
 913     }
 914 
 915     switch(s->params.cleaning_policy) {
 916         case DISKSIM_SSD_CLEANING_POLICY_RANDOM:
 917             cost = ssd_clean_blocks_random(-1, elem_num, s);
 918             break;
 919 
 920         case DISKSIM_SSD_CLEANING_POLICY_GREEDY_WEAR_AGNOSTIC:
 921         case DISKSIM_SSD_CLEANING_POLICY_GREEDY_WEAR_AWARE:
 922             cost = ssd_clean_blocks_greedy(-1, elem_num, s);
 923             break;
 924 
 925         default:
 926             fprintf(stderr, "Error: invalid cleaning policy %d\n",
 927                 s->params.cleaning_policy);
 928             exit(1);
 929     }
 930 
 931     return cost;
 932 }

 956 /*

 957  * 1. find a plane to clean in each of the parallel unit

 958  * 2. invoke copyback cleaning on all such planes simultaneously

 959  */
 960 double ssd_clean_element_copyback(int elem_num, ssd_t *s)
 961 {
 962     // this means that some (or all) planes require cleaning

 963     int clean_req = 0;
 964     int i;
 965     double max_cleaning_cost = 0;
 966     int plane_to_clean[SSD_MAX_PARUNITS_PER_ELEM];
 967     ssd_element_metadata *metadata = &s->elements[elem_num].metadata;
 968     int tot_cleans = 0;
 969 
 970     for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 971         if ((plane_to_clean[i] = ssd_start_cleaning_parunit(i, elem_num, s)) != -1) {
 972             clean_req = 1;
 973         }
 974     }
 975 
 976     if (clean_req) {
 977         for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 978             double cleaning_cost = 0;
 979             int plane_num = plane_to_clean[i];
 980 
 981             if (plane_num == -1) {
 982                 // don't force cleaning

 983                 continue;
 984             }
 985 
 986             if (metadata->plane_meta[plane_num].clean_in_progress) {
 987                 metadata->plane_meta[plane_num].clean_in_progress = 0;
 988                 metadata->plane_meta[plane_num].clean_in_block = -1;
 989             }
 990 
 991             metadata->active_page = metadata->plane_meta[plane_num].active_page;
 992             cleaning_cost = ssd_clean_plane_copyback(plane_num, elem_num, s);
 993 
 994             tot_cleans ++;
 995 
 996             if (max_cleaning_cost < cleaning_cost) {
 997                 max_cleaning_cost = cleaning_cost;
 998             }
 999         }
1000     }
1001 
1002     return max_cleaning_cost;
1003 }

1016 int ssd_next_plane_in_parunit(int plane_num, int parunit_num, int elem_num, ssd_t *s)
1017 {
1018     return (parunit_num*SSD_PLANES_PER_PARUNIT(s) + (plane_num+1)%SSD_PLANES_PER_PARUNIT(s));

1021 int ssd_start_cleaning_parunit(int parunit_num, int elem_num, ssd_t *s)
1022 {
1023     int i;
1024     int start;
1025 
1026     start = s->elements[elem_num].metadata.parunits[parunit_num].plane_to_clean;
1027     i = start;
1028     do {
1029         ASSERT(parunit_num == s->elements[elem_num].metadata.plane_meta[i].parunit_num);
1030         if (ssd_start_cleaning(i, elem_num, s)) {
1031             s->elements[elem_num].metadata.parunits[parunit_num].plane_to_clean = \
1032                 ssd_next_plane_in_parunit(i, parunit_num, elem_num, s);
1033             return i;
1034         }
1035 
1036         i = ssd_next_plane_in_parunit(i, parunit_num, elem_num, s);
1037     } while (i != start);
1038 
1039     return -1;

1042 /*

1043  * invoke cleaning when the number of free blocks drop below a

1044  * certain threshold (in a plane or an element).

1045  */
1046 int ssd_start_cleaning(int plane_num, int elem_num, ssd_t *s)
1047 {
1048     if (plane_num == -1) {
1049         unsigned int low = (unsigned int)LOW_WATERMARK_PER_ELEMENT(s);
1050         return (s->elements[elem_num].metadata.tot_free_blocks <= low);
1051     } else {
1052         int low = (int)LOW_WATERMARK_PER_PLANE(s);
1053         return (s->elements[elem_num].metadata.plane_meta[plane_num].free_blocks <= low);
1054     }
1055 }

1057 /*

1058  * stop cleaning when sufficient number of free blocks

1059  * are generated (in a plane or an element).

1060  */
1061 int ssd_stop_cleaning(int plane_num, int elem_num, ssd_t *s)
1062 {
1063     if (plane_num == -1) {
1064         unsigned int high = (unsigned int)HIGH_WATERMARK_PER_ELEMENT(s);
1065         return (s->elements[elem_num].metadata.tot_free_blocks > high);
1066     } else {
1067         int high = (int)HIGH_WATERMARK_PER_PLANE(s);
1068         return (s->elements[elem_num].metadata.plane_meta[plane_num].free_blocks > high);
1069     }
1070 }
