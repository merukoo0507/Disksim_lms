  58 /*

  59  * this is the metadata stored on each block.

  60  * it contains a list of valid pages (in the real system,

  61  * we can find if a page in a block is valid by comparing

  62  * a version number stored on each block) and some stat

  63  * about its remaining lifetime to help wear-leveling and

  64  * cleaning program.

  65  */
  66 typedef struct _block_metadata {
  67     int         block_num;              //

  68     int         plane_num;              // the plane to which this block belongs to

  69 
  70     int         *page;                  // size of the array = pages_per_block

  71                                         // holds all the valid logical page numbers present

  72                                         // in a block

  73 
  74     int         num_valid;              // represents the no of valid pages in the block (that is,

  75                                         // the no of non '-1' entries in the above page array)

  76                                         // don't use it as an index to add pages in the above array.

  77 
  78     int         rem_lifetime;           // remaining life time for this block. when the system

  79                                         // starts for the first time, this value will be initialized

  80                                         // to total num of erasures that can be performed on a block.

  81                                         // after every time this block is erased, this count is

  82                                         // decremented.

  83 
  84     double      time_of_last_erasure;   // holds the time when the last erasure was performed.

  85 
  86     int         state;                  // state of the block

  87                                         // a block could be in one of the following states:

  88                                         // CLEAN - when the block is erased and clean to take new writes

  89                                         // INUSE - when its pages are being filled

  90                                         // SEALED - when all the free pages are written and the block

  91                                         // is sealed for future writes

  92 
  93     unsigned int    bsn;                // block sequence number (version number for blocks)

  94 } block_metadata;

 178 /*

 179  * the ssd element (a single package) internally consists of multiple

 180  * planes and the entire package has only one set of pins. so, when

 181  * transfering data to/from the package, the access to the pins must

 182  * be serialized. however, operations on multiple planes can be

 183  * performed concurrently.

 184  */
 185 typedef struct _ssd_element {
 186    struct ioq *queue;
 187    int media_busy;
 188    ssd_element_metadata metadata;               // contains the mapping info b/w lba and flash pages

 189    ssd_element_stat stat;                       // statistics about each ssd element

 190 
 191    int pin_busy;                                // state to hold the busy state of the package pins

 192    int num_planes;                              // number of planes in this package

 193    ssd_plane plane[SSD_MAX_PLANES_PER_ELEM];    // an array of flash planes

 194 } ssd_element;

 232 /////////////////////////////////////////////////////////////////

 233 //vp - blocks in a plane can be arranged in different ways

 234 
 235 // blocks can be concatenated (chained) from each plane

 236 //

 237 // plane 0    plane 1    plane 2    plane 3

 238 // ------------------------------------------

 239 // blk 0      blk 2048   blk 4096   blk 6144

 240 // blk 1      blk 2049   blk 4097   blk 6145

 241 // ...        ...

 242 // blk 2047   blk 4095   blk 6143   blk 8191

 243 #define PLANE_BLOCKS_CONCAT                 1
 244 
 245 
 246 // blocks can be stripped across every pair of planes

 247 // this is how samsung has designed its dies.

 248 // according to the samsung chip design, blocks are

 249 // stripped across every pair of planes. that is, the

 250 // block numbers are arranged as follows.

 251 //

 252 // plane 0    plane 1    plane 2    plane 3

 253 // ------------------------------------------

 254 // blk 0      blk 1      blk 4096   blk 4097

 255 // blk 2      blk 3      blk 4098   blk 4099

 256 // ...        ...

 257 // blk 4094   blk 4095   blk 8190   blk 8191

 258 #define PLANE_BLOCKS_PAIRWISE_STRIPE        2
 259 
 260 
 261 // blocks can be stripped across all the planes

 262 //

 263 // plane 0    plane 1    plane 2    plane 3

 264 // ------------------------------------------

 265 // blk 0      blk 1      blk 2      blk 3

 266 // blk 4      blk 5      blk 6      blk 7

 267 // ...        ...

 268 // blk 8188   blk 8189   blk 8190   blk 8191

 269 //

 270 #define PLANE_BLOCKS_FULL_STRIPE            3

 286 typedef struct _ssd_timing_params {
 287     int    ssd_model;                   // e.g. SSD_*_MODEL above

 288     int    nelements;                   // how many concurrent entities (chips)

 291     int    blocks_per_element;          // total blocks per chip

 292     int    element_stride_pages;        // usually 1 or 2

 293 
 294     //vp - changing the chip_xfer_latency to per byte transfer cost

 295     //instead of per sector transfer cost to make more accurate estimate

 296     double chip_xfer_latency;           // time to get a byte to/from the chip register

 297     double page_read_latency;           // time to read a page into chip register

 298     double page_write_latency;          // time to write a page from chip register

 299     double block_erase_latency;         // time to erase a block

 300 
 301     int     write_policy;               // policy followed when writing a block

 302                                         // follow the above definitions

 303                                         // (e.g., DISKSIM_SSD_WRITE_POLICY_SIMPLE)

 304 
 305     int     reserve_blocks;             // percentage of blocks to reserve

 306 
 307     int     min_freeblks_percent;       // min free blocks percentage

 308 
 309     int     cleaning_policy;            // cleaning & wear-leveling policy used to

 310                                         // clean blocks

 311 
 312     int     planes_per_pkg;             // num of planes in a single SSD package

 313 
 314     unsigned int    blocks_per_plane;   // num of flash blocks in a single plane

 315 
 316     int     plane_block_mapping;        // how blocks are mapped in a plane

 317 
 318     int     copy_back;      // how active pages are assigned in an element

 319 
 320     int     num_parunits;               // number of parallel units in each element

 321 
 322     int     elements_per_gang;          // number of packages in a gang

 323 
 324     int     gang_share;                 // is this a shared bus or shared control gang?

 325 
 326     int     cleaning_in_background;     // do we want to do the cleaning in foreground/background?

 328     int     alloc_pool_logic;           // static or dynamic allocation
