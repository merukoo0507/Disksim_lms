  35 /*

  36  * implements the simple write policy wrote by ted.

  37  */
  38 static double ssd_write_policy_simple(int blkno, int count, int elem_num, ssd_t *s)
  39 {
  40     double cost;
  41     int pages_moved;        // for statistics

  42     struct my_timing_t *tt = (struct my_timing_t *)(s->timing_t);
  43 
  44     // the page position on-chip is calculated as follows (N is the number of chips per SSD):

  45     //               [integer division is assumed]

  46     //    absolute page number (APN) = block number / page_size

  47     //    PN = ((APN - (APN % (Stride * N))) / N) + (APN % Stride)

  48     //

  49     int blockpos, lastpos;
  50     int ppb = tt->params->pages_per_block;
  51     int last_pn = tt->next_write_page[elem_num];
  52     int apn = blkno/tt->params->page_size;
  53     int pn = ((apn - (apn % (tt->params->element_stride_pages*tt->params->nelements)))/
  54                       tt->params->nelements) + (apn % tt->params->element_stride_pages);
  55 
  56     blockpos = pn % ppb;
  57     lastpos = last_pn % ppb;
  58 
  59     if (last_pn >= 0 && pn >= last_pn && (pn/ppb == last_pn/ppb)) {
  60         // the cost of one page write

  61         cost = tt->params->page_write_latency;
  62 
  63         // plus the cost of copying the intermediate pages

  64         cost += (pn-last_pn) * (tt->params->page_read_latency + tt->params->page_write_latency);
  65 
  66         // stat - pages moved other than the page we wrote

  67         pages_moved = pn - last_pn;
  68     } else {
  69         // the cost of one page write

  70         cost = tt->params->page_write_latency;
  71 
  72         // plus the cost of an erase

  73         cost += tt->params->block_erase_latency;
  74 
  75         // stat

  76         s->elements[elem_num].stat.num_clean ++;
  77 
  78         // plus the cost of copying the blocks prior to this pn

  79         cost += blockpos * (tt->params->page_read_latency + tt->params->page_write_latency);
  80 
  81         // stat

  82         pages_moved = blockpos;
  83 
  84         if (last_pn >= 0) {
  85             // plus the cost of copying remaining pages in last written block

  86             cost += (ppb-lastpos) *
  87                 (tt->params->page_read_latency + tt->params->page_write_latency);
  88 
  89             // stat

  90             pages_moved += ppb - lastpos;
  91         }
  92     }
  93 
  94     pn = pn + 1;
  95     if ((blockpos) == 0)
  96         tt->next_write_page[elem_num] = -1;
  97     else
  98         tt->next_write_page[elem_num] = pn;
  99 
 100     // stat

 101     s->elements[elem_num].stat.pages_moved += pages_moved;
 102 
 103     // plus the cost of moving the required sectors on/off chip

 104     cost += ssd_data_transfer_cost(s, count);
 105 
 106     return cost;
 107 }

 121 /*

 122  * returns a count of the number of free bits in a plane.

 123  */
 124 int ssd_free_bits(int plane_num, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
 125 {
 126     int i;
 127     int freecount = 0;
 128     int start = plane_num * s->params.blocks_per_plane;
 129     for (i = start; i < start + (int)s->params.blocks_per_plane; i ++) {
 130         if (!ssd_bit_on(metadata->free_blocks, i)) {
 131             freecount ++;
 132         }
 133     }
 134 
 135     return freecount;
 136 }

 209 int ssd_bitpos_to_block(int bitpos, ssd_t *s)
 210 {
 211     int block = -1;
 212 
 213     switch(s->params.plane_block_mapping) {
 214         case PLANE_BLOCKS_CONCAT:
 215             block = bitpos;
 216             break;
 217 
 218         case PLANE_BLOCKS_PAIRWISE_STRIPE:
 219         {
 220             int plane_pair = bitpos / (s->params.blocks_per_plane*2);
 221             int stripe_no = (bitpos/2) % s->params.blocks_per_plane;
 222             int offset = bitpos % 2;
 223 
 224             block = plane_pair * (s->params.blocks_per_plane*2)  + \
 225                                                         (stripe_no * 2) + \
 226                                                         offset;
 227         }
 228         break;
 229 
 230         case PLANE_BLOCKS_FULL_STRIPE:
 231         {
 232             int stripe_no = bitpos % s->params.blocks_per_plane;
 233             int col_no = bitpos / s->params.blocks_per_plane;
 234             block = (stripe_no * s->params.planes_per_pkg) + col_no;
 235         }
 236         break;
 237 
 238         default:
 239             fprintf(stderr, "Error: unknown plane_block_mapping %d\n", s->params.plane_block_mapping);
 240             exit(1);
 241     }
 242 
 243     return block;
 244 }

 246 int ssd_block_to_bitpos(ssd_t *currdisk, int block)
 247 {
 248     int bitpos = -1;
 249 
 250     // mark the bit position as allocated.

 251     // we need to first find the bit that corresponds to the block

 252     switch(currdisk->params.plane_block_mapping) {
 253         case PLANE_BLOCKS_CONCAT:
 254             bitpos = block;
 255         break;
 256 
 257         case PLANE_BLOCKS_PAIRWISE_STRIPE:
 258         {
 259             int block_index = block % (2*currdisk->params.blocks_per_plane);
 260             int pairs_to_skip = block / (2*currdisk->params.blocks_per_plane);
 261             int planes_to_skip = block % 2;
 262             int blocks_to_skip = block_index / 2;
 263 
 264             bitpos = pairs_to_skip * (2*currdisk->params.blocks_per_plane) + \
 265                         planes_to_skip * currdisk->params.blocks_per_plane + \
 266                         blocks_to_skip;
 267         }
 268         break;
 269 
 270         case PLANE_BLOCKS_FULL_STRIPE:
 271         {
 272             int planes_to_skip = block % currdisk->params.planes_per_pkg;
 273             int blocks_to_skip = block / currdisk->params.planes_per_pkg;
 274             bitpos = planes_to_skip * currdisk->params.blocks_per_plane + blocks_to_skip;
 275         }
 276         break;
 277 
 278         default:
 279             fprintf(stderr, "Error: unknown plane_block_mapping %d\n",
 280                 currdisk->params.plane_block_mapping);
 281             exit(1);
 282     }
 283 
 284     ASSERT((bitpos >= 0) && (bitpos < currdisk->params.blocks_per_element));
 285 
 286     return bitpos;
 287 }

 295 /*

 296  * writes a single page into the active block of a ssd element.

 297  * this code assumes that there is a valid active page where the write can go

 298  * without invoking new cleaning.

 299  */
 300 double _ssd_write_page_osr(ssd_t *s, ssd_element_metadata *metadata, int lpn)
 301 {
 302     double cost;
 303     unsigned int active_page = metadata->active_page;
 304     unsigned int active_block = SSD_PAGE_TO_BLOCK(active_page, s);
 305     unsigned int pagepos_in_block = active_page % s->params.pages_per_block;
 306     unsigned int active_plane = metadata->block_usage[active_block].plane_num;
 307 
 308     // see if this logical page no is already mapped.

 309     if (metadata->lba_table[lpn] != -1) {
 310 
 311         // since the lpn is going to be written to a new location,

 312         // its previous copy is invalid now. therefore reduce the block

 313         // usage of the previous copy's block.

 314 
 315         unsigned int prev_page = metadata->lba_table[lpn];
 316         unsigned int prev_block = SSD_PAGE_TO_BLOCK(prev_page, s);
 317         unsigned int pagepos_in_prev_block = prev_page % s->params.pages_per_block;
 318         unsigned int prev_plane = metadata->block_usage[prev_block].plane_num;
 319 
 320         // make sure the version numbers are correct

 321         ssd_assert_page_version(prev_page, active_page, metadata, s);
 322 
 323         if (metadata->block_usage[prev_block].page[pagepos_in_prev_block] != lpn) {
 324             fprintf(stderr, "Error: lpn %d not found in prev block %d pos %d\n",
 325                 lpn, prev_block, pagepos_in_prev_block);
 326             ASSERT(0);
 327         } else {
 328             metadata->block_usage[prev_block].page[pagepos_in_prev_block] = -1;
 329             metadata->block_usage[prev_block].num_valid --;
 330             metadata->plane_meta[prev_plane].valid_pages --;
 331             ssd_assert_valid_pages(prev_plane, metadata, s);
 332         }
 333     } else {
 334         fprintf(stderr, "Error: This case should not be executed\n");
 335     }
 336 
 337     // add the entry to the lba table

 338     metadata->lba_table[lpn] = active_page;
 339 
 340     // increment the usage count on the active block

 341     metadata->block_usage[active_block].page[pagepos_in_block] = lpn;
 342     metadata->block_usage[active_block].num_valid ++;
 343     metadata->plane_meta[active_plane].valid_pages ++;
 344     ssd_assert_valid_pages(active_plane, metadata, s);
 345 
 346     // some sanity checking

 347     if (metadata->block_usage[active_block].num_valid >= s->params.pages_per_block) {
 348         fprintf(stderr, "Error: len %d of block %d is greater than or equal to pages per block %d\n",
 349             metadata->block_usage[active_block].num_valid, active_block, s->params.pages_per_block);
 350         exit(1);
 351     }
 352 
 353     // add the cost of the write

 354     cost = s->params.page_write_latency;
 355     //printf("lpn %d active pg %d\n", lpn, active_page);

 356 
 357     // go to the next free page

 358     metadata->active_page = active_page + 1;
 359     metadata->plane_meta[active_plane].active_page = metadata->active_page;
 360 
 361     // if this is the last data page on the block, let us write the

 362     // summary page also

 363     if (ssd_last_page_in_block(metadata->active_page, s)) {
 364         // cost of transferring the summary page data

 365         cost += ssd_data_transfer_cost(s, SSD_SECTORS_PER_SUMMARY_PAGE);
 366 
 367         // cost of writing the summary page data

 368         cost += s->params.page_write_latency;
 369 
 370         // seal the last summary page. since we use the summary page

 371         // as a metadata, we don't count it as a valid data page.

 372         metadata->block_usage[active_block].page[s->params.pages_per_block - 1] = -1;
 373         metadata->block_usage[active_block].state = SSD_BLOCK_SEALED;
 374         //printf("SUMMARY: lpn %d active pg %d\n", lpn, active_page);

 375     }
 376 
 377     return cost;
 378 }

 380 /*

 381  * description: this routine finds the next free block inside a ssd

 382  * element/plane and returns it for writing. this routine can be called

 383  * during cleaning and therefore, must be guaranteed to find a free block

 384  * without invoking any further cleaning (otherwise there will be a

 385  * circular dep). we can ensure this by invoking the cleaning algorithm

 386  * when the num of free blocks is high enough to help in cleaning.

 387  */
 388 void _ssd_alloc_active_block(int plane_num, int elem_num, ssd_t *s)
 389 {
 390     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
 391     unsigned char *free_blocks = metadata->free_blocks;
 392     int active_block = -1;
 393     int prev_pos;
 394     int bitpos;
 395 
 396     if (plane_num != -1) {
 397         prev_pos = metadata->plane_meta[plane_num].block_alloc_pos;
 398     } else {
 399         prev_pos = metadata->block_alloc_pos;
 400     }
 401 
 402     // find a free bit

 403     bitpos = ssd_find_zero_bit(free_blocks, s->params.blocks_per_element, prev_pos);
 404     ASSERT((bitpos >= 0) && (bitpos < s->params.blocks_per_element));
 405 
 406     // check if we found the free bit in the plane we wanted to

 407     if (plane_num != -1) {
 408         if (ssd_bitpos_to_plane(bitpos, s) != plane_num) {
 409 
 410             //printf("Error: We wanted a free block in plane %d but found another in %d\n",

 411             //  plane_num, ssd_bitpos_to_plane(bitpos, s));

 412             //printf("So, starting the search again for plane %d\n", plane_num);

 413 
 414             // start from the beginning

 415             metadata->plane_meta[plane_num].block_alloc_pos = plane_num * s->params.blocks_per_plane;
 416             prev_pos = metadata->plane_meta[plane_num].block_alloc_pos;
 417             bitpos = ssd_find_zero_bit(free_blocks, s->params.blocks_per_element, prev_pos);
 418             ASSERT((bitpos >= 0) && (bitpos < s->params.blocks_per_element));
 419 
 420             if (ssd_bitpos_to_plane(bitpos, s) != plane_num) {
 421                 printf("Error: this plane %d is full\n", plane_num);
 422                 printf("this case is not yet handled\n");
 423                 exit(1);
 424             }
 425         }
 426 
 427         metadata->plane_meta[plane_num].block_alloc_pos = \
 428             (plane_num * s->params.blocks_per_plane) + ((bitpos+1) % s->params.blocks_per_plane);
 429     } else {
 430         metadata->block_alloc_pos = (bitpos+1) % s->params.blocks_per_element;
 431     }
 432 
 433     // find the block num

 434     active_block = ssd_bitpos_to_block(bitpos, s);
 435 
 436     if (active_block != -1) {
 437         plane_metadata *pm;
 438 
 439         // make sure we're doing the right thing

 440         ASSERT(metadata->block_usage[active_block].plane_num == ssd_bitpos_to_plane(bitpos, s));
 441         ASSERT(metadata->block_usage[active_block].bsn == 0);
 442         ASSERT(metadata->block_usage[active_block].num_valid == 0);
 443         ASSERT(metadata->block_usage[active_block].state == SSD_BLOCK_CLEAN);
 444 
 445         if (plane_num == -1) {
 446             plane_num = metadata->block_usage[active_block].plane_num;
 447         } else {
 448             ASSERT(plane_num == metadata->block_usage[active_block].plane_num);
 449         }
 450 
 451         pm = &metadata->plane_meta[plane_num];
 452 
 453         // reduce the total number of free blocks

 454         metadata->tot_free_blocks --;
 455         pm->free_blocks --;
 456 
 457         //assertion

 458         ssd_assert_free_blocks(s, metadata);
 459 
 460         // allocate the block

 461         ssd_set_bit(free_blocks, bitpos);
 462         metadata->block_usage[active_block].state = SSD_BLOCK_INUSE;
 463         metadata->block_usage[active_block].bsn = metadata->bsn ++;
 464 
 465         // start from the first page of the active block

 466         pm->active_page = active_block * s->params.pages_per_block;
 467         metadata->active_page = pm->active_page;
 468 
 469         //ssd_assert_plane_freebits(plane_num, elem_num, metadata, s);

 470     } else {
 471         fprintf(stderr, "Error: cannot find a free block in ssd element %d\n", elem_num);
 472         exit(-1);
 473     }
 474 }

 476 listnode **ssd_pick_parunits(ssd_req **reqs, int total, int elem_num, ssd_element_metadata *metadata, ssd_t *s)
 477 {
 478     int i;
 479     int lpn;
 480     int prev_page;
 481     int prev_block;
 482     int plane_num;
 483     int parunit_num;
 484     listnode **parunits;
 485     int filled = 0;
 486 
 487     // parunits is an array of linked list structures, where

 488     // each entry in the array corresponds to one parallel unit

 489     parunits = (listnode **)malloc(SSD_PARUNITS_PER_ELEM(s) * sizeof(listnode *));
 490     for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 491         ll_create(&parunits[i]);
 492     }
 493 
 494     // first, fill in the reads

 495     for (i = 0; i < total; i ++) {
 496         if (reqs[i]->is_read) {
 497             // get the logical page number corresponding to this blkno

 498             lpn = ssd_logical_pageno(reqs[i]->blk, s);
 499             prev_page = metadata->lba_table[lpn];
 500             ASSERT(prev_page != -1);
 501             prev_block = SSD_PAGE_TO_BLOCK(prev_page, s);
 502             plane_num = metadata->block_usage[prev_block].plane_num;
 503             parunit_num = metadata->plane_meta[plane_num].parunit_num;
 504             reqs[i]->plane_num = plane_num;
 505             ll_insert_at_tail(parunits[parunit_num], (void*)reqs[i]);
 506             filled ++;
 507         }
 508     }
 509 
 510     // if all the reqs are reads, return

 511     if (filled == total) {
 512         return parunits;
 513     }
 514 
 515     for (i = 0; i < total; i ++) {
 516 
 517         // we need to find planes for the writes

 518         if (!reqs[i]->is_read) {
 519             int j;
 520             int prev_bsn = -1;
 521             int min_valid;
 522 
 523             plane_num = -1;
 524             lpn = ssd_logical_pageno(reqs[i]->blk, s);
 525             prev_page = metadata->lba_table[lpn];
 526             ASSERT(prev_page != -1);
 527             prev_block = SSD_PAGE_TO_BLOCK(prev_page, s);
 528             prev_bsn = metadata->block_usage[prev_block].bsn;
 529 
 530             if (s->params.alloc_pool_logic == SSD_ALLOC_POOL_PLANE) {
 531                 plane_num = metadata->block_usage[prev_block].plane_num;
 532             } else {
 533                 // find a plane with the max no of free blocks

 534                 j = metadata->plane_to_write;
 535                 do {
 536                     int active_block;
 537                     int active_bsn;
 538                     plane_metadata *pm = &metadata->plane_meta[j];
 539 
 540                     active_block = SSD_PAGE_TO_BLOCK(pm->active_page, s);
 541                     active_bsn = metadata->block_usage[active_block].bsn;
 542 
 543                     // see if we can write to this block

 544                     if ((active_bsn > prev_bsn) ||
 545                         ((active_bsn == prev_bsn) && (pm->active_page > (unsigned int)prev_page))) {
 546                         int free_pages_in_act_blk;
 547                         int k;
 548                         int p;
 549                         int tmp;
 550                         int size;
 551                         int additive = 0;
 552 
 553                         p = metadata->plane_meta[j].parunit_num;
 554                         size = ll_get_size(parunits[p]);
 555 
 556                         // check if this plane has been already selected for writing

 557                         // in this parallel unit

 558                         for (k = 0; k < size; k ++) {
 559                             ssd_req *r;
 560                             listnode *n = ll_get_nth_node(parunits[p], k);
 561                             ASSERT(n != NULL);
 562 
 563                             r = ((ssd_req *)n->data);
 564 
 565                             // is this plane has been already selected for writing?

 566                             if ((r->plane_num == j) &&
 567                                 (!r->is_read)) {
 568                                 additive ++;
 569                             }
 570                         }
 571 
 572                         // select a plane with the most no of free pages

 573                         free_pages_in_act_blk = s->params.pages_per_block - ((pm->active_page%s->params.pages_per_block) + additive);
 574                         tmp = pm->free_blocks * s->params.pages_per_block + free_pages_in_act_blk;
 575 
 576                         if (plane_num == -1) {
 577                             // this is the first plane that satisfies the above criterion

 578                             min_valid = tmp;
 579                             plane_num = j;
 580                         } else {
 581                             if (min_valid < tmp) {
 582                                 min_valid = tmp;
 583                                 plane_num = j;
 584                             }
 585                         }
 586                     }
 587 
 588                     // check out the next plane

 589                     j = (j+1) % s->params.planes_per_pkg;
 590                 } while (j != metadata->plane_to_write);
 591             }
 592 
 593             if (plane_num != -1) {
 594                 // start searching from the next plane

 595                 metadata->plane_to_write = (plane_num+1) % s->params.planes_per_pkg;
 596 
 597                 reqs[i]->plane_num = plane_num;
 598                 parunit_num = metadata->plane_meta[plane_num].parunit_num;
 599                 ll_insert_at_tail(parunits[parunit_num], (void *)reqs[i]);
 600                 filled ++;
 601             } else {
 602                 fprintf(stderr, "Error: cannot find a plane to write\n");
 603                 exit(1);
 604             }
 605         }
 606     }
 607 
 608     ASSERT(filled == total);
 609 
 610     return parunits;
 611 }

 613 static double ssd_issue_overlapped_ios(ssd_req **reqs, int total, int elem_num, ssd_t *s)
 614 {
 615     double max_cost = 0;
 616     double parunit_op_cost[SSD_MAX_PARUNITS_PER_ELEM];
 617     double parunit_tot_cost[SSD_MAX_PARUNITS_PER_ELEM];
 618     ssd_element_metadata *metadata;
 619     int lpn;
 620     int i;
 621     int read_cycle = 0;
 622     listnode **parunits;
 623 
 624     // all the requests must be of the same type

 625     for (i = 1; i < total; i ++) {
 626         ASSERT(reqs[i]->is_read == reqs[0]->is_read);
 627     }
 628 
 629     // is this a set of read requests?

 630     if (reqs[0]->is_read) {
 631         read_cycle = 1;
 632     }
 633 
 634     memset(parunit_tot_cost, 0, sizeof(double)*SSD_MAX_PARUNITS_PER_ELEM);
 635 
 636     // find the planes to which the reqs are to be issued

 637     metadata = &(s->elements[elem_num].metadata);
 638     parunits = ssd_pick_parunits(reqs, total, elem_num, metadata, s);
 639 
 640     // repeat until we've served all the requests

 641     while (1) {
 642         //double tot_xfer_cost = 0;

 643         double max_op_cost = 0;
 644         int active_parunits = 0;
 645         int op_count = 0;
 646 
 647         // do we still have any request to service?

 648         for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 649             if (ll_get_size(parunits[i]) > 0) {
 650                 active_parunits ++;
 651             }
 652         }
 653 
 654         // no more requests -- get out

 655         if (active_parunits == 0) {
 656             break;
 657         }
 658 
 659         // clear this arrays for storing costs

 660         memset(parunit_op_cost, 0, sizeof(double)*SSD_MAX_PARUNITS_PER_ELEM);
 661 
 662         // begin a round of serving. we serve one request per

 663         // parallel unit. if an unit has more than one request

 664         // in the list, they have to be serialized.

 665         max_cost = 0;
 666         for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 667             int size;
 668 
 669             size = ll_get_size(parunits[i]);
 670             if (size > 0) {
 671                 // this parallel unit has a request to serve

 672                 ssd_req *r;
 673                 listnode *n = ll_get_nth_node(parunits[i], 0);
 674 
 675                 op_count ++;
 676                 ASSERT(op_count <= active_parunits);
 677 
 678                 // get the request

 679                 r = (ssd_req *)n->data;
 680                 lpn = ssd_logical_pageno(r->blk, s);
 681 
 682                 if (r->is_read) {
 683                     parunit_op_cost[i] = s->params.page_read_latency;
 684                 } else {
 685                     int plane_num = r->plane_num;
 686                     // if this is the last page on the block, allocate a new block

 687                     if (ssd_last_page_in_block(metadata->plane_meta[plane_num].active_page, s)) {
 688                         _ssd_alloc_active_block(plane_num, elem_num, s);
 689                     }
 690 
 691                     // issue the write to the current active page.

 692                     // we need to transfer the data across the serial pins for write.

 693                     metadata->active_page = metadata->plane_meta[plane_num].active_page;
 694                     //printf("elem %d plane %d ", elem_num, plane_num);

 695                     parunit_op_cost[i] = _ssd_write_page_osr(s, metadata, lpn);
 696                 }
 697 
 698                 ASSERT(r->count <= s->params.page_size);
 699 
 700                 // calc the cost: the access time should be something like this

 701                 // for read

 702                 if (read_cycle) {
 703                     if (SSD_PARUNITS_PER_ELEM(s) > 4) {
 704                         printf("modify acc time here ...\n");
 705                         ASSERT(0);
 706                     }
 707                     if (op_count == 1) {
 708                         r->acctime = parunit_op_cost[i] + ssd_data_transfer_cost(s,s->params.page_size);
 709                         r->schtime = parunit_tot_cost[i] + (op_count-1)*ssd_data_transfer_cost(s,s->params.page_size) + r->acctime;
 710                     } else {
 711                         r->acctime = ssd_data_transfer_cost(s,s->params.page_size);
 712                         r->schtime = parunit_tot_cost[i] + op_count*ssd_data_transfer_cost(s,s->params.page_size) + parunit_op_cost[i];
 713                     }
 714                 } else {
 715                     // for write

 716                     r->acctime = parunit_op_cost[i] + ssd_data_transfer_cost(s,s->params.page_size);
 717                     r->schtime = parunit_tot_cost[i] + (op_count-1)*ssd_data_transfer_cost(s,s->params.page_size) + r->acctime;
 718                 }
 719 
 720 
 721                 // find the maximum cost for this round of operations

 722                 if (max_cost < r->schtime) {
 723                     max_cost = r->schtime;
 724                 }
 725 
 726                 // release the node from the linked list

 727                 ll_release_node(parunits[i], n);
 728             }
 729         }
 730 
 731         // we can start the next round of operations only after all

 732         // the operations in the first round are over because we're

 733         // limited by the one set of pins to all the parunits

 734         for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 735             parunit_tot_cost[i] = max_cost;
 736         }
 737     }
 738 
 739     for (i = 0; i < SSD_PARUNITS_PER_ELEM(s); i ++) {
 740         ll_release(parunits[i]);
 741     }
 742     free(parunits);
 743 
 744     return max_cost;
 745 }

 747 static double ssd_write_one_active_page(int blkno, int count, int elem_num, ssd_t *s)
 748 {
 749     double cost = 0;
 750     int cleaning_invoked = 0;
 751     ssd_element_metadata *metadata;
 752     int lpn;
 753 
 754     metadata = &(s->elements[elem_num].metadata);
 755 
 756     // get the logical page number corresponding to this blkno

 757     lpn = ssd_logical_pageno(blkno, s);
 758 
 759     // see if there are any free pages left inside the active block.

 760     // as per the osr design, the last page is used as a summary page.

 761     // so if the active_page is already pointing to the summary page,

 762     // then we need to find another free block to act as active block.

 763     if (ssd_last_page_in_block(metadata->active_page, s)) {
 764 
 765         // do we need to create more free blocks for future writes?

 766         if (ssd_start_cleaning(-1, elem_num, s)) {
 767 
 768             printf ("We should not clean here ...\n");
 769             ASSERT(0);
 770 
 771             // if we're cleaning in the background, this should

 772             // not get executed

 773             if (s->params.cleaning_in_background) {
 774                 exit(1);
 775             }
 776 
 777             cleaning_invoked = 1;
 778             cost += ssd_clean_element_no_copyback(elem_num, s);
 779         }
 780 
 781         // if we had invoked the cleaning, we must again check if we

 782         // need an active block before allocating one. this check is

 783         // needed because the above cleaning procedure might have

 784         // allocated new active blocks during the process of cleaning,

 785         // which might still have free pages for writing.

 786         if (!cleaning_invoked ||
 787             ssd_last_page_in_block(metadata->active_page, s)) {
 788             _ssd_alloc_active_block(-1, elem_num, s);
 789         }
 790     }
 791 
 792 
 793     // issue the write to the current active page

 794     cost += _ssd_write_page_osr(s, metadata, lpn);
 795     cost += ssd_data_transfer_cost(s, count);
 796 
 797     return cost;
 798 }

 800 /*

 801  * computes the time taken to read/write data in a flash element

 802  * using just one active page.

 803  */
 804 static void ssd_compute_access_time_one_active_page
 805 (ssd_req **reqs, int total, int elem_num, ssd_t *s)
 806 {
 807     // we assume that requests have been broken down into page sized chunks

 808     double cost;
 809     int blkno;
 810     int count;
 811     int is_read;
 812 
 813     // we can serve only one request at a time

 814     ASSERT(total == 1);
 815 
 816     blkno = reqs[0]->blk;
 817     count = reqs[0]->count;
 818     is_read = reqs[0]->is_read;
 819 
 820     if (is_read) {
 821         // there are no separate read policies.

 822         switch(s->params.write_policy) {
 823             case DISKSIM_SSD_WRITE_POLICY_SIMPLE:
 824             case DISKSIM_SSD_WRITE_POLICY_OSR:
 825                 // the cost of one page read

 826                 cost = ssd_read_policy_simple(count, s);
 827                 break;
 828 
 829             default:
 830                 fprintf(stderr, "Error: unknown write policy %d in ssd_compute_access_time\n",
 831                     s->params.write_policy);
 832                 exit(1);
 833         }
 834     } else {
 835         switch(s->params.write_policy) {
 836             case DISKSIM_SSD_WRITE_POLICY_SIMPLE:
 837                 cost = ssd_write_policy_simple(blkno, count, elem_num, s);
 838                 break;
 839 
 840             case DISKSIM_SSD_WRITE_POLICY_OSR:
 841                 cost = ssd_write_one_active_page(blkno, count, elem_num, s);
 842                 break;
 843 
 844             default:
 845                 fprintf(stderr, "Error: unknown write policy %d in ssd_compute_access_time\n",
 846                     s->params.write_policy);
 847                 exit(1);
 848         }
 849     }
 850 
 851     reqs[0]->acctime = cost;
 852     reqs[0]->schtime = cost;
 853 }

 855 /*

 856  * with multiple planes, we can support either single active page or multiple

 857  * active pages.

 858  *

 859  * one-active-page-per-element: we can just have one active page per

 860  * element and move this active page to different blocks whenever a block

 861  * gets full. this is the basic osr design.

 862  *

 863  * one-active-page-per-plane: since each ssd element consists of multiple

 864  * parallel planes, a more dynamic design would have one active page

 865  * per plane. but this slightly complicates the design due to page versions.

 866  *

 867  * page versions are identified using a block sequence number (bsn) and page

 868  * number. bsn is a monotonically increasing num stored in the first page of

 869  * each block. so, if a same logical page 'p' is stored on two different physical

 870  * pages P and P', we find the latest version by comparing <BSN(P), P>

 871  * and <BSN(P'), P'>. in the simple, one-active-page-per-element design

 872  * whenever a block gets full and the active page is moved to a different

 873  * block, the bsn is increased. this way, we can always ensure that the latest

 874  * copy of a page will have its <BSN, P'> higher than its previous versions

 875  * (because BSN is monotonically increasing and physical pages in a block are

 876  * written only in increasing order). however, with multiple active pages say,

 877  * P0 to Pi with their corresponding bsns BSN0 to BSNi, we have to make sure

 878  * that the invariant is still maintained.

 879  */
 880 void ssd_compute_access_time(ssd_t *s, int elem_num, ssd_req **reqs, int total)
 881 {
 882     if (total == 0) {
 883         return;
 884     }
 885 
 886     if (s->params.copy_back == SSD_COPY_BACK_DISABLE) {
 887         ssd_compute_access_time_one_active_page(reqs, total, elem_num, s);
 888     } else {
 889         ssd_issue_overlapped_ios(reqs, total, elem_num, s);
 890     }
 891 }
