  41     int skip_by = ppage % currdisk->params.pages_per_block;

  59 /*

  60  * vp

  61  * description: this routine allocates and initializes the ssd element metadata

  62  * structures. FIXME: if the systems is powered up, this init routine has to

  63  * populate the structures by scanning the summary pages (to implement this,

  64  * we can read from a disk checkpoint file). but, this is future work.

  65 */
  66 void ssd_element_metadata_init(int elem_number, ssd_element_metadata *metadata, ssd_t *currdisk)
  67 {
  68     gang_metadata *g;
  69     unsigned int ppage;
  70     unsigned int i;
  71     unsigned int bytes_to_alloc;
  72     unsigned int tot_blocks = currdisk->params.blocks_per_element;
  73     unsigned int tot_pages = tot_blocks * currdisk->params.pages_per_block;
  74     unsigned int reserved_blocks, usable_blocks, export_size;
  75     unsigned int reserved_blocks_per_plane, usable_blocks_per_plane;
  76     unsigned int bitpos;
  77     unsigned int active_block;
  78     unsigned int elem_index;
  79     unsigned int bsn = 1;
  80     int plane_block_mapping = currdisk->params.plane_block_mapping;
  81 
  82     //////////////////////////////////////////////////////////////////////////////

  83     // active page starts at the 1st page on the reserved section

  84     reserved_blocks_per_plane = (currdisk->params.reserve_blocks * currdisk->params.blocks_per_plane) / 100;
  85     usable_blocks_per_plane = currdisk->params.blocks_per_plane - reserved_blocks_per_plane;
  86     reserved_blocks = reserved_blocks_per_plane * currdisk->params.planes_per_pkg;
  87     usable_blocks = usable_blocks_per_plane * currdisk->params.planes_per_pkg;
  88 
  89     //////////////////////////////////////////////////////////////////////////////

  90     // initialize the free blocks and free pages

  91     metadata->tot_free_blocks = reserved_blocks;
  92 
  93     //////////////////////////////////////////////////////////////////////////////

  94     // assign the gang and init the element's free pages

  95     metadata->gang_num = elem_number / currdisk->params.elements_per_gang;
  96     currdisk->gang_meta[metadata->gang_num].elem_free_pages[elem_number] = \
  97         metadata->tot_free_blocks * SSD_DATA_PAGES_PER_BLOCK(currdisk);
  98     g = &currdisk->gang_meta[metadata->gang_num];
  99     elem_index = elem_number % currdisk->params.elements_per_gang;
 100 
 101     //////////////////////////////////////////////////////////////////////////////

 102     // let's begin cleaning with the first plane

 103     metadata->plane_to_clean = 0;
 104     metadata->plane_to_write = 0;
 105     metadata->block_alloc_pos = 0;
 106     metadata->reqs_waiting = 0;
 107     metadata->tot_migrations = 0;
 108     metadata->tot_pgs_migrated = 0;
 109     metadata->mig_cost = 0;
 110 
 111     //////////////////////////////////////////////////////////////////////////////

 112     // init the plane metadata

 113     for (i = 0; i < (unsigned int)currdisk->params.planes_per_pkg; i ++) {
 114         int blocks_to_skip;
 115 
 116         switch(plane_block_mapping) {
 117             case PLANE_BLOCKS_CONCAT:
 118                 blocks_to_skip = (i*currdisk->params.blocks_per_plane + usable_blocks_per_plane);
 119                 break;
 120 
 121             case PLANE_BLOCKS_PAIRWISE_STRIPE:
 122                 blocks_to_skip = (i/2)*(2*currdisk->params.blocks_per_plane) + (2*usable_blocks_per_plane) + i%2;
 123                 break;
 124 
 125             case PLANE_BLOCKS_FULL_STRIPE:
 126                 blocks_to_skip = (currdisk->params.planes_per_pkg * usable_blocks_per_plane) + i;
 127                 break;
 128 
 129             default:
 130                 fprintf(stderr, "Error: unknown plane_block_mapping %d\n", plane_block_mapping);
 131                 exit(1);
 132         }
 133 
 134         metadata->plane_meta[i].active_page = blocks_to_skip*currdisk->params.pages_per_block;
 135         metadata->plane_meta[i].free_blocks = reserved_blocks_per_plane;
 136         metadata->plane_meta[i].valid_pages = 0;
 137         metadata->plane_meta[i].clean_in_progress = 0;
 138         metadata->plane_meta[i].clean_in_block = -1;
 139         metadata->plane_meta[i].block_alloc_pos = i*currdisk->params.blocks_per_plane;
 140         metadata->plane_meta[i].parunit_num = i / SSD_PLANES_PER_PARUNIT(currdisk);
 141         metadata->plane_meta[i].num_cleans = 0;
 142     }
 143 
 144     //////////////////////////////////////////////////////////////////////////////

 145     // init the next plane to clean in a parunit

 146     for (i = 0; i < (unsigned int) SSD_PARUNITS_PER_ELEM(currdisk); i ++) {
 147         metadata->parunits[i].plane_to_clean = SSD_PLANES_PER_PARUNIT(currdisk)*i;
 148     }
 149 
 150     //////////////////////////////////////////////////////////////////////////////

 151     // init the element's active page

 152     switch(plane_block_mapping) {
 153         case PLANE_BLOCKS_CONCAT:
 154             metadata->active_page = usable_blocks_per_plane * currdisk->params.pages_per_block;
 155             break;
 156 
 157         case PLANE_BLOCKS_PAIRWISE_STRIPE:
 158             metadata->active_page = (2 * usable_blocks_per_plane) * currdisk->params.pages_per_block;
 159             break;
 160 
 161         case PLANE_BLOCKS_FULL_STRIPE:
 162             metadata->active_page = (currdisk->params.planes_per_pkg * usable_blocks_per_plane) * currdisk->params.pages_per_block;
 163             break;
 164 
 165         default:
 166             fprintf(stderr, "Error: unknown plane_block_mapping %d\n", plane_block_mapping);
 167             exit(1);
 168     }
 169 
 170     ASSERT(metadata->active_page == metadata->plane_meta[0].active_page);
 171     active_block = metadata->active_page / currdisk->params.pages_per_block;
 172 
 173     // since we reserve one page out of every block to store the summary info,

 174     // the size exported by the flash disk is little less.

 175     export_size = usable_blocks * SSD_DATA_PAGES_PER_BLOCK(currdisk);
 176     currdisk->data_pages_per_elem = export_size;
 177     //printf("res blks = %d, use blks = %d act page = %d exp size = %d\n",

 178     //  reserved_blocks, usable_blocks, metadata->active_page, export_size);

 179 
 180     //////////////////////////////////////////////////////////////////////////////

 181     // allocate the lba table

 182     if ((metadata->lba_table = (int *)malloc(export_size * sizeof(int))) == NULL) {
 183         fprintf(stderr, "Error: malloc to lba table in ssd_element_metadata_init failed\n");
 184         fprintf(stderr, "Allocation size = %d\n", export_size * sizeof(int));
 185         exit(1);
 186     }
 187 
 188     //////////////////////////////////////////////////////////////////////////////

 189     // allocate the free blocks bit map

 190     // what if the no of blocks is not divisible by 8?

 191     if ((tot_blocks % (sizeof(unsigned char) * 8)) != 0) {
 192         fprintf(stderr, "This case is not yet handled\n");
 193         exit(1);
 194     }
 195 
 196     bytes_to_alloc = tot_blocks / (sizeof(unsigned char) * 8);
 197     if (!(metadata->free_blocks = (unsigned char *)malloc(bytes_to_alloc))) {
 198         fprintf(stderr, "Error: malloc to free_blocks in ssd_element_metadata_init failed\n");
 199         fprintf(stderr, "Allocation size = %d\n", bytes_to_alloc);
 200         exit(1);
 201     }
 202     bzero(metadata->free_blocks, bytes_to_alloc);
 203 
 204     //////////////////////////////////////////////////////////////////////////////

 205     // allocate the block usage array and initialize it

 206     if (!(metadata->block_usage = (block_metadata *)malloc(tot_blocks * sizeof(block_metadata)))) {
 207         fprintf(stderr, "Error: malloc to block_usage in ssd_element_metadata_init failed\n");
 208         fprintf(stderr, "Allocation size = %d\n", tot_blocks * sizeof(block_metadata));
 209         exit(1);
 210     }
 211     bzero(metadata->block_usage, tot_blocks * sizeof(block_metadata));
 212 
 213     for (i = 0; i < tot_blocks; i ++) {
 214         int j;
 215 
 216         metadata->block_usage[i].block_num = i;
 217         metadata->block_usage[i].page = (int*)malloc(sizeof(int) * currdisk->params.pages_per_block);
 218 
 219         for (j = 0; j < currdisk->params.pages_per_block; j ++) {
 220             metadata->block_usage[i].page[j] = -1;
 221         }
 222 
 223         // assign the plane number to each block

 224         switch(plane_block_mapping) {
 225             case PLANE_BLOCKS_CONCAT:
 226                 metadata->block_usage[i].plane_num = i / currdisk->params.blocks_per_plane;
 227                 break;
 228 
 229             case PLANE_BLOCKS_PAIRWISE_STRIPE:
 230                 metadata->block_usage[i].plane_num = (i/(2*currdisk->params.blocks_per_plane))*2 + i%2;
 231                 break;
 232 
 233             case PLANE_BLOCKS_FULL_STRIPE:
 234                 metadata->block_usage[i].plane_num = i % currdisk->params.planes_per_pkg;
 235                 break;
 236 
 237             default:
 238                 fprintf(stderr, "Error: unknown plane_block_mapping %d\n", plane_block_mapping);
 239                 exit(1);
 240         }
 241 
 242         // set the remaining life time and time of last erasure

 243         metadata->block_usage[i].rem_lifetime = SSD_MAX_ERASURES;
 244         metadata->block_usage[i].time_of_last_erasure = simtime;
 245 
 246         // set the block state

 247         metadata->block_usage[i].state = SSD_BLOCK_CLEAN;
 248 
 249         // init the bsn to be zero

 250         metadata->block_usage[i].bsn = 0;
 251     }
 252 
 253     //////////////////////////////////////////////////////////////////////////////

 254     // initially, we assume that every logical page is mapped

 255     // onto a physical page. we start from the first phy page

 256     // and continue to map, leaving the last page of every block

 257     // to store the summary information.

 258     ppage = 0;
 259     i = 0;
 260     while (i < export_size) {
 261         int pgnum_in_gang;
 262         int pp_index;
 263         int plane_num;
 264         unsigned int block = SSD_PAGE_TO_BLOCK(ppage, currdisk);
 265 
 266         ASSERT(block < (unsigned int)currdisk->params.blocks_per_element);
 267 
 268         // if this is the last page in the block

 269         if (ssd_last_page_in_block(ppage, currdisk)) {
 270             // leave this physical page for summary page and

 271             // seal the block

 272             metadata->block_usage[block].state = SSD_BLOCK_SEALED;
 273 
 274             // go to next block

 275             ppage ++;
 276             block = SSD_PAGE_TO_BLOCK(ppage, currdisk);
 277         }
 278 
 279         // if this block is in the reserved section, skip it

 280         // and go to the next block.

 281         switch(plane_block_mapping) {
 282             case PLANE_BLOCKS_CONCAT:
 283             {
 284                 unsigned int block_index = block % currdisk->params.blocks_per_plane;
 285                 if ((block_index >= usable_blocks_per_plane) && (block_index < currdisk->params.blocks_per_plane)) {
 286                     // go to next block

 287                     ppage = ssd_first_page_in_next_block(ppage, currdisk);
 288                     continue;
 289                 }
 290             }
 291             break;
 292 
 293             case PLANE_BLOCKS_PAIRWISE_STRIPE:
 294             {
 295                 unsigned int block_index = block % (2*currdisk->params.blocks_per_plane);
 296                 if ((block_index >= 2*usable_blocks_per_plane) && (block_index < 2*currdisk->params.blocks_per_plane)) {
 297                     ppage = ssd_first_page_in_next_block(ppage, currdisk);
 298                     continue;
 299                 }
 300             }
 301             break;
 302 
 303             case PLANE_BLOCKS_FULL_STRIPE:
 304                 // ideally the control should not come here ...

 305                 if ((block >= usable_blocks) && (block < (unsigned int)currdisk->params.blocks_per_element)) {
 306                     printf("Error: the control should not come here ...\n");
 307                     ppage = ssd_first_page_in_next_block(ppage, currdisk);
 308                     continue;
 309                 }
 310             break;
 311 
 312             default:
 313                 fprintf(stderr, "Error: unknown plane_block_mapping %d\n", plane_block_mapping);
 314                 exit(1);
 315         }
 316 
 317         // when the control comes here, 'ppage' contains the next page

 318         // that can be assigned to a logical page.

 319         // find the index of the phy page within the block

 320         pp_index = ppage % currdisk->params.pages_per_block;
 321 
 322         // populate the lba table

 323         metadata->lba_table[i] = ppage;
 324         pgnum_in_gang = elem_index * export_size + i;
 325         g->pg2elem[pgnum_in_gang].e = elem_number;
 326 
 327         // mark this block as not free and its state as 'in use'.

 328         // note that a block could be not free and its state be 'sealed'.

 329         // it is enough if we set it once while working on the first phy page.

 330         // also increment the block sequence number.

 331         if (pp_index == 0) {
 332             bitpos = ssd_block_to_bitpos(currdisk, block);
 333             ssd_set_bit(metadata->free_blocks, bitpos);
 334             metadata->block_usage[block].state = SSD_BLOCK_INUSE;
 335             metadata->block_usage[block].bsn = bsn ++;
 336         }
 337 
 338         // increase the usage count per block

 339         plane_num = metadata->block_usage[block].plane_num;
 340         metadata->block_usage[block].page[pp_index] = i;
 341         metadata->block_usage[block].num_valid ++;
 342         metadata->plane_meta[plane_num].valid_pages ++;
 343 
 344         // go to the next physical page

 345         ppage ++;
 346 
 347         // go to the next logical page

 348         i ++;
 349     }
 350 
 351     //////////////////////////////////////////////////////////////////////////////

 352     // mark the block that corresponds to the active page

 353     // as not free and 'in_use'.

 354     switch(currdisk->params.copy_back) {
 355         case SSD_COPY_BACK_DISABLE:
 356             bitpos = ssd_block_to_bitpos(currdisk, active_block);
 357             ssd_set_bit(metadata->free_blocks, bitpos);
 358             metadata->block_usage[active_block].state = SSD_BLOCK_INUSE;
 359             metadata->block_usage[active_block].bsn = bsn ++;
 360         break;
 361 
 362         case SSD_COPY_BACK_ENABLE:
 363             for (i = 0; i < (unsigned int)currdisk->params.planes_per_pkg; i ++) {
 364                 int plane_active_block = SSD_PAGE_TO_BLOCK(metadata->plane_meta[i].active_page, currdisk);
 365 
 366                 bitpos = ssd_block_to_bitpos(currdisk, plane_active_block);
 367                 ssd_set_bit(metadata->free_blocks, bitpos);
 368                 metadata->block_usage[plane_active_block].state = SSD_BLOCK_INUSE;
 369                 metadata->block_usage[plane_active_block].bsn = bsn ++;
 370                 metadata->tot_free_blocks --;
 371                 metadata->plane_meta[i].free_blocks --;
 372             }
 373         break;
 374 
 375         default:
 376             fprintf(stderr, "Error: invalid copy back policy %d\n",
 377                 currdisk->params.copy_back);
 378             exit(1);
 379     }
 380 
 381     //////////////////////////////////////////////////////////////////////////////

 382     // set the bsn for the ssd element

 383     metadata->bsn = bsn;
 384     //printf("set the bsn to %d\n", bsn);

 385 }

 387 void ssd_plane_init(ssd_element *elem, ssd_t *s, int devno)
 388 {
 389     int i;
 390 
 391     // set the num of planes per package

 392     elem->num_planes = s->params.planes_per_pkg;
 393 
 394     // init all the planes

 395     for (i = 0; i < elem->num_planes; i ++) {
 396         elem->plane[i].media_busy = FALSE;
 397         elem->plane[i].num_blocks = s->params.blocks_per_plane;
 398 
 399         // just flip the LSB to find the pair

 400         elem->plane[i].pair_plane = i ^ 0x1;
 401     }
 402 }

 424 void ssd_alloc_queues(ssd_t *t)
 425 {
 426    // gross hack !!!!!

 427    // copy the queue for the gangs and element queues from the

 428    // main queue

 429    // printf("Total gang = %d\n", SSD_NUM_GANG(result));

 528 void ssd_resetstats (void)
 529 {
 530    int i;
 531 
 532    for (i=0; i<MAXDEVICES; i++) {
 533       ssd_t *currdisk = getssd (i);
 534       if (currdisk) {
 535          int j;
 536          ioqueue_resetstats(currdisk->queue);
 537          for (j=0; j<currdisk->params.nelements; j++) {
 538              ioqueue_resetstats(currdisk->elements[j].queue);
 539          }
 540          ssd_statinit(i, 0);
 541       }
 542    }
