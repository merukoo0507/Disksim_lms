 120  * Acquires the bus (if not already acquired), then uses bus_delay to

 121  * send the event up the path.

 122  *

 123  * If the bus is already owned by this device or can be acquired

 124  * immediately (interleaved bus), the event is sent immediately.

 125  * Otherwise, ssd_bus_ownership_grant will later send the event.

 128 {
 129    ssd_t *currdisk;
 130    int busno;
 131    int slotno;

 133    // fprintf (outputfile, "ssd_send_event_up_path - devno %d, type %d, cause %d, blkno %d\n", curr->devno, curr->type, curr->cause, curr->blkno);

 134 
 135    currdisk = getssd (curr->devno);
 136 
 137    ssd_assert_current_activity(currdisk, curr);
 138 
 139    busno = ssd_get_busno(curr);
 140    slotno = currdisk->slotno[0];
 141 
 142    /* Put new request at head of buswait queue */
 143    curr->next = currdisk->buswait;
 144    currdisk->buswait = curr;
 145 
 146    curr->tempint1 = busno;
 147    curr->time = delay;
 148    if (currdisk->busowned == -1) {
 149 
 150       // fprintf (outputfile, "Must get ownership of the bus first\n");

 151 
 152       if (curr->next) {
 153          //fprintf(stderr,"Multiple bus requestors detected in ssd_send_event_up_path\n");

 154          /* This should be ok -- counting on the bus module to sequence 'em */
 155       }
 156       if (bus_ownership_get(busno, slotno, curr) == FALSE) {
 157          /* Remember when we started waiting (only place this is written) */
 158          currdisk->stat.requestedbus = simtime;
 159       } else {
 160          currdisk->busowned = busno;
 161          bus_delay(busno, DEVICE, curr->devno, delay, curr); /* Never for SCSI */
 162       }

 171    }

 174 /* The idea here is that only one request can "possess" the channel back to the

 175    controller at a time. All others are enqueued on queue of pending activities.

 176    "Completions" ... those operations that need only be signaled as done to the

 177    controller ... are given on this queue.  The "channel_activity" field indicates

 178    whether any operation currently possesses the channel.

 179 

 180    It is our hope that new requests cannot enter the system when the channel is

 181    possessed by another operation.  This would not model reality!!  However, this

 182    code (and that in ssd_request_arrive) will handle this case "properly" by enqueuing

 183    the incoming request.  */

 185 static void ssd_check_channel_activity (ssd_t *currdisk)
 186 {
 187    while (1) {
 188        ioreq_event *curr = currdisk->completion_queue;
 189        currdisk->channel_activity = curr;
 190        if (curr != NULL) {
 191            currdisk->completion_queue = curr->next;
 192            if (currdisk->neverdisconnect) {
 193                /* already connected */
 194                if (curr->flags & READ) {
 195                    /* transfer data up the line: curr->bcount, which is still set to */
 196                    /* original requested value, indicates how many blks to transfer. */
 197                    curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 198                    ssd_send_event_up_path(curr, (double) 0.0);
 199                } else {
 200                    ssd_request_complete (curr);
 201                }
 202            } else {
 203                /* reconnect to controller */
 204                curr->type = IO_INTERRUPT_ARRIVE;
 205                curr->cause = RECONNECT;
 206                ssd_send_event_up_path (curr, currdisk->bus_transaction_latency);
 207                currdisk->reconnect_reason = DEVICE_ACCESS_COMPLETE;
 208            }
 209        } else {
 210            curr = ioqueue_get_next_request(currdisk->queue);
 211            currdisk->channel_activity = curr;
 212            if (curr != NULL) {
 213                if (curr->flags & READ) {
 214                    if (!currdisk->neverdisconnect) {
 215                        ssd_media_access_request(ioreq_copy(curr));
 216                        curr->type = IO_INTERRUPT_ARRIVE;
 217                        curr->cause = DISCONNECT;
 218                        ssd_send_event_up_path (curr, currdisk->bus_transaction_latency);
 219                    } else {
 220                        ssd_media_access_request(curr);
 221                        continue;
 222                    }
 223                } else {
 224                    curr->cause = RECONNECT;
 225                    curr->type = IO_INTERRUPT_ARRIVE;
 226                    currdisk->reconnect_reason = IO_INTERRUPT_ARRIVE;
 227                    ssd_send_event_up_path (curr, currdisk->bus_transaction_latency);
 228                }
 229            }
 230        }
 231        break;
 232    }
 233 }

 237  * Calls bus_delay to handle the event that the disk has been granted the bus.  I believe

 240 void ssd_bus_ownership_grant (int devno, ioreq_event *curr, int busno, double arbdelay)
 241 {
 242    ssd_t *currdisk;
 243    ioreq_event *tmp;
 244 
 245    currdisk = getssd (devno);

 352 /*

 353  * returns the logical page number within an element given a block number as

 354  * issued by the file system

 355  */
 356 int ssd_logical_pageno(int blkno, ssd_t *s)
 357 {
 358     int apn;
 359     int lpn;
 360 
 361     // absolute page number is the block number as written by the above layer

 362     apn = blkno/s->params.page_size;
 363 
 364     // find the logical page number within the ssd element. we maintain the

 365     // mapping between the logical page number and the actual physical page

 366     // number. an alternative is that we could maintain the mapping between

 367     // apn we calculated above and the physical page number. but the range

 368     // of apn is several times bigger and so we chose to go with the mapping

 369     // b/w lpn --> physical page number

 370     lpn = ((apn - (apn % (s->params.element_stride_pages * s->params.nelements)))/
 371                       s->params.nelements) + (apn % s->params.element_stride_pages);
 372 
 373     return lpn;
 374 }

 438 static void ssd_activate_elem(ssd_t *currdisk, int elem_num)
 439 {
 440     ioreq_event *req;
 441     ssd_req **read_reqs;
 442     ssd_req **write_reqs;
 443     int i;
 444     int read_total = 0;
 445     int write_total = 0;
 446     double schtime = 0;
 447     int max_reqs;
 448     int tot_reqs_issued;
 449     double max_time_taken = 0;
 450 
 451 
 452     ssd_element *elem = &currdisk->elements[elem_num];
 453 
 454     // if the media is busy, we can't do anything, so return

 455     if (elem->media_busy == TRUE) {
 456         return;
 457     }
 458 
 459     ASSERT(ioqueue_get_reqoutstanding(elem->queue) == 0);
 460 
 461     // we can invoke cleaning in the background whether there

 462     // is request waiting or not

 463     if (currdisk->params.cleaning_in_background) {
 464         // if cleaning was invoked, wait until

 465         // it is over ...

 466         if (ssd_invoke_element_cleaning(elem_num, currdisk)) {
 467             return;
 468         }
 469     }
 470 
 471     ASSERT(elem->metadata.reqs_waiting == ioqueue_get_number_in_queue(elem->queue));
 472 
 473     if (elem->metadata.reqs_waiting > 0) {
 474 
 475         // invoke cleaning in foreground when there are requests waiting

 476         if (!currdisk->params.cleaning_in_background) {
 477             // if cleaning was invoked, wait until

 478             // it is over ...

 479             if (ssd_invoke_element_cleaning(elem_num, currdisk)) {
 480                 return;
 481             }
 482         }
 483 
 484         // how many reqs can we issue at once

 485         if (currdisk->params.copy_back == SSD_COPY_BACK_DISABLE) {
 486             max_reqs = 1;
 487         } else {
 488             if (currdisk->params.num_parunits == 1) {
 489                 max_reqs = 1;
 490             } else {
 491                 max_reqs = MAX_REQS_ELEM_QUEUE;
 492             }
 493         }
 494 
 495         // ideally, we should issue one req per plane, overlapping them all.

 496         // in order to simplify the overlapping strategy, let's issue

 497         // requests of the same type together.

 498 
 499         read_reqs = (ssd_req **) malloc(max_reqs * sizeof(ssd_req *));
 500         write_reqs = (ssd_req **) malloc(max_reqs * sizeof(ssd_req *));
 501 
 502         // collect the requests

 503         while ((req = ioqueue_get_next_request(elem->queue)) != NULL) {
 504             int found = 0;
 505 
 506             elem->metadata.reqs_waiting --;
 507 
 508             // see if we already have the same request in the list.

 509             // this usually doesn't happen -- but on synthetic traces

 510             // this weird case can occur.

 511             if (req->flags & READ) {
 512                 found = ssd_already_present(read_reqs, read_total, req);
 513             } else {
 514                 found = ssd_already_present(write_reqs, write_total, req);
 515             }
 516 
 517             if (!found) {
 518                 // this is a valid request

 519                 ssd_req *r = malloc(sizeof(ssd_req));
 520                 r->blk = req->blkno;
 521                 r->count = req->bcount;
 522                 r->is_read = req->flags & READ;
 523                 r->org_req = req;
 524                 r->plane_num = -1; // we don't know to which plane this req will be directed at

 525 
 526                 if (req->flags & READ) {
 527                     read_reqs[read_total] = r;
 528                     read_total ++;
 529                 } else {
 530                     write_reqs[write_total] = r;
 531                     write_total ++;
 532                 }
 533 
 534                 // if we have more reqs than we can handle, quit

 535                 if ((read_total >= max_reqs) ||
 536                     (write_total >= max_reqs)) {
 537                     break;
 538                 }
 539             } else {
 540                 // throw this request -- it doesn't make sense

 541                 stat_update (&currdisk->stat.acctimestats, 0);
 542                 req->time = simtime;
 543                 req->ssd_elem_num = elem_num;
 544                 req->type = DEVICE_ACCESS_COMPLETE;
 545                 addtointq ((event *)req);
 546             }
 547         }
 548 
 549         if (read_total > 0) {
 550             // first issue all the read requests (it doesn't matter what we

 551             // issue first). i chose read because reads are mostly synchronous.

 552             // find the time taken to serve these requests.

 553             ssd_compute_access_time(currdisk, elem_num, read_reqs, read_total);
 554 
 555             // add an event for each request completion

 556             for (i = 0; i < read_total; i ++) {
 557               elem->media_busy = TRUE;
 558 
 559               // find the maximum time taken by a request

 560               if (schtime < read_reqs[i]->schtime) {
 561                   schtime = read_reqs[i]->schtime;
 562               }
 563 
 564               stat_update (&currdisk->stat.acctimestats, read_reqs[i]->acctime);
 565               read_reqs[i]->org_req->time = simtime + read_reqs[i]->schtime;
 566               read_reqs[i]->org_req->ssd_elem_num = elem_num;
 567               read_reqs[i]->org_req->type = DEVICE_ACCESS_COMPLETE;
 568 
 569               //printf("R: blk %d elem %d acctime %f simtime %f\n", read_reqs[i]->blk,

 570                 //  elem_num, read_reqs[i]->acctime, read_reqs[i]->org_req->time);

 571 
 572               addtointq ((event *)read_reqs[i]->org_req);
 573               free(read_reqs[i]);
 574             }
 575         }
 576 
 577         free(read_reqs);
 578 
 579         max_time_taken = schtime;
 580 
 581         if (write_total > 0) {
 582             // next issue the write requests

 583             ssd_compute_access_time(currdisk, elem_num, write_reqs, write_total);
 584 
 585             // add an event for each request completion.

 586             // note that we can issue the writes only after all the reads above are

 587             // over. so, include the maximum read time when creating the event.

 588             for (i = 0; i < write_total; i ++) {
 589               elem->media_busy = TRUE;
 590 
 591               stat_update (&currdisk->stat.acctimestats, write_reqs[i]->acctime);
 592               write_reqs[i]->org_req->time = simtime + schtime + write_reqs[i]->schtime;
 593               //printf("blk %d elem %d acc time %f\n", write_reqs[i]->blk, elem_num, write_reqs[i]->acctime);

 594 
 595               if (max_time_taken < (schtime+write_reqs[i]->schtime)) {
 596                   max_time_taken = (schtime+write_reqs[i]->schtime);
 597               }
 598 
 599               write_reqs[i]->org_req->ssd_elem_num = elem_num;
 600               write_reqs[i]->org_req->type = DEVICE_ACCESS_COMPLETE;
 601               //printf("W: blk %d elem %d acctime %f simtime %f\n", write_reqs[i]->blk,

 602                 //  elem_num, write_reqs[i]->acctime, write_reqs[i]->org_req->time);

 603 
 604               addtointq ((event *)write_reqs[i]->org_req);
 605               free(write_reqs[i]);
 606             }
 607         }
 608 
 609         free(write_reqs);
 610 
 611         // statistics

 612         tot_reqs_issued = read_total + write_total;
 613         ASSERT(tot_reqs_issued > 0);
 614         currdisk->elements[elem_num].stat.tot_reqs_issued += tot_reqs_issued;
 615         currdisk->elements[elem_num].stat.tot_time_taken += max_time_taken;
 616     }
 617 }

 620 static void ssd_media_access_request_element (ioreq_event *curr)
 621 {
 622    ssd_t *currdisk = getssd(curr->devno);
 623    int blkno = curr->blkno;
 624    int count = curr->bcount;
 625 
 626    /* **** CAREFUL ... HIJACKING tempint2 and tempptr2 fields here **** */
 627    curr->tempint2 = count;
 628    while (count != 0) {
 629 
 630        // find the element (package) to direct the request

 631        int elem_num = currdisk->timing_t->choose_element(currdisk->timing_t, blkno);
 632        ssd_element *elem = &currdisk->elements[elem_num];
 633 
 634        // create a new sub-request for the element

 635        ioreq_event *tmp = (ioreq_event *)getfromextraq();
 636        tmp->devno = curr->devno;
 637        tmp->busno = curr->busno;
 638        tmp->flags = curr->flags;
 639        tmp->blkno = blkno;
 640        tmp->bcount = ssd_choose_aligned_count(currdisk->params.page_size, blkno, count);
 641        ASSERT(tmp->bcount == currdisk->params.page_size);
 642 
 643        tmp->tempptr2 = curr;
 644        blkno += tmp->bcount;
 645        count -= tmp->bcount;
 646 
 647        elem->metadata.reqs_waiting ++;
 648 
 649        // add the request to the corresponding element's queue

 650        ioqueue_add_new_request(elem->queue, (ioreq_event *)tmp);
 651        ssd_activate_elem(currdisk, elem_num);
 652    }
 653 }

 680 {
 681    ssd_t *currdisk;
 682 
 683    // fprintf (outputfile, "Entering ssd_reconnect_done for disk %d: %12.6f\n", curr->devno, simtime);

 684 
 685    currdisk = getssd (curr->devno);

 687 
 688    if (curr->flags & READ) {
 689       if (currdisk->neverdisconnect) {

 691          /* media access is complete.                                    */
 692          addtoextraq((event *) curr);
 693          ssd_check_channel_activity (currdisk);
 694       } else {

 699       }
 700 
 701    } else {
 702       if (currdisk->reconnect_reason == DEVICE_ACCESS_COMPLETE) {
 703          ssd_request_complete (curr);
 704 
 705       } else {
 706          /* data transfer: curr->bcount, which is still set to original */
 707          /* requested value, indicates how many blks to transfer.       */
 708          curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 709          ssd_send_event_up_path(curr, (double) 0.0);
 710       }

 714 static void ssd_request_arrive (ioreq_event *curr)
 715 {
 716    ssd_t *currdisk;
 717 
 718    // fprintf (outputfile, "Entering ssd_request_arrive: %12.6f\n", simtime);

 719    // fprintf (outputfile, "ssd = %d, blkno = %d, bcount = %d, read = %d\n",curr->devno, curr->blkno, curr->bcount, (READ & curr->flags));

 720 
 721    currdisk = getssd(curr->devno);
 722 
 723    /* verify that request is valid. */
 724    if ((curr->blkno < 0) || (curr->bcount <= 0) ||
 725        ((curr->blkno + curr->bcount) > currdisk->numblocks)) {
 726       fprintf(stderr, "Invalid set of blocks requested from ssd - blkno %d, bcount %d, numblocks %d\n", curr->blkno, curr->bcount, currdisk->numblocks);
 727       exit(1);
 728    }
 729 
 730    /* create a new request, set it up for initial interrupt */
 731    ioqueue_add_new_request(currdisk->queue, curr);
 732    if (currdisk->channel_activity == NULL) {
 733 
 734       curr = ioqueue_get_next_request(currdisk->queue);
 735       currdisk->busowned = ssd_get_busno(curr);
 736       currdisk->channel_activity = curr;
 737       currdisk->reconnect_reason = IO_INTERRUPT_ARRIVE;
 738 
 739       if (curr->flags & READ) {
 740           if (!currdisk->neverdisconnect) {
 741               ssd_media_access_request (ioreq_copy(curr));
 742               curr->cause = DISCONNECT;
 743               curr->type = IO_INTERRUPT_ARRIVE;
 744               ssd_send_event_up_path(curr, currdisk->bus_transaction_latency);
 745           } else {
 746               ssd_media_access_request (curr);
 747               ssd_check_channel_activity(currdisk);
 748           }
 749       } else {
 750          curr->cause = READY_TO_TRANSFER;
 751          curr->type = IO_INTERRUPT_ARRIVE;
 752          ssd_send_event_up_path(curr, currdisk->bus_transaction_latency);
 753       }
 754    }

 777 void ssd_complete_parent(ioreq_event *curr, ssd_t *currdisk)
 778 {
 779     ioreq_event *parent;
 780 
 781     /* **** CAREFUL ... HIJACKING tempint2 and tempptr2 fields here **** */
 782     parent = curr->tempptr2;
 783     parent->tempint2 -= curr->bcount;
 784 
 785     if (parent->tempint2 == 0) {
 786       ioreq_event *prev;
 787 
 788       assert(parent != currdisk->channel_activity);
 789       prev = currdisk->completion_queue;
 790       if (prev == NULL) {
 791          currdisk->completion_queue = parent;
 792          parent->next = prev;
 793       } else {
 794          while (prev->next != NULL)
 795             prev = prev->next;
 796             parent->next = prev->next;
 797             prev->next = parent;
 798       }
 799       if (currdisk->channel_activity == NULL) {
 800          ssd_check_channel_activity (currdisk);
 801       }
 802     }
 803 }

 946       case IO_INTERRUPT_COMPLETE:

1167 void ssd_print_block_lifetime_distribution(int elem_num, ssd_t *s, int ssdno, double avg_lifetime, char *sourcestr)
1168 {
1169     const int bucket_size = 20;
1170     int no_buckets = (100/bucket_size + 1);
1171     int i;
1172     int *hist;
1173     int dead_blocks = 0;
1174     int n;
1175     double sum;
1176     double sum_sqr;
1177     double mean;
1178     double variance;
1179     ssd_element_metadata *metadata = &(s->elements[elem_num].metadata);
1180 
1181     // allocate the buckets

1182     hist = (int *) malloc(no_buckets * sizeof(int));
1183     memset(hist, 0, no_buckets * sizeof(int));
1184 
1185     // to calc the variance

1186     n = s->params.blocks_per_element;
1187     sum = 0;
1188     sum_sqr = 0;
1189 
1190     for (i = 0; i < s->params.blocks_per_element; i ++) {
1191         int bucket;
1192         int rem_lifetime = metadata->block_usage[i].rem_lifetime;
1193         double perc = (rem_lifetime * 100.0) / avg_lifetime;
1194 
1195         // find out how many blocks have completely been erased.

1196         if (metadata->block_usage[i].rem_lifetime == 0) {
1197             dead_blocks ++;
1198         }
1199 
1200         if (perc >= 100) {
1201             // this can happen if a particular block was not

1202             // cleaned at all and so its remaining life time

1203             // is greater than the average life time. put these

1204             // blocks in the last bucket.

1205             bucket = no_buckets - 1;
1206         } else {
1207             bucket = (int) perc / bucket_size;
1208         }
1209 
1210         hist[bucket] ++;
1211 
1212         // calculate the variance

1213         sum = sum + rem_lifetime;
1214         sum_sqr = sum_sqr + (rem_lifetime*rem_lifetime);
1215     }
1216 
1217

1389    struct ioq * queueset[MAXDEVICES*SSD_MAX_ELEMENTS];
1390    int set[MAXDEVICES];
1391    int i,j;
1392    int reqcnt = 0;

1398    fprintf(outputfile, "---------------------\n\n");
1399 
1400    sprintf_s3(prefix, 80, "ssd ");
1401 
1402    diskcnt = 0;
1403    queuecnt = 0;
1404    for (i=0; i<MAXDEVICES; i++) {
1405       ssd_t *currdisk = getssd (i);
1406       if (currdisk) {
1407          struct ioq *q = currdisk->queue;
1408          queueset[queuecnt] = q;
1409          queuecnt++;
1410          reqcnt += ioqueue_get_number_of_requests(q);
1411          diskcnt++;
1412       }
1413    }
1414    assert (diskcnt == numssds);
1415 
1416    if (reqcnt == 0) {
1417       fprintf(outputfile, "No ssd requests encountered\n");
1418       return;
1419    }
1420 
1421    ioqueue_printstats(queueset, queuecnt, prefix);
1422 
1423    diskcnt = 0;
1424    for (i=0; i<MAXDEVICES; i++) {
1425       ssd_t *currdisk = getssd (i);
1426       if (currdisk) {
1427          set[diskcnt] = i;
1428          diskcnt++;
1429       }
1430    }
1431    assert (diskcnt == numssds);
1432 
1433    ssd_acctime_printstats(set, numssds, prefix);
1434    ssd_other_printstats(set, numssds, prefix);
1435 
1436    ssd_printcleanstats(set, numssds, prefix);
1437

1439 
1440    for (i=0; i<numssds; i++) {
1441       ssd_t *currdisk = getssd (set[i]);
1442       if (currdisk->printstats == FALSE) {
1443           continue;
1444       }
1445       reqcnt = 0;
1446       {
1447           struct ioq *q = currdisk->queue;
1448           reqcnt += ioqueue_get_number_of_requests(q);
1449       }
1450       if (reqcnt == 0) {
1451           fprintf(outputfile, "No requests for ssd #%d\n\n\n", set[i]);
1452           continue;
1453       }
1454       fprintf(outputfile, "ssd #%d:\n\n", set[i]);
1455       sprintf_s4(prefix, 80, "ssd #%d ", set[i]);
1456       {
1457           struct ioq *q;
1458           q = currdisk->queue;
1459           ioqueue_printstats(&q, 1, prefix);
1460       }
1461       for (j=0;j<currdisk->params.nelements;j++) {
1462           char pprefix[100];
1463           struct ioq *q;

1466           ioqueue_printstats(&q, 1, pprefix);
1467       }
1468       ssd_acctime_printstats(&set[i], 1, prefix);
1469       ssd_other_printstats(&set[i], 1, prefix);

1471    }
