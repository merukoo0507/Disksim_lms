 118 static void pf_allocate_process_structs()
 119 {
 120    int i;
 121    process *temp = (process *) DISKSIM_malloc(ALLOCSIZE);
 122 
 123    ASSERT(temp != NULL);
 124    for (i=0; i<((ALLOCSIZE/sizeof(process))-1); i++) {
 125       temp[i].next = &temp[i+1];
 126    }
 127    temp[((ALLOCSIZE/sizeof(process))-1)].next = NULL;
 128    extra_process_q = temp;
 129    extra_process_qlen = ALLOCSIZE/sizeof(process);
 130 }

 144 process *pf_getfromextra_process_q()
 145 {
 146    process *temp = extra_process_q;
 147 
 148    if (extra_process_qlen == 0) {
 149       pf_allocate_process_structs();
 150       temp = extra_process_q;
 151       extra_process_q = extra_process_q->next;
 152    } else if (extra_process_qlen == 1) {
 153       extra_process_q = NULL;
 154    } else {
 155       extra_process_q = extra_process_q->next;
 156    }
 157    extra_process_qlen--;
 158    temp->pfflags = 0;
 159    temp->stat = PROC_RUN;
 160    temp->idler = 0;
 161    temp->runcpu = 0;
 162    temp->flags = 0;
 163    temp->chan = 0;
 164    temp->ios = 0;
 165    temp->ioreads = 0;
 166    temp->cswitches = 0;
 167    temp->active = 0;
 168    temp->sleeps = 0;
 169    temp->iosleep = 0;
 170    temp->iosleeps = 0;
 171    temp->runiosleep = 0.0;
 172    temp->lastsleep = 0.0;
 173    temp->runsleep = 0.0;
 174    temp->runtime = 0.0;
 175    temp->falseidletime = 0.0;
 176    temp->lasteventtime = -1.0;
 177    stat_initialize(statdeffile, "Time limit duration", &temp->readtimelimitstats);
 178    stat_initialize(statdeffile, "Time limit duration", &temp->writetimelimitstats);
 179    stat_initialize(statdeffile, "Time limit duration", &temp->readmisslimitstats);
 180    stat_initialize(statdeffile, "Time limit duration", &temp->writemisslimitstats);
 181    temp->ioreq = NULL;
 182    temp->eventlist = NULL;
 183    temp->space = NULL;
 184    temp->link = NULL;
 185    temp->next = NULL;
 186    return(temp);

 228 static double pf_add_false_idle_time (double length)
 229 {
 230    double ret = 0.0;
 231    process *procp = process_livelist;
 232    while (procp != NULL) {
 233       if ((procp->stat == PROC_SLEEP) && (procp->iosleep)) {
 234          procp->falseidletime += length;
 235          ret = length;
 236       } else if ((procp->stat == PROC_RUN) && (procp->pfflags & PF_SLEEP_FLAG) && (procp->iosleep)) {
 237          procp->falseidletime += length;
 238          ret = length;
 239       } else if ((procp->stat == PROC_ONPROC) && (cpus[procp->runcpu].state == CPU_IDLE) && (procp->pfflags & PF_SLEEP_FLAG) && (procp->iosleep)) {
 240          procp->falseidletime += length;
 241          ret = length;
 242       }
 243       procp = procp->livelist;
 244    }
 245    return(ret);
 246 }

 266 event * pf_io_done_notify (ioreq_event *curr, void *ctx)
 267 {
 268    ioreq_event *tmp = pendiolist;
 269    ioreq_event *tmp2;
 270    wakeup_event *tmpwake = NULL;
 271 
 272    if(disksim->external_control){
 273      disksim->external_io_done_notify (curr, ctx);
 274    }
 275 
 276    if (pf_printhack) {
 277      fprintf (outputfile, "pf_io_done_notify: curr->buf %p, curr->opid %x, curr->blkno %d\n", curr->buf, curr->opid, curr->blkno);
 278    }
 279 
 280    ASSERT(pendiolist != NULL);
 281    if ((tmp->buf == curr->buf) && (tmp->opid == curr->opid)) {
 282       pendiolist = tmp->next;
 283    } 
 284    else {
 285      while ((tmp->next) && 
 286             ((tmp->next->buf != curr->buf) || 
 287              (tmp->next->opid != curr->opid))) 
 288        {
 289          tmp = tmp->next;
 290        }
 291      
 292      ASSERT(tmp->next != NULL);
 293      tmp2 = tmp->next;
 294      tmp->next = tmp2->next;
 295      tmp = tmp2;
 296    }
 297 
 298    if (tmp->flags & PF_BUF_WANTED) {
 299       tmpwake = (wakeup_event *) getfromextraq();
 300       tmpwake->type = WAKEUP_EVENT;
 301       /* tmpwake->time = PF_IO_WAKEUP_TIME;         gets overridden anyway */
 302       tmpwake->next = NULL;
 303       tmpwake->chan = tmp->buf;
 304       tmpwake->dropit = 0;
 305    }
 306 
 307    if (curr->flags & TIME_CRITICAL) {
 308       stat_update(&timecritrespstats, (simtime - tmp->time));
 309    } 
 310    else if (curr->flags & TIME_LIMITED) {
 311       stat_update(&timelimitrespstats, (simtime - tmp->time));
 312    } 
 313    else {
 314       stat_update(&timenoncritrespstats, (simtime - tmp->time));
 315    }
 316 
 317    if (!(curr->flags & TIME_LIMITED) || (tmpwake)) {
 318       addtoextraq((event *) tmp);
 319    } 
 320    else {
 321       tmp->next = doneiolist;
 322       doneiolist = tmp;
 323    }
 324    return((event *) tmpwake);
 325 }

 328 static int pf_iowait (void *chan, process *procp)
 329 {
 330    double tmptime;
 331    ioreq_event *tmp = pendiolist;
 332 
 333    while ((tmp) && (tmp->buf != chan)) {
 334       tmp = tmp->next;
 335    }
 336    if (tmp == NULL) {
 337       ioreq_event *prev = 0;
 338       tmp = doneiolist;
 339       while ((tmp) && (tmp->buf != chan)) {
 340          prev = tmp;
 341          tmp = tmp->next;
 342       }
 343       if (tmp) {
 344          if (doneiolist == tmp) {
 345             doneiolist = tmp->next;
 346          } else {
 347             prev->next = tmp->next;
 348          }
 349          if (tmp->flags & READ) {
 350             stat_update(&procp->readtimelimitstats, (simtime - tmp->time));
 351          } else {
 352             stat_update(&procp->writetimelimitstats, (simtime - tmp->time));
 353          }
 354          addtoextraq((event *) tmp);
 355       }
 356       return(FALSE);
 357    }
 358 
 359 if (pf_printhack)
 360 fprintf (outputfile, "pf_iowait: chan %p, read %d, crit %d, opid %d, blkno %x\n", tmp->buf, (tmp->flags & READ), (tmp->flags & (TIME_LIMITED|TIME_CRITICAL)), tmp->opid, tmp->blkno);
 361 
 362    if (tmp->flags & TIME_LIMITED) {
 363       if (tmp->flags & READ) {
 364          stat_update(&procp->readtimelimitstats, (simtime - tmp->time));
 365          stat_update(&procp->readmisslimitstats, (simtime - tmp->time));
 366       } else {
 367          stat_update(&procp->writetimelimitstats, (simtime - tmp->time));
 368          stat_update(&procp->writemisslimitstats, (simtime - tmp->time));
 369       }
 370    }
 371    tmp->flags |= PF_BUF_WANTED;
 372    tmptime = io_raise_priority(tmp->opid, tmp->devno, tmp->blkno, tmp->buf);
 373    if (tmptime != 0.0) {
 374       fprintf(stderr, "Haven't handled non-zero bufferwait time yet\n");
 375       exit(1);
 376    }
 377    return(TRUE);
 378 }

 381 static void pf_add_cswitch_event (process *procp, process *newprocp, cpu_event *cpu_ev)
 382 {
 383    cpu *currcpu = &cpus[(cpu_ev->cpunum)];
 384    cswitch_event *new = (cswitch_event *) getfromextraq();
 385    new->type = CSWITCH_EVENT;
 386    new->newprocp = newprocp;
 387    if (procp != newprocp) {
 388       new->time = PF_CSWITCH_TIME;
 389       currcpu->cswitches++;
 390    } else {
 391       new->time = 0.0;
 392    }
 393    currcpu->runswitchtime += new->time;
 394    if (procp) {
 395       pf_add_to_process_eventlist(procp, (event *) new);
 396    } else {
 397       new->next = currcpu->idleevents;
 398       currcpu->idleevents = (event *) new;
 399    }
 400 }

 432 void pf_idle_cpu_recheck_dispq (int cpunum)
 433 {
 434   ASSERT(cpus[cpunum].current->procp == NULL);
 435 
 436    if (cpus[cpunum].idleevents == NULL) {
 437       event *new = getfromextraq();
 438       new->type = IDLELOOP_EVENT;
 439       new->time = PF_IDLE_CHECK_DISPQ_TIME;
 440       new->next = cpus[cpunum].idleevents;
 441       cpus[cpunum].idleevents = new;
 442 
 443       if (cpus[cpunum].current->intrp == NULL) {
 444         cpus[cpunum].idletime += simtime - cpus[cpunum].idlestart;
 445         cpus[cpunum].falseidletime += pf_add_false_idle_time(simtime - cpus[cpunum].idlestart);
 446         cpus[cpunum].current->time = simtime + (cpus[cpunum].scale * new->time);
 447         new->time = simtime;
 448         cpus[cpunum].state = CPU_IDLE_WORK;
 449         addtointq((event *) cpus[cpunum].current);
 450       }
 451    }
 452 }

 455 static void pf_handle_cswitch_event (cswitch_event *curr, cpu_event *cpu_ev)
 456 {
 457    process *procp = NULL;
 458 
 459    if (pf_printhack)
 460    fprintf (outputfile, "%.3f\tCONTEXT SWITCH cpu=%d pid=%d, newpid=%d\n", simtime, cpu_ev->cpunum, ((cpu_ev->procp) ? cpu_ev->procp->pid : 0), ((curr->newprocp) ? curr->newprocp->pid : 0));
 461 
 462    if (cpu_ev->procp) {
 463       if (cpu_ev->procp->idler == 0) {
 464          lastuser = simtime;
 465       }
 466       cpu_ev->procp->cswitches++;
 467    }
 468 
 469    /* Used to identify how long the system has been completely idle   */
 470    /* Simulation should stop rather than continuing to idle endlessly */
 471    if ((idlereset == 0) && 
 472        ((curr->newprocp == NULL) || 
 473         (curr->newprocp->idler != 0))) {
 474       idlein = simtime;
 475       idlereset = 1;
 476    } 
 477    else {
 478       idlereset = 0;
 479    }
 480 
 481    cpu_ev->procp = curr->newprocp;
 482    cpus[(cpu_ev->cpunum)].state = (curr->newprocp) ? CPU_PROCESS : CPU_IDLE;
 483    if (cpu_ev->procp) {
 484       cpu_ev->procp->pfflags &= ~PF_SLEEP_FLAG;
 485    }
 486 
 487    if ((cpu_ev->procp == NULL) && (pf_dispq)) {
 488       procp = pf_dispatch(cpu_ev->cpunum);
 489       if (procp != cpu_ev->procp) {
 490          pf_add_cswitch_event(cpu_ev->procp, procp, cpu_ev);
 491       }
 492    } else {
 493       addlisttoextraq((event **) &cpus[(cpu_ev->cpunum)].idleevents);
 494    }
 495 }

 513 static void pf_handle_sleep_event (sleep_event *curr, cpu_event *cpu_ev)
 514 {
 515    process *procp = cpu_ev->procp;
 516    process *newprocp;
 517 
 518    if (pf_printhack)
 519    fprintf (outputfile, "%f\tSLEEP cpu=%d pid=%d chan=%p info=%d iosleep=%d\n", simtime, cpu_ev->cpunum, procp->pid, curr->chan, curr->info, curr->iosleep);
 520 
 521    if (pf_iowait(curr->chan, procp)) {
 522       ASSERT(curr->iosleep > 0);
 523       procp->iosleep = TRUE;
 524       procp->iosleeps++;
 525    } else if (curr->iosleep) {
 526       return;   /* to support synthetic TIME_LIMIT'd requests */
 527    } else {
 528       ASSERT (0);
 529    }
 530    procp->pfflags |= PF_SLEEP_FLAG;
 531    procp->chan = curr->chan;
 532    newprocp = pf_disp_sleep(procp);
 533    pf_add_cswitch_event(procp, newprocp, cpu_ev);
 534    procp->lastsleep = simtime;
 535    procp->sleeps++;
 536 }

 539 static void pf_handle_wakeup_event(wakeup_event *curr, cpu_event *cpu_ev)
 540 {
 541    process *procp;
 542 
 543    if (pf_printhack) {
 544      fprintf (outputfile, "%f\tWAKEUP cpu=%d chan=%p info=%d\n", 
 545               simtime, cpu_ev->cpunum, curr->chan, curr->info);
 546    }
 547 
 548    while ((procp = pf_disp_get_from_sleep_queue(curr->chan))) {
 549 
 550      if (pf_printhack) {
 551        fprintf (outputfile, "Waking up process %d\n", procp->pid);
 552      }
 553 
 554      procp->runsleep += simtime - procp->lastsleep;
 555      if (procp->iosleep == 1) {
 556        procp->runiosleep += simtime - procp->lastsleep;
 557      }
 558      pf_disp_wakeup(procp);
 559    }
 560 
 561    procp = cpu_ev->procp;
 562    if ((procp) 
 563        && ((procp->pfflags & PF_SLEEP_FLAG) 
 564            && (procp->chan == curr->chan))) 
 565      {
 566        procp->pfflags |= PF_WAKED;
 567      }
 568 }

 581    fprintf (outputfile, "%f\tIOREQ cpu=%d opid=%d buf=%p blkno=%x flags=%x, bcount=%d\n", simtime, cpu_ev->cpunum, curr->opid, curr->buf, curr->blkno, curr->flags, curr->bcount);
 582 
 583    curr->flags &= ~(TIMED_OUT|HALF_OUT); /* hack to help out ioqueue.c */
 584 
 585    pf_add_to_pendiolist(procp, curr);
 586 
 587    if (cpu_ev->intrp == NULL) {
 588       procp->ios++;
 589       if (curr->flags & READ) {
 590          procp->ioreads++;
 591       }
 592    }
 593    curr->cause = 0;   /* To avoid a stupid problem in logorg.c --- TEMP  */
 594    new = (ioreq_event *) io_request(curr);
 595    if (new) {
 596       tmp = (ioreq_event *) procp->eventlist;
 597       procp->eventlist = (event *) new;
 598       while (new->next) {
 599          new = new->next;
 600       }
 601       new->next = tmp;
 602    }
 603 }

 646 static void pf_handle_intend_event (intend_event *curr, cpu_event *cpu_ev)
 647 {
 648    intr_event *intrp = cpu_ev->intrp;
 649    process *procp = cpu_ev->procp;
 650    cpu *currcpu = &cpus[(cpu_ev->cpunum)];
 651 
 652    if (pf_printhack)
 653    fprintf (outputfile, "%f\tINTEND cpu=%d\n", simtime, cpu_ev->cpunum);
 654 
 655    cpu_ev->intrp = (intr_event *) intrp->next;
 656    currcpu->state = intrp->oldstate;
 657    currcpu->intrs++;
 658    currcpu->runintrtime += intrp->runtime;
 659    if (intrp->vector == IO_INTERRUPT) {
 660       currcpu->iointrs++;
 661       currcpu->runiointrtime += intrp->runtime;
 662       io_interrupt_complete ((ioreq_event *) intrp);
 663    } else if (intrp->vector == CLOCK_INTERRUPT) {
 664       currcpu->clockintrs++;
 665       currcpu->runclockintrtime += intrp->runtime;
 666       pf_clock_interrupt_complete(intrp);
 667    } else {
 668       fprintf (stderr, "pf_handle_intend_event: unknown interrupt vector (%d)\n", intrp->vector);
 669       exit(1);
 670    }
 671    intrp = cpu_ev->intrp;
 672    if (intrp == NULL) {
 673       if ((procp == NULL) && (pf_dispq)) {
 674          procp = pf_dispatch(cpu_ev->cpunum);
 675          assert (procp != NULL);
 676          pf_add_cswitch_event (NULL, procp, cpu_ev);
 677       }
 678       if ((procp == NULL) || (procp->idler != 0)) {
 679          if ((idlereset) && ((simtime - idlein) > (double) 50000.0)) {
 680             fprintf (outputfile, "Idle process has been running for more than 50 seconds\n");
 681             disksim_simstop();
 682          }
 683       }
 684    }
 685 }

 700 static void pf_handle_event (event *curr, cpu_event *cpu_ev)
 701 {
 702   if (pf_printhack) {
 703     fprintf (outputfile, "%f, entered pf_handle_event with event type %d\n", simtime, curr->type);
 704     fflush(outputfile);
 705   }
 706 
 707   switch (curr->type) {
 708   case IDLELOOP_EVENT:
 709     pf_handle_idleloop_event((idleloop_event *)curr, cpu_ev);
 710     break;
 711   case CSWITCH_EVENT:
 712     pf_handle_cswitch_event((cswitch_event *)curr, cpu_ev);
 713     break;
 714   case SLEEP_EVENT:
 715     pf_handle_sleep_event((sleep_event *)curr, cpu_ev);
 716     break;
 717   case WAKEUP_EVENT:
 718     pf_handle_wakeup_event((wakeup_event *)curr, cpu_ev);
 719     break;
 720   case IOREQ_EVENT:
 721     pf_handle_ioreq_event((ioreq_event *)curr, cpu_ev);
 722     curr = NULL;
 723     break;
 724   case IOACC_EVENT:
 725     pf_handle_ioacc_event((ioreq_event *)curr, cpu_ev);
 726     curr = NULL;
 727     break;
 728   case INTEND_EVENT:
 729     pf_handle_intend_event((intend_event *)curr, cpu_ev);
 730     break;
 731   case SYNTHIO_EVENT:
 732     pf_handle_synthio_event(curr, cpu_ev);
 733     curr = NULL;
 734     break;
 735   default:
 736     if ((curr->type >= IO_MIN_EVENT) && (curr->type <= IO_MAX_EVENT)) {
 737       pf_handle_io_internal_event((ioreq_event *)curr, cpu_ev);
 738       curr = NULL;
 739       break;
 740     }
 741     fprintf(stderr, "Trying to handle unknown pf event - %d\n", curr->type);
 742     exit(1);
 743   }
 744   addtoextraq((event *) curr);
 745 }

 789 void pf_handle_intr_event (intr_event *intrp, int cpunum)
 790 {
 791    cpu *currcpu = &cpus[cpunum];
 792    cpu_event *cpu_ev = currcpu->current;
 793    process *procp = cpu_ev->procp;
 794 
 795    if (pf_printhack)
 796    fprintf (outputfile, "%f\tINTR cpu=%d vector=%d cpustate=%d\n", simtime, cpunum, intrp->vector, currcpu->state);
 797 
 798    intrp->runtime = 0.0;
 799    intrp->flags = 0;
 800    intrp->next = cpu_ev->intrp;
 801    if (currcpu->state == CPU_IDLE) {
 802       currcpu->idletime += simtime - currcpu->idlestart;
 803       currcpu->falseidletime += pf_add_false_idle_time(simtime - currcpu->idlestart);
 804    } else {
 805       if (removefromintq((event *)cpu_ev) == 0) {
 806          fprintf(stderr, "Event from active CPU not pending on internal queue\n");
 807          exit(1);
 808       }
 809       if (intrp->next) {
 810          intrp->next->runtime += simtime - intrp->next->eventlist->time;
 811          intrp->next->eventlist->time = (cpu_ev->time - simtime) / currcpu->scale;
 812       } else if (procp == NULL) {
 813          currcpu->idleworktime += simtime - currcpu->idleevents->time;
 814          currcpu->idleevents->time = (cpu_ev->time - simtime) / currcpu->scale;
 815       } else {
 816          procp->runtime += simtime - procp->eventlist->time;
 817          procp->eventlist->time = (cpu_ev->time - simtime) / currcpu->scale;
 818       }
 819    }
 820    intrp->oldstate = currcpu->state;
 821    currcpu->state = CPU_INTR;
 822    cpu_ev->intrp = intrp;
 823    if (intrp->vector == IO_INTERRUPT) {
 824       pf_handle_io_intr_arrive(intrp, cpu_ev);
 825    } else if (intrp->vector == CLOCK_INTERRUPT) {
 826       pf_handle_clock_intr_arrive(intrp, cpu_ev);
 827    } else {
 828       fprintf(stderr, "Unknown interrupt vector: %d\n", intrp->vector);
 829       exit(1);
 830    }
 831    cpu_ev->time = currcpu->scale * pf_get_time_to_next_intr_event(intrp);
 832    cpu_ev->time += simtime;
 833    addtointq((event *) cpu_ev);
 834 /*
 835 fprintf (outputfile, "Exit pf_handle_intr_event\n");
 836 */
 837 }

 840 static void pf_handle_cpu_event (cpu_event *curr)
 841 {
 842    event *tmp;
 843    cpu *currcpu = &cpus[(curr->cpunum)];
 844 
 845    /*
 846     * if (simtime > 29000.0) {
 847     *   pf_printhack = 1;
 848     * }
 849     */
 850 
 851    if(pf_printhack) {
 852      fprintf (outputfile, "Entered handle_cpu_event: %d\n", curr->type);
 853      fflush(outputfile);
 854    }
 855 
 856    if(curr->intrp) {
 857       tmp = curr->intrp->eventlist;
 858       curr->intrp->runtime += simtime - tmp->time;
 859       curr->intrp->eventlist = tmp->next;
 860    } 
 861    else if(curr->procp) {
 862       curr->procp->lasteventtime = simtime;
 863       tmp = curr->procp->eventlist;
 864       curr->procp->runtime += simtime - tmp->time;
 865       curr->procp->eventlist = tmp->next;
 866    } 
 867    else {
 868       tmp = currcpu->idleevents;
 869       currcpu->idleworktime += simtime - tmp->time;
 870       currcpu->idleevents = tmp->next;
 871    }
 872 
 873    ASSERT(tmp != NULL);
 874    tmp->next = NULL;
 875    pf_handle_event(tmp, curr);
 876 
 877    if(curr->intrp) {
 878       curr->time = pf_get_time_to_next_intr_event(curr->intrp);
 879    } 
 880    else if(curr->procp == NULL) {
 881      if (currcpu->idleevents == NULL) {
 882        currcpu->state = CPU_IDLE;
 883        currcpu->idlestart = simtime;
 884        return;
 885      }
 886      curr->time = currcpu->idleevents->time;
 887      currcpu->idleevents->time = simtime;
 888      currcpu->state = CPU_IDLE_WORK;
 889    } 
 890    else {
 891      curr->time = pf_get_time_to_next_process_event(curr->procp);
 892    }
 893 
 894    curr->time = simtime + (currcpu->scale * curr->time);
 895    addtointq((event *) curr);
 896 
 897    
 898    if(pf_printhack) {
 899      fprintf(outputfile, "Exited handle_cpu_event\n");
 900      fflush(outputfile);
 901    }
 902 }

1065 void pf_cleanstats()
1066 {
1067    int i;
1068    intr_event *intrp;
1069    process *procp;
1070 
1071    for (i=0; i<numcpus; i++) {
1072       intrp = cpus[i].current->intrp;
1073       procp = cpus[i].current->procp;
1074       switch (cpus[i].state) {
1075          case CPU_IDLE:
1076                       cpus[i].idletime += simtime - cpus[i].idlestart;
1077                       cpus[i].falseidletime += pf_add_false_idle_time(simtime - cpus[i].idlestart);
1078                       break;
1079          case CPU_IDLE_WORK:
1080                       cpus[i].idleworktime += simtime - cpus[i].idleevents->time;
1081                       break;
1082          case CPU_PROCESS:
1083                       if (procp->idler == 0) {
1084                          lastuser = simtime;
1085                       }
1086                       procp->runtime += simtime - procp->eventlist->time;
1087                       break;
1088          case CPU_INTR:
1089                       intrp->runtime += simtime - intrp->eventlist->time;
1090                       break;
1091          default:
1092                       fprintf(stderr, "Unknown CPU state at pf_cleanstats - %d\n", cpus[i].state);
1093                       exit(1);
1094       }
1095    }
1096    procp = process_livelist;
1097    while (procp) {
1098       if (procp->stat == PROC_SLEEP) {
1099          procp->runsleep += simtime - procp->lastsleep;
1100          if (procp->iosleep) {
1101             procp->runiosleep += simtime - procp->lastsleep;
1102          }
1103       }
1104       procp = procp->livelist;
1105    }
1106 }

1279 {
1280    int i;
1281    double idletime = 0.0;
1282    double falseidletime = 0.0;
1283    double idleworktime = 0.0;

1287    double runswitchtime = 0.0;

1290    char cpustr2[81];
1291 
1292    if (startcpu == stopcpu) {
1293       sprintf(cpustr, "CPU #%d ", startcpu);
1294    } else {
1295       sprintf(cpustr, "CPU ");
1296    }
1297 
1298    cpucnt = stopcpu - startcpu + 1;
1299    for (i=startcpu; i<=stopcpu; i++) {
1300       idletime += cpus[i].idletime;
