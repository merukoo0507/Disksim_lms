 127  * segment in the cache -rcohen

 130 int disk_buffer_overlap (segment *seg, ioreq_event *curr)
 131 {
 132    int tmp;
 133 
 134    if ((curr->blkno >= seg->startblkno) && (curr->blkno < seg->endblkno)) {
 135       return(TRUE);
 136    }
 137 
 138    tmp = curr->blkno + curr->bcount;
 139    if ((tmp > seg->startblkno) && (tmp <= seg->endblkno)) {
 140       return(TRUE);
 141    }
 142 
 143    if ((curr->blkno <= seg->startblkno) && (tmp >= seg->endblkno)) {
 144       return(TRUE);
 145    }
 146 
 147    return(FALSE);
 148 }

 176    currdiskreq->seg = NULL;

 182 /* return real or effective owner of segment or NULL.  For effective, a 
 183    READ segment OWNER may be overridden by a READ with an outblkno that 
 184    is less than the (effective) OWNER's outblkno
 185 */
 186 
 187 diskreq* disk_buffer_seg_owner (segment *seg, int effective)
 188 {
 189    diskreq *seg_owner = seg->diskreqlist;
 190    diskreq *tmpdiskreq = seg->diskreqlist;
 191 
 192    while (seg_owner) {
 193       if (seg_owner->flags & SEG_OWNED) {
 194          break;
 195       }
 196       seg_owner = seg_owner->seg_next;
 197    }
 198 
 199    if (effective && seg->access && (seg->access->flags & READ)) {
 200       while (tmpdiskreq) {
 201          if ((seg->recyclereq != tmpdiskreq) &&
 202              (!tmpdiskreq->ioreqlist || 
 203               (tmpdiskreq->ioreqlist->flags & READ))) {
 204             if (!seg_owner || tmpdiskreq->outblkno < seg_owner->outblkno) {
 205                seg_owner = tmpdiskreq;
 206 /*
 207                fprintf(stderr,"%.6f  effective seg_owner found\n",simtime);fflush(stderr);
 208 */
 209             }
 210          }
 211          tmpdiskreq = tmpdiskreq->seg_next;
 212       }
 213    }
 214 
 215    return(seg_owner);
 216 }

 224 int disk_buffer_reusable_segment_check (disk *currdisk, segment *currseg)
 225 {
 226    diskreq *currdiskreq = currseg->diskreqlist;
 227    ioreq_event *currioreq;
 228 
 229 if (disk_printhack && (simtime >= disk_printhacktime)) {
 230 fprintf (outputfile, "%12.6f            Entering disk_buffer_reusable_segment_check\n",simtime);
 231 fflush(outputfile);
 232 }
 233 
 234    if (currdisk->acctime < 0.0 && !currseg->recyclereq && currdiskreq && 
 235        !currdiskreq->seg_next && currdiskreq->ioreqlist && 
 236        !(currdiskreq->flags & COMPLETION_RECEIVED)) {
 237       if (currdiskreq->ioreqlist->flags & READ) {
 238          currioreq = currdiskreq->ioreqlist;
 239          while (currioreq->next) {
 240             currioreq = currioreq->next;
 241          }
 242          if ((currdiskreq->outblkno >= currseg->startblkno) &&
 243              (currdiskreq->outblkno <  currseg->endblkno) &&
 244              ((currioreq->blkno + currioreq->bcount) > currseg->startblkno) &&
 245              ((currioreq->blkno + currioreq->bcount) <= currseg->endblkno)) {
 246             if (!(currdiskreq->flags & SEG_OWNED)) {
 247                disk_buffer_attempt_seg_ownership(currdisk,currdiskreq);
 248             }
 249 
 250 if (disk_printhack && (simtime >= disk_printhacktime)) {
 251 fprintf (outputfile, "                        segment %8p is reusable\n",currseg);
 252 fflush(outputfile);
 253 }
 254 
 255             return(TRUE);
 256          }
 257       } else {                  /* WRITE */
 258          currioreq = currdiskreq->ioreqlist;
 259          while (currioreq->next) {
 260             currioreq = currioreq->next;
 261          }
 262          if (currdiskreq->inblkno >= (currioreq->blkno + currioreq->bcount)) {
 263             if (!(currdiskreq->flags & SEG_OWNED)) {
 264                disk_buffer_attempt_seg_ownership(currdisk,currdiskreq);
 265             }
 266 
 267 if (disk_printhack && (simtime >= disk_printhacktime)) {
 268 fprintf (outputfile, "                        segment %8p is reusable\n",currseg);
 269 fflush(outputfile);
 270 }
 271 
 272             return(TRUE);
 273          }
 274       }
 275    }
 276 
 277 if (disk_printhack && (simtime >= disk_printhacktime)) {
 278 fprintf (outputfile, "                        segment %8p is NOT reusable\n",currseg);
 279 fflush(outputfile);
 280 }
 281 
 282    return(FALSE);
 283 }

 391 /* if BUFFER_READING, checks to see if cache can and should "fit in" blkno.
 392    if BUFFER_WRITING, checks to see if blkno is in the cache and can be
 393      written to media.
 394 */
 395 
 396 int disk_buffer_block_available (disk *currdisk, segment *seg, int blkno)
 397 {
 398    diskreq *seg_owner;
 399 
 400    if (blkno < 0) {
 401       return(FALSE);
 402    }
 403    if (seg->state == BUFFER_READING) {
 404       if ((blkno < seg->endblkno) && (blkno >= seg->startblkno)) {
 405          return(FALSE);
 406       }
 407       if ((blkno < seg->startblkno) && 
 408           ((!currdisk->readanyfreeblocks) ||
 409            (seg->endblkno != seg->startblkno))) {
 410          return(FALSE);
 411       }
 412       seg_owner = disk_buffer_seg_owner(seg,TRUE);
 413       if (seg_owner && !(seg_owner->flags & COMPLETION_SENT) && 
 414           ((blkno - seg_owner->outblkno) >= seg->size)) {
 415          return(FALSE);
 416       }
 417    } else if (seg->state == BUFFER_WRITING) {
 418       if ((blkno < currdisk->effectivehda->inblkno) || 
 419           (blkno >= seg->endblkno)) {
 420          return(FALSE);
 421       }
 422       if ((currdisk->effectivehda->ioreqlist) && 
 423           (currdisk->extra_write_disconnect)) {
 424          if (currdisk->effectivehda->ioreqlist->bcount < 45) {
 425             if ((seg->endblkno - seg->startblkno) < 
 426                 min(currdisk->effectivehda->ioreqlist->bcount, 10)) {
 427                return(FALSE);
 428             }
 429          } else {
 430             if ((seg->endblkno - seg->startblkno) < 15) {
 431                return(FALSE);
 432             }
 433          }
 434       }
 435    } else {
 436       fprintf(stderr, "Not actively using disk at disk_buffer_block_available\n");
 437       exit(1);
 438    }
 439    return(TRUE);
 440 }

 464 static void disk_buffer_stats (disk *currdisk, ioreq_event *curr, segment *seg, int hittype)
 465 {
 466    double size;
 467 
 468    if (!device_printbufferstats) {
 469       return;
 470    }
 471 
 472    /* Add stats for read hits on write data? */
 473 
 474    switch (hittype) {
 475 
 476       case BUFFER_NOMATCH:
 477          if (curr->flags & READ) {
 478             currdisk->stat.readmisses++;
 479          } else {
 480             currdisk->stat.writemisses++;
 481          }
 482          break;
 483 
 484       case BUFFER_WHOLE:
 485          currdisk->stat.fullreadhits++;
 486          break;
 487 
 488       case BUFFER_APPEND:
 489          currdisk->stat.appendhits++;
 490          break;
 491 
 492       case BUFFER_PREPEND:
 493          currdisk->stat.prependhits++;
 494          break;
 495 
 496       case BUFFER_PARTIAL:
 497          if (seg->startblkno <= curr->blkno) {
 498             size = (double) (seg->endblkno - curr->blkno);
 499          } else {
 500 /*
 501             size = (double) (curr->blkno + curr->bcount - seg->startblkno);
 502 */
 503             fprintf(stderr, "Tail hits not currently allowed.\n");
 504             exit(1);
 505          }
 506          if (seg->state == BUFFER_READING) {
 507 /*
 508       fprintf (outputfile, "Ongoing hit: blkno %d  bcount %d  segstart %d  segstop %d  read %d\n", curr->blkno, curr->bcount, seg->startblkno, seg->endblkno, seg->access->type);
 509 */
 510             currdisk->stat.readinghits++;
 511             currdisk->stat.runreadingsize += size;
 512             currdisk->stat.remreadingsize += curr->bcount - size;
 513          } else {
 514 /*
 515       fprintf (outputfile, "Partial hit: blkno %d  bcount %d  segstart %d  segstop %d  read %d\n", curr->blkno, curr->bcount, seg->startblkno, seg->endblkno, ((curr->flags & READ) && (seg->state == BUFFER_CLEAN)));
 516 */
 517             currdisk->stat.parthits++;
 518             currdisk->stat.runpartsize += size;
 519             currdisk->stat.rempartsize += curr->bcount - size;
 520          }
 521          break;
 522 
 523       default:
 524          fprintf(stderr, "Invalid hittype in disk_buffer_stats - blkno %d, bcount %d, state %d\n", curr->blkno, curr->bcount, seg->state);
 525          exit(1);
 526    }
 527    return;
 528 }

 531 /* segment selection priority for read requests:
 532    11   BUFFER_WHOLE,   BUFFER_DIRTY   (write data is freshest, less chance of 
 533                                         data scrolling off the segment quickly)
 534    10                   BUFFER_WRITING (write data is freshest)
 535    9                    BUFFER_READING (possible prefetch-stream hit)
 536    8                    BUFFER_CLEAN
 537    7    BUFFER_PARTIAL, BUFFER_DIRTY   (write data is freshest)
 538    6                    BUFFER_WRITING (write data is freshest)
 539    5                    BUFFER_READING (possible prefetch-stream hit, includes
 540                                         case of current prefetch block is first
 541                                         block in the new request)
 542    4                    BUFFER_CLEAN
 543    3    BUFFER_NOMATCH, BUFFER_EMPTY   (doesn't trash other data)
 544    2                    BUFFER_CLEAN
 545    1                    BUFFER_READING (must be last place for preempt check)
 546    
 547       not usable:
 548         BUFFER_NOMATCH, BUFFER_DIRTY
 549                         BUFFER_WRITING
 550                         BUFFER_CLEAN   (if pending requests)
 551                         BUFFER_READING (if not pre-emptable or pending requests)
 552         recycled segs
 553         dedicatedwriteseg
 554 
 555       not usable if !enablecache
 556         BUFFER_DIRTY
 557         BUFFER_WRITING
 558         BUFFER_READING (if non-stoppable)
 559 
 560       not usable if no read hits on write data:
 561         BUFFER_WHOLE,   BUFFER_DIRTY
 562                         BUFFER_WRITING
 563                         BUFFER_CLEAN (if previous request was a write)
 564         BUFFER_PARTIAL, BUFFER_DIRTY
 565                         BUFFER_WRITING
 566                         BUFFER_CLEAN (if previous request was a write)
 567 
 568       NO seg usable if overlapping dirty data exists which cannot be "hit"
 569         -- return BUFFER_COLLISION hittype if seg in question is not "reusable".
 570 
 571       NO seg usable if two segments are found with overlapping dirty data
 572         -- return BUFFER_COLLISION hittype.
 573 */

 609                tmpioreq = currdiskreq->ioreqlist;
 610                while (tmpioreq) {
 611                  if (disk_buffer_overlap(seg,tmpioreq)) {
 612                    if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 613                      currdiskreq->hittype = BUFFER_COLLISION;
 614                      currdiskreq->seg = NULL;
 615                    }
 616                    break;
 617                  }
 618                  tmpioreq = tmpioreq->next;
 619                }
 620              }
 621            } 
 622       else if (seg->state == BUFFER_EMPTY) {
 623         curr_value = 3;
 624         curr_hittype = BUFFER_NOMATCH;
 625       } 
 626       else if ((first_ioreq->blkno < seg->startblkno) 
 627                || (first_ioreq->blkno >= seg->endblkno)) 
 628       {
 629         curr_hittype = BUFFER_NOMATCH;
 630         if ((seg->state == BUFFER_CLEAN) && !(seg->diskreqlist)) {
 631           curr_value = 2;
 632         } 
 633         else if (seg->state == BUFFER_READING) {
 634           if ((seg->endblkno == first_ioreq->blkno) && 
 635               currdisk->almostreadhits) {
 636             /* special case: block currently being prefetched is the
 637              * first block of this request.  We count this as a
 638              * partial read hit.  
 639              */
 640             curr_hittype = BUFFER_PARTIAL;
 641             curr_value = 5;
 642           } 
 643           else {
 644             if (!seg->diskreqlist->ioreqlist) {
 645               curr_value = 1;
 646             }
 647           }
 648         } 
 649         else if ((first_ioreq->blkno < seg->startblkno) 
 650                  && ((seg->state == BUFFER_DIRTY) 
 651                      || (seg->state == BUFFER_WRITING))) 
 652         {
 653           /* check for collision with dirty data */
 654           tmpioreq = currdiskreq->ioreqlist;
 655           while (tmpioreq) {
 656             if (disk_buffer_overlap(seg,tmpioreq)) {
 657               if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 658                 currdiskreq->hittype = BUFFER_COLLISION;
 659                 currdiskreq->seg = NULL;
 660               }
 661               break;
 662             }
 663             tmpioreq = tmpioreq->next;
 664           }
 665         }
 666       } 
 667       else {
 668         switch (seg->state) {
 669         case BUFFER_DIRTY:
 670           if (currdisk->enablecache) {
 671             if ((currdisk->readhitsonwritedata) 
 672                 && (best_value < 10)) {
 673               curr_value = 11;
 674             } 
 675             else {
 676               if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 677                 currdiskreq->hittype = BUFFER_COLLISION;
 678                 currdiskreq->seg = NULL;
 679               }
 680             }
 681           }
 682           break;
 683         case BUFFER_WRITING:
 684           if (currdisk->enablecache) {
 685             if ((currdisk->readhitsonwritedata) 
 686                 && (best_value < 10)) 
 687             {
 688               curr_value = 10;
 689             } 
 690             else {
 691               if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 692                 currdiskreq->hittype = BUFFER_COLLISION;
 693                 currdiskreq->seg = NULL;
 694               }
 695             }
 696           }
 697           break;
 698         case BUFFER_READING:
 699           if (currdisk->enablecache) {
 700             curr_value = 9;
 701           }
 702           break;
 703         case BUFFER_CLEAN:
 704           if (currdisk->readhitsonwritedata 
 705               || (seg->access && (seg->access->flags & READ))) 
 706           {
 707             curr_value = 8;
 708           }
 709           break;
 710 
 711          default:
 712            ddbg_assert(0);
 713         }
 714 
 715         if (seg->endblkno >= (first_ioreq->blkno + first_ioreq->bcount)) {
 716           curr_hittype = BUFFER_WHOLE;
 717         } 
 718         else {
 719           curr_hittype = BUFFER_PARTIAL;
 720           curr_value -= 4;
 721         }
 722       }
 723 
 724       if (curr_value > best_value) {
 725          currdiskreq->seg = seg;
 726          currdiskreq->hittype = curr_hittype;
 727          best_value = curr_value;
 728       }
 729       seg = seg->next;
 730    }
 731 
 732    /* If BUFFER_NOMATCH && BUFFER_READING, perform preemption check.
 733     * Note that currdiskreq->seg should be set before the call.  
 734     */
 735 
 736    if ((best_value == 1) 
 737        && !disk_buffer_stopable_access(currdisk,currdiskreq)) 
 738    {
 739      currdiskreq->seg = NULL;
 740    }
 741    
 742 if (disk_printhack && (simtime >= disk_printhacktime)) {
 743 fprintf (outputfile, "                        segment = %8p, hittype = %d\n",currdiskreq->seg,currdiskreq->hittype);
 744 fflush(outputfile);
 745 }
 746 
 747  return currdiskreq->seg;
 748 }

 751 /* segment selection priority for write requests:
 752    7    BUFFER_APPEND,  BUFFER_WRITING (keep the stream going)
 753    6                    BUFFER_DIRTY
 754    5    BUFFER_PREPEND, BUFFER_DIRTY
 755    4    BUFFER_NOMATCH, BUFFER_CLEAN, overlapping (will be emptied anyways)
 756    3                    BUFFER_EMPTY   (don't trash other data)
 757    2                    BUFFER_CLEAN
 758    1    BUFFER_NOMATCH, BUFFER_READING (must be last place for preempt check)
 759 
 760    
 761       not usable:
 762         BUFFER_APPEND                  (if no combining writes)
 763         BUFFER_PREPEND                 (if no combining writes)
 764         BUFFER_NOMATCH, BUFFER_DIRTY  
 765                         BUFFER_WRITING
 766                         BUFFER_READING (if not pre-emptable or pending requests)
 767                         BUFFER_CLEAN   (if pending requests)
 768         recycled segs
 769 
 770       NO seg usable if overlapping data exists which cannot be "cleaned"
 771         -- return BUFFER_COLLISION hittype if seg is question isn't "reusable".
 772 */

 776 static segment* disk_buffer_select_write_segment(disk *currdisk, diskreq *currdiskreq)

 782    ioreq_event *last_ioreq;

 806           (seg != currdisk->dedicatedwriteseg)) {

 809               (seg->diskreqlist)) ||
 810              ((seg->state == BUFFER_READING) && 
 811               ((seg->diskreqlist->ioreqlist) || 
 812                (seg->diskreqlist->seg_next) ||
 813                !disk_buffer_stopable_access(currdisk,currdiskreq)))) {
 814             tmpioreq = currdiskreq->ioreqlist;

 820                   }
 821                   break;
 822                }
 823                tmpioreq = tmpioreq->next;

 825          }
 826       } else if (seg->state == BUFFER_EMPTY) {
 827          if (currdisk->numdirty < currdisk->numwritesegs) {

 829              || reusable_dirty_segment) {
 830 */

 832          }
 833       } else if (seg->state == BUFFER_READING) {
 834          if (!seg->diskreqlist->ioreqlist && !seg->diskreqlist->seg_next) {
 835             if (currdisk->numdirty < currdisk->numwritesegs) {
 836 /*
 837                 || reusable_dirty_segment) {
 838 */
 839                curr_value = 1;
 840             }
 841          } else {
 842             tmpioreq = currdiskreq->ioreqlist;
 843             while (tmpioreq) {
 844                if (disk_buffer_overlap(seg,tmpioreq)) {
 845                   if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 846                      currdiskreq->hittype = BUFFER_COLLISION;
 847                      currdiskreq->seg = NULL;
 848                   }
 849                   break;
 850                }
 851                tmpioreq = tmpioreq->next;
 852             }
 853          }
 854       } else if (seg->state == BUFFER_CLEAN) {
 855          if (!seg->diskreqlist) {
 856             if (currdisk->numdirty < currdisk->numwritesegs) {
 857 /*
 858                 || reusable_dirty_segment) {
 859 */
 860                curr_value = 2;
 861                first_ioreq = currdiskreq->ioreqlist;
 862                while (first_ioreq) {
 863                   if (disk_buffer_overlap(seg,first_ioreq)) {
 864                      curr_value = 4;
 865                      break;
 866                   }
 867                   first_ioreq = first_ioreq->next;
 868                }
 869             }
 870          } else {
 871             /* check for collision with uncleanable data */
 872             tmpioreq = currdiskreq->ioreqlist;
 873             while (tmpioreq) {
 874                if (disk_buffer_overlap(seg,tmpioreq)) {
 875                   if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 876                      currdiskreq->hittype = BUFFER_COLLISION;
 877                      currdiskreq->seg = NULL;
 878                   }
 879                   break;
 880                }
 881                tmpioreq = tmpioreq->next;
 882             }
 883          }
 884       } else if ((!currdisk->writecomb) || 
 885                  (seg->outstate == BUFFER_TRANSFERING)) {
 886          /* no PREPEND's or APPEND's allowed */
 887          /* check for collision with uncleanable data */
 888          tmpioreq = currdiskreq->ioreqlist;
 889          while (tmpioreq) {
 890             if (disk_buffer_overlap(seg,tmpioreq)) {
 891                if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 892                   currdiskreq->hittype = BUFFER_COLLISION;
 893                   currdiskreq->seg = NULL;
 894                }
 895                break;
 896             }
 897             tmpioreq = tmpioreq->next;
 898          }
 899 /*
 900          if ((currdisk->numdirty >= currdisk->numwritesegs) &&
 901              !reusable_dirty_segment &&
 902              disk_buffer_reusable_segment_check(currdisk, seg)) {
 903             reusable_dirty_segment = TRUE;
 904             currdiskreq->seg = NULL;
 905             currdiskreq->hittype = BUFFER_NOMATCH;
 906             seg = currdisk->seglist;
 907             best_value = 0;
 908             continue;
 909          }
 910 */
 911       } else {
 912          first_ioreq = currdiskreq->ioreqlist;
 913          tmp_diskreq = holddiskreq = seg->diskreqlist;
 914          while (tmp_diskreq->seg_next) {
 915             tmp_diskreq = tmp_diskreq->seg_next;
 916             if (!(tmp_diskreq->ioreqlist->flags & READ)) {
 917                holddiskreq = tmp_diskreq;
 918             }
 919          }
 920          last_ioreq = holddiskreq->ioreqlist;
 921          while (last_ioreq && last_ioreq->next) {
 922             last_ioreq = last_ioreq->next;
 923          }
 924          if ((first_ioreq->blkno == (last_ioreq->blkno + last_ioreq->bcount)) &&
 925              (first_ioreq->blkno == seg->endblkno) &&
 926              (seg->size > (first_ioreq->blkno - (disk_buffer_seg_owner(seg,FALSE))->inblkno))) {
 927             curr_hittype = BUFFER_APPEND;
 928             curr_value = ((seg->state == BUFFER_WRITING) ? 7 : 6);
 929          } else if (seg->state == BUFFER_DIRTY) {
 930             int total_bcount = currdiskreq->ioreqlist->bcount;
 931 
 932             last_ioreq = currdiskreq->ioreqlist;
 933             while (last_ioreq->next) {
 934                last_ioreq = last_ioreq->next;
 935                total_bcount += last_ioreq->bcount;
 936             }
 937             holddiskreq = seg->diskreqlist;
 938             while (holddiskreq->ioreqlist->flags & READ) {
 939                holddiskreq = holddiskreq->seg_next;
 940             }
 941             first_ioreq = holddiskreq->ioreqlist;
 942             if ((first_ioreq->blkno == (last_ioreq->blkno + last_ioreq->bcount)) &&
 943                 (first_ioreq->blkno == seg->startblkno) &&
 944                 ((seg->startblkno == seg->endblkno) || (seg->hold_bcount == 0)) &&
 945                 (seg->size >= ((seg->endblkno - seg->startblkno) + total_bcount)) &&
 946                 (!currdisk->extradisc_diskreq || 
 947                  (currdisk->extradisc_diskreq->seg != seg))) {
 948                curr_hittype = BUFFER_PREPEND;
 949                curr_value = 5;
 950             }
 951          }
 952          if (curr_hittype == BUFFER_NOMATCH) {
 953             /* check for collision with uncleanable data */
 954             tmpioreq = currdiskreq->ioreqlist;
 955             while (tmpioreq) {
 956                if (disk_buffer_overlap(seg,tmpioreq)) {
 957                   if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 958                      currdiskreq->hittype = BUFFER_COLLISION;
 959                      currdiskreq->seg = NULL;
 960                   }
 961                   break;
 962                }
 963                tmpioreq = tmpioreq->next;
 964             }
 965 /*
 966             if ((currdisk->numdirty >= currdisk->numwritesegs) &&
 967                 !reusable_dirty_segment &&
 968                 disk_buffer_reusable_segment_check(currdisk, seg)) {
 969                reusable_dirty_segment = TRUE;
 970                currdiskreq->seg = NULL;
 971                currdiskreq->hittype = BUFFER_NOMATCH;
 972                seg = currdisk->seglist;
 973                best_value = 0;
 974                continue;
 975             }
 976 */
 977          }
 978       }
 979       if (curr_value > best_value) {
 980          currdiskreq->seg = seg;
 981          currdiskreq->hittype = curr_hittype;
 982          best_value = curr_value;
 983       }
 984       seg = seg->next;
 985    }
 986 
 987    /* If BUFFER_NOMATCH && BUFFER_READING, perform preemption check.
 988       Note that currdiskreq->seg should be set before the call.
 989    */
 990 
 991    if ((best_value == 1) &&
 992        !disk_buffer_stopable_access(currdisk,currdiskreq)) {
 993       /* check for collision with uncleanable data */
 994       tmpioreq = currdiskreq->ioreqlist;
 995       while (tmpioreq) {
 996          if (disk_buffer_overlap(currdiskreq->seg,tmpioreq)) {
 997             if (!disk_buffer_reusable_segment_check(currdisk, seg)) {
 998                currdiskreq->hittype = BUFFER_COLLISION;
 999             }
1000             break;
1001          }
1002          tmpioreq = tmpioreq->next;
1003       }
1004       currdiskreq->seg = NULL;
1005    }
1006 
1007 if (disk_printhack && (simtime >= disk_printhacktime)) {
1008 fprintf (outputfile, "                        segment = %8p, hittype = %d\n",currdiskreq->seg,currdiskreq->hittype);
1009 fflush(outputfile);
1010 }
1011 
1012    return(currdiskreq->seg);
1013 }

1060 /* check for BUFFER hit (and no COLLISIONS's) */
1061 
1062 static int disk_buffer_check_read_segments (disk *currdisk, ioreq_event *currioreq, int* buffer_reading)
1063 {
1064    segment *seg;
1065    int return_hittype = BUFFER_NOMATCH;
1066    int read_hit_on_dirty_data = FALSE;
1067 
1068 if ((disk_printhack > 1) && (simtime >= disk_printhacktime)) {
1069 fprintf (outputfile, "%12.6f  %8p  Entering disk_buffer_check_read_segments\n",simtime,currioreq);
1070 fflush(outputfile);
1071 }
1072 
1073    if (!currdisk->enablecache) {
1074       return(BUFFER_NOMATCH);
1075    }
1076 
1077    seg = currdisk->seglist;
1078    while (seg) {
1079       if (seg->recyclereq) {
1080       } else if (seg->state == BUFFER_EMPTY) {
1081       } else if ((currdisk->dedicatedwriteseg) && 
1082           (seg == currdisk->dedicatedwriteseg)) {
1083          /* check for collision with dirty data */
1084          if ((seg->state == BUFFER_DIRTY) || (seg->state == BUFFER_WRITING)) {
1085             if (disk_buffer_overlap(seg,currioreq) &&
1086                 !disk_buffer_reusable_segment_check(currdisk, seg)) {
1087                return(BUFFER_COLLISION);
1088             }
1089          }
1090       } else if ((currioreq->blkno < seg->startblkno) || 
1091                  (currioreq->blkno >= seg->endblkno)) {
1092          if (seg->state == BUFFER_CLEAN) {
1093          } else if (seg->state == BUFFER_READING) {
1094             if ((seg->endblkno == currioreq->blkno) && 
1095                 currdisk->almostreadhits) {
1096                /* special case:  block currently being prefetched is
1097                   the first block of this request.  We count this as
1098                   a partial read hit.
1099                */
1100                if (return_hittype == BUFFER_NOMATCH) {
1101                   *buffer_reading = TRUE;
1102                   return_hittype = BUFFER_PARTIAL;
1103                }
1104             }
1105          } else if (currioreq->blkno < seg->startblkno) {
1106             /* check for collision with dirty data */
1107             if (disk_buffer_overlap(seg,currioreq) &&
1108                 !disk_buffer_reusable_segment_check(currdisk, seg)) {
1109                return(BUFFER_COLLISION);
1110             }
1111          }
1112       } else {
1113          switch (seg->state) {
1114          case BUFFER_DIRTY:
1115          case BUFFER_WRITING:
1116            if (read_hit_on_dirty_data ||
1117                (!currdisk->readhitsonwritedata &&
1118                 !disk_buffer_reusable_segment_check(currdisk, seg))) {
1119              return(BUFFER_COLLISION);
1120            }
1121            read_hit_on_dirty_data = TRUE;
1122            break;
1123          case BUFFER_READING:
1124            break;
1125          case BUFFER_CLEAN:
1126                if (!currdisk->readhitsonwritedata &&
1127                    seg->access && !(seg->access->flags & READ)) {
1128                  continue;
1129                }
1130                break;
1131          default:
1132            ddbg_assert(0);
1133          }
1134          if (seg->endblkno >= (currioreq->blkno + currioreq->bcount)) {
1135             return_hittype = BUFFER_WHOLE;
1136             *buffer_reading = (seg->state == BUFFER_READING);
1137          } else if (return_hittype != BUFFER_WHOLE) {
1138             return_hittype = BUFFER_PARTIAL;
1139             *buffer_reading = (seg->state == BUFFER_READING);
1140          }
1141       }
1142       seg = seg->next;
1143    }
1144 
1145 if ((disk_printhack > 1) && (simtime >= disk_printhacktime) && 
1146     (return_hittype != BUFFER_NOMATCH)) {
1147 fprintf (outputfile, "                        HITable segment found\n");
1148 fflush(outputfile);
1149 }
1150 
1151    return(return_hittype);
1152 
1153 }

1207          if (disk_buffer_overlap(seg,currioreq) &&
1208              !disk_buffer_reusable_segment_check(currdisk, seg)) {
1209             return(BUFFER_COLLISION);
1210          }
1211       } 
1212       else {
1213          tmp_diskreq = holddiskreq = seg->diskreqlist;
1214          while (tmp_diskreq->seg_next) {
1215             tmp_diskreq = tmp_diskreq->seg_next;
1216             if (!(tmp_diskreq->ioreqlist->flags & READ)) {
1217                holddiskreq = tmp_diskreq;
1218             }
1219          }
1220          last_ioreq = holddiskreq->ioreqlist;
1221          while (last_ioreq && last_ioreq->next) {
1222             last_ioreq = last_ioreq->next;
1223          }
1224          if ((currioreq->blkno == (last_ioreq->blkno + last_ioreq->bcount)) &&
1225              (currioreq->blkno == seg->endblkno) &&
1226              (seg->size > (currioreq->blkno - (disk_buffer_seg_owner(seg,FALSE))->inblkno))) {
1227             return_hittype = BUFFER_APPEND;
1228          } 
1229          else {
1230             /* check for collision with uncleanable data */
1231             if (disk_buffer_overlap(seg,currioreq) &&
1232                 !disk_buffer_reusable_segment_check(currdisk, seg)) {
1233                return(BUFFER_COLLISION);
1234             }
1235          }
1236       }
1237       seg = seg->next;
1238    }
1239 
1240 if ((disk_printhack > 1) && (simtime >= disk_printhacktime) && 
1241     (return_hittype == BUFFER_APPEND)) {
1242 fprintf (outputfile, "                        APPENDable segment found\n");
1243 fflush(outputfile);
1244 }
1245 
1246    return(return_hittype);
1247 }

1263 void disk_buffer_set_segment (disk *currdisk, diskreq *currdiskreq)
1264 {
1265    segment     *seg = currdiskreq->seg;
1266    segment     *tmp_seg;
1267    diskreq     *tmp_diskreq;
1268    ioreq_event *tmp_ioreq;
1269    int          is_read = (currdiskreq->ioreqlist->flags & READ);
1270 
1271 if (disk_printhack && (simtime >= disk_printhacktime)) {
1272 fprintf (outputfile, "%12.6f  %8p  Entering disk_buffer_set_segment\n",simtime,currdiskreq);
1273 fflush(outputfile);
1274 }
1275 
1276    if (!seg) {
1277       fprintf(stderr, "diskreq has NULL segment in disk_buffer_set_segment\n");
1278       exit(1);
1279    }
1280 
1281    /* check to make sure segment/diskreq combination is valid */
1282 
1283    if (is_read && (seg == currdisk->dedicatedwriteseg)) {
1284       fprintf(stderr, "Read request about to use the write segment\n");
1285       exit(1);
1286    }
1287 
1288    if (seg->recyclereq && (currdiskreq != seg->recyclereq)) {
1289       fprintf(stderr, "non-recyclereq detected for recycled segment in disk_buffer_set_segment\n");
1290       exit(1);
1291    }
1292 
1293    switch (seg->state) {
1294       case BUFFER_EMPTY:
1295          if (currdiskreq == seg->recyclereq) {
1296             fprintf(stderr, "seg->state of BUFFER_EMPTY for recyclereq in disk_buffer_set_segment\n");
1297             exit(1);
1298          }
1299          break;
1300 
1301       case BUFFER_CLEAN:
1302          if (currdiskreq == seg->recyclereq) {
1303             fprintf(stderr, "seg->state of BUFFER_CLEAN for recyclereq in disk_buffer_set_segment\n");
1304             exit(1);
1305          }
1306          if (is_read && currdiskreq->hittype != BUFFER_NOMATCH && !currdisk->readhitsonwritedata && seg->access && !(seg->access->flags & READ)) {
1307             fprintf(stderr, "Unallowed read hit on write data disk_buffer_set_segment\n");
1308             exit(1);
1309          }
1310          break;
1311 
1312       case BUFFER_DIRTY:
1313          if (currdiskreq == seg->recyclereq) {
1314             fprintf(stderr, "seg->state of BUFFER_DIRTY for recyclereq in disk_buffer_set_segment\n");
1315             exit(1);
1316          }
1317          if (currdiskreq->hittype == BUFFER_NOMATCH) {
1318             fprintf(stderr, "Attempt to overwrite dirty data in disk_buffer_set_segment\n");
1319             exit(1);
1320          } else if (is_read && !currdisk->readhitsonwritedata) {
1321             fprintf(stderr, "Unallowed read hit on write data disk_buffer_set_segment\n");
1322             exit(1);
1323          }
1324          if (!is_read) {
1325             currdisk->stat.writecombs++;
1326          }
1327          break;
1328 
1329       case BUFFER_READING:
1330          if (currdiskreq == seg->recyclereq) {
1331             if (!(currdiskreq->ioreqlist->flags & READ)) {
1332                fprintf(stderr, "seg->state of BUFFER_READING for write recyclereq in disk_buffer_set_segment\n");
1333                exit(1);
1334             }
1335          } else if (!is_read || (currdiskreq->hittype == BUFFER_NOMATCH)) {
1336             if (!disk_buffer_stopable_access(currdisk,currdiskreq)) {
1337                fprintf(stderr, "Attempt to re-target active read segment in disk_buffer_set_segment\n");
1338                exit(1);
1339             }
1340          }
1341          break;
1342 
1343       case BUFFER_WRITING:
1344          if (currdiskreq == seg->recyclereq) {
1345             if (currdiskreq->ioreqlist->flags & READ) {
1346                fprintf(stderr, "seg->state of BUFFER_WRITING for read recyclereq in disk_buffer_set_segment\n");
1347                exit(1);
1348             }
1349          } else {
1350             if ((currdiskreq->hittype == BUFFER_NOMATCH) || 
1351                 (currdiskreq->hittype == BUFFER_PREPEND)) {
1352                fprintf(stderr, "Attempt to re-target active write segment in disk_buffer_set_segment\n");
1353                exit(1);
1354             }
1355             if (!is_read) {
1356                currdisk->stat.writecombs++;
1357             }
1358          }
1359          break;
1360 
1361       default:
1362          fprintf(stderr, "Invalid segment state in disk_buffer_set_segment - blkno %d, bcount %d, state %d\n", currdiskreq->ioreqlist->blkno, currdiskreq->ioreqlist->bcount, seg->state);
1363          exit(1);
1364    }
1365 
1366    /* end of sanity checks, now do the work */
1367 
1368    disk_buffer_stats(currdisk,currdiskreq->ioreqlist,seg,currdiskreq->hittype);
1369 
1370    if (LRU_at_seg_list_head) {
1371       /* place segment at beginning of disk's segment list (for LRU) */
1372       if (seg->prev) {
1373          seg->prev->next = seg->next;
1374          if (seg->next) {
1375             seg->next->prev = seg->prev;
1376          }
1377          seg->next = currdisk->seglist;
1378          currdisk->seglist->prev = seg;
1379          seg->prev = NULL;
1380          currdisk->seglist = seg;
1381       }
1382    } else {
1383       /* place segment at end of disk's segment list (for LRU) */
1384       if (seg->next) {
1385          if (seg->prev) {
1386             seg->prev->next = seg->next;
1387          } else {
1388          currdisk->seglist = seg->next;
1389          }
1390          seg->next->prev = seg->prev;
1391          tmp_seg = currdisk->seglist;
1392          while (tmp_seg->next) {
1393          tmp_seg = tmp_seg->next;
1394          }
1395          tmp_seg->next = seg;
1396          seg->prev = tmp_seg;
1397          seg->next = NULL;
1398       }
1399    }
1400 
1401    /* clear out clean, non-active, overlapping segments if write */
1402 
1403    if (!is_read) {
1404       tmp_seg = currdisk->seglist;
1405       while (tmp_seg) {
1406          if (tmp_seg != seg) {
1407             tmp_ioreq= currdiskreq->ioreqlist;
1408             while (tmp_ioreq) {
1409                if ((tmp_seg->state != BUFFER_EMPTY) &&
1410                    disk_buffer_overlap(tmp_seg,tmp_ioreq)) {
1411                   if ((tmp_seg->state == BUFFER_CLEAN) &&
1412                       (tmp_seg->diskreqlist == NULL)) {
1413                      tmp_seg->state = BUFFER_EMPTY;
1414 /*
1415                   } else {
1416                      fprintf(stderr, "Request overlaps with non-clean segment in disk_buffer_set_segment\n");
1417                      exit(1);
1418 */
1419                   }
1420                }
1421                tmp_ioreq = tmp_ioreq->next;
1422             }
1423          }
1424          tmp_seg = tmp_seg->next;
1425       }
1426    }
1427 
1428 
1429    /* add diskreq to seg's diskreq list, sorted in ascending blkno order */
1430 
1431    if (!seg->diskreqlist) {
1432       if (seg->access == NULL) {
1433          seg->access = ioreq_copy(currdiskreq->ioreqlist);
1434          seg->access->type = NULL_EVENT;
1435       }
1436       seg->diskreqlist = currdiskreq;
1437       currdiskreq->seg_next = NULL;
1438    } else {                             /* non-empty list */
1439       diskreq *prev_diskreq = 0;
1440       tmp_diskreq = seg->diskreqlist;
1441       while (tmp_diskreq) {
1442          if (tmp_diskreq->ioreqlist && 
1443              (currdiskreq->ioreqlist->blkno < tmp_diskreq->ioreqlist->blkno)) {
1444             currdiskreq->seg_next = tmp_diskreq;
1445             if (seg->diskreqlist == tmp_diskreq) {
1446                seg->diskreqlist = currdiskreq;
1447             } else {
1448                prev_diskreq->seg_next = currdiskreq;
1449             }
1450             break;
1451          }
1452          prev_diskreq = tmp_diskreq;
1453          tmp_diskreq = tmp_diskreq->seg_next;
1454       }
1455       if (!tmp_diskreq) {
1456          prev_diskreq->seg_next = currdiskreq;
1457          currdiskreq->seg_next = NULL;
1458       }
1459    }
1460 
1461    /* set watermark */
1462    /* I'm still figuring out this watermark business -rcohen */
1463 
1464    tmp_ioreq = currdiskreq->ioreqlist;
1465 
1466    if (is_read) {
1467       if (currdisk->reqwater) {
1468          currdiskreq->watermark = max(1, (int) ((double) min(seg->size, tmp_ioreq->bcount) * currdisk->readwater));
1469       } else {
1470          currdiskreq->watermark = (int) (((double) seg->size * currdisk->readwater) + (double) 0.9999999999);
1471       }
1472    } else {                     /* WRITE */
1473       if (currdisk->reqwater) {
1474          currdiskreq->watermark = max(1, (int) ((double) min(seg->size, tmp_ioreq->bcount) * currdisk->writewater));
1475       } else {
1476          currdiskreq->watermark = (int) (((double) seg->size * currdisk->writewater) + (double) 0.9999999999);
1477       }
1478    }
1479 }

1482 int disk_buffer_attempt_seg_ownership (disk *currdisk, diskreq *currdiskreq)
1483 {
1484    segment     *seg = currdiskreq->seg;
1485    diskreq     *tmp_diskreq;
1486    ioreq_event *currioreq = currdiskreq->ioreqlist;
1487    int         currdiskreq_found = FALSE;
1488    int         write_found = FALSE;
1489    int         write_incomplete = 0;
1490    ioreq_event *tmpioreq;
1491 
1492 if (disk_printhack && (simtime >= disk_printhacktime)) {
1493 fprintf (outputfile, "%12.6f  %8p  Entering disk_buffer_attempt_seg_ownership\n",simtime,currdiskreq);
1494 fflush(outputfile);
1495 }
1496 
1497    if (!seg) {
1498       fprintf(stderr, "diskreq has NULL segment in disk_buffer_attempt_seg_ownership\n");
1499       exit(1);
1500    }
1501 
1502    if (currdiskreq->flags & SEG_OWNED) {
1503       fprintf(stderr, "diskreq already owns seg in disk_buffer_attempt_seg_ownership\n");
1504       exit(1);
1505    }
1506 
1507    tmp_diskreq = seg->diskreqlist;
1508    while (tmp_diskreq) {
1509       if (tmp_diskreq->flags & SEG_OWNED) {
1510          if (currdiskreq->hittype == BUFFER_PREPEND) {
1511             if (seg->state != BUFFER_DIRTY) {
1512                fprintf(stderr, "PREPEND of active request detected in disk_buffer_attempt_seg_ownership\n");
1513                exit(1);
1514             }
1515             /* release ownership to new request and set up hold areas */
1516             tmp_diskreq->flags &= ~SEG_OWNED;
1517             tmp_diskreq->hittype = BUFFER_APPEND;
1518             if (seg->endblkno != seg->startblkno) {
1519                if (seg->hold_bcount != 0) {
1520                   fprintf(stderr, "PREPEND of segment already 'holding' data detected in disk_buffer_attempt_seg_ownership\n");
1521                   exit(1);
1522                }
1523                seg->hold_blkno = seg->startblkno;
1524                seg->hold_bcount = seg->endblkno - seg->startblkno;
1525             }
1526          } else {
1527             break;
1528          }
1529       }
1530       if (tmp_diskreq == currdiskreq) {
1531          currdiskreq_found = TRUE;
1532       } else if (!currdiskreq_found && tmp_diskreq->ioreqlist && 
1533                  !(tmp_diskreq->ioreqlist->flags & READ)) {
1534          write_found = TRUE;
1535          tmpioreq = tmp_diskreq->ioreqlist;
1536          while (tmpioreq->next) {
1537             tmpioreq = tmpioreq->next;
1538          }
1539          if (tmp_diskreq->inblkno < (tmpioreq->blkno + tmpioreq->bcount)) {
1540             write_incomplete = TRUE;
1541          }
1542       }
1543       tmp_diskreq = tmp_diskreq->seg_next;
1544    }
1545 
1546 
1547    if (!tmp_diskreq) {
1548       ASSERT(!write_found || !write_incomplete);
1549       currdiskreq->flags |= SEG_OWNED;
1550       if (currioreq &&
1551           (!currdisk->effectivehda || 
1552            (currdisk->effectivehda == currdiskreq) ||
1553            (currdisk->effectivehda->seg != seg))) {
1554          seg->access->flags = currioreq->flags;
1555       }
1556 
1557       /* set state, startblkno, endblkno, and readahead fields */
1558 
1559       if (!currioreq || (currioreq->flags & READ)) {
1560 
1561          /* check to see if the cache segment needs to be reset */
1562 
1563 #if 0   /* GROK: removed on 10/10/99, because it causes problems, but */
1564         /* we are not sure that removing this won't break something   */
1565         /* else down the road...                                      */
1566          if ((currdiskreq->hittype != BUFFER_NOMATCH) &&
1567              (seg->startblkno > currdiskreq->outblkno)) {
1568             currdiskreq->hittype = BUFFER_NOMATCH;
1569          }
1570 #endif
1571 
1572          if (currdiskreq->hittype == BUFFER_NOMATCH) {
1573             ASSERT(currioreq != NULL);
1574             ASSERT1(((seg->access->type == NULL_EVENT) ||
1575                      (seg->recyclereq == currdiskreq)),"seg->access->type",seg->access->type);
1576             if (currdiskreq != seg->recyclereq) {
1577                seg->state = BUFFER_CLEAN;
1578                seg->access->blkno = currdiskreq->outblkno;
1579             }
1580             seg->startblkno = seg->endblkno = currdiskreq->outblkno;
1581             seg->minreadaheadblkno = seg->maxreadaheadblkno = -1;
1582             seg->hold_bcount = 0;
1583          }
1584 
1585          if (currioreq) {
1586             while (currioreq->next) {
1587                currioreq = currioreq->next;
1588             }
1589             seg->minreadaheadblkno = max(seg->minreadaheadblkno, min((currioreq->blkno + currioreq->bcount + currdisk->minreadahead), currdisk->model->dm_sectors));
1590 
1591             seg->maxreadaheadblkno = max(seg->maxreadaheadblkno, min(disk_buffer_get_max_readahead(currdisk,seg,currioreq), currdisk->model->dm_sectors));
1592 
1593             if ((seg->endblkno >= (currioreq->blkno + currioreq->bcount)) &&
1594                 (currdisk->effectivehda == currdiskreq)) {
1595                seg->access->flags |= BUFFER_BACKGROUND;
1596             }
1597          }
1598 
1599       } else {                  /* WRITE */
1600          if (!currdisk->effectivehda || 
1601              (currdisk->effectivehda == currdiskreq) ||
1602              (currdisk->effectivehda->seg != seg)) {
1603             if ((currdiskreq != seg->recyclereq) &&
1604                 (seg->access->type == NULL_EVENT)){
1605                seg->access->blkno = currdiskreq->inblkno;
1606             }
1607             if (currdiskreq->flags & COMPLETION_RECEIVED) {
1608                seg->access->flags |= BUFFER_BACKGROUND;
1609             }
1610          }
1611          if (currdiskreq->hittype == BUFFER_NOMATCH) {
1612             seg->minreadaheadblkno = seg->maxreadaheadblkno = -1;
1613             seg->hold_bcount = 0;
1614             seg->startblkno = seg->endblkno = currdiskreq->inblkno;
1615             if (currdiskreq != seg->recyclereq) {
1616                seg->state = BUFFER_DIRTY;
1617                currdisk->numdirty++;
1618 if ((disk_printhack > 1) && (simtime >= disk_printhacktime)) {
1619 fprintf (outputfile, "                        numdirty++ = %d\n",currdisk->numdirty);
1620 fflush(outputfile);
1621 }
1622                ASSERT1(((currdisk->numdirty >= 0) && (currdisk->numdirty <= currdisk->numwritesegs)),"numdirty",currdisk->numdirty);
1623             }
1624          } else if (currdiskreq->hittype == BUFFER_PREPEND) {
1625             seg->startblkno = seg->endblkno = currdiskreq->inblkno;
1626          }
1627       }
1628    }
1629 if (disk_printhack && (simtime >= disk_printhacktime)) {
1630 fprintf (outputfile, "                        Ownership = %d\n",(currdiskreq->flags & SEG_OWNED));
1631 fflush(outputfile);
1632 }
1633 
1634    return(currdiskreq->flags & SEG_OWNED);
1635 }

1639 {
1640    int endreq;
1641 
1642    if (!(curr->flags & READ)) {
1643       fprintf(stderr, "No read-ahead for write accesses, in disk_buffer_get_max_readahead\n");
1644       exit(1);

1646 
1647    endreq = curr->blkno + curr->bcount;
1648 
1649    if (currdisk->contread == BUFFER_NO_READ_AHEAD) {
1650       return(endreq);
1651    } else if (currdisk->contread == BUFFER_DEC_PREFETCH_SCHEME) {
1652       int startlbn;
1653       int endlbn;
1654       struct dm_pbn pbn;
1655       //      band* bandptr;

1658

1661       currdisk->model->layout->dm_translate_ltop(currdisk->model,

1666

1668                                                        &pbn,
1669                                                        &startlbn,

1692         }

1694    }
1695 
1696    return(endreq + seg->size);
1697 }
