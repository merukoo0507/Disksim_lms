 131 int ioreq_compare (ioreq_event *first, ioreq_event *second)
 132 {
 133    int diff = 0;
 134    diff |= first->time != second->time;
 135    diff |= first->type != second->type;
 136    //diff |= first->next != second->next;
 137    //diff |= first->prev != second->prev;
 138    diff |= first->bcount != second->bcount;
 139    diff |= first->blkno != second->blkno;
 140    diff |= first->flags != second->flags;
 141    diff |= first->busno != second->busno;
 142    diff |= first->slotno != second->slotno;
 143    diff |= first->devno != second->devno;
 144    diff |= first->opid != second->opid;
 145    diff |= first->buf != second->buf;
 146    diff |= first->cause != second->cause;
 147    diff |= first->tempint1 != second->tempint1;
 148    diff |= first->tempint2 != second->tempint2;
 149    diff |= first->tempptr1 != second->tempptr1;
 150    diff |= first->tempptr2 != second->tempptr2;
 151    return(diff);
 152 }

 155 void io_catch_stray_events (ioreq_event *curr)
 156 {
 157    switch (curr->type) {
 158    case DEVICE_DATA_TRANSFER_COMPLETE:
 159      curr->time = device_get_blktranstime(curr);
 160      curr->time = simtime + (curr->time * (double) curr->bcount);
 161      addtointq((event *) curr);
 162      break;
 163      
 164    default:
 165      fprintf(stderr, "Unknown event type at io_catch_stray_events\n");
 166      exit(1);
 167    }

 211 void io_internal_event(ioreq_event *curr)
 212 {
 213    ASSERT(curr != NULL);
 214 /*
 215 fprintf (outputfile, "%f: io_internal_event entered with event type %d, %f\n", curr->time, curr->type, simtime);
 216 */
 217    switch (curr->type) {
 218    case IO_REQUEST_ARRIVE:
 219      iodriver_request(0, curr);
 220      break;
 221      
 222    case IO_ACCESS_ARRIVE:
 223      iodriver_schedule(0, curr);
 224      break;
 225      
 226    case IO_INTERRUPT_ARRIVE:
 227      iodriver_interrupt_arrive(0, (intr_event *) curr);
 228      break;
 229 
 230    case IO_RESPOND_TO_DEVICE:
 231      iodriver_respond_to_device(0, (intr_event *) curr->tempptr1);
 232      addtoextraq((event *) curr);
 233      break;
 234      
 235    case IO_ACCESS_COMPLETE:
 236      iodriver_access_complete(0, (intr_event *) curr->tempptr1);
 237      addtoextraq((event *) curr);
 238      break;
 239      
 240    case IO_INTERRUPT_COMPLETE:
 241      iodriver_interrupt_complete(0, (intr_event *) curr);
 242      break;
 243 
 244    case DEVICE_OVERHEAD_COMPLETE:
 245    case DEVICE_ACCESS_COMPLETE:
 246    case DEVICE_DATA_TRANSFER_COMPLETE:
 247    case DEVICE_PREPARE_FOR_DATA_TRANSFER:
 248    case DEVICE_BUFFER_SEEKDONE:
 249    case DEVICE_BUFFER_TRACKACC_DONE:
 250    case DEVICE_BUFFER_SECTOR_DONE:
 251    case DEVICE_GOT_REMAPPED_SECTOR:
 252    case DEVICE_GOTO_REMAPPED_SECTOR:
 253    case MEMS_SLED_SCHEDULE:
 254    case MEMS_SLED_SEEK:
 255    case MEMS_SLED_SERVO:
 256    case MEMS_SLED_DATA:
 257    case MEMS_SLED_UPDATE:
 258    case MEMS_BUS_INITIATE:
 259    case MEMS_BUS_TRANSFER:
 260    case MEMS_BUS_UPDATE:
 261    case SSD_CLEAN_ELEMENT:
 262    case SSD_CLEAN_GANG:
 263      device_event_arrive(curr);
 264      break;
 265 
 266    case BUS_OWNERSHIP_GRANTED:
 267    case BUS_DELAY_COMPLETE:
 268      bus_event_arrive(curr);
 269      break;
 270 
 271    case CONTROLLER_DATA_TRANSFER_COMPLETE:
 272      controller_event_arrive(curr->tempint2, curr);
 273      break;
 274 
 275    case TIMESTAMP_LOGORG:
 276      logorg_timestamp(curr);
 277      break;
 278 
 279    case IO_TRACE_REQUEST_START:
 280      iodriver_trace_request_start(0, curr);
 281      break;
 282 
 283    default:
 284      fprintf(stderr, "Unknown event type passed to io_internal_events\n");
 285      exit(1);
 286    }
 287    /*
 288      fprintf (outputfile, "Leaving io_internal_event\n");
 289    */
 290 }

 293 void iosim_get_path_to_controller (int iodriverno, int ctlno, intchar *buspath, intchar *slotpath)
 294 {
 295    int depth;
 296    int currbus;
 297    char inslotno;
 298    char outslotno;
 299    int master;
 300 
 301    depth = controller_get_depth(ctlno);
 302    currbus = controller_get_inbus(ctlno);
 303    inslotno = (char) controller_get_slotno(ctlno);
 304    master = controller_get_bus_master(currbus);
 305    while (depth > 0) {
 306 /*
 307 fprintf (outputfile, "ctlno %d, depth %d, currbus %d, inslotno %d, master %d\n", ctl->ctlno, depth, currbus, inslotno, master);
 308 */
 309       outslotno = (char) controller_get_outslot(master, currbus);
 310       buspath->byte[depth] = (char) currbus;
 311       slotpath->byte[depth] = (inslotno & 0x0F) | (outslotno << 4);
 312       depth--;
 313       currbus = controller_get_inbus(master);
 314       inslotno = (char) controller_get_slotno(master);
 315       master = controller_get_bus_master(currbus);
 316           /* Bus must have at least one controller */
 317       ASSERT1((master != -1) || (depth <= 0), "currbus", currbus);
 318    }
 319    outslotno = (char) iodriverno;
 320    buspath->byte[depth] = currbus;
 321    slotpath->byte[depth] = (inslotno & 0x0F) | (outslotno << 4);
 322 }

 359 static int iosim_load_map(struct lp_block *b, int n) {
 360   int c;
 361   int i = 0;
 362   char *s = 0; 
 363 
 364 
 365   //#include "modules/disksim_iomap_param.c"
 366   lp_loadparams((void *)n, b, &disksim_iomap_mod);
 367 
 368   if (tracemap2[n] == 512) {
 369     tracemap2[n] = 0;
 370   } 
 371   else if (tracemap2[n] > 512) {
 372     tracemap2[n] = -(tracemap2[n] / 512);
 373   } 
 374   else {
 375     tracemap2[n] = 512 / tracemap2[n];
 376   }
 377   
 378 
 379   return 1;
 380 }

 382 int iosim_load_mappings(struct lp_list *l) {
 383   int c; 
 384   int mapno = 0;
 385 
 386   if(l->values_len > TRACEMAPPINGS) {
 387     fprintf(stderr, "*** error: too many io mappings.\n");
 388     return -1;
 389   }
 390 
 391   for(c = 0; c < l->values_len; c++) {
 392     if(!l->values[c]) continue;
 393 
 394     if(l->values[c]->t == BLOCK) {
 395       assert(iosim_load_map(l->values[c]->v.b, mapno));
 396       tracemappings++;
 397     }
 398 
 399     mapno++;
 400   }
 401 
 402   return 1;
 403 }

 480 void io_validate_do_stats1 ()
 481 {
 482    int i;
 483 
 484    if (tracestats2 == NULL) {
 485       tracestats2 = (statgen *)DISKSIM_malloc(sizeof(statgen));
 486       tracestats3 = (statgen *)DISKSIM_malloc(sizeof(statgen));
 487       tracestats4 = (statgen *)DISKSIM_malloc(sizeof(statgen));
 488       tracestats5 = (statgen *)DISKSIM_malloc(sizeof(statgen));
 489 
 490       stat_initialize(statdeffile, statdesc_traceaccstats, tracestats2);
 491       stat_initialize(statdeffile, statdesc_traceaccdiffstats, tracestats3);
 492       stat_initialize(statdeffile, statdesc_traceaccwritestats, tracestats4);
 493       stat_initialize(statdeffile, statdesc_traceaccdiffwritestats, tracestats5);
 494       for (i=0; i<10; i++) {
 495          validatebuf[i] = 0;
 496       }
 497    } else {
 498       stat_update(tracestats3, (validate_lastserv - disksim->lastphystime));
 499       if (!validate_lastread) {
 500          stat_update(tracestats5, (validate_lastserv - disksim->lastphystime));
 501       }
 502    }
 503 }

 506 void io_validate_do_stats2 (ioreq_event *new)
 507 {
 508    stat_update(tracestats2, validate_lastserv);
 509    if (new->flags == WRITE) {
 510       stat_update(tracestats4, validate_lastserv);
 511    }
 512    if (strcmp(validate_buffaction, "Doub") == 0) {
 513       validatebuf[0]++;
 514    } else if (strcmp(validate_buffaction, "Trip") == 0) {
 515       validatebuf[1]++;
 516    } else if (strcmp(validate_buffaction, "Miss") == 0) {
 517       validatebuf[2]++;
 518    } else if (strcmp(validate_buffaction, "Hit") == 0) {
 519       validatebuf[3]++;
 520    } else {
 521       fprintf(stderr, "Unrecognized buffaction in validate trace: %s\n", validate_buffaction);
 522       exit(1);
 523    }
 524 }

 549 void io_map_trace_request (ioreq_event *temp)
 550 {
 551    int i;
 552 
 553    for (i=0; i<tracemappings; i++) {
 554       if (temp->devno == tracemap[i]) {
 555          temp->devno = tracemap1[i];
 556          if (tracemap2[i]) {
 557             if (tracemap2[i] < 1) {
 558                temp->blkno *= -tracemap2[i];
 559             } else {
 560                if (temp->blkno % tracemap2[i]) {
 561                   fprintf(stderr, "Small sector size disk using odd sector number: %d\n", temp->blkno);
 562                   exit(1);
 563                }
 564 /*
 565                fprintf (outputfile, "mapping block number %d to %d\n", temp->blkno, (temp->blkno / tracemap2[i]));
 566 */
 567                temp->blkno /= tracemap2[i];
 568             }
 569          }
 570          temp->bcount *= tracemap3[i];
 571          temp->blkno += tracemap4[i];
 572          if (tracestats) {
 573             stat_update(&tracestats[i], ((double) temp->tempint1 / (double) 1000));
 574             stat_update(&tracestats1[i],((double) (temp->tempint1 + temp->tempint2) / (double) 1000));
 575             stat_update(&tracestats2[i],((double) temp->tempint2 / (double) 1000));
 576             stat_update(&tracestats3[i], (double) temp->slotno);
 577             if (temp->slotno == 1) {
 578                stat_update(&tracestats4[i], ((double) temp->tempint1 / (double) 1000));
 579             }
 580          }
 581          return;
 582       }
 583    }
 584 /*
 585    fprintf(stderr, "Requested device not mapped - %x\n", temp->devno);
 586    exit(1);
 587 */
 588 }

 595    ASSERT(io_extq == NULL);
 596 
 597    //fprintf (outputfile, "Near beginning of io_get_next_external_event\n");
 598 
 599    temp = (ioreq_event *) getfromextraq();
 600 
 601    switch (disksim->traceformat) {
 602       case VALIDATE: io_validate_do_stats1();
 603                      break;
 604       case HPL: io_hpl_do_stats1();
 605                 break;
 606    }
 607 
 608    temp = iotrace_get_ioreq_event(iotracefile, disksim->traceformat, temp);
 609    if (temp) {
 610       switch (disksim->traceformat) {
 611          case VALIDATE: io_validate_do_stats2 (temp);
 612                         break;
 613       }
 614       temp->type = IO_REQUEST_ARRIVE;
 615       if (constintarrtime > 0.0) {
 616          temp->time = last_request_arrive + constintarrtime;
 617          last_request_arrive = temp->time;
 618       }
 619       temp->time = (temp->time * ioscale) + tracebasetime;
 620       if ((temp->time < simtime) && (!disksim->closedios)) {
 621          fprintf(stderr, "Trace event appears out of time order in trace - simtime %f, time %f\n", simtime, temp->time);
 622          fprintf(stderr, "ioscale %f, tracebasetime %f\n", ioscale, tracebasetime);
 623          fprintf(stderr, "devno %d, blkno %d, bcount %d, flags %d\n", temp->devno, temp->blkno, temp->bcount, temp->flags);
 624          exit(1);
 625       }
 626       if (tracemappings) {
 627          io_map_trace_request(temp);
 628       }
 629       io_extq = (event *)temp;
 630       io_extq_type = temp->type;
 631    }
 632 /*
 633 fprintf (outputfile, "leaving io_get_next_external_event\n");
 634 */
 635    return ((event *)temp);
 636 }

 639 int io_using_external_event (event *curr)
 640 {
 641    if (io_extq == curr) {
 642       curr->type = io_extq_type;
 643       io_extq = NULL;
 644       return(1);
 645    }
 646    return(0);
 647 }

 650 void io_printstats()
 651 {
 652    int i;
 653    int cnt = 0;
 654    char prefix[80];
 655 
 656    fprintf (outputfile, "\nSTORAGE SUBSYSTEM STATISTICS\n");
 657    fprintf (outputfile, "----------------------------\n");
 658 
 659    iotrace_printstats (outputfile);
 660 
 661    if ((tracestats) && (PRINTTRACESTATS)) {
 662       /* info relevant to HPL traces */
 663       for (i=0; i<tracemappings; i++) {
 664          sprintf(prefix, "Mapped disk #%d ", i);
 665          stat_print(&tracestats[i], prefix);
 666          stat_print(&tracestats1[i], prefix);
 667          stat_print(&tracestats2[i], prefix);
 668          stat_print(&tracestats3[i], prefix);
 669          stat_print(&tracestats4[i], prefix);
 670       }
 671    } else if ((tracestats2) && (PRINTTRACESTATS)) {
 672       /* info to help with validation */
 673       fprintf (outputfile, "\n");
 674       stat_print(tracestats2, "VALIDATE ");
 675       stat_print(tracestats3, "VALIDATE ");
 676       stat_print(tracestats4, "VALIDATE ");
 677       stat_print(tracestats5, "VALIDATE ");
 678       for (i=0; i<10; i++) {
 679          cnt += validatebuf[i];
 680       }
 681       fprintf (outputfile, "VALIDATE double disconnects:  %5d  \t%f\n", validatebuf[0], ((double) validatebuf[0] / (double) cnt));
 682       fprintf (outputfile, "VALIDATE triple disconnects:  %5d  \t%f\n", validatebuf[1], ((double) validatebuf[1] / (double) cnt));
 683       fprintf (outputfile, "VALIDATE read buffer hits:    %5d  \t%f\n", validatebuf[3], ((double) validatebuf[3] / (double) cnt));
 684       fprintf (outputfile, "VALIDATE buffer misses:       %5d  \t%f\n", validatebuf[2], ((double) validatebuf[2] / (double) cnt));
 685    }
 686 
 687    iodriver_printstats();
 688    device_printstats();
 689    controller_printstats();
 690    bus_printstats();
 691 }
