 102 /*
 103  *
 104  * This file offers a suggested interface between disksim and a full
 105  * system simulator.  Using this interface (which assumes only a few
 106  * characteristics of the system simulator), disksim will act as a
 107  * slave of the system simulator, providing disk request completion
 108  * indications in time for an interrupt to be generated in the system
 109  * simulation.  Specifically, disksim code will only be executed when
 110  * invoked by one of the procedures in this file.
 111  *
 112  * Using ths interface requires only two significant functionalities
 113  * of the system simulation environment:
 114  *
 115  * 1. the ability to function correctly without knowing when a disk
 116  * request will complete at the time that it is initiated.  Via the
 117  * code in this file, the system simulation will be informed at some
 118  * later point in its view of time that the request is completed.  (At
 119  * this time, the appropriate "disk request completion" interrupt
 120  * could be inserted into the system simulation.)
 121  *
 122  * 2. the ability for disksim to register a callback with the system
 123  * simulation environment.  That is, this client code must be able to
 124  * say, "please invoke this function when the simulated time reaches
 125  * X".  It is also helpful to be able to "deschedule" a callback at a
 126  * later time -- but lack of this support can certainly be worked
 127  * around.
 128  *
 129  * To use the external interface, your program should link with
 130  * libdisksim.a and disksim_interface.o.  See syssim.* for a good
 131  * example.
 132  *
 133  */

 156 /* This is the disksim callback for reporting completion of a disk
 157  * request to the system-level simulation -- the system-level *
 158  * simulation should incorporate this completion as appropriate *
 159  * (probably by inserting a simulated "disk completion interrupt" at *
 160  * the specified simulated time).  Based on the requestdesc pointed to
 161  * * by "curr->buf" (below), the system-level simulation should be
 162  * able * to determine which request completed.  (A ptr to the
 163  * system-level * simulator's request structure is a reasonable use of
 164  * "curr->buf".)
 165  */
 166 static event * 
 167 disksim_interface_io_done_notify (ioreq_event *curr, void *ctx)
 168 {
 169   struct disksim_interface *iface = (struct disksim_interface *)ctx;
 170   struct disksim_request *req = (struct disksim_request *) curr->buf;
 171 
 172   // wrong ctx -- should be the per-req one ... how to demux?
 173   iface->complete_fn(simtime, req, iface->ctx);
 174   return 0;
 175 }

 178 static int
 179 disksim_interface_initialize_latency (struct disksim_interface *iface,
 180                                       const char *pfile, 
 181                                       const char *ofile, 
 182                                       int latency_weight, 
 183                                       char *paramval, 
 184                                       char *paramname, 
 185                                       int synthio, 
 186                                       char *sched_alg,
 187                                       int over_argc,
 188                                       char **over_argv)
 189 {
 190   char **argv;
 191   int argc = 6;
 192   int i = argc;
 193 
 194   argc += over_argc;
 195   
 196   // fprintf (stder, "disksim_initialize\n");
 197   
 198   disksim_initialize_disksim_structure(iface->disksim);
 199 
 200   if(latency_weight) {
 201     argc += 12;
 202   }
 203 
 204   argv = calloc(argc, sizeof(char *));
 205 
 206   if(latency_weight){
 207     //     iodriver_param_override(paramname,paramval,-1,-1);
 208     
 209     
 210     argv[i++] = "driver*";
 211     argv[i++] = paramname;
 212     argv[i++] = paramval;
 213 
 214 
 215     /* note: these parameter names must match those from
 216      * modules/disksim_ioqueue_param.h, etc 
 217      */
 218     
 219     argv[i++] = "driver*";
 220     argv[i++] = "Scheduler:Scheduling policy";
 221     argv[i++] = sched_alg;
 222     
 223     argv[i++] = "driver*";
 224     argv[i++] = "Scheduler:Scheduling priority scheme";
 225     argv[i++] = sched_alg;
 226 
 227     argv[i++] = "driver*";
 228     argv[i++] = "Scheduler:Timeout scheduling";
 229     argv[i++] = sched_alg;
 230   }
 231 
 232   memcpy(argv + i, over_argv, over_argc * sizeof(char *));
 233 
 234   argv[0] = "disksim";
 235   argv[1] = (char *) pfile;
 236   argv[2] = (char *) ofile;
 237   argv[3] = "external";
 238   argv[4] = "0";
 239   if(synthio == 1){
 240     argv[5] = "1";
 241   }else{
 242     argv[5] = "0";
 243   }
 244   disksim_setup_disksim (argc, argv);
 245   
 246   /* Note that this call must be redone anytime a disksim checkpoint is 
 247    * restored -- this prevents old function addresses from polluting    
 248    * the restored execution...                                          
 249    */
 250   disksim_set_external_io_done_notify(disksim_interface_io_done_notify);
 251   
 252   // fprintf (stder, "disksim_initialize done\n");
 253   
 254   return 0;
 255 }

 259 struct disksim_interface * 
 260 disksim_interface_initialize (const char *pfile, 
 261                               const char *ofile,
 262                               disksim_interface_complete_t comp,
 263                               disksim_interface_sched_t sched,
 264                               disksim_interface_desched_t desched,
 265                               void *ctx,
 266                               int argc,
 267                               char **argv)
 268 {
 269   struct disksim_interface *iface = calloc(1, sizeof(struct disksim_interface));
 270   ddbg_assert(iface != 0);
 271 
 272   iface->complete_fn = comp;
 273   iface->sched_fn = sched;
 274   iface->desched_fn = desched;
 275   iface->ctx = ctx;
 276   iface->disksim = calloc(1, sizeof(struct disksim));
 277 
 278   disksim = iface->disksim;
 279 
 280 
 281   disksim_interface_initialize_latency(iface, 
 282                                        pfile, 
 283                                        ofile, 
 284                                        0,         // latency_weight
 285                                        NULL,      // paramval
 286                                        NULL,      // paramname
 287                                        0,         // synthio
 288                                        NULL,      // sched_alg
 289                                        argc,
 290                                        argv);
 291 
 292   disksim->notify_ctx = iface;
 293 
 294   return iface;
 295 }

 301 void 
 302 disksim_interface_shutdown (struct disksim_interface *iface,
 303                             double syssimtime)
 304 {
 305    double curtime = syssimtime;
 306 
 307    disksim = iface->disksim;
 308 
 309    // fprintf (stderr, "disksim_shutdown\n");
 310 
 311    if ((curtime + 0.0001) < simtime) {
 312       fprintf (stderr, "external time is ahead of disksim time: %f > %f\n", curtime, simtime);
 313       exit(1);
 314    }
 315 
 316    simtime = curtime;
 317    disksim_printstats ();
 318 
 319    // fprintf (stderr, "disksim_shutdown done\n");
 320 }

 323 /* Prints the current disksim statistics.  This call does not reset the    */
 324 /* statistics, so the simulation can continue to run and calculate running */
 325 /* totals.  "syssimtime" should be the current simulated time of the       */
 326 /* system-level simulation.                                                */
 327 
 328 void 
 329 disksim_interface_dump_stats (struct disksim_interface *iface,
 330                               double syssimtime)
 331 {
 332    double curtime = syssimtime;
 333 
 334    disksim = iface->disksim;
 335 
 336    // fprintf (stderr, "disksim_dump_stats\n");
 337 
 338    if ((disksim->intq) && (disksim->intq->time < curtime) && ((disksim->intq->time + 0.0001) >= curtime)) {
 339       curtime = disksim->intq->time;
 340    }
 341    if (((curtime + 0.0001) < simtime) 
 342        || ((disksim->intq) 
 343            && (disksim->intq->time < curtime))) 
 344    {
 345      fprintf (stderr, "external time is mismatched with disksim time: %f vs. %f (%f)\n", curtime, simtime, ((disksim->intq) ? disksim->intq->time : 0.0));
 346      exit (1);
 347    }
 348 
 349    simtime = curtime;
 350    disksim_cleanstats();
 351    disksim_printstats();
 352 
 353    // fprintf (stderr, "disksim_dump_stats done\n");
 354 }

 359 /* This is the callback for handling internal disksim events while running */
 360 /* as a slave of a system-level simulation.  "syssimtime" should be the    */
 361 /* current simulated time of the system-level simulation.                  */
 362 
 363 void 
 364 disksim_interface_internal_event (struct disksim_interface *iface, 
 365                                   double syssimtime,
 366                                   void *junk)
 367 {
 368    double curtime = syssimtime;
 369    disksim = iface->disksim;
 370 
 371    // fprintf (stderr, "disksim_internal_event\n");
 372 
 373    /* if next event time is less than now, error.  Also, if no event is  */
 374    /* ready to be handled, then this is a spurious callback -- it should */
 375    /* not be possible with the descheduling below (allow it if it is not */
 376    /* possible to deschedule.                                            */
 377 
 378    if (disksim->intq != NULL 
 379        && (disksim->intq->time + 0.0001) < curtime) 
 380    {
 381      fprintf (stderr, "external time is ahead of disksim time: %f > %f\n", curtime, disksim->intq->time);
 382      exit (1);
 383    }
 384 
 385    // fprintf(stderr, "disksim_internal_event: intq->time=%f curtime=%f\n", disksim->intq->time, curtime);
 386 
 387    /* while next event time is same as now, handle next event */
 388    if(disksim->intq != NULL){
 389      ASSERT (disksim->intq->time >= simtime);
 390    }
 391 
 392    while ((disksim->intq != NULL) 
 393           && (disksim->intq->time <= (curtime + 0.0001))) 
 394    {
 395        
 396      // fprintf (stderr, "handling internal event: type %d\n", disksim->intq->type);
 397      
 398      disksim_simulate_event(event_count++);
 399    }
 400 
 401    if (disksim->intq != NULL) {
 402       /* Note: this could be a dangerous operation when employing checkpoint */
 403       /* and, specifically, restore -- functions move around when programs   */
 404       /* are changed and recompiled...                                       */
 405 
 406       iface->sched_fn(disksim_interface_internal_event, 
 407                       disksim->intq->time,
 408                       iface->ctx);
 409    }
 410 
 411    // fprintf (stderr, "disksim_internal_event done\n");
 412 }

 415 /* This function should be called by the system-level simulation when    */
 416 /* it wants to issue a request into disksim.  "syssimtime" should be the */
 417 /* system-level simulation time at which the request should "arrive".    */
 418 /* "requestdesc" is the system-level simulator's description of the      */
 419 /* request (device number, block number, length, etc.).                  */
 420 
 421 void 
 422 disksim_interface_request_arrive (struct disksim_interface *iface,
 423                                   double syssimtime, 
 424                                   struct disksim_request *requestdesc)
 425 {
 426    ioreq_event *new;
 427 
 428    double curtime = syssimtime;
 429    disksim = iface->disksim;
 430 
 431    new = (ioreq_event *) getfromextraq();
 432 
 433    assert (new != NULL);
 434    new->type = IO_REQUEST_ARRIVE;
 435    new->time = curtime;
 436    new->busno = 0;
 437    new->devno = requestdesc->devno;
 438    new->blkno = requestdesc->blkno;
 439    new->flags = requestdesc->flags;
 440    new->bcount = requestdesc->bytecount / 512;
 441    new->batchno = requestdesc->batchno;
 442    new->batch_complete = requestdesc->batch_complete;
 443       
 444    new->flags |= TIME_CRITICAL;
 445    
 446    new->cause = 0;
 447    new->opid = 0;
 448    new->buf = requestdesc;
 449 
 450    io_map_trace_request (new);
 451 
 452    /* issue it into simulator */
 453    if (disksim->intq) {
 454      iface->desched_fn(0.0, iface->ctx);
 455    }
 456    addtointq ((event *)new);
 457 
 458    /* while next event time is same as now, handle next event */
 459    while ((disksim->intq != NULL) 
 460           && (disksim->intq->time <= (curtime + 0.0001))) 
 461    {
 462      disksim_simulate_event (event_count++);
 463    }
 464 
 465    if (disksim->intq) {
 466       /* Note: this could be a dangerous operation when employing checkpoint */
 467       /* and, specifically, restore -- functions move around when programs   */
 468       /* are changed and recompiled...                                       */
 469 
 470       iface->sched_fn(disksim_interface_internal_event, 
 471                       disksim->intq->time,
 472                       iface->ctx);
 473    }
 474 }
