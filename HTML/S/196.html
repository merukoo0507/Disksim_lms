 165 {

 167   int cnt;
 168

 176   currdisk->depth[cnt] = depth;
 177   currdisk->slotno[cnt] = slotno;

 235 /*
 236  **- disk_send_event_up_path
 237 
 238  Acquires the bus (if not already acquired), then uses bus_delay to
 239  send the event up the path.
 240 
 241  If the bus is already owned by this device or can be acquired
 242  immediately (interleaved bus), the event is sent immediately.
 243  Otherwise, disk_bus_ownership_grant will later send the event.  
 244 */
 245   
 246 static void disk_send_event_up_path (ioreq_event *curr, double delay)
 247 {
 248   disk *currdisk = getdisk (curr->devno);
 249   int busno;
 250   int slotno;
 251 
 252   disksim_inst_enter();
 253 
 254   /*
 255     fprintf (outputfile, "Disk_send_event_up_path - devno %d, type %d, cause %d, blkno %d\n", curr->devno, curr->type, curr->cause, curr->blkno);
 256   */
 257 
 258   busno = disk_get_busno(curr);
 259   slotno = currdisk->slotno[0];
 260 
 261   /* Put new request at head of buswait queue */
 262   curr->next = currdisk->buswait;
 263   currdisk->buswait = curr;
 264 
 265   curr->tempint1 = busno;
 266   curr->time = delay;
 267   if(currdisk->busowned == -1) {
 268     /* fprintf (outputfile, "Must get ownership of the bus first\n"); */
 269     if(curr->next) {
 270       fprintf(stderr,"Multiple bus requestors detected in "
 271               "disk_send_event_up_path\n");
 272       /* Is this OK?  Strange that the new requester is put on the head?  */
 273     }
 274     if(bus_ownership_get(busno, slotno, curr) == FALSE) {
 275       /* Remember when we started waiting (only place this is written) */
 276       currdisk->stat.requestedbus = simtime;
 277     } 
 278     else {
 279       bus_delay(busno, DEVICE, curr->devno, delay, curr); /* Never for SCSI */
 280     }
 281   } 
 282   else if(currdisk->busowned == busno) {
 283     /*
 284       fprintf (outputfile, "Already own bus - so send it on up\n");
 285     */
 286     bus_delay(busno, DEVICE, curr->devno, delay, curr);
 287   } 
 288   else {
 289     ddbg_assert2(0, "Wrong bus owned for transfer desired");
 290   }
 291 }

 294 /*
 295  * disk_bus_ownership_grant
 296  *
 297  * Calls bus_delay to handle the event that the disk has been granted
 298  * the bus.  I believe this is always initiated by a call to
 299  * disk_send_even_up_path.
 300  */
 301 
 302 void 
 303 disk_bus_ownership_grant(int devno, 
 304                          ioreq_event *curr, 
 305                          int busno, 
 306                          double arbdelay)
 307 {
 308   disk *currdisk = getdisk (devno);
 309   ioreq_event *tmp;
 310 
 311   tmp = currdisk->buswait;
 312   while ((tmp != NULL) && (tmp != curr)) {
 313     tmp = tmp->next;
 314   }
 315 
 316   ddbg_assert3(tmp != 0, ("Bus ownership granted to unknown disk request - "
 317                         "devno %d, busno %d", 
 318                         devno, busno));
 319   
 320   currdisk->busowned = busno;
 321   currdisk->stat.waitingforbus += simtime - currdisk->stat.requestedbus;
 322   ddbg_assert(arbdelay == (simtime - currdisk->stat.requestedbus));
 323   currdisk->stat.numbuswaits++;
 324   bus_delay(busno, DEVICE, devno, tmp->time, tmp);
 325 }

 328 void 
 329 disk_bus_delay_complete(int devno, 
 330                         ioreq_event *curr, 
 331                         int sentbusno)
 332 {
 333   disk *currdisk = getdisk (devno);
 334   intchar slotno;
 335   intchar busno;
 336   int depth;
 337 
 338   /*
 339    * fprintf (outputfile, "Entered disk_bus_delay_complete\n");
 340    */
 341   if(curr == currdisk->buswait) {
 342     currdisk->buswait = curr->next;
 343   } else {
 344     ioreq_event *tmp = currdisk->buswait;
 345     while ((tmp->next != NULL) && (tmp->next != curr)) {
 346       tmp = tmp->next;
 347     }
 348 
 349     ddbg_assert3(tmp->next == curr, ("Bus delay complete for unknown disk "
 350                                    "request - devno %d, busno %d", 
 351                                    devno, busno.value));
 352 
 353     tmp->next = curr->next;
 354   }
 355   busno.value = curr->busno;
 356   slotno.value = curr->slotno;
 357   depth = currdisk->depth[0];
 358   slotno.byte[depth] = slotno.byte[depth] >> 4;
 359   curr->time = 0.0;
 360   if(depth == 0) {
 361     intr_request ((event *)curr);
 362   } 
 363   else {
 364     bus_deliver_event(busno.byte[depth], slotno.byte[depth], curr);
 365   }
 366 }

 391 static void disk_check_prefetch_swap (disk *currdisk)
 392 {
 393   ioreq_event *nextioreq;
 394   diskreq     *nextdiskreq;
 395   int           setseg = FALSE;
 396 
 397   if(disk_printhack && (simtime >= disk_printhacktime)) {
 398     fprintf (outputfile, "%12.6f            Entering disk_check_prefetch_swap for disk %d\n", simtime, currdisk->devno);
 399     fflush(outputfile);
 400   }
 401 
 402   ddbg_assert(currdisk->effectivehda != 0);
 403 
 404   ddbg_assert(currdisk->effectivehda->seg != 0);
 405 
 406   {
 407     int ispureprefetch = 
 408       (currdisk->effectivehda->seg->access->flags & BUFFER_BACKGROUND) 
 409       && (currdisk->effectivehda->seg->access->flags & READ);
 410 
 411     ddbg_assert2(ispureprefetch, "effectivehda->seg->access is not a pure "
 412                "prefetch");
 413 
 414   }
 415 
 416   nextioreq = ioqueue_show_next_request(currdisk->queue);
 417   if(nextioreq) {
 418     nextdiskreq = nextioreq->ioreq_hold_diskreq;
 419     if(!nextdiskreq->seg) {
 420       setseg = TRUE;
 421       disk_buffer_select_segment(currdisk,nextdiskreq,FALSE);
 422     }
 423     if(nextioreq->flags & READ) {
 424       disk_activate_read(currdisk,nextdiskreq,setseg,TRUE);
 425     } 
 426     else {
 427       disk_activate_write(currdisk,nextdiskreq,setseg,TRUE);
 428     }
 429   }
 430 }

 435 static void disk_request_complete(disk *currdisk, 
 436                                   diskreq *currdiskreq, 
 437                                   ioreq_event *curr)
 438 {
 439   ioreq_event *tmpioreq = currdiskreq->ioreqlist;
 440   diskreq     *nextdiskreq;
 441   double       delay;
 442 
 443   if(disk_printhack && (simtime >= disk_printhacktime)) {
 444     fprintf (outputfile, "%12.6f  %8p "
 445              " Entering disk_request_complete\n", simtime, currdiskreq);
 446     fflush(outputfile);
 447   }
 448 
 449 
 450   //  printf("disk_request_complete %f\n", simtime);
 451 
 452   ddbg_assert(currdisk->effectivebus == currdiskreq);
 453 
 454 
 455   if(curr->blkno != tmpioreq->blkno) {
 456     addtoextraq((event *) curr);
 457     curr = ioreq_copy(tmpioreq);
 458   }
 459 
 460   currdisk->lastflags = curr->flags;
 461 
 462   ddbg_assert(currdisk->outstate == DISK_TRANSFERING);
 463 
 464   currdiskreq->flags |= COMPLETION_SENT;
 465 
 466   if(currdisk->const_acctime) {
 467 
 468     ddbg_assert2(!(currdiskreq->flags & HDA_OWNED),
 469                "HDA_OWNED set for fixed-access-time disk");
 470 
 471 
 472     ddbg_assert2(ioqueue_get_specific_request(currdisk->queue,tmpioreq),
 473                "ioreq_event not found");
 474 
 475     disk_interferestats(currdisk, tmpioreq);
 476     /* GROK: this would seem to leak (or worse) for concatenating schedulers */
 477     ddbg_assert2(ioqueue_physical_access_done(currdisk->queue,tmpioreq),
 478                "ioreq_event not found");
 479 
 480     currdisk->currentbus = 
 481       currdisk->currenthda = 
 482       currdisk->effectivehda = NULL;
 483 
 484     tmpioreq = ioqueue_show_next_request(currdisk->queue);
 485     if(tmpioreq) {
 486       nextdiskreq = (diskreq*) tmpioreq->ioreq_hold_diskreq;
 487       if(nextdiskreq->ioreqlist->flags & READ) {
 488         disk_activate_read(currdisk, nextdiskreq, FALSE, TRUE);
 489       } 
 490       else {
 491         disk_activate_write(currdisk, nextdiskreq, FALSE, TRUE);
 492       }
 493     }
 494   } 
 495   else if(tmpioreq->flags & READ) {
 496     while (tmpioreq) {
 497       int evtfound = 0;
 498       if(!(currdiskreq->flags & HDA_OWNED)) {
 499 
 500         ddbg_assert2(ioqueue_get_specific_request(currdisk->queue,tmpioreq),
 501                    "ioreq_event not found");
 502         
 503         disk_interferestats(currdisk, tmpioreq);
 504       }
 505 
 506       evtfound = (int)ioqueue_physical_access_done(currdisk->queue,tmpioreq);
 507 
 508       ddbg_assert2(evtfound != 0, "ioreq_event not found");
 509 
 510       tmpioreq = tmpioreq->next;
 511     }
 512     if((currdisk->keeprequestdata == -1) &&
 513         !currdiskreq->seg->diskreqlist->seg_next)
 514       {
 515         ddbg_assert2((currdiskreq->outblkno <= currdiskreq->seg->endblkno),
 516                    "Unable to erase request data from segment");
 517 
 518         currdiskreq->seg->startblkno = currdiskreq->outblkno;
 519       }
 520 
 521     
 522     if((currdiskreq == currdisk->effectivehda) ||
 523         (currdiskreq == currdisk->currenthda)) 
 524       {
 525         if(currdiskreq->seg->access->type == NULL_EVENT) {
 526           disk_release_hda(currdisk,currdiskreq);
 527         } 
 528         else {
 529           if(currdisk->preseeking != NO_PRESEEK) {
 530             disk_check_prefetch_swap(currdisk);
 531           }
 532         }
 533       }
 534   } 
 535   else {                        /* WRITE */
 536     if((currdiskreq == currdisk->effectivehda) 
 537        || (currdiskreq == currdisk->currenthda)) 
 538       {
 539         if(currdiskreq->seg->access->type == NULL_EVENT) {
 540           disk_release_hda(currdisk,currdiskreq);
 541         }
 542       }
 543   }
 544 
 545   curr->time = simtime;
 546   curr->type = IO_INTERRUPT_ARRIVE;
 547   curr->cause = COMPLETION;
 548 
 549   if(currdisk->const_acctime) {
 550     delay = 0.0;
 551   }
 552   else {
 553     if(curr->flags & READ) {
 554       delay = currdisk->overhead_complete_read;
 555     }
 556     else {
 557       delay = currdisk->overhead_complete_write;
 558     }
 559   }
 560 
 561   disk_send_event_up_path(curr, (delay * currdisk->timescale));
 562   currdisk->outstate = DISK_WAIT_FOR_CONTROLLER;
 563 }

 566 static void disk_reconnection_or_transfer_complete (ioreq_event *curr)
 567 {
 568   disk        *currdisk = getdisk(curr->devno);
 569   diskreq     *currdiskreq;
 570   ioreq_event *tmpioreq;
 571   double delay;
 572 
 573   disksim_inst_enter();
 574 
 575   currdiskreq = currdisk->effectivebus;
 576   if(!currdiskreq) {
 577     currdiskreq = currdisk->effectivebus = currdisk->currentbus;
 578   }
 579 
 580   ddbg_assert2(currdiskreq != 0, "effectivebus and currentbus are NULL");
 581 
 582 
 583   if(disk_printhack && (simtime >= disk_printhacktime)) {
 584     fprintf (outputfile, "%12.6f  %8p  "
 585              "Entering disk_reconnection_or_transfer_complete for disk %d\n", 
 586              simtime, currdiskreq, curr->devno);
 587     fflush(outputfile);
 588   }
 589 
 590   tmpioreq = currdiskreq->ioreqlist;
 591   while (tmpioreq) {
 592     if(tmpioreq->blkno == curr->blkno) {
 593       break;
 594     }
 595     tmpioreq = tmpioreq->next;
 596   }
 597 
 598   ddbg_assert2(tmpioreq != 0, "ioreq_event not found in effectivebus");
 599 
 600   switch (currdisk->outstate) {
 601   case DISK_WAIT_FOR_CONTROLLER:
 602   case DISK_TRANSFERING:
 603     break;
 604   default:
 605     ddbg_assert3(0, ("Disk not waiting to transfer - devno %d, state %d\n", 
 606                curr->devno, currdisk->outstate));
 607   }
 608   
 609   currdisk->effectivebus = currdiskreq;
 610   currdisk->outstate = DISK_TRANSFERING;
 611   curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 612   curr = disk_buffer_transfer_size(currdisk, currdiskreq, curr);
 613 
 614   if(disk_printhack && (simtime >= disk_printhacktime)) {
 615     fprintf (outputfile, "                        disk_buffer_transfer_size set bcount to %d\n", curr->bcount);
 616     fflush(outputfile);
 617   }
 618 
 619   if(curr->bcount == -2) {
 620     ddbg_assert2(currdisk->outwait == 0, "non-NULL outwait found");
 621     currdisk->outwait = curr;
 622   } 
 623   else if(curr->bcount == -1) {
 624     curr->type = IO_INTERRUPT_ARRIVE;
 625     curr->cause = DISCONNECT;
 626     if(tmpioreq->flags & READ) {
 627       delay = ((currdisk->lastflags & READ) 
 628                ? currdisk->overhead_disconnect_read_afterread 
 629                : currdisk->overhead_disconnect_read_afterwrite);
 630     } 
 631     else {
 632       delay = ((currdiskreq->flags & EXTRA_WRITE_DISCONNECT) 
 633                ? currdisk->extradisc_disconnect2 
 634                : currdisk->overhead_disconnect_write);
 635     }
 636 
 637     disk_send_event_up_path(curr, (delay * currdisk->timescale));
 638     currdisk->outstate = DISK_WAIT_FOR_CONTROLLER;
 639   } 
 640   else if(curr->bcount == 0) {
 641     disk_request_complete(currdisk, currdiskreq, curr);
 642   } 
 643   else if(curr->bcount > 0) {
 644     currdisk->starttrans = simtime;
 645     currdisk->blksdone = 0;
 646     disk_send_event_up_path(curr, (double) 0.0);
 647   }
 648 }

 660 
 661   if(disk_printhack && (simtime >= disk_printhacktime)) {

 664   }

 666   ddbg_assert(seg != NULL);

 669   while (currdiskreq) {
 670     if(currdiskreq->flags & SEG_OWNED) {
 671       return;

 674         (currdiskreq->flags & HDA_OWNED) && (currdiskreq->ioreqlist) &&

 703 /*
 704  * If pure prefetch, release the hda and free the diskreq.
 705  * If active read, release the hda if preseeking level is appropriate.
 706  * If fast write, release the hda, remove the event(s) from the ioqueue,
 707  *   and, if COMPLETION_RECEIVED, free all structures.  If
 708  *   LIMITED_FASTWRITE and an appended request exists, make it the
 709  *   next effectivehda.
 710  * If slow write, call the completion routine and release the hda if
 711  *   preseeking level is appropriate.
 712  */
 713 
 714 static void disk_release_hda(disk *currdisk, diskreq *currdiskreq)
 715 {
 716   ioreq_event *tmpioreq;
 717   ioreq_event *holdioreq;
 718   segment *seg = currdiskreq->seg;
 719   diskreq *tmpdiskreq;
 720 
 721   diskreq *nextdiskreq = NULL;
 722   int release_hda = FALSE;
 723   int free_structs = FALSE;
 724 
 725   if(disk_printhack && (simtime >= disk_printhacktime)) {
 726     fprintf (outputfile, "%12.6f  %8p  Entering disk_release_hda for disk %d\n", simtime, currdiskreq,currdisk->devno);
 727     fflush(outputfile);
 728   }
 729 
 730   ddbg_assert(currdiskreq != NULL);
 731 
 732   ddbg_assert3((currdiskreq == currdisk->effectivehda), 
 733              ("currdiskreq 0x%x, effectivehda 0x%x", 
 734               currdiskreq, currdisk->effectivehda));
 735 
 736   ddbg_assert(seg != NULL);
 737 
 738   if(!currdiskreq->ioreqlist) {
 739     ddbg_assert2((currdiskreq->flags & COMPLETION_RECEIVED),
 740                "COMPLETION_RECEIVED not flagged for pure prefetch");
 741 
 742     release_hda = TRUE;
 743     free_structs = TRUE;
 744     seg->state = 
 745       ((currdisk->enablecache || seg->diskreqlist->seg_next) 
 746        ? BUFFER_CLEAN 
 747        : BUFFER_EMPTY);
 748   }
 749   else if(currdiskreq->ioreqlist->flags & READ) {
 750     if((currdisk->preseeking == PRESEEK_BEFORE_COMPLETION) 
 751        || ((currdisk->preseeking == PRESEEK_DURING_COMPLETION) 
 752            && (currdiskreq->flags & COMPLETION_SENT)))
 753       {
 754         int cleanOrEmpty;
 755         release_hda = TRUE;
 756         
 757         cleanOrEmpty = !(currdiskreq->flags & COMPLETION_SENT)
 758           ? BUFFER_CLEAN 
 759           : BUFFER_EMPTY;
 760 
 761         seg->state = (currdisk->enablecache 
 762                        || seg->diskreqlist->seg_next 
 763                        || cleanOrEmpty);
 764       }
 765   }
 766   else if((currdiskreq->flags & COMPLETION_RECEIVED) 
 767           || (currdisk->preseeking == PRESEEK_BEFORE_COMPLETION) 
 768           || ((currdisk->preseeking == PRESEEK_DURING_COMPLETION) 
 769               && (currdiskreq->flags & COMPLETION_SENT)))
 770     {
 771       release_hda = TRUE;
 772       if(currdiskreq->flags & COMPLETION_RECEIVED) {
 773         free_structs = TRUE;
 774       }
 775         
 776       /* check to see if seg should be left DIRTY */
 777         
 778       tmpdiskreq = seg->diskreqlist;
 779       while (tmpdiskreq) {
 780         if(tmpdiskreq != currdiskreq) {
 781           tmpioreq = tmpdiskreq->ioreqlist;
 782           ddbg_assert(tmpioreq != NULL);
 783           if(!(tmpioreq->flags & READ)) {
 784             while (tmpioreq->next) {
 785               tmpioreq = tmpioreq->next;
 786             }
 787             if(tmpdiskreq->inblkno < (tmpioreq->blkno + tmpioreq->bcount)) {
 788               seg->state = BUFFER_DIRTY;
 789               if((tmpdiskreq->hittype == BUFFER_APPEND) &&
 790                  (currdisk->fastwrites == LIMITED_FASTWRITE)) {
 791                 tmpioreq = currdiskreq->ioreqlist;
 792                 ddbg_assert(tmpioreq != NULL);
 793                 while (tmpioreq->next) {
 794                   tmpioreq = tmpioreq->next;
 795                 }
 796                 if(tmpdiskreq->ioreqlist->blkno == 
 797                    (tmpioreq->blkno + tmpioreq->bcount)) 
 798                   {
 799                     nextdiskreq = tmpdiskreq;
 800                   }
 801               }
 802               break;
 803             }
 804           }
 805         }
 806         tmpdiskreq = tmpdiskreq->seg_next;
 807       }
 808         
 809       if(!tmpdiskreq) {
 810         if(seg->diskreqlist->seg_next) {
 811           seg->state = BUFFER_CLEAN;
 812         } 
 813         else {
 814           int clean = (currdisk->enablecache && currdisk->readhitsonwritedata);
 815           if(clean) {
 816             seg->state = BUFFER_CLEAN;
 817           }
 818           else {
 819             seg->state = BUFFER_EMPTY;
 820           }
 821         }
 822         currdisk->numdirty--;
 823         if((disk_printhack > 1) 
 824            && (simtime >= disk_printhacktime)) 
 825           {
 826             fprintf (outputfile, "                        numdirty-- = %d\n",currdisk->numdirty);
 827             fflush(outputfile);
 828           }
 829         ddbg_assert3(((currdisk->numdirty >= 0) 
 830                     && (currdisk->numdirty <= currdisk->numwritesegs)),
 831                    ("numdirty: %d", currdisk->numdirty));
 832       }
 833     }
 834 
 835  
 836   if(disk_printhack && (simtime >= disk_printhacktime)) {
 837     fprintf (outputfile, "                        free_structs = %d, release_hda = %d\n", free_structs, release_hda);
 838     fflush(outputfile);
 839   }
 840   
 841   if(free_structs) {
 842     disk_buffer_remove_from_seg(currdiskreq);
 843     tmpioreq = currdiskreq->ioreqlist;
 844     while (tmpioreq) {
 845       holdioreq = tmpioreq->next;
 846       addtoextraq((event *) tmpioreq);
 847       tmpioreq = holdioreq;
 848     }
 849     addtoextraq((event *) currdiskreq);
 850     if(seg->recyclereq) {
 851       /* I don't think the following code ever gets used... */
 852       fprintf(stderr,"GOT HERE!  SURPRISE!\n");
 853       
 854       ddbg_assert2(seg->diskreqlist == 0,
 855                  "non-NULL diskreqlist found for recycled seg");
 856       
 857       nextdiskreq = seg->recyclereq;
 858       disk_buffer_set_segment(currdisk,seg->recyclereq);
 859       disk_buffer_attempt_seg_ownership(currdisk,seg->recyclereq);
 860       seg->recyclereq = NULL;
 861     } 
 862     else if(seg->diskreqlist) {
 863       disk_find_new_seg_owner(currdisk,seg);
 864     }
 865   }
 866   
 867   if(release_hda) {
 868     if(currdisk->currenthda == currdisk->effectivehda) {
 869       currdisk->currenthda = NULL;
 870     }
 871     currdisk->effectivehda = NULL;
 872     ddbg_assert2(seg->access->type == NULL_EVENT,
 873                "non-NULL seg->access->type found upon releasing hda");
 874     
 875     disk_check_hda(currdisk, nextdiskreq, TRUE);
 876   }
 877   
 878 }

 881 /* Set up an ioreq_event for a request if it needs bus service.  If
 882  *  data is ready for transfer, bcount is set to the amount.  
 883  */
 884 
 885 static ioreq_event * 
 886 disk_request_needs_bus(disk *currdisk, 
 887                        diskreq *currdiskreq,
 888                        int check_watermark)
 889 {
 890   ioreq_event *busioreq = NULL;
 891   ioreq_event *tmpioreq = currdiskreq->ioreqlist;
 892   diskreq     *seg_owner;
 893   segment     *seg = currdiskreq->seg;
 894 
 895   if((currdisk->outstate != DISK_IDLE) && !currdisk->outwait) {
 896     return 0;
 897   }
 898 
 899   if((currdiskreq->flags & FINAL_WRITE_RECONNECTION_1) &&
 900       !(currdiskreq->flags & FINAL_WRITE_RECONNECTION_2)) {
 901     return 0;
 902   }
 903 
 904   if(currdisk->const_acctime) {
 905     if((currdiskreq == currdisk->currenthda) && 
 906         (currdiskreq->overhead_done <= simtime)) {
 907 
 908       if(currdisk->outwait) {
 909         busioreq = currdisk->outwait;
 910         currdisk->outwait = NULL;
 911         busioreq->time = simtime;
 912       } 
 913       else {
 914         busioreq = ioreq_copy(tmpioreq);
 915         busioreq->time = simtime;
 916         busioreq->type = IO_INTERRUPT_ARRIVE;
 917         busioreq->cause = RECONNECT;
 918       }
 919 
 920       busioreq->bcount = 0;
 921     }
 922     return busioreq;
 923   }
 924 
 925   if(seg 
 926      && (currdiskreq != seg->recyclereq) 
 927      && tmpioreq 
 928      && !(currdiskreq->flags & COMPLETION_SENT)) 
 929     {
 930       while (tmpioreq && tmpioreq->next) {
 931         if(currdiskreq->outblkno < (tmpioreq->blkno + tmpioreq->bcount)) {
 932           break;
 933         }
 934         tmpioreq = tmpioreq->next;
 935       }
 936       if(currdiskreq->ioreqlist->flags & READ) {
 937         ddbg_assert2(currdiskreq->outblkno < 
 938                    (tmpioreq->blkno + tmpioreq->bcount),
 939                    "read completion detected in disk_request_needs_bus");
 940 
 941         if((seg->endblkno > currdiskreq->outblkno) 
 942            && (seg->startblkno <= currdiskreq->outblkno) 
 943            && (!check_watermark 
 944                || (seg->endblkno >= (tmpioreq->blkno + tmpioreq->bcount)) 
 945                || ((seg->endblkno - currdiskreq->outblkno) >= 
 946                    currdiskreq->watermark))) 
 947           {
 948             if(currdisk->outwait) {
 949               busioreq = currdisk->outwait;
 950               currdisk->outwait = NULL;
 951               busioreq->time = simtime;
 952             } 
 953             else {
 954               busioreq = ioreq_copy(tmpioreq);
 955               busioreq->time = simtime;
 956               busioreq->type = IO_INTERRUPT_ARRIVE;
 957               busioreq->cause = RECONNECT;
 958             }
 959             busioreq->bcount = min(seg->endblkno,(tmpioreq->blkno + tmpioreq->bcount)) - currdiskreq->outblkno;
 960           }
 961       } 
 962       else {                    /* WRITE */
 963         seg_owner = disk_buffer_seg_owner(seg,FALSE);
 964         if(!seg_owner) {
 965           seg_owner = currdiskreq;
 966         }
 967         if((currdiskreq->inblkno >= (tmpioreq->blkno + tmpioreq->bcount)) 
 968            || ((currdiskreq->outblkno == currdiskreq->ioreqlist->blkno) 
 969                && (currdiskreq->hittype != BUFFER_APPEND)) 
 970            || ((seg->endblkno == currdiskreq->outblkno) 
 971                && (seg->endblkno < (tmpioreq->blkno + tmpioreq->bcount)) 
 972                && ((seg->endblkno - seg_owner->inblkno) < seg->size) 
 973                && (!check_watermark 
 974                    || ((seg->endblkno - currdiskreq->inblkno) <= 
 975                        currdiskreq->watermark)))) 
 976           {
 977             if(currdisk->outwait) {
 978               busioreq = currdisk->outwait;
 979               currdisk->outwait = NULL;
 980               busioreq->time = simtime;
 981             } 
 982             else {
 983               busioreq = ioreq_copy(tmpioreq);
 984               busioreq->time = simtime;
 985               busioreq->type = IO_INTERRUPT_ARRIVE;
 986               busioreq->cause = RECONNECT;
 987             }
 988             if((currdiskreq->outblkno == currdiskreq->ioreqlist->blkno) 
 989                && (currdiskreq->hittype != BUFFER_APPEND)) 
 990               {
 991                 busioreq->bcount = min(tmpioreq->bcount,seg->size);
 992               } 
 993             else {
 994               int i1 = tmpioreq->blkno + tmpioreq->bcount - 
 995                 currdiskreq->outblkno;
 996 
 997               int i2 = seg->size - seg->endblkno + seg_owner->inblkno;
 998 
 999               busioreq->bcount = min(i1, i2);
1000             }
1001           }
1002       }
1003     }
1004   
1005   return busioreq;
1006 }

1009 /*
1010  * Queue priority list:
1011  * 10  HDA_OWNED, !currenthda & !effectivehda (completions and full read hits)
1012  *  9             effectivehda
1013  *  8             currenthda
1014  *  7  Appending limited fastwrite (LIMITED_FASTWRITE and BUFFER_APPEND)
1015  *  6  Full sneakyintermediateread with seg
1016  *  5  Partial sneakyintermediateread with seg
1017  *  4  Full sneakyintermediateread currently without seg
1018  *  3  Partial sneakyintermediateread currently without seg
1019  *  2  Write prebuffer to effectivehda->seg
1020  *  1  Write prebuffer to currenthda->seg
1021  *  0  Write prebuffer with other seg
1022  *  -1 Write prebuffer currently without seg
1023  *
1024  * Not usable:
1025  *    requests which don't need bus service
1026  *    requests using recycled segs
1027  *    requests with no available segs
1028  *
1029  * If a sneakyintermediateread is detected which has a seg, was marked as 
1030  * a read hit, has not transferred any data yet, and is no longer a hit, 
1031  * remove the diskreq from the segment.
1032  *
1033  * Possible improvements:
1034  *   un-set segment if read hit is no longer a hit and no data has been
1035  *   transferred yet.
1036  *
1037  *   reverse queue order (oldest first) so that we don't have to peruse
1038  *   the entire queue if a 10 is found
1039  */
1040 
1041 static diskreq * 
1042 disk_select_bus_request(disk *currdisk, ioreq_event **busioreq)
1043 {
1044   diskreq *currdiskreq = currdisk->pendxfer;
1045   diskreq *bestdiskreq = NULL;
1046   ioreq_event *currioreq = 0;
1047   ioreq_event *tmpioreq;
1048   int curr_value;
1049   int best_value = -99;
1050   int curr_set_segment;
1051   int best_set_segment = FALSE;
1052 
1053   if(disk_printhack && (simtime >= disk_printhacktime)) {
1054     fprintf (outputfile, "%12.6f         Entering disk_select_bus_request for disk %d\n", simtime, currdisk->devno);
1055     fflush(outputfile);
1056   }
1057 
1058   while (currdiskreq) {
1059     curr_value = -100;
1060     curr_set_segment = FALSE;
1061     ddbg_assert2(currdiskreq->ioreqlist != 0, 
1062                "diskreq with NULL ioreqlist found in bus queue");
1063 
1064     if(currdiskreq->seg 
1065        && (currdiskreq->seg->recyclereq == currdiskreq)) 
1066       {
1067         // dont do anything
1068       } 
1069     else if(currdiskreq->flags & HDA_OWNED) {
1070       currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1071       if(currioreq) {
1072         if(currdiskreq == currdisk->effectivehda) {
1073           curr_value = 9;
1074         } 
1075         else if(currdiskreq == currdisk->currenthda) {
1076           curr_value = 8;
1077         } 
1078         else {
1079           curr_value = 10;
1080         }
1081       }
1082     } 
1083     else if((best_value <= 7) 
1084             && (currdisk->fastwrites == LIMITED_FASTWRITE) 
1085             && (currdiskreq->hittype == BUFFER_APPEND)) 
1086       {
1087         currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1088         if(currioreq) {
1089           curr_value = 7;
1090         }
1091       } 
1092     else if((best_value <= 6) 
1093             && (currdisk->sneakyintermediatereadhits) 
1094             && (currdiskreq->ioreqlist->flags & READ)) 
1095       {
1096         if(currdiskreq->seg) {
1097           currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1098           if(currioreq) {
1099             tmpioreq = currdiskreq->ioreqlist;
1100             while (tmpioreq->next) {
1101               tmpioreq = tmpioreq->next;
1102             }
1103 
1104             curr_value = ((currdiskreq->seg->endblkno >= (tmpioreq->blkno + tmpioreq->bcount)) ? 6 : 5);
1105           } 
1106           else if(currdiskreq->outblkno == currdiskreq->ioreqlist->blkno) {
1107             /* Not entirely sure this works or is appropriate */
1108             currioreq = disk_request_needs_bus(currdisk,currdiskreq,FALSE);
1109             if(currioreq) {
1110               addtoextraq((event *)currioreq);
1111             } 
1112             else {
1113 
1114 
1115               if(disk_printhack && (simtime >= disk_printhacktime)) {
1116                 fprintf (outputfile, "%12.6f         "
1117                          "sneakyintermediatereadhits removed diskreq "
1118                          "from seg\n",
1119                          simtime);
1120 
1121                 fprintf (outputfile, "                       "
1122                          "seg = %d-%d\n", 
1123                          currdiskreq->seg->startblkno, 
1124                          currdiskreq->seg->endblkno);
1125 
1126                 fprintf (outputfile, "                       "
1127                          "diskreq = %d, %d, %d (1==R)\n",
1128                          currdiskreq->ioreqlist->blkno, 
1129                          currdiskreq->ioreqlist->bcount, 
1130                          (currdiskreq->ioreqlist->flags & READ));
1131 
1132                 fflush(outputfile);
1133               }
1134 
1135 
1136 
1137               disk_buffer_remove_from_seg(currdiskreq);
1138             }
1139           }
1140         } 
1141         if(!currdiskreq->seg && (best_value <= 4)) {
1142           disk_buffer_select_segment(currdisk, currdiskreq, FALSE);
1143           if(currdiskreq->seg) {
1144             if(currdiskreq->hittype == BUFFER_NOMATCH) {
1145               currdiskreq->seg = NULL;
1146             } 
1147             else {
1148               if(currdisk->reqwater) {
1149                 currdiskreq->watermark = max(1, (int) ((double) min(currdiskreq->seg->size, currdiskreq->ioreqlist->bcount) * currdisk->readwater));
1150               } 
1151               else {
1152                 currdiskreq->watermark = (int) (((double) currdiskreq->seg->size * currdisk->readwater) + (double) 0.9999999999);
1153               }
1154 
1155               currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1156               if(!currioreq) {
1157                 currdiskreq->seg = NULL;
1158                 currdiskreq->hittype = BUFFER_NOMATCH;
1159               } 
1160               else {
1161                 curr_value = ((currdiskreq->hittype == BUFFER_WHOLE) ? 4 : 3);
1162                 curr_set_segment = TRUE;
1163               }
1164             }
1165           }
1166         }
1167       }
1168     else if((best_value <= 2) 
1169             && (currdisk->writeprebuffering) 
1170             && !(currdiskreq->ioreqlist->flags & READ)) 
1171       {
1172         if(currdiskreq->seg) {
1173           currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1174           if(currioreq) {
1175             tmpioreq = currdiskreq->ioreqlist;
1176             while (tmpioreq->next) {
1177               tmpioreq = tmpioreq->next;
1178             }
1179             if(currdisk->effectivehda && 
1180                (currdisk->effectivehda->seg == currdiskreq->seg)) {
1181               curr_value = 2;
1182             } 
1183             else if(currdisk->currenthda && 
1184                     (currdisk->currenthda->seg == currdiskreq->seg)) {
1185               curr_value = 1;
1186             } 
1187             else {
1188               curr_value = 0;
1189             }
1190           }
1191         } 
1192         else if(best_value <= -1) {
1193           disk_buffer_select_segment(currdisk,currdiskreq,FALSE);
1194           if(currdiskreq->seg) {
1195             if(currdisk->reqwater) {
1196               currdiskreq->watermark = max(1, (int) ((double) min(currdiskreq->seg->size, currdiskreq->ioreqlist->bcount) * currdisk->writewater));
1197             } 
1198             else {
1199               currdiskreq->watermark = (int) (((double) currdiskreq->seg->size * currdisk->writewater) + (double) 0.9999999999);
1200             }
1201             currioreq = disk_request_needs_bus(currdisk,currdiskreq,TRUE);
1202             if(currioreq) {
1203               curr_value = -1;
1204               curr_set_segment = TRUE;
1205             } 
1206             else {
1207               currdiskreq->seg = NULL;
1208               currdiskreq->hittype = BUFFER_NOMATCH;
1209             }
1210           }
1211         }
1212       }
1213 
1214     if(curr_value >= best_value) {
1215       if(*busioreq) {
1216         addtoextraq((event *) *busioreq);
1217         if(best_set_segment) {
1218           bestdiskreq->seg = NULL;
1219           bestdiskreq->hittype = BUFFER_NOMATCH;
1220         }
1221       }
1222       best_value = curr_value;
1223       best_set_segment = curr_set_segment;
1224       bestdiskreq = currdiskreq;
1225       *busioreq = currioreq;
1226     } else if(curr_set_segment) {
1227       currdiskreq->seg = NULL;
1228       currdiskreq->hittype = BUFFER_NOMATCH;
1229     }
1230 
1231     currdiskreq = currdiskreq->bus_next;
1232   }
1233 
1234   if(best_set_segment) {
1235     disk_buffer_set_segment(currdisk,bestdiskreq);
1236   }
1237 
1238   return(bestdiskreq);
1239 } // disk_select_bus_request()

1242 /*
1243  * Attempt to start bus activity.  If currdiskreq doesn't match a
1244  * non-NULL effectivebus, do nothing.  Otherwise, if currdiskreq
1245  * doesn't match a non-NULL currentbus, do nothing.  Otherwise, if
1246  * there is already a request outstanding (buswait), do nothing.
1247  * Otherwise, check to see if currdiskreq needs some bus activity.  If
1248  * currdiskreq is NULL, attempt to find the appropriate next request
1249  * from the queue.
1250  *
1251  * Queue priority list:
1252  *    effectivebus (if non-NULL, this is the only choice possible)
1253  *    currentbus   (if non-NULL, this is the only choice possible)
1254  *    see disk_select_bus_request 
1255  */
1256 
1257 static void 
1258 disk_check_bus (disk *currdisk, diskreq *currdiskreq)
1259 {
1260   diskreq     *nextdiskreq = currdiskreq;
1261   diskreq     *tmpdiskreq;
1262   ioreq_event *busioreq = NULL;
1263   double        delay;
1264 
1265   disksim_inst_enter();
1266 
1267   if(disk_printhack && (simtime >= disk_printhacktime)) {
1268     fprintf (outputfile, "%12.6f  %8p  Entering disk_check_bus for disk %d\n", simtime, currdiskreq,currdisk->devno);
1269     fflush(outputfile);
1270   }
1271 
1272   if(currdisk->buswait) {
1273     return;
1274   }
1275 
1276   /* add checking here for acctime >= 0.0 */
1277 
1278   if(currdiskreq) {
1279     if(currdisk->effectivebus) {
1280       if(currdisk->effectivebus != currdiskreq) {
1281         nextdiskreq = NULL;
1282       }
1283     } 
1284     else if(currdisk->currentbus) {
1285       if(currdisk->currentbus != currdiskreq) {
1286         nextdiskreq = NULL;
1287       }
1288     }
1289     if(nextdiskreq) {
1290 
1291       /* Only truly affected call is the one from disk_check_hda.  It
1292        * needs to be TRUE in order to match disk_completion's call.
1293        * Otherwise, preseeking is disadvantaged...  
1294        */
1295       busioreq = disk_request_needs_bus(currdisk,nextdiskreq,TRUE);
1296       if(!busioreq) {
1297         nextdiskreq = NULL;
1298       }
1299     }
1300   } 
1301   else {
1302     if(currdisk->effectivebus) {
1303       busioreq = disk_request_needs_bus(currdisk,currdisk->effectivebus,TRUE);
1304       if(busioreq) {
1305         nextdiskreq = currdisk->effectivebus;
1306       }
1307     } 
1308     else if(currdisk->currentbus) {
1309       busioreq = disk_request_needs_bus(currdisk,currdisk->currentbus,TRUE);
1310       if(busioreq) {
1311         nextdiskreq = currdisk->currentbus;
1312       }
1313     } 
1314     else {
1315       nextdiskreq = disk_select_bus_request(currdisk,&busioreq);
1316     }
1317   }
1318 
1319   if(nextdiskreq) {
1320     if(disk_printhack && (simtime >= disk_printhacktime)) {
1321       fprintf (outputfile, "                        nextdiskreq = %8p\n", nextdiskreq);
1322       fflush(outputfile);
1323     }
1324 
1325     if((nextdiskreq != currdisk->currentbus) 
1326        && (nextdiskreq != currdisk->effectivebus)) 
1327       {
1328         /* remove nextdiskreq from bus queue */
1329         if(currdisk->pendxfer == nextdiskreq) {
1330           currdisk->pendxfer = nextdiskreq->bus_next;
1331         } 
1332         else {
1333           tmpdiskreq = currdisk->pendxfer;
1334           while (tmpdiskreq && (tmpdiskreq->bus_next != nextdiskreq)) {
1335             tmpdiskreq = tmpdiskreq->bus_next;
1336           }
1337 
1338           ddbg_assert2(tmpdiskreq != 0, 
1339                      "Next bus request not found in bus queue");
1340 
1341           tmpdiskreq->bus_next = nextdiskreq->bus_next;
1342         }
1343         nextdiskreq->bus_next = NULL;
1344       }
1345 
1346     currdisk->currentbus = currdisk->effectivebus = nextdiskreq;
1347 
1348     if(busioreq->cause == RECONNECT) {
1349       if(nextdiskreq->ioreqlist->flags & READ) {
1350         disk_activate_read(currdisk, nextdiskreq, FALSE, FALSE);
1351       } 
1352       else {
1353         disk_activate_write(currdisk, nextdiskreq, FALSE, FALSE);
1354       }
1355 
1356       if(currdisk->extradisc_diskreq == nextdiskreq) {
1357         currdisk->extradisc_diskreq = NULL;
1358         delay = currdisk->overhead_reselect_first;
1359       } 
1360       else {
1361         delay = ((nextdiskreq->flags & EXTRA_WRITE_DISCONNECT) ? currdisk->overhead_reselect_other : currdisk->overhead_reselect_first);
1362       }
1363 
1364       if(busioreq->bcount > 0) {
1365         delay += currdisk->overhead_data_prep;
1366       }
1367 
1368       if(currdisk->const_acctime) {
1369         delay = 0.0;
1370       } 
1371       else {
1372         nextdiskreq->seg->outstate = BUFFER_CONTACTING;
1373       }
1374 
1375       currdisk->outstate = DISK_WAIT_FOR_CONTROLLER;
1376       disk_send_event_up_path(busioreq, (delay * currdisk->timescale));
1377     } 
1378 
1379     else if(busioreq->cause == DEVICE_DATA_TRANSFER_COMPLETE) {
1380       disk_reconnection_or_transfer_complete(busioreq);
1381     } 
1382     else {
1383       ddbg_assert2(0, "unexpected busioreq->cause in disk_check_bus");
1384     }
1385   }
1386 } // disk_check_bus()

1623 /* Attempt to start hda activity.  
1624  * 1.  If effectivehda already set, do nothing.
1625  * 2.  If currdiskreq is NULL, attempt to find the appropriate next
1626  * request from the queue.  
1627  */
1628 
1629 static void
1630 disk_get_effectivehda(disk *currdisk,
1631                       diskreq *currdiskreq)
1632 {
1633   diskreq     *nextdiskreq = currdiskreq;
1634   ioreq_event *nextioreq;
1635 
1636   // if we have a currenthda, make that the effectivehda
1637   if(currdisk->currenthda) {
1638     if(currdiskreq && (currdiskreq != currdisk->currenthda)) {
1639       fprintf(stderr, "currdiskreq != currenthda in disk_check_hda\n");
1640       assert(0);
1641     }
1642     currdisk->effectivehda = currdisk->currenthda;
1643   } 
1644 
1645   // No currenthda.  Let's find a "next" io request
1646   // (nextdiskreq) and make that the current/effective hda.
1647   else { 
1648     if(nextdiskreq && nextdiskreq->ioreqlist) {
1649       nextioreq = ioreq_copy(nextdiskreq->ioreqlist);
1650       nextioreq->ioreq_hold_diskreq = nextdiskreq;
1651       nextioreq->ioreq_hold_disk = currdisk;
1652       if(!disk_enablement_function(nextioreq)) {
1653         nextdiskreq = NULL;
1654       }
1655       addtoextraq((event *)nextioreq);
1656     }
1657 
1658     // we don't have a nextdiskreq yet.  check the disk's ioqueue for
1659     // one.
1660     if(!nextdiskreq) {
1661       nextioreq = ioqueue_show_next_request(currdisk->queue);
1662       if(nextioreq) {
1663         nextdiskreq = (diskreq*) nextioreq->ioreq_hold_diskreq;
1664       }
1665     }
1666 
1667     // if we don't have a nextdiskreq by now, we're going to fall out
1668     // of this without having figured out an effectivehda and bail
1669     // back in disk_check_hda()
1670     if(nextdiskreq) {
1671       if(nextdiskreq->flags & HDA_OWNED) {
1672         fprintf(stderr, "disk_check_hda:"
1673                 "  diskreq was already HDA_OWNED\n");
1674         assert(0);
1675       }
1676 
1677       // does this request have a disk cache segment associated with
1678       // it?
1679       if(nextdiskreq->seg && 
1680           !(nextdiskreq->flags & SEG_OWNED) &&
1681           (nextdiskreq->seg->recyclereq != nextdiskreq)) 
1682       {
1683         disk_buffer_attempt_seg_ownership(currdisk, nextdiskreq);
1684       }
1685       
1686 
1687       if(!nextdiskreq->seg) {
1688         // hmmm ... no cache segment ... let's try to find one
1689         disk_buffer_select_segment(currdisk,nextdiskreq,FALSE);
1690         if(nextdiskreq->seg) {
1691           disk_buffer_set_segment(currdisk, nextdiskreq);
1692           disk_buffer_attempt_seg_ownership(currdisk, nextdiskreq);
1693         } 
1694         else { 
1695           // still no cache segment.  try to "recycle" one.
1696           nextdiskreq->seg = 
1697             disk_buffer_recyclable_segment(currdisk,
1698                                            (nextdiskreq->ioreqlist->flags & READ));
1699           if(nextdiskreq->seg) {
1700             nextdiskreq->seg->recyclereq = nextdiskreq;
1701           }
1702 
1703         } // else { // nextdiskreq->seg == 0
1704       } // if(!nextdiskreq->seg)
1705          
1706       // So if we succeeded in finding a cache segment above, 
1707       // do something with the ioreqs on the nextdiskreq->ioreqlist.
1708       if(nextdiskreq->seg) {
1709         nextioreq = nextdiskreq->ioreqlist;
1710         while(nextioreq) { 
1711           if(!ioqueue_get_specific_request(currdisk->queue,
1712                                            nextioreq)) 
1713           {
1714             fprintf(stderr, "disk_get_effectivehda()"
1715                     "  ioreq_event not found by"
1716                     " ioqueue_get_specific_request call\n");
1717             assert(0);
1718           }
1719           
1720           nextioreq = nextioreq->next;
1721         }
1722       } // if(nextdiskreq->seg)
1723       // what if we didn't find a segment?
1724 
1725       currdisk->currenthda = currdisk->effectivehda = nextdiskreq;
1726       nextdiskreq->flags |= HDA_OWNED;
1727      
1728     } // if(nextdiskreq)
1729   } // else { // currdisk->currenthda == 0
1730 
1731   if(disk_printhack && (simtime >= disk_printhacktime)) {
1732     fprintf (outputfile, "                        "
1733              "effectivehda = %8p\n", currdisk->effectivehda);
1734     fflush(outputfile);
1735   }
1736 } // disk_get_effectivehda()

1741 // handle the case for disk_check_hda when requests are pending
1742 // (i.e. not pure prefetch)
1743 static void 
1744 disk_check_hda_pending(disk *currdisk, 
1745                        ioreq_event *nextioreq, 
1746                        diskreq *nextdiskreq, 
1747                        segment *seg, 
1748                        int ok_to_check_bus,
1749                        double *mintime,
1750                        int *immediate_release,
1751                        int *initiate_seek) 
1752 {
1753   disksim_inst_enter();
1754 
1755   nextioreq = nextdiskreq->ioreqlist;
1756   while (nextioreq->next) {
1757     nextioreq = nextioreq->next;
1758   }
1759   seg->access->flags = nextioreq->flags;
1760 
1761   // is the request a READ?
1762   if(nextioreq->flags & READ) {
1763     if(seg->access->type != NULL_EVENT) {
1764       /*
1765        *        if((seg->access->blkno != seg->endblkno) &&
1766        *        (seg->access->blkno != (seg->endblkno + 1)) &&
1767        *        (seg->access->blkno !=
1768        *        currdisk->firstblkontrack)) { fprintf(stderr,
1769        *        "non-NULL seg->access->type with incorrect
1770        *        seg->access->blkno found for read segment in
1771        *        disk_check_hda\n"); exit(1); } 
1772        */
1773     } 
1774     else {
1775       // here's our bug ... we have some bogus seg that ends on
1776       // the 1-past-the-end-of-the-disk block and we try
1777       // to access it here
1778       seg->access->blkno = seg->endblkno;
1779     }
1780 
1781     if(!seg->recyclereq) {
1782       seg->minreadaheadblkno = 
1783         max(seg->minreadaheadblkno, 
1784             min((nextioreq->blkno + 
1785                  nextioreq->bcount + 
1786                  currdisk->minreadahead), 
1787                 currdisk->model->dm_sectors));
1788 
1789       seg->maxreadaheadblkno = 
1790         max(seg->maxreadaheadblkno, 
1791             min(disk_buffer_get_max_readahead(currdisk,seg,nextioreq),
1792                 currdisk->model->dm_sectors));
1793     }
1794 
1795     currdisk->immed = currdisk->immedread;
1796     seg->state = BUFFER_READING;
1797     if(seg->recyclereq) {
1798       seg->access->blkno = nextdiskreq->outblkno;
1799       *initiate_seek = TRUE;
1800     } 
1801     else if((seg->endblkno < (nextioreq->blkno + 
1802                               nextioreq->bcount)) 
1803             || (seg->startblkno > nextdiskreq->outblkno)) 
1804     {
1805       *initiate_seek = TRUE;
1806     } 
1807     else if(seg->endblkno < 
1808             max(seg->maxreadaheadblkno, 
1809                 disk_buffer_get_max_readahead(currdisk, 
1810                                               seg, 
1811                                               nextioreq))) 
1812     {
1813       seg->access->flags |= BUFFER_BACKGROUND;

1900  * 2.  If currdiskreq is NULL, attempt to find the appropriate next

1904                            diskreq *currdiskreq, 
1905                            int ok_to_check_bus)
1906 {

1918   // dest track.

1921   //                                              currdiskreq->inblkno);
1922 
1923   if(disk_printhack && (simtime >= disk_printhacktime)) {
1924     fprintf (outputfile, "%12.6f  %8p  Entering disk_check_hda"

1926     fflush(outputfile);
1927   }
1928 
1929   if(currdisk->const_acctime) {

1931   }

1933   // 1.  If effectivehda already set, do nothing.
1934   if(currdisk->effectivehda) {
1935     return;
1936   }
1937   else {
1938     // try to get an effectivehda

1953                            nextioreq, 
1954                            nextdiskreq, 
1955                            seg, 
1956                            ok_to_check_bus,
1957                            &mintime,
1958                            &immediate_release,
1959                            &initiate_seek);
1960   } 
1961   else {
1962     // there aren't any requests so we're only prefetching
1963     seg->state = BUFFER_READING;
1964     if(seg->endblkno < seg->maxreadaheadblkno) {
1965       if(seg->access->type != NULL_EVENT) {
1966         /* 
1967          * if((seg->access->blkno != seg->endblkno) &&
1968          * (seg->access->blkno != (seg->endblkno + 1)) &&
1969          * (seg->access->blkno != currdisk->firstblkontrack)) {
1970          * fprintf(stderr, "non-NULL seg->access->type with
1971          * incorrect seg->access->blkno found for pure read
1972          * segment in disk_check_hda\n"); exit(1); } 
1973          */
1974       } 
1975       else {
1976         seg->access->blkno = seg->endblkno;
1977       }
1978 
1979       seg->access->flags = READ | BUFFER_BACKGROUND;
1980       currdisk->immed = currdisk->immedread;
1981 
1982       if(seg->access->type == NULL_EVENT) {
1983         initiate_seek = TRUE;
1984       }
1985     } 
1986     else {
1987       immediate_release = TRUE;
1988     }
1989   } // pure prefetch case
1990 
1991   if(disk_printhack && (simtime >= disk_printhacktime)) {
1992     fprintf (outputfile, "                        "
1993              "initiate_seek = %d, immediate_release = %d\n", 
1994              initiate_seek, immediate_release);
1995     fflush(outputfile);
1996   }
1997 
1998   if(initiate_seek) {
1999     if(nextdiskreq->overhead_done > simtime) {
2000       delay += nextdiskreq->overhead_done - simtime;
2001     }
2002      
2003     if(disk_seek_stoptime > simtime) {
2004       delay += disk_seek_stoptime - simtime;
2005     }
2006      
2007     if(!disk_initiate_seek(currdisk,
2008                            seg, 
2009                            seg->access, 
2010                            TRUE, 
2011                            delay, 
2012                            mintime)) 
2013     {
2014       disk_release_hda(currdisk, nextdiskreq);
2015     }
2016   }
2017   else if(immediate_release) {
2018     disk_release_hda(currdisk, nextdiskreq);
2019   }
2020 
2021   if(!immediate_release 
2022      && (seg->state == BUFFER_READING) 
2023      && (seg->access->flags & BUFFER_BACKGROUND)) 
2024   {
2025     disk_check_prefetch_swap(currdisk);
2026   }

2430    * select_seg (even if seg == NULL, hittype should be set to
2431    * BUFFER_NOMATCH) 
2432    */
2433 
2434   if(flags & READ) {
2435     disk_read_arrive(currdisk, curr, new_diskreq, seg, intrp);
2436   }
2437   else {                                
2438     disk_write_arrive(currdisk, curr, new_diskreq, seg, intrp);

2443 static void 
2444 disk_read_arrive(disk *currdisk,
2445                  ioreq_event *curr,
2446                  diskreq *new_diskreq,
2447                  segment *seg,
2448                  ioreq_event *intrp)
2449 {
2450   double delay;
2451   int enough_ready_to_go = TRUE;
2452 
2453   disksim_inst_enter();
2454   
2455   /* set up overhead_done to delay any prefetch or completion read
2456    * activity.  readmiss overheads are used to be compatible with
2457    * the old simulator.  
2458    */
2459  
2460   if(currdisk->lastflags & READ) {
2461     new_diskreq->overhead_done = 
2462       currdisk->overhead_command_readmiss_afterread;
2463   }
2464   else {
2465     new_diskreq->overhead_done = 
2466       currdisk->overhead_command_readmiss_afterwrite;
2467   }
2468   new_diskreq->overhead_done *= currdisk->timescale;
2469   new_diskreq->overhead_done += simtime;
2470 
2471   switch (new_diskreq->hittype) {
2472   case BUFFER_PARTIAL:
2473     if(!currdisk->immedtrans_any_readhit) {
2474       int readwater;
2475 
2476       if(currdisk->reqwater) {
2477         readwater = max(1, (int)((double)min(seg->size,curr->bcount) 
2478                                  * currdisk->readwater));
2479       } 
2480       else {
2481         readwater = (int)((((double) seg->size * currdisk->readwater)) + (double) 0.9999999999);
2482       }
2483       if((seg->endblkno <= curr->blkno) 
2484          || ((seg->endblkno - curr->blkno) < min(readwater,curr->bcount))) 
2485         {
2486           enough_ready_to_go = FALSE;
2487         }
2488     }
2489     /* FALLTHROUGH */
2490 
2491   case BUFFER_COLLISION:
2492   case BUFFER_NOMATCH:
2493     if(new_diskreq->hittype != BUFFER_PARTIAL) {
2494       enough_ready_to_go = FALSE;
2495     }
2496     /* FALLTHROUGH */
2497 
2498   case BUFFER_WHOLE:
2499 
2500     /* IF(never_disconnect 
2501          OR (enough_ready_to_go 
2502            AND (sneaky read hits allowed 
2503               OR (no request is active
2504                  OR active request is preemptable)
2505                   )
2506               )
2507           )
2508        THEN
2509          Perform appropriate read hit overhead and prepare to 
2510          transfer the data.
2511        ELSE
2512          Otherwise, perform the initial command overhead portion,
2513          place the request on the hda and bus queues, and disconnect.
2514     */
2515 
2516     if(currdisk->neverdisconnect 
2517        || (enough_ready_to_go 
2518            && !currdisk->currentbus 
2519            && !currdisk->const_acctime
2520            && ((currdisk->sneakyfullreadhits 
2521                 && (new_diskreq->hittype == BUFFER_WHOLE)) 
2522                || (currdisk->sneakypartialreadhits 
2523                    && (new_diskreq->hittype == BUFFER_PARTIAL)) 
2524                || (!currdisk->pendxfer 
2525                    && (!currdisk->effectivehda 
2526                        || disk_buffer_stopable_access(currdisk,new_diskreq)
2527                        )
2528                    )
2529                )
2530            )
2531        )
2532     {
2533       intrp->cause = READY_TO_TRANSFER;
2534       currdisk->currentbus = new_diskreq;
2535       if(seg) {
2536         disk_buffer_set_segment(currdisk,new_diskreq);
2537         seg->outstate = BUFFER_CONTACTING;
2538       }
2539 
2540       if(!currdisk->const_acctime) {
2541         if(new_diskreq->hittype == BUFFER_NOMATCH) {
2542           if(currdisk->lastflags & READ) {
2543             delay = currdisk->overhead_command_readmiss_afterread;
2544           }
2545           else {
2546             delay = currdisk->overhead_command_readmiss_afterwrite;
2547           }
2548         } 
2549         else {
2550           if(currdisk->lastflags & READ) {
2551             delay = currdisk->overhead_command_readhit_afterread; 
2552           }
2553           else {
2554             delay = currdisk->overhead_command_readhit_afterwrite; 
2555           }
2556         }
2557         delay += currdisk->overhead_data_prep;
2558       }
2559       else {
2560         delay = 0.0;
2561       }
2562 
2563       disk_send_event_up_path(intrp, (delay * currdisk->timescale));
2564       disk_activate_read(currdisk, new_diskreq, FALSE, TRUE);
2565     }
2566     else {
2567       intrp->cause = DISCONNECT;
2568       if(!currdisk->const_acctime) {
2569         if(currdisk->lastflags & READ) {
2570           delay = currdisk->overhead_command_readmiss_afterread +
2571             currdisk->overhead_disconnect_read_afterread;
2572         }
2573         else {
2574           delay = currdisk->overhead_command_readmiss_afterwrite +
2575             currdisk->overhead_disconnect_read_afterwrite;
2576         }
2577       }
2578       else {
2579         delay = 0.0;
2580       }
2581       disk_send_event_up_path(intrp, (delay * currdisk->timescale));
2582       disk_activate_read(currdisk, new_diskreq, TRUE, TRUE);
2583     }
2584     break;
2585       
2586   default:
2587     ddbg_assert3(0, ("Invalid read hittype in disk_request_arrive - "
2588                "blkno %d, bcount %d, hittype %d\n", 
2589                curr->blkno, curr->bcount, new_diskreq->hittype));
2590   }
2591 }

2601 
2602   disksim_inst_enter();
2603 
2604   /* set up overhead_done to delay any mechanical activity */
2605   if(currdisk->fastwrites == NO_FASTWRITE) {
2606     if(currdisk->lastflags & READ) {
2607       new_diskreq->overhead_done = 
2608         currdisk->overhead_command_writemiss_afterread;

2610     else {

2617       new_diskreq->overhead_done =

2619     }
2620     else {
2621       new_diskreq->overhead_done =
2622         currdisk->overhead_command_writehit_afterwrite;
2623     }
2624   }
2625    
2626   new_diskreq->overhead_done *= currdisk->timescale;
2627   new_diskreq->overhead_done += simtime;
2628 
2629 
2630   /* LIMITED_FASTWRITE checking is done for "free", as there will be
2631    * no currentbus or pendxfer and the effectivehda access will be
2632    * "stopable" in appropriate cases.  
2633    */

2635   if(currdisk->neverdisconnect 
2636      || (seg 
2637          && !currdisk->currentbus 
2638          && !currdisk->const_acctime
2639          && (currdisk->writeprebuffering 
2640              || (!currdisk->pendxfer 
2641                  && (!currdisk->effectivehda

2643                      )
2644                  )
2645              )
2646          )
2647      )
2648     {
2649       if((new_diskreq->flags & EXTRA_WRITE_DISCONNECT) 
2650          && !currdisk->const_acctime) 
2651         {
2652           /* re-set up overhead_done to delay any mechanical activity */

2654             currdisk->extradisc_command + currdisk->extradisc_seekdelta;
2655           
2656           new_diskreq->overhead_done *= currdisk->timescale;
2657           new_diskreq->overhead_done += simtime;

2661             currdisk->extradisc_command + currdisk->extradisc_disconnect1;
2662           
2663           extra_write_disconnects++;
2664           if(seg) {
2665             seg->outstate = BUFFER_IDLE;
2666           }
2667         }
2668       else {
2669         intrp->cause = READY_TO_TRANSFER;
2670         if(!currdisk->const_acctime) {
2671           if(currdisk->fastwrites == NO_FASTWRITE) {
2672             if(currdisk->lastflags & READ) {
2673               delay = currdisk->overhead_command_writemiss_afterread; 
2674             }
2675             else {
2676               delay = currdisk->overhead_command_writemiss_afterwrite; 
2677             }
2678           } 
2679           else {
2680             if(currdisk->lastflags & READ) {
2681               delay = currdisk->overhead_command_writehit_afterread;
2682             }
2683             else {
2684               delay = currdisk->overhead_command_writehit_afterwrite;
2685             }
2686           }
2687           delay += currdisk->overhead_data_prep;
2688         }
2689         else {
2690           delay = 0.0;
2691         }
2692 
2693         if(seg) {
2694           seg->outstate = BUFFER_CONTACTING;
2695         }
2696       }
2697       currdisk->currentbus = new_diskreq;
2698       
2699       
2700       if(seg) {
2701         disk_buffer_set_segment(currdisk,new_diskreq);
2702       }
2703 
2704       disk_send_event_up_path(intrp, (delay * currdisk->timescale));
2705       disk_activate_write(currdisk, new_diskreq, FALSE, TRUE);

2707   else {
2708     new_diskreq->flags &= ~EXTRA_WRITE_DISCONNECT;
2709     intrp->cause = DISCONNECT;

2711       if(currdisk->lastflags & READ) {
2712         delay = currdisk->overhead_command_writemiss_afterread;
2713       }
2714       else {
2715         delay = currdisk->overhead_command_writemiss_afterwrite;
2716       }
2717     } 
2718     else {
2719       if(!currdisk->const_acctime) { 
2720         // i.e. not constant acctime
2721         if(currdisk->lastflags & READ) {
2722           delay = currdisk->overhead_command_writehit_afterread;
2723         } 
2724         else {
2725             delay = currdisk->overhead_command_writehit_afterwrite;
2726         }
2727         delay += currdisk->overhead_disconnect_write;
2728       }
2729       else {
2730         delay = 0.0;
2731       }
2732         
2733       disk_send_event_up_path(intrp, (delay * currdisk->timescale));
2734       disk_activate_write(currdisk, new_diskreq, TRUE, TRUE);

2827     fflush(outputfile);
2828   }
2829 
2830   ddbg_assert2(currdiskreq != 0, "effectivebus is NULL");
2831 
2832   ddbg_assert2((currdiskreq->flags & COMPLETION_SENT),
2833              "effectivebus isn't marked COMPLETION_SENT");
2834

2836   ddbg_assert2((seg != 0) || (currdisk->const_acctime >= 0),
2837              "NULL seg found in effectivebus");
2838 
2839   tmpioreq = currdiskreq->ioreqlist;
2840   while (tmpioreq) {
2841     if(tmpioreq->blkno == curr->blkno) {
2842       break;
2843     }
2844     tmpioreq = tmpioreq->next;
2845   }
2846   ddbg_assert2(tmpioreq != 0, "ioreq_event not found in effectivebus");
2847 
2848   while (tmpioreq->next) {
2849     tmpioreq = tmpioreq->next;
2850   }
2851 
2852   currdiskreq->flags |= COMPLETION_RECEIVED;
2853   addtoextraq((event *) curr);
2854   currdisk->effectivebus = NULL;
2855   if(currdiskreq == currdisk->currentbus) {
2856     currdisk->currentbus = NULL;
2857   }
2858 
2859   if(currdisk->busowned != -1) {
2860     bus_ownership_release(currdisk->busowned);
2861     currdisk->busowned = -1;
2862     currdisk->outstate = DISK_IDLE;
2863   }
2864 
2865   if(currdisk->const_acctime) {
2866     addtoextraq((event *) currdiskreq->ioreqlist);
2867     addtoextraq((event *) currdiskreq);
2868     return;
2869   } 
2870   else if(currdiskreq->ioreqlist->flags & READ) {
2871     while (currdiskreq->ioreqlist) {
2872       tmpioreq = currdiskreq->ioreqlist;
2873       currdiskreq->ioreqlist = tmpioreq->next;
2874       addtoextraq((event *) tmpioreq);
2875     }
2876     if((currdiskreq == currdisk->effectivehda) ||
2877        (currdiskreq == currdisk->currenthda)) {
2878       if(seg->access->type == NULL_EVENT) {
2879         disk_release_hda(currdisk,currdiskreq);
2880       }
2881       else {
2882         disk_check_prefetch_swap(currdisk);
2883       }
2884     }
2885     else {
2886       disk_buffer_remove_from_seg(currdiskreq);
2887       addtoextraq((event *) currdiskreq);
2888       if(seg->recyclereq) {
2889         ddbg_assert2(seg->diskreqlist == 0,
2890                    "non-NULL diskreqlist found for recycled seg");
2891

2893         disk_buffer_attempt_seg_ownership(currdisk,seg->recyclereq);
2894         seg->recyclereq = NULL;
2895       } 
2896       else if(seg->diskreqlist) {
2897         disk_find_new_seg_owner(currdisk,seg);
2898       } 
2899       else if(!currdisk->enablecache) {
2900         seg->state = BUFFER_EMPTY;
2901       }
2902     }
2903   } 
2904   else {                        /* WRITE */
2905     if(currdiskreq->inblkno >= (tmpioreq->blkno + tmpioreq->bcount)) {
2906       if((currdiskreq == currdisk->effectivehda) ||
2907          (currdiskreq == currdisk->currenthda)) {
2908         disk_release_hda(currdisk,currdiskreq);
2909       } 
2910       else {
2911         while (currdiskreq->ioreqlist) {
2912           tmpioreq = currdiskreq->ioreqlist;
2913           currdiskreq->ioreqlist = tmpioreq->next;
2914           addtoextraq((event *) tmpioreq);
2915         }
2916         disk_buffer_remove_from_seg(currdiskreq);
2917         addtoextraq((event *) currdiskreq);
2918         if(seg->recyclereq) {
2919           ddbg_assert2(seg->diskreqlist == 0,
2920                      "non-NULL diskreqlist found for recycled seg");
2921 
2922           disk_buffer_set_segment(currdisk,seg->recyclereq);
2923           disk_buffer_attempt_seg_ownership(currdisk,seg->recyclereq);
2924           seg->recyclereq = NULL;
2925         } else if(seg->diskreqlist) {
2926           disk_find_new_seg_owner(currdisk,seg);
2927         }
2928       }
2929     } 
2930     else if(currdiskreq->flags & SEG_OWNED) {
2931       seg->access->flags |= BUFFER_BACKGROUND;
2932     }
2933   }
2934 
2935   disk_check_hda(currdisk, NULL, TRUE);
2936   disk_check_bus(currdisk, NULL);
2937 }

3715  * sequential writes.

3718  * appropriate, or if its seg is marked as recycled, or if there is a

3721  * ioqueue_show_next_request.

3867

3869 
3870           end.head = pbn.head;
3871           end.cyl = pbn.cyl;

3873 
3874           // seg->access->time = seg->time + 
3875           //   diskacctime(currdisk, 
3876           //               seg->access->tempptr1, 
3877           //               DISKSEEK, 
3878           //               (seg->access->flags & READ), 
3879           //               seg->time,

3881           //               global_currsurface,
3882           //               seg->access->cause,
3883           //               seg->access->bcount, 0);

3885

3888                         &currdisk->mech_state,
3889                         &end,

3910       }

3912 
3913   }
3914   else if(seg->state == BUFFER_WRITING) {

3916        (currdiskreq->seg == seg) &&
3917        (seg->endblkno == currdiskreq->ioreqlist->blkno) &&
3918        ((currdisk->fastwrites != LIMITED_FASTWRITE) || 
3919         (ioqueue_get_number_pending(currdisk->queue) == 1))) {

3921       
3922       if(!bestioreq || (bestioreq->ioreq_hold_diskreq == currdiskreq)) {
3923         return(TRUE);
3924       }
3925     }
3926   }
3927

3929
