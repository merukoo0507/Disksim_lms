 104 static void
 105 determine_zones(disk *currdisk, sfreq_t *sectfreq, int capacity)
 106 {
 107   typedef struct sp {
 108     int spt;
 109     int count;
 110     int avgcyl;
 111     struct sp *next;
 112   } spt_t;
 113 
 114   spt_t *sectspertrk, *currspt, *temp;
 115   spt_t *prev = 0;
 116   int i;
 117 
 118   int sectpercyl_sum = 0;
 119 
 120   if (!(sectspertrk = (spt_t *) malloc(sizeof(spt_t))))
 121     fprintf(stderr,"Could not allocate memory!\n");
 122 
 123   sectspertrk->spt = 0;
 124   sectspertrk->count = 0;
 125   sectspertrk->avgcyl = 0;
 126   sectspertrk->next = NULL;
 127 
 128   for(i=1;i<MAXSECTORS;i++) {
 129     if (sectfreq[i].count > 0) {
 130       int avg = sectfreq[i].cyls / sectfreq[i].count;
 131       currspt = sectspertrk;
 132       do {
 133         if (abs(avg - currspt->avgcyl) > (currspt->avgcyl * PERCENT)) {
 134           if (sectfreq[i].count >= (COUNT_THRSHLD * capacity)) {
 135             if (!(temp = (spt_t *) malloc(sizeof(spt_t)))) {
 136               fprintf(stderr,"Could not allocate memory!\n");
 137             }
 138             temp->spt = i;
 139             temp->count = sectfreq[i].count;
 140             temp->avgcyl = sectfreq[i].cyls / sectfreq[i].count;
 141             temp->next = currspt;
 142             if (currspt == sectspertrk) {
 143               sectspertrk = temp;
 144             }
 145             else {
 146               prev->next = temp;
 147             }
 148           }
 149           break;
 150         } 
 151         else {  /* within an allowed limit*/
 152           if (sectfreq[i].count > currspt->count) {
 153             currspt->spt = i;
 154             currspt->count = sectfreq[i].count;
 155             currspt->avgcyl = sectfreq[i].cyls / sectfreq[i].count;
 156             break;
 157           }
 158         }
 159         prev = currspt;
 160       } while ((currspt = currspt->next));
 161     }
 162 #ifdef PRINT_LAYOUT_INFO
 163     if (i == 1) {
 164       printf("\tSPT\tCount\tAvg. Cyl\n");
 165       printf("=============================\n");
 166     }
 167     if (sectfreq[i].count > 0) {
 168       printf("\t%d\t%d\t%d\n",i,sectfreq[i].count, 
 169              sectfreq[i].cyls / sectfreq[i].count);
 170     }
 171 #endif
 172   }
 173   
 174   currspt = sectspertrk;
 175 
 176 #ifdef PRINT_LAYOUT_INFO
 177   printf("\nSPT\tCount\tAvg. Cyl\tMin. Cyl\tMax. Cyl\n");
 178   printf("========================================================\n");
 179 #endif
 180 
 181   i = 0;
 182   do {
 183     if (currspt->count > 0) {
 184       currdisk->bands[i].startcyl = sectfreq[currspt->spt].mincyl;
 185       currdisk->bands[i].endcyl = sectfreq[currspt->spt].maxcyl;
 186       currdisk->bands[i].blkspertrack = currspt->spt;
 187       currdisk->bands[i].blksinband = 
 188         currdisk->rlcyls[currdisk->bands[i].endcyl].lastlbn - 
 189         currdisk->rlcyls[currdisk->bands[i].startcyl].firstlbn + 1;
 190 
 191       currdisk->bands[i].firstblkno = 0.0;
 192       currdisk->bands[i].deadspace = 0;
 193       currdisk->bands[i].sparecnt = 0;
 194       currdisk->bands[i].numslips = 0;
 195 
 196       sectpercyl_sum += currspt->spt * currdisk->numsurfaces;
 197 
 198 #ifdef PRINT_LAYOUT_INFO
 199       printf("%3d\t%6d\t%6d",currspt->spt,currspt->count,currspt->avgcyl);
 200       printf("\t\t%6d\t\t%6d\n",sectfreq[currspt->spt].mincyl,
 201              sectfreq[currspt->spt].maxcyl);
 202 #endif
 203       i++;
 204     }
 205     temp = currspt;
 206     free(temp);
 207     currspt = currspt->next;
 208   } while (currspt);
 209   currdisk->numbands = i;
 210 
 211   currdisk->sectpercyl = sectpercyl_sum / currdisk->numbands;
 212 
 213 #ifdef PRINT_LAYOUT_INFO
 214   printf("\n\n");
 215   printf("SPT\tStart LBN\tEnd LBN  \tMin. Cyl\tMax. Cyl\n");
 216   printf("================================================================\n");
 217   {
 218     int total = 0;
 219     for(i=0;i< currdisk->numbands;i++) {
 220       printf("%3d\t%8d\t%8d",currdisk->bands[i].blkspertrack,
 221              currdisk->rlcyls[currdisk->bands[i].startcyl].firstlbn,
 222              currdisk->rlcyls[currdisk->bands[i].startcyl].firstlbn +
 223              currdisk->bands[i].blksinband - 1);
 224       printf("\t%6d\t\t%6d\n",
 225              /*  currdisk->bands[i].blksinband, */
 226              currdisk->bands[i].startcyl,
 227              currdisk->bands[i].endcyl);
 228       total += currdisk->bands[i].blksinband;
 229     }
 230     printf("\nTotal = %d\n",total);
 231   }
 232 #endif
 233 }

 238 int
 239 read_raw_layout(disk *currdisk,char *rlfilename) {
 240 #define LINELEN      100
 241   char line[LINELEN];
 242   int rot, numsurfaces;
 243   int lbn, cyl, head, sect, seqcnt;
 244   int linenum,retval,currcyl,currhead,valid_idx,i,spt;
 245   int range = 0;
 246   FILE *fp;
 247   int maxlbn,blocksize,numcyls;
 248 
 249   int *sf_cyls;
 250   sfreq_t *sectfreq;
 251 #ifndef DISKSIM
 252   int cnt;
 253 #endif
 254 
 255   if (!(fp = fopen (rlfilename, "r"))) {
 256     error_handler("Error opening raw layout file %s!\n",rlfilename);
 257     return -1;
 258   }
 259   
 260   if (!(sectfreq = (sfreq_t *) malloc(MAXSECTORS*sizeof(sfreq_t)))) 
 261     internal_error("Could not allocate sectfreq structure!\n");
 262   memset(sectfreq,0,sizeof(sectfreq));
 263   for(i=1;i<MAXSECTORS;i++)
 264     sectfreq[i].mincyl = NOVAL;
 265 
 266   if (!(sf_cyls = (int *) malloc(MAXSECTORS*sizeof(int)))) 
 267     internal_error("Could not allocate sf_cyls structure!\n");
 268   for(i=0;i<MAXSECTORS;i++)
 269     sf_cyls[i] = NOVAL;
 270 
 271   fgets(line,LINELEN,fp);
 272   retval = fscanf(fp,"maxlbn %d, blocksize %d\n",&maxlbn,&blocksize);
 273   retval = fscanf (fp, "%d cylinders, %d rot, %d heads\n", &numcyls,&rot, 
 274                    &numsurfaces);
 275 
 276   if (retval != 3) {
 277     error_handler("Cannot read .layout.mappings file!\n");
 278     return -1;
 279   }
 280 
 281   linenum = 4;
 282 
 283   /*  currdisk->sectsize = blocksize; */
 284   currdisk->numsurfaces = numsurfaces;
 285   currdisk->numblocks = maxlbn+1;
 286   currdisk->numcyls = numcyls;
 287   currdisk->mapping = LAYOUT_RAWFILE;
 288   currdisk->sparescheme = NO_SPARING;
 289 
 290   if (!(currdisk->lbnrange = (lbn_rawlayout_t *) malloc(currdisk->numcyls * 
 291         currdisk->numsurfaces * sizeof(lbn_rawlayout_t))))
 292     internal_error("Could not allocate lbn_rawlayout_t structure!\n");
 293 
 294   if (!(currdisk->rlcyls = (cyl_rawlayout_t *)
 295         malloc(currdisk->numcyls * sizeof(cyl_rawlayout_t))))
 296     internal_error("Could not allocate cyl_rawlayout_t structure!\n");
 297 
 298   memset(currdisk->rlcyls,0,currdisk->numcyls * sizeof(cyl_rawlayout_t));
 299   for(i=0;i<currdisk->numcyls;i++)
 300     currdisk->rlcyls[i].firstlbn = NOVAL;
 301 
 302   currcyl = NOVAL;
 303   currhead = NOVAL;
 304 
 305   /*  printf("Reading layout for %s %s\n", manufacturer, product); */
 306 
 307   fscanf (fp, "lbn %d --> cyl %d, head %d, sect %d, seqcnt %d\n", 
 308           &lbn, &cyl, &head, &sect, &seqcnt);
 309   do { 
 310     /* while (5 == fscanf (fp, "lbn %d --> cyl %d, head %d, sect %d, seqcnt %d\n", 
 311        &lbn, &cyl, &head, &sect, &seqcnt)) { */
 312     if (currdisk->numcyls <= cyl) {
 313       error_handler("Cyl %d on line %d is bigger than the max value of %d!\n",
 314                     cyl,linenum,currdisk->numcyls-1);
 315       return -1;
 316     }
 317     if (currdisk->numsurfaces <= head) {
 318       error_handler("Head %d on line %d is bigger than the max value of %d!\n",
 319                     head,linenum,currdisk->numsurfaces-1);
 320       return -1;
 321     }
 322 
 323     if ((currhead != head) || (currcyl != cyl)){
 324       if (currhead != NOVAL) {
 325         /* record the last lbn for the the surface */
 326         currdisk->rlcyls[currcyl].surface[currhead].lastlbn = lbn - 1;
 327 
 328         /* increment the count of tracks with # of sectors */
 329         spt = currdisk->rlcyls[currcyl].surface[currhead].lastlbn - 
 330           currdisk->rlcyls[currcyl].surface[currhead].firstlbn + 1;
 331 
 332         if (spt >= MAXSECTORS) {
 333           internal_error("Sector count exceeds %d (MAXSECTORS).\n",MAXSECTORS);
 334         }
 335 
 336         sectfreq[spt].count++;
 337         sectfreq[spt].cyls += /*  currcyl */ cyl;
 338 
 339         if (sf_cyls[spt] != NOVAL) {
 340           if (abs(/*  currcyl */ cyl - sf_cyls[spt]) > MIN_CYL_DISTANCE) {
 341             sectfreq[spt].mincyl = /*  currcyl */ cyl; 
 342 
 343             sectfreq[spt].count = 1;
 344             sectfreq[spt].cyls = /*  currcyl */ cyl;
 345           }
 346         }
 347         sf_cyls[spt] = /*  currcyl */ cyl;
 348 
 349         /* record min and max cylinder for the given number of blocks */
 350         if ((sectfreq[spt].mincyl>/*  currcyl */ cyl) || 
 351             (sectfreq[spt].mincyl==NOVAL)){
 352           sectfreq[spt].mincyl = /*  currcyl */ cyl;
 353         }
 354         if (sectfreq[spt].maxcyl < /*  currcyl */ cyl)
 355           sectfreq[spt].maxcyl = /*  currcyl */ cyl;
 356       }
 357       currhead = head;
 358     }
 359     if (currcyl != cyl) {
 360       if (currcyl != NOVAL) {
 361         currdisk->rlcyls[currcyl].lastlbn = lbn-1;
 362         currdisk->lbnrange[range-1].lastlbn = lbn-1;
 363       }
 364       currcyl = cyl;
 365       currdisk->rlcyls[currcyl].validcyl = 1;
 366 
 367       currdisk->lbnrange[range].firstlbn = lbn;
 368       currdisk->lbnrange[range].cyl = cyl;
 369       range++;
 370 
 371       if ((currdisk->rlcyls[currcyl].firstlbn > lbn) || 
 372           (currdisk->rlcyls[currcyl].firstlbn == NOVAL)){
 373         currdisk->rlcyls[currcyl].firstlbn = lbn;
 374       }
 375     }
 376     /* !!! coalesce sequences if sect == 0 */
 377     valid_idx = currdisk->rlcyls[currcyl].surface[head].valid++;
 378 
 379     if (!valid_idx) {
 380       currdisk->rlcyls[currcyl].surface[head].firstlbn = lbn;
 381     }
 382 #ifdef DETAILED_RAW_LAYOUT
 383     /* put in the sector and seqcnt values to the appropriate values */
 384     currdisk->rlcyls[currcyl].surface[head].sect[valid_idx] = sect;
 385     currdisk->rlcyls[currcyl].surface[head].seqcnt[valid_idx] = seqcnt;
 386 #endif
 387     linenum++;
 388   } while (5 == fscanf (fp, "lbn %d --> cyl %d, head %d, sect %d, seqcnt %d\n",
 389                         &lbn, &cyl, &head, &sect, &seqcnt));
 390 
 391   currdisk->rlcyls[currcyl].lastlbn = lbn+seqcnt;;
 392   currdisk->rlcyls[currcyl].surface[head].lastlbn = lbn+seqcnt;
 393 
 394   /* put in the values for range */
 395   currdisk->lbnrange[range-1].lastlbn = lbn+seqcnt;
 396   currdisk->numlbnranges=range;
 397 #ifndef DISKSIM
 398   currdisk->defect_lists = (defect_t *) malloc (0x80000);
 399   memset((void *)currdisk->defect_lists,0,0x80000);
 400   cnt = 1;
 401   do {
 402     retval = fscanf (fp, "Defect at cyl %d, head %d, sect %d\n",
 403                      &currdisk->defect_lists[cnt].cyl,
 404                      &currdisk->defect_lists[cnt].head,
 405                      &currdisk->defect_lists[cnt].sect);
 406     if (retval == 3) {
 407       /*  printf ("Defect at cyl %d, head %d, sect %d\n", currdisk->defect_lists[cnt].cyl, currdisk->defect_lists[cnt].head, currdisk->defect_lists[cnt].sect); */
 408       cnt++;
 409     }
 410   /* prune duplicated list entries if they are in mappings file */
 411     if ((cnt > 2) && (currdisk->defect_lists[cnt-1].cyl == currdisk->defect_lists[cnt-2].cyl) &&
 412         (currdisk->defect_lists[cnt-1].head == currdisk->defect_lists[cnt-2].head) &&
 413         (currdisk->defect_lists[cnt-1].sect == currdisk->defect_lists[cnt-2].sect)) {
 414       cnt--;
 415     }
 416   } while (retval == 3);
 417   currdisk->defect_lists[0].cyl = cnt-1;
 418 #endif  
 419 
 420   determine_zones(currdisk,sectfreq,(int) (((double) currdisk->numblocks) * 
 421                   ((double) blocksize) /1024 /1024 /1024));
 422   free(sectfreq);
 423   free(sf_cyls);
 424   fclose(fp);
 425 
 426 
 427   return 0;
 428 }
