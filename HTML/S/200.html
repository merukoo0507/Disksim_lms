 187 void bus_set_depths()
 188 {
 189    int depth;
 190    int busno;
 191    int slotno;
 192    intchar ret;
 193    int i;
 194    u_int devno;
 195    
 196    for (depth = 0; depth < MAXDEPTH; depth++) {
 197      /* fprintf (outputfile, "At depth %d\n", depth); */
 198 
 199      for (busno = 0; busno < numbuses; busno++) {
 200        struct bus *currbus = getbus(busno);
 201        /* fprintf (outputfile, "At busno %d\n", busno); */
 202 
 203        if (currbus->depth == depth) {
 204          /* fprintf (outputfile, "Busno %d is at this depth - %d\n",
 205             busno, depth); */
 206 
 207          for (slotno = 0; slotno < currbus->numslots; slotno++) {
 208            /* fprintf (outputfile, "Deal with slotno %d of busno
 209               %d\n", slotno, busno); */
 210 
 211            devno = currbus->slots[slotno].devno;
 212 
 213            switch (currbus->slots[slotno].devtype) {
 214            case CONTROLLER:
 215              /* fprintf (outputfile, "Slotno %d contains controller
 216                 number %d\n", slotno, devno); */
 217              ret.value = controller_set_depth(devno, busno, depth, slotno);
 218              break;
 219 
 220            case DEVICE:
 221              /* fprintf (outputfile, "Slotno %d contains disk number
 222                 %d\n", slotno, devno); */
 223              ret.value = device_set_depth(devno, busno, depth, slotno);
 224              break;
 225 
 226            default:         
 227              fprintf(stderr, "*** Invalid device type in bus slot\n");
 228              ddbg_assert(0);
 229            }
 230 
 231            /* fprintf (outputfile, "Back from setting device depth\n"); */
 232            if (ret.value != 0) {
 233              /* fprintf (outputfile, "Non-zero return - %x\n", ret.value); */
 234              for (i = 0; i < MAXDEPTH; i++) {
 235                if (ret.byte[i] != 0) {
 236                  struct bus *currbus2 = getbus(ret.byte[i]);
 237                  /* fprintf (outputfile, "Set busno %x to have depth
 238                     %d\n", ret.byte[i], (depth+1));*/
 239                  currbus2->depth = depth + 1;
 240                }
 241              }
 242            }
 243          }
 244        }
 245      }
 246    }

 278 void bus_event_arrive(ioreq_event *ptr)
 279 {
 280    bus_event *curr = (bus_event *) ptr;
 281 
 282    if (curr->type == BUS_OWNERSHIP_GRANTED) {
 283       struct bus *currbus = getbus(curr->busno);
 284       ASSERT(curr == currbus->arbwinner);
 285       currbus->arbwinner = NULL;
 286       stat_update (&currbus->arbwaitstats, (simtime - curr->wait_start));
 287    }
 288 
 289    switch (curr->devtype) {
 290 
 291       case CONTROLLER:
 292          if (curr->type == BUS_DELAY_COMPLETE) {
 293             controller_bus_delay_complete (curr->devno, curr->delayed_event, curr->busno);
 294          } else {
 295             controller_bus_ownership_grant (curr->devno, curr->delayed_event, curr->busno, (simtime - curr->wait_start));
 296          }
 297          break;
 298 
 299       case DEVICE:
 300          if (curr->type == BUS_DELAY_COMPLETE) {
 301             device_bus_delay_complete (curr->devno, curr->delayed_event, curr->busno);
 302          } else {
 303             device_bus_ownership_grant (curr->devno, curr->delayed_event, curr->busno, (simtime - curr->wait_start));
 304          }
 305          break;
 306 
 307       default:
 308          fprintf(stderr, "Unknown device type at bus_event_arrive: %d, type %d\n", curr->devtype, curr->type);
 309          exit(1);
 310    }
 311    addtoextraq((event *)curr);
 312 }

 315 static bus_event * bus_fifo_arbitration(bus *currbus)
 316 {
 317    bus_event *ret = currbus->owners;
 318    ASSERT(currbus->owners != NULL);
 319    currbus->owners = ret->next;
 320    return(ret);

 324 static bus_event * bus_slotpriority_arbitration(bus *currbus)
 325 {
 326    bus_event *ret = currbus->owners;
 327 
 328    ASSERT(currbus->owners != NULL);
 329    if (currbus->owners->next == NULL) {
 330       currbus->owners = NULL;
 331       return(ret);
 332    } else {
 333       bus_event *trv = ret;
 334       while (trv->next) {
 335          if (trv->next->slotno > ret->slotno) {
 336             ret = trv->next;
 337          }
 338          trv = trv->next;
 339       }
 340       if (ret == currbus->owners) {
 341          currbus->owners = ret->next;
 342          ret->next = NULL;
 343       } else {
 344          trv = currbus->owners;
 345          while ((trv->next != NULL) && (trv->next != ret)) {
 346             trv = trv->next;
 347          }
 348          ASSERT(trv->next == ret);
 349          trv->next = ret->next;
 350          ret->next = NULL;
 351       }
 352    }
 353    return(ret);
 354 }

 374 /*
 375  * int bus_ownership_get(int busno, int slotno, ioreq_event *curr)
 376  *
 377  * Returns TRUE if the bus is immediately acquired, which only happens
 378  * for an interleaved bus.  In this case, this function does nothing
 379  * else.
 380  *
 381  * Returns FALSE if there is a delay (always true with SCSI). A
 382  * bus_event of type BUS_OWNERSHIP_GRANTED for this request is created.
 383  * If the bus is initially BUS_BUSY, the event is added to the
 384  * buses.owners queue If the bus is initially BUS_FREE, its state is
 385  * changed to BUS_BUSY and is granted to this request.  (I think:
 386  * buses.arbwinner is set to this event).  The event is scheduled on
 387  * the intq for the current time plus buses.arbtime.
 388  *
 389  * Note that I don't think this really how SCSI works: If a higher
 390  * priority device arbitrates for the bus within the arbitration time,
 391  * it will win.  Maybe this window is small enough to be ignored for
 392  * this level of simulation.
 393  *
 394  */
 395 
 396 int bus_ownership_get(int busno, int slotno, ioreq_event *curr)
 397 {
 398    bus_event *tmp;
 399    struct bus *currbus = getbus (busno);
 400 
 401    if (currbus->type == INTERLEAVED) {
 402       return(TRUE);
 403    }
 404    tmp = (bus_event *) getfromextraq();
 405    tmp->type = BUS_OWNERSHIP_GRANTED;
 406    tmp->devno = currbus->slots[slotno].devno;
 407    tmp->busno = busno;
 408    tmp->delayed_event = curr;
 409    tmp->wait_start = simtime;
 410    if (currbus->state == BUS_FREE) {
 411       tmp->devtype = currbus->slots[slotno].devtype;
 412       tmp->time = simtime + currbus->arbtime;
 413 /*
 414 fprintf (outputfile, "Granting ownership immediately - devno %d, devtype %d, busno %d\n", tmp->devno, tmp->devtype, tmp->busno);
 415 */
 416       currbus->arbwinner = tmp;
 417       addtointq((event *) tmp);
 418       currbus->state = BUS_OWNED;
 419       currbus->runidletime += simtime - currbus->lastowned;
 420       stat_update (&currbus->busidlestats, (simtime - currbus->lastowned));
 421    } else {
 422       tmp->slotno = slotno;
 423 /*
 424 fprintf (outputfile, "Must wait for bus to become free - devno %d, slotno %d, busno %d\n", tmp->devno, tmp->slotno, tmp->busno);
 425 */
 426       tmp->next = NULL;
 427       if (currbus->owners) {
 428          bus_event *tail = currbus->owners;
 429          while (tail->next) {
 430             tail = tail->next;
 431          }
 432          tail->next = tmp;
 433       } else {
 434          currbus->owners = tmp;
 435       }
 436    }
 437    return(FALSE);
 438 }

 441 void bus_ownership_release(int busno)
 442 {
 443    bus_event *tmp;
 444    struct bus *currbus = getbus(busno);
 445 /*
 446 fprintf (outputfile, "Bus ownership being released - %d\n", busno);
 447 */
 448    ASSERT(currbus->arbwinner == NULL);
 449    if (currbus->owners == NULL) {
 450 /*
 451 fprintf (outputfile, "Bus has become free - %d\n", busno);
 452 */
 453       currbus->state = BUS_FREE;
 454       currbus->lastowned = simtime;
 455    } else {
 456       tmp = bus_arbitrate_for_ownership(currbus);
 457       tmp->devtype = currbus->slots[tmp->slotno].devtype;
 458       tmp->time = simtime + currbus->arbtime;
 459 /*
 460 fprintf (outputfile, "Bus ownership transfered - devno %d, devtype %d, busno %d\n", tmp->devno, tmp->devtype, tmp->busno);
 461 */
 462       currbus->arbwinner = tmp;
 463       addtointq((event *) tmp);
 464    }
 465 }

 468 void bus_remove_from_arbitration(int busno, ioreq_event *curr)
 469 {
 470    bus_event *tmp;
 471    bus_event *trv;
 472    struct bus *currbus = getbus(busno);
 473 
 474    if (currbus->arbwinner) {
 475       if (curr == currbus->arbwinner->delayed_event) {
 476          if (removefromintq((event *)currbus->arbwinner) != TRUE) {
 477             fprintf(stderr, "Ongoing arbwinner not in internal queue, at bus_remove_from_arbitration\n");
 478             exit(1);
 479          }
 480          addtoextraq((event *) currbus->arbwinner);
 481          currbus->arbwinner = NULL;
 482          bus_ownership_release(busno);
 483          return;
 484       }
 485    }
 486    if (curr == currbus->owners->delayed_event) {
 487       tmp = currbus->owners;
 488       currbus->owners = tmp->next;
 489    } else {
 490       trv = currbus->owners;
 491       while ((trv->next) && (curr != trv->next->delayed_event)) {
 492          trv = trv->next;
 493       }
 494       ASSERT(trv->next != NULL);
 495       tmp = trv->next;
 496       trv->next = tmp->next;
 497    }
 498    addtoextraq((event *) tmp);
 499 }

 502 void bus_deliver_event(int busno, int slotno, ioreq_event *curr)
 503 {
 504    int devno;
 505    struct bus *currbus = getbus(busno);
 506 
 507    ASSERT2((slotno >= 0) && (slotno < currbus->numslots), 
 508            "slotno", slotno, "busno", busno);
 509 
 510 /*
 511 fprintf (outputfile, "In middle of bus_deliver_event\n");
 512 */
 513    devno = currbus->slots[slotno].devno;
 514    switch (currbus->slots[slotno].devtype) {
 515 
 516    case CONTROLLER:  
 517      controller_event_arrive(devno, curr);
 518      break;
 519 
 520    case DEVICE:      
 521      ASSERT(devno == curr->devno);
 522      device_event_arrive(curr);
 523      break;
 524 
 525    default:          
 526      fprintf(stderr, "Invalid device type in bus slot\n");
 527      assert(0);
 528    }
 529 }

 532 int bus_get_data_transfered(ioreq_event *curr, int depth)
 533 {
 534    intchar slotno;
 535    intchar busno;
 536    int checkdevno;
 537    int ret;
 538 /*
 539 fprintf (outputfile, "Entered bus_get_data_transfered - devno %d, depth %d, busno %x, slotno %x\n", curr->devno, depth, curr->busno, curr->slotno);
 540 */
 541    busno.value = curr->busno;
 542    slotno.value = curr->slotno;
 543    while (depth) {
 544       struct bus *currbus = getbus(busno.byte[depth]);
 545       checkdevno = currbus->slots[(slotno.byte[depth] >> 4)].devno;
 546       switch (currbus->slots[(slotno.byte[depth] >> 4)].devtype) {
 547    
 548          case CONTROLLER:       ret = controller_get_data_transfered(checkdevno, curr->devno);
 549                                 break;
 550    
 551          default:       fprintf(stderr, "Invalid device type in bus_get_data_transfered\n");
 552                         exit(1);
 553       }
 554       if (ret != -1) {
 555          return(ret);
 556       }
 557       depth--;
 558    }
 559    return(-1);
 560 }

 636          sprintf (prefix, "Bus #%d (%s) ", i, currbus->name);
 637          fprintf (outputfile, "Bus #%d\n", i);
 638          fprintf (outputfile, "Bus #%d Total utilization time: \t%.2f   \t%6.5f\n", i, (simtime - warmuptime - currbus->runidletime), ((simtime - warmuptime - currbus->runidletime) / (simtime - warmuptime)));
 639          if (disksim->businfo->bus_printidlestats) {
 640             stat_print (&currbus->busidlestats, prefix);
 641          }
 642          if (disksim->businfo->bus_printarbwaitstats) {
 643             fprintf (outputfile, "Bus #%d Number of arbitrations: \t%d\n", i, stat_get_count (&currbus->arbwaitstats));
 644             stat_print (&currbus->arbwaitstats, prefix);
 645          }
 646          fprintf (outputfile, "\n");
 647       }
 648    }
 649 }

 653 {
 654    int i;
 655 
 656    for (i=0; i<numbuses; i++) {
 657       struct bus *currbus = getbus(i);
 658       if (currbus->state == BUS_FREE) {
 659          currbus->runidletime += simtime - currbus->lastowned;
 660          stat_update (&currbus->busidlestats, (simtime - currbus->lastowned));

 662    }

 668   else return 0;

 671 /* bus constructor using new parser.  Returns null on failure
 672  * or a pointer to an allocated/initialized bus struct on success 
 673  * if parent is non-null, it will be copied and the new structure
 674  * initialized from it, i.e. you only need to provide parameters in 
 675  * b that you want to override from parent
 676  */
 677 bus *disksim_bus_loadparams(struct lp_block *b, 
 678                     int *num)
 679 {
 680   struct bus *result;
 681   int c;
 682 
 683   if(!disksim->businfo) {
 684     disksim->businfo = malloc(sizeof(businfo_t));
 685     bzero(disksim->businfo, sizeof(businfo_t));
 686   }
 687 
 688   //   disksim->businfo->bus_printidlestats = disksim_bus_printidlestats;
 689   //   disksim->businfo->bus_printarbwaitstats = disksim_bus_printarbwaitstats;
 690 
 691 
 692    for(c = 0; c < numbuses; c++) {
 693      if(!disksim->businfo->buses[c]) { break; } 
 694    }
 695    /* didn't find a free slot */
 696    if(c == disksim->businfo->buses_len) {
 697      int newlen = c ? 2*c : 2;
 698      
 699      disksim->businfo->buses = realloc(disksim->businfo->buses,
 700                                        newlen * sizeof(int*));
 701      if(newlen > 2)
 702        bzero(disksim->businfo->buses + c, (newlen/2)*sizeof(int *));
 703      else 
 704        bzero(disksim->businfo->buses, 2 * sizeof(int *));
 705 
 706      disksim->businfo->buses_len = newlen;
 707 
 708    }
 709    result = malloc(sizeof(struct bus));
 710    if(!result) { return 0; }
 711    numbuses++;
 712    
 713    disksim->businfo->buses[c] = result;
 714    bzero(result, sizeof(struct bus));
 715    if(num) *num = c;
 716 
 717    result->name = strdup(b->name);
 718 
 719 
 720    //#include "modules/disksim_bus_param.c"
 721   lp_loadparams(result, b, &disksim_bus_mod);
 722 
 723    result->owners = NULL;
 724    result->depth = -1;
 725    
 726    return result;
 727 }

 731 int load_bus_topo(struct lp_topospec *t, int *parentctlno) {
 732   int c, d;
 733   struct bus *b;
 734   struct lp_topospec *ts = 0;
 735   int busno;
 736   int slotnum = 0;
 737   int type;
 738 
 739   assert(!strcmp(t->type, disksim_mods[DISKSIM_MOD_BUS]->name));
 740 
 741   /* lookup bus */
 742 #ifdef WIN32
 743    /* XXX fixed by dnarayan; don't put side-effects in the assert arg, they might get macro'd out */
 744    /* original code: assert(b = getbusbyname(t->name, &busno)); */
 745    b = getbusbyname(t->name, &busno);
 746    assert(b);
 747 #else
 748    assert(b = getbusbyname(t->name, &busno));
 749 #endif
 750 
 751   b->numslots = 0;
 752   for(c = 0; c < t->l->values_len; c++)
 753     if(!t->l->values[c]) continue;
 754   /*      else if(t->l->values[c]->t != TOPOSPEC) b->numslots++; */
 755     else for(d = 0; d < t->l->values[c]->v.t.len; d++)
 756       if(t->l->values[c]->v.t.l[d].name || t->l->values[c]->v.t.l[d].l) b->numslots++;
 757 
 758   /* if this bus is the child of a controller */
 759   if(parentctlno) { b->numslots++; }
 760 
 761   b->slots = malloc(b->numslots * sizeof(slot));
 762   bzero(b->slots, b->numslots * sizeof(slot));
 763 
 764   if(parentctlno) {
 765     b->slots[0].devtype = CONTROLLER;
 766     b->slots[0].devno = *parentctlno;
 767     slotnum = 1;
 768   }
 769 
 770   for(c = 0; c < t->l->values_len; c++) {
 771     if(!t->l->values[c]) continue;
 772     /* t->l->values[c] should be a topospec */
 773     if(t->l->values[c]->t != TOPOSPEC) {
 774       fprintf(stderr, "*** error loading topology spec -- bad child device spec.\n");
 775       return 0;
 776     }
 777 
 778     for(d = 0; d < t->l->values[c]->v.t.len; d++) {
 779       ts = &t->l->values[c]->v.t.l[d];
 780       if(!ts->type) continue;
 781     }
 782  
 783     type = lp_mod_name(ts->type);
 784     if(type == lp_mod_name("disksim_ctlr")) {
 785       b->slots[slotnum].devtype = CONTROLLER;
 786         
 787       if(!getctlrbyname(ts->name, &b->slots[slotnum].devno)) {
 788         fprintf(stderr, "*** error: failed to load controller \"%s\" into slot %d of bus %s: no such controller \"%s\".\n", ts->name, c, t->name, ts->name);
 789         return 0;
 790       } 
 791         
 792       if(!load_ctlr_topo(ts, &busno)) {
 793         fprintf(stderr, "*** error: failed to load controller topology spec for bus %s slot %d\n", t->name, c);
 794         return 0;
 795       }
 796     }
 797     else if ((type == lp_mod_name("disksim_disk"))
 798         ||
 799         (type == lp_mod_name("disksim_simpledisk"))
 800         ||
 801         (type == lp_mod_name("memsmodel_mems"))
 802         || /*SSD:*/
 803         (type == lp_mod_name("ssdmodel_ssd"))
 804         )
 805       {
 806         b->slots[slotnum].devtype = DEVICE;
 807         
 808         if(!getdevbyname(ts->name, &b->slots[slotnum].devno, 0, 0)) {
 809           fprintf(stderr, "*** error: failed to load device \"%s\" into slot %d of bus %s: no such device \"%s\".\n",
 810                   ts->name, c, t->name, ts->name);
 811           return 0;
 812         }
 813       }
 814     else {
 815       fprintf(stderr, "Error loading topology spec -- slot must contain a device or a controller (not %s).\n", ts->type);
 816       return 0;
 817       break;
 818     }
 819     /* eventually want to add pointer to device here to avoid table
 820      * lookups elsewhere */
 821       
 822     slotnum++;
 823       
 824   }
 825   
 826   return 1;
 827   
 828 }
