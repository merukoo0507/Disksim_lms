 126 int simpledisk_set_depth (int devno, int inbusno, int depth, int slotno)
 127 {
 128    simpledisk_t *currdisk;
 129    int cnt;
 130 
 131    currdisk = getsimpledisk (devno);
 132    assert(currdisk);
 133    cnt = currdisk->numinbuses;
 134    currdisk->numinbuses++;
 135    if ((cnt + 1) > MAXINBUSES) {
 136       fprintf(stderr, "Too many inbuses specified for simpledisk %d - %d\n", devno, (cnt+1));
 137       exit(1);
 138    }
 139    currdisk->inbuses[cnt] = inbusno;
 140    currdisk->depth[cnt] = depth;
 141    currdisk->slotno[cnt] = slotno;
 142    return(0);
 143 }

 205 /*
 206  * simpledisk_send_event_up_path()
 207  *
 208  * Acquires the bus (if not already acquired), then uses bus_delay to
 209  * send the event up the path.
 210  *
 211  * If the bus is already owned by this device or can be acquired
 212  * immediately (interleaved bus), the event is sent immediately.
 213  * Otherwise, disk_bus_ownership_grant will later send the event.
 214  */
 215   
 216 static void simpledisk_send_event_up_path (ioreq_event *curr, double delay)
 217 {
 218    simpledisk_t *currdisk;
 219    int busno;
 220    int slotno;
 221 
 222    // fprintf (outputfile, "simpledisk_send_event_up_path - devno %d, type %d, cause %d, blkno %d\n", curr->devno, curr->type, curr->cause, curr->blkno);
 223 
 224    currdisk = getsimpledisk (curr->devno);
 225 
 226    busno = simpledisk_get_busno(curr);
 227    slotno = currdisk->slotno[0];
 228 
 229    /* Put new request at head of buswait queue */
 230    curr->next = currdisk->buswait;
 231    currdisk->buswait = curr;
 232 
 233    curr->tempint1 = busno;
 234    curr->time = delay;
 235    if (currdisk->busowned == -1) {
 236 
 237       // fprintf (outputfile, "Must get ownership of the bus first\n");
 238 
 239       if (curr->next) {
 240          //fprintf(stderr,"Multiple bus requestors detected in disk_send_event_up_path\n");
 241          /* This should be ok -- counting on the bus module to sequence 'em */
 242       }
 243       if (bus_ownership_get(busno, slotno, curr) == FALSE) {
 244          /* Remember when we started waiting (only place this is written) */
 245          currdisk->stat.requestedbus = simtime;
 246       } else {
 247          bus_delay(busno, DEVICE, curr->devno, delay, curr); /* Never for SCSI */
 248       }
 249    } else if (currdisk->busowned == busno) {
 250 
 251       //fprintf (outputfile, "Already own bus - so send it on up\n");
 252 
 253       bus_delay(busno, DEVICE, curr->devno, delay, curr);
 254    } else {
 255       fprintf(stderr, "Wrong bus owned for transfer desired\n");
 256       exit(1);
 257    }
 258 }

 261 /*
 262   **-simpledisk_bus_ownership_grant
 263 
 264   Calls bus_delay to handle the event that the disk has been granted the bus.  I believe
 265   this is always initiated by a call to disk_send_even_up_path.
 266 
 267   */
 268 
 269 void simpledisk_bus_ownership_grant (int devno, ioreq_event *curr, int busno, double arbdelay)
 270 {
 271    simpledisk_t *currdisk;
 272    ioreq_event *tmp;
 273 
 274    currdisk = getsimpledisk (devno);
 275 
 276    tmp = currdisk->buswait;
 277    while ((tmp != NULL) && (tmp != curr)) {
 278       tmp = tmp->next;
 279    }
 280    if (tmp == NULL) {
 281       fprintf(stderr, "Bus ownership granted to unknown simpledisk request - devno %d, busno %d\n", devno, busno);
 282       exit(1);
 283    }
 284    currdisk->busowned = busno;
 285    currdisk->stat.waitingforbus += arbdelay;
 286    //ASSERT (arbdelay == (simtime - currdisk->stat.requestedbus));
 287    currdisk->stat.numbuswaits++;
 288    bus_delay(busno, DEVICE, devno, tmp->time, tmp);
 289 }

 292 void simpledisk_bus_delay_complete (int devno, ioreq_event *curr, int sentbusno)
 293 {
 294    simpledisk_t *currdisk;
 295    intchar slotno;
 296    intchar busno;
 297    int depth;
 298 
 299    currdisk = getsimpledisk (devno);
 300 
 301    // fprintf (outputfile, "Entered disk_bus_delay_complete\n");
 302 
 303    if (curr == currdisk->buswait) {
 304       currdisk->buswait = curr->next;
 305    } else {
 306       ioreq_event *tmp = currdisk->buswait;
 307       while ((tmp->next != NULL) && (tmp->next != curr)) {
 308          tmp = tmp->next;
 309       }
 310       if (tmp->next != curr) {
 311          fprintf(stderr, "Bus delay complete for unknown simpledisk request - devno %d, busno %d\n", devno, busno.value);
 312          exit(1);
 313       }
 314       tmp->next = curr->next;
 315    }
 316    busno.value = curr->busno;
 317    slotno.value = curr->slotno;
 318    depth = currdisk->depth[0];
 319    slotno.byte[depth] = slotno.byte[depth] >> 4;
 320    curr->time = 0.0;
 321    if (depth == 0) {
 322       intr_request ((event *)curr);
 323    } else {
 324       bus_deliver_event(busno.byte[depth], slotno.byte[depth], curr);
 325    }
 326 }

 331 static void simpledisk_request_complete(ioreq_event *curr)
 332 {
 333    simpledisk_t *currdisk;
 334 
 335    // fprintf (outputfile, "Entering simpledisk_request_complete: %12.6f\n", simtime);
 336 
 337    currdisk = getsimpledisk (curr->devno);
 338 
 339    if ((curr = ioqueue_physical_access_done(currdisk->queue,curr)) == NULL) {
 340       fprintf(stderr, "simpledisk_request_complete:  ioreq_event not found by ioqueue_physical_access_done call\n");
 341       exit(1);
 342    }
 343 
 344    /* send completion interrupt */
 345    curr->type = IO_INTERRUPT_ARRIVE;
 346    curr->cause = COMPLETION;
 347    simpledisk_send_event_up_path(curr, currdisk->bus_transaction_latency);
 348 }

 351 static void simpledisk_bustransfer_complete (ioreq_event *curr)
 352 {
 353    simpledisk_t *currdisk;
 354 
 355    // fprintf (outputfile, "Entering simpledisk_bustransfer_complete for disk %d: %12.6f\n", curr->devno, simtime);
 356 
 357    currdisk = getsimpledisk (curr->devno);
 358 
 359    if (curr->flags & READ) {
 360       simpledisk_request_complete (curr);
 361 
 362    } else {
 363       simpledisk_t *currdisk = getsimpledisk (curr->devno);
 364 
 365       if (currdisk->neverdisconnect == FALSE) {
 366          /* disconnect from bus */
 367          ioreq_event *tmp = ioreq_copy (curr);
 368          tmp->type = IO_INTERRUPT_ARRIVE;
 369          tmp->cause = DISCONNECT;
 370          simpledisk_send_event_up_path (tmp, currdisk->bus_transaction_latency);
 371       }
 372 
 373       /* do media access */
 374       currdisk->media_busy = TRUE;
 375       stat_update (&currdisk->stat.acctimestats, currdisk->acctime);
 376       curr->time = simtime + currdisk->acctime;
 377       curr->type = DEVICE_ACCESS_COMPLETE;
 378       addtointq ((event *) curr);
 379    }
 380 }

 383 static void simpledisk_reconnect_done (ioreq_event *curr)
 384 {
 385    simpledisk_t *currdisk;
 386 
 387    // fprintf (outputfile, "Entering simpledisk_reconnect_done for disk %d: %12.6f\n", curr->devno, simtime);
 388 
 389    currdisk = getsimpledisk (curr->devno);
 390 
 391    if (curr->flags & READ) {
 392       if (currdisk->neverdisconnect) {
 393          /* Just holding on to bus; data transfer will be initiated when */
 394          /* media access is complete.                                    */
 395          addtoextraq((event *) curr);
 396 
 397       } else {
 398          /* data transfer: curr->bcount, which is still set to original */
 399          /* requested value, indicates how many blks to transfer.       */
 400          curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 401          simpledisk_send_event_up_path(curr, (double) 0.0);
 402       }
 403 
 404    } else {
 405       if (currdisk->reconnect_reason == DEVICE_ACCESS_COMPLETE) {
 406          simpledisk_request_complete (curr);
 407 
 408       } else {
 409          /* data transfer: curr->bcount, which is still set to original */
 410          /* requested value, indicates how many blks to transfer.       */
 411          curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 412          simpledisk_send_event_up_path(curr, (double) 0.0);
 413       }
 414    }
 415 }

 418 static void simpledisk_request_arrive (ioreq_event *curr)
 419 {
 420    ioreq_event *intrp;
 421    simpledisk_t *currdisk;
 422 
 423    // fprintf (outputfile, "Entering simpledisk_request_arrive: %12.6f\n", simtime);
 424    // fprintf (outputfile, "simpledisk = %d, blkno = %d, bcount = %d, read = %d\n",curr->devno, curr->blkno, curr->bcount, (READ & curr->flags));
 425 
 426    currdisk = getsimpledisk(curr->devno);
 427 
 428    /* verify that request is valid. */
 429    if ((curr->blkno < 0) || (curr->bcount <= 0) ||
 430        ((curr->blkno + curr->bcount) > currdisk->numblocks)) {
 431       fprintf(stderr, "Invalid set of blocks requested from simpledisk - blkno %d, bcount %d, numblocks %d\n", curr->blkno, curr->bcount, currdisk->numblocks);
 432       exit(1);
 433    }
 434 
 435    /* create a new request, set it up for initial interrupt */
 436    currdisk->busowned = simpledisk_get_busno(curr);
 437 
 438    if (ioqueue_get_reqoutstanding (currdisk->queue) == 0) {
 439       ioqueue_add_new_request(currdisk->queue, curr);
 440       curr = ioqueue_get_next_request (currdisk->queue);
 441       intrp = curr;
 442 
 443       /* initiate media access if request is a READ */
 444       if (curr->flags & READ) {
 445          ioreq_event *tmp = ioreq_copy (curr);
 446          currdisk->media_busy = TRUE;
 447          stat_update (&currdisk->stat.acctimestats, currdisk->acctime);
 448          tmp->time = simtime + currdisk->acctime;
 449          tmp->type = DEVICE_ACCESS_COMPLETE;
 450          addtointq ((event *)tmp);
 451       }
 452 
 453       /* if not disconnecting, then the READY_TO_TRANSFER is like a RECONNECT */
 454       currdisk->reconnect_reason = IO_INTERRUPT_ARRIVE;
 455       if (curr->flags & READ) {
 456          intrp->cause = (currdisk->neverdisconnect) ? READY_TO_TRANSFER : DISCONNECT;
 457       } else {
 458          intrp->cause = READY_TO_TRANSFER;
 459       }
 460 
 461    } else {
 462       intrp = ioreq_copy(curr);
 463       ioqueue_add_new_request(currdisk->queue, curr);
 464       intrp->cause = DISCONNECT;
 465    }
 466 
 467    intrp->type = IO_INTERRUPT_ARRIVE;
 468    simpledisk_send_event_up_path(intrp, currdisk->bus_transaction_latency);
 469 }

 472 static void simpledisk_access_complete (ioreq_event *curr)
 473 {
 474    simpledisk_t *currdisk;
 475 
 476    // fprintf (outputfile, "Entering simpledisk_access_complete: %12.6f\n", simtime);
 477 
 478    currdisk = getsimpledisk (curr->devno);
 479    currdisk->media_busy = FALSE;
 480 
 481    if (currdisk->neverdisconnect) {
 482       /* already connected */
 483       if (curr->flags & READ) {
 484          /* transfer data up the line: curr->bcount, which is still set to */
 485          /* original requested value, indicates how many blks to transfer. */
 486          curr->type = DEVICE_DATA_TRANSFER_COMPLETE;
 487          simpledisk_send_event_up_path(curr, (double) 0.0);
 488 
 489       } else {
 490          simpledisk_request_complete (curr);
 491       }
 492 
 493    } else {
 494       /* reconnect to controller */
 495       curr->type = IO_INTERRUPT_ARRIVE;
 496       curr->cause = RECONNECT;
 497       simpledisk_send_event_up_path (curr, currdisk->bus_transaction_latency);
 498       currdisk->reconnect_reason = DEVICE_ACCESS_COMPLETE;
 499    }
 500 }

 522 /* completion disconnect done */
 523 
 524 static void simpledisk_completion_done (ioreq_event *curr)
 525 {
 526    simpledisk_t *currdisk = getsimpledisk (curr->devno);
 527 
 528    // fprintf (outputfile, "Entering simpledisk_completion for disk %d: %12.6f\n", currdisk->devno, simtime);
 529 
 530    addtoextraq((event *) curr);
 531 
 532    if (currdisk->busowned != -1) {
 533       bus_ownership_release(currdisk->busowned);
 534       currdisk->busowned = -1;
 535    }
 536 
 537    /* check for and start next queued request, if any */
 538    curr = ioqueue_get_next_request(currdisk->queue);
 539    if (curr != NULL) {
 540       ASSERT (currdisk->media_busy == FALSE);
 541       if (curr->flags & READ) {
 542          currdisk->media_busy = TRUE;
 543          stat_update (&currdisk->stat.acctimestats, currdisk->acctime);
 544          curr->time = simtime + currdisk->acctime;
 545          curr->type = DEVICE_ACCESS_COMPLETE;
 546          addtointq ((event *)curr);
 547 
 548       } else {
 549          curr->type = IO_INTERRUPT_ARRIVE;
 550          curr->cause = RECONNECT;
 551          simpledisk_send_event_up_path (curr, currdisk->bus_transaction_latency);
 552          currdisk->reconnect_reason = IO_INTERRUPT_ARRIVE;
 553       }
 554    }
 555 }

 558 static void simpledisk_interrupt_complete (ioreq_event *curr)
 559 {
 560    // fprintf (outputfile, "Entered simpledisk_interrupt_complete - cause %d\n", curr->cause);
 561 
 562    switch (curr->cause) {
 563 
 564       case RECONNECT:
 565          simpledisk_reconnect_done(curr);
 566          break;
 567 
 568       case DISCONNECT:
 569          simpledisk_disconnect_done(curr);
 570          break;
 571 
 572       case COMPLETION:
 573          simpledisk_completion_done(curr);
 574          break;
 575 
 576       default:
 577          ddbg_assert2(0, "bad event type");
 578    }
 579 }

 585 
 586    // fprintf (outputfile, "Entered simpledisk_event_arrive: time %f (simtime %f)\n", curr->time, simtime);
 587    // fprintf (outputfile, " - devno %d, blkno %d, type %d, cause %d, read = %d\n", curr->devno, curr->blkno, curr->type, curr->cause, curr->flags & READ);
 588 
 589    currdisk = getsimpledisk (curr->devno);
 590 
 591    switch (curr->type) {
 592 
 593       case IO_ACCESS_ARRIVE:
 594          curr->time = simtime + currdisk->overhead;
 595          curr->type = DEVICE_OVERHEAD_COMPLETE;
 596          addtointq((event *) curr);
 597          break;
 598 
 599       case DEVICE_OVERHEAD_COMPLETE:
 600          simpledisk_request_arrive(curr);
 601          break;
 602 
 603       case DEVICE_ACCESS_COMPLETE:
 604          simpledisk_access_complete (curr);
 605          break;
 606 
 607       case DEVICE_DATA_TRANSFER_COMPLETE:
 608          simpledisk_bustransfer_complete(curr);
 609          break;
 610 
 611       case IO_INTERRUPT_COMPLETE:
 612          simpledisk_interrupt_complete(curr);
 613          break;
 614 
 615       case IO_QLEN_MAXCHECK:
 616          /* Used only at initialization time to set up queue stuff */
 617          curr->tempint1 = -1;
 618          curr->tempint2 = simpledisk_get_maxoutstanding(curr->devno);
 619          curr->bcount = 0;
 620          break;
 621 
 622       default:
 623          fprintf(stderr, "Unrecognized event type at simpledisk_event_arrive\n");
 624          exit(1);
 625    }
 626 
 627    // fprintf (outputfile, "Exiting simpledisk_event_arrive\n");
 628 }

 795 int simpledisk_add(struct simpledisk *d) {
 796   int c;
 797 
 798   if(!disksim->simplediskinfo) simpledisk_initialize_diskinfo();
 799   
 800   for(c = 0; c < disksim->simplediskinfo->simpledisks_len; c++) {
 801     if(!disksim->simplediskinfo->simpledisks[c]) {
 802       disksim->simplediskinfo->simpledisks[c] = d;
 803       numsimpledisks++;
 804       return c;
 805     }
 806   }
 807 
 808   /* note that numdisks must be equal to diskinfo->disks_len */
 809   disksim->simplediskinfo->simpledisks = 
 810     realloc(disksim->simplediskinfo->simpledisks, 
 811             2 * c * sizeof(struct simpledisk *));
 812 
 813   bzero(disksim->simplediskinfo->simpledisks + numsimpledisks, 
 814         numsimpledisks);
 815 
 816   disksim->simplediskinfo->simpledisks[c] = d;
 817   numsimpledisks++;
 818   disksim->simplediskinfo->simpledisks_len *= 2;
 819   return c;
 820 }

 921 void simpledisk_printstats (void)
 922 {
 923    struct ioq * queueset[MAXDEVICES];
 924    int set[MAXDEVICES];
 925    int i;
 926    int reqcnt = 0;
 927    char prefix[80];
 928    int diskcnt;
 929 
 930    fprintf(outputfile, "\nSIMPLEDISK STATISTICS\n");
 931    fprintf(outputfile, "---------------------\n\n");
 932 
 933    sprintf(prefix, "Simpledisk ");
 934 
 935    diskcnt = 0;
 936    for (i=0; i<MAXDEVICES; i++) {
 937       simpledisk_t *currdisk = getsimpledisk (i);
 938       if (currdisk) {
 939          queueset[diskcnt] = currdisk->queue;
 940          reqcnt += ioqueue_get_number_of_requests(currdisk->queue);
 941          diskcnt++;
 942       }
 943    }
 944    assert (diskcnt == numsimpledisks);
 945 
 946    if (reqcnt == 0) {
 947       fprintf(outputfile, "No simpledisk requests encountered\n");
 948       return;
 949    }
 950 
 951    ioqueue_printstats(queueset, numsimpledisks, prefix);
 952 
 953    diskcnt = 0;
 954    for (i=0; i<MAXDEVICES; i++) {
 955       simpledisk_t *currdisk = getsimpledisk (i);
 956       if (currdisk) {
 957          set[diskcnt] = i;
 958          diskcnt++;
 959       }
 960    }
 961    assert (diskcnt == numsimpledisks);
 962 
 963    simpledisk_acctime_printstats(set, numsimpledisks, prefix);
 964    simpledisk_other_printstats(set, numsimpledisks, prefix);
 965    fprintf (outputfile, "\n\n");
 966 
 967    if (numsimpledisks <= 1) {
 968       return;
 969    }
 970 
 971    for (i=0; i<numsimpledisks; i++) {
 972       simpledisk_t *currdisk = getsimpledisk (set[i]);
 973       if (currdisk->printstats == FALSE) {
 974          continue;
 975       }
 976       if (ioqueue_get_number_of_requests(currdisk->queue) == 0) {
 977          fprintf(outputfile, "No requests for simpledisk #%d\n\n\n", set[i]);
 978          continue;
 979       }
 980       fprintf(outputfile, "Simpledisk #%d:\n\n", set[i]);
 981       sprintf(prefix, "Simpledisk #%d ", set[i]);
 982       ioqueue_printstats(&currdisk->queue, 1, prefix);
 983       simpledisk_acctime_printstats(&set[i], 1, prefix);
 984       simpledisk_other_printstats(&set[i], 1, prefix);
 985       fprintf (outputfile, "\n\n");
 986    }
 987 }
