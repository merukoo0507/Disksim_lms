 121 struct iodriver *getiodriverbyname(char *name, int *num) {
 122   int c;
 123   if(!disksim->iodriver_info) return 0;
 124 
 125   for(c = 0; c < numiodrivers; c++) {
 126     if(iodrivers[c])
 127       if(!strcmp(iodrivers[c]->name, name)) {
 128         if(num) *num = c;
 129         return iodrivers[c];
 130       }
 131   }
 132 
 133   return 0;
 134 }

 159 static void schedule_disk_access (iodriver *curriodriver, ioreq_event *curr)
 160 {
 161    int queuectlr = curriodriver->devices[(curr->devno)].queuectlr;
 162    if (queuectlr != -1) {
 163       curriodriver->ctlrs[queuectlr].numoutstanding++;
 164    }
 165    curriodriver->devices[(curr->devno)].busy = TRUE;
 166 }

 169 static int check_send_out_request (iodriver *curriodriver, int devno)
 170 {
 171    int numout;
 172 
 173    if ((curriodriver->consttime == IODRIVER_TRACED_QUEUE_TIMES) || (curriodriver->consttime == IODRIVER_TRACED_BOTH_TIMES)) {
 174       return(FALSE);
 175    }
 176    numout = ioqueue_get_reqoutstanding(curriodriver->devices[devno].queue);
 177    if (curriodriver->usequeue == TRUE) {
 178       int queuectlr = curriodriver->devices[devno].queuectlr;
 179       if (queuectlr != -1) {
 180          numout = curriodriver->ctlrs[queuectlr].numoutstanding;
 181 /*
 182 fprintf (outputfile, "Check send_out_req: queuectlr %d, numout %d, maxout %d, send %d\n", queuectlr, numout, curriodriver->ctlrs[queuectlr].maxoutstanding, (numout < curriodriver->ctlrs[queuectlr].maxoutstanding));
 183 */
 184          return(numout < curriodriver->ctlrs[queuectlr].maxoutstanding);
 185       } else {
 186          return(numout < curriodriver->devices[devno].maxoutstanding);
 187       }
 188    } else {
 189       return(!numout);
 190    }
 191 }

 194 static ioreq_event * handle_new_request (iodriver *curriodriver, ioreq_event *curr)
 195 {
 196    struct ioq *queue = curriodriver->devices[(curr->devno)].queue;
 197    ioreq_event *ret = NULL;
 198 
 199    /*
 200    fprintf(outputfile, "\n***  handle_new_request::  time %f, devno %d, blkno %d, bcount %d\n\n",
 201            simtime, curr->devno, curr->blkno, curr->bcount);
 202 
 203    fprintf(outputfile, "handle_new_request::  calling ioqueue_add_new_request\n");
 204    */
 205 
 206    ioqueue_add_new_request(queue, curr);
 207    if (check_send_out_request(curriodriver, curr->devno)) {
 208      /*
 209      fprintf(outputfile, "handle_new_request::  calling ioqueue_get_next_request\n");
 210      */
 211       ret = ioqueue_get_next_request(queue);
 212       if (ret != NULL) {
 213          schedule_disk_access(curriodriver, ret);
 214          ret->time = IODRIVER_IMMEDSCHED_TIME * curriodriver->scale;
 215       }
 216    }
 217    return(ret);
 218 }

 221 void iodriver_schedule (int iodriverno, ioreq_event *curr)
 222 {
 223    ctlr *ctl;
 224 /*
 225 fprintf (outputfile, "Entered iodriver_schedule - devno %d, blkno %d, bcount %d, read %d\n", curr->devno, curr->blkno, curr->bcount, (curr->flags & READ));
 226 */
 227    ASSERT1(curr->type == IO_ACCESS_ARRIVE, "curr->type", curr->type);
 228 
 229    if ((iodrivers[iodriverno]->consttime != 0.0) 
 230        && (iodrivers[iodriverno]->consttime != IODRIVER_TRACED_QUEUE_TIMES)) 
 231    {
 232       curr->type = IO_INTERRUPT;
 233       if (iodrivers[iodriverno]->consttime > 0.0) {
 234          curr->time = iodrivers[iodriverno]->consttime;
 235       } 
 236       else {
 237          curr->time = ((double) curr->tempint2 / (double) 1000);
 238       }
 239       curr->cause = COMPLETION;
 240       intr_request((event *) curr);
 241       return;
 242    }
 243 
 244    ctl = iodrivers[iodriverno]->devices[(curr->devno)].ctl;
 245 
 246    if ((ctl) && (ctl->flags & DRIVER_C700)) {
 247       if ((ctl->pendio) 
 248           && ((curr->devno != ctl->pendio->next->devno) 
 249               || (curr->opid != ctl->pendio->next->opid) 
 250               || (curr->blkno != ctl->pendio->next->blkno))) 
 251       {
 252          curr->next = ctl->pendio->next;
 253          ctl->pendio->next = curr;
 254          ctl->pendio = curr;
 255          return;
 256       } 
 257       else if (ctl->pendio == NULL) {
 258          ctl->pendio = ioreq_copy(curr);
 259          ctl->pendio->next = ctl->pendio;
 260       }
 261       if (ctl->flags & DRIVER_CTLR_BUSY) {
 262          addtoextraq((event *) curr);
 263          return;
 264       }
 265    }
 266    curr->busno = iodrivers[iodriverno]->devices[(curr->devno)].buspath.value;
 267    curr->slotno = iodrivers[iodriverno]->devices[(curr->devno)].slotpath.value;
 268    if (iodrivers[iodriverno]->devices[(curr->devno)].queuectlr != -1) {
 269       int ctlrno = iodrivers[iodriverno]->devices[(curr->devno)].queuectlr;
 270       ctl = &iodrivers[iodriverno]->ctlrs[ctlrno];
 271       if ((ctl->maxreqsize) && (curr->bcount > ctl->maxreqsize)) {
 272          ioreq_event *totalreq = ioreq_copy(curr);
 273 /*
 274 fprintf (outputfile, "%f, oversized request: opid %d, blkno %d, bcount %d, maxreqsize %d\n", simtime, curr->opid, curr->blkno, curr->bcount, ctl->maxreqsize);
 275 */
 276          curr->bcount = ctl->maxreqsize;
 277          if (ctl->oversized) {
 278             totalreq->next = ctl->oversized->next;
 279             ctl->oversized->next = totalreq;
 280          } else {
 281             totalreq->next = totalreq;
 282             ctl->oversized = totalreq;
 283          }
 284       }
 285    }
 286    iodriver_send_event_down_path(curr);
 287 /*
 288 fprintf (outputfile, "Leaving iodriver_schedule\n");
 289 */
 290 }

 298 static void iodriver_check_c700_based_status (iodriver *curriodriver, int devno, int cause, int type, int blkno)
 299 {
 300    ctlr *ctl;
 301    ioreq_event *tmp;
 302 
 303    ctl = curriodriver->devices[devno].ctl;
 304    if ((ctl == NULL) || (!(ctl->flags & DRIVER_C700))) {
 305       return;
 306    }
 307    if (type == IO_INTERRUPT_ARRIVE) {
 308       if (ctl->pendio != NULL) {
 309          tmp = ctl->pendio->next;
 310          if ((tmp->devno == devno) && (tmp->blkno == blkno)) {
 311             if (ctl->pendio == tmp) {
 312                ctl->pendio = NULL;
 313             } else {
 314                ctl->pendio->next = tmp->next;
 315             }
 316             addtoextraq((event *) tmp);
 317          }
 318       }
 319       switch (cause) {
 320          case COMPLETION:
 321          case DISCONNECT:
 322          case RECONNECT:
 323          case READY_TO_TRANSFER:
 324                  ctl->flags |= DRIVER_CTLR_BUSY;
 325                  break;
 326          default:
 327                  fprintf(stderr, "Unknown interrupt cause at iodriver_check_c700_based_status - cause %d\n", cause);
 328                  exit(1);
 329       }
 330    } else {
 331       switch (cause) {
 332          case COMPLETION:
 333          case DISCONNECT:
 334                            ctl->flags &= ~DRIVER_CTLR_BUSY;
 335                            if (ctl->pendio != NULL) {
 336                               tmp = ioreq_copy(ctl->pendio->next);
 337                               tmp->time = simtime;
 338                               addtointq((event *) tmp);
 339                            }
 340                            break;
 341          case RECONNECT:
 342          case READY_TO_TRANSFER:
 343                            ctl->flags |= DRIVER_CTLR_BUSY;
 344                            break;
 345          default:
 346                            fprintf(stderr, "Unknown interrupt cause at iodriver_check_c700_based_status - cause %d\n", cause);
 347                            exit(1);
 348       }
 349    }
 350 }

 355 static void iodriver_add_to_intrp_eventlist (intr_event *intrp, event *curr, double scale)
 356 {
 357    if (curr) {
 358       event *tmp = intrp->eventlist;
 359       int lasttype = curr->type;
 360       intrp->eventlist = curr;
 361       if (curr->type == IO_ACCESS_ARRIVE) {
 362          curr->time = IODRIVER_COMPLETION_RESPTOSCHED_TIME;
 363       } else if (curr->type == IO_REQUEST_ARRIVE) {
 364          curr->time = IODRIVER_COMPLETION_RESPTOREQ_TIME;
 365       } else {       /* Assume wakeup */
 366          curr->time = IODRIVER_COMPLETION_RESPTOWAKEUP_TIME;
 367       }
 368       curr->time *= scale;
 369       while (curr->next) {
 370          curr = curr->next;
 371          curr->time = 0.0;
 372       }
 373       curr->next = tmp;
 374       if (lasttype != IO_REQUEST_ARRIVE) {
 375          while (tmp) {
 376             if (tmp->type == INTEND_EVENT) {
 377                ASSERT(tmp->next == NULL);
 378                tmp->time = (lasttype == IO_ACCESS_ARRIVE) ? IODRIVER_COMPLETION_SCHEDTOEND_TIME : IODRIVER_COMPLETION_WAKEUPTOEND_TIME;
 379                tmp->time *= scale;
 380             } else if (tmp->type != IO_REQUEST_ARRIVE) {    /* assume wakeup */
 381                tmp->time = (lasttype == IO_ACCESS_ARRIVE) ? IODRIVER_COMPLETION_SCHEDTOWAKEUP_TIME : 0.0;
 382                lasttype = tmp->type;
 383                tmp->time *= scale;
 384             }
 385             tmp = tmp->next;
 386          }
 387       }
 388    }
 389 }

 392 void iodriver_access_complete (int iodriverno, intr_event *intrp)
 393 {
 394    int i;
 395    int numreqs;
 396    ioreq_event *tmp;
 397    ioreq_event *del;
 398    ioreq_event *req;
 399    int devno;
 400    int skip = 0;
 401    ctlr *ctl = NULL;
 402 
 403    if (iodrivers[iodriverno]->type == STANDALONE) {
 404       req = ioreq_copy((ioreq_event *) intrp->infoptr);
 405    } 
 406    else {
 407       req = (ioreq_event *) intrp->infoptr;
 408    }
 409 
 410    
 411    disksim_exectrace("Request completion: simtime %f, devno %d, blkno %d, time %f\n", 
 412                      simtime, req->devno, req->blkno);
 413 
 414    if (iodrivers[iodriverno]->devices[(req->devno)].queuectlr != -1) 
 415    {
 416       int ctlrno = iodrivers[iodriverno]->devices[(req->devno)].queuectlr;
 417       ctl = &iodrivers[iodriverno]->ctlrs[ctlrno];
 418       tmp = ctl->oversized;
 419       numreqs = 1;
 420 
 421 
 422       while (((numreqs) || (tmp != ctl->oversized)) 
 423              && (tmp) 
 424              && (tmp->next) 
 425              && ((tmp->next->devno != req->devno) 
 426                  || (tmp->next->opid != req->opid) 
 427                  || (req->blkno < tmp->next->blkno) 
 428                  || (req->blkno >= (tmp->next->blkno + tmp->next->bcount)))) 
 429       {
 430         
 431         // fprintf (outputfile, "oversized request in list: opid %d, blkno %d, bcount %d\n", tmp->opid, tmp->blkno, tmp->bcount);
 432         
 433         numreqs = 0;
 434         tmp = tmp->next;
 435       }
 436 
 437 
 438       if ((tmp) 
 439           && (tmp->next->devno == req->devno) 
 440           && (tmp->next->opid == req->opid) 
 441           && (req->blkno >= tmp->next->blkno) 
 442           && (req->blkno < (tmp->next->blkno + tmp->next->bcount))) 
 443       {
 444         fprintf (outputfile, "%f, part of oversized request completed: opid %d, blkno %d, bcount %d, maxreqsize %d\n", simtime, req->opid, req->blkno, req->bcount, ctl->maxreqsize);
 445 
 446         if ((req->blkno + ctl->maxreqsize) < (tmp->next->blkno + tmp->next->bcount)) 
 447         {
 448           fprintf (outputfile, "more to go\n");
 449           req->blkno += ctl->maxreqsize;
 450           req->bcount = min(ctl->maxreqsize, 
 451                             (tmp->next->blkno + tmp->next->bcount - req->blkno));
 452           goto schedule_next;
 453         } 
 454         else {
 455           fprintf (outputfile, "done for real\n");
 456 
 457           addtoextraq((event *) req);
 458           req = tmp->next;
 459           tmp->next = tmp->next->next;
 460 
 461           if (ctl->oversized == req) {
 462             ctl->oversized = (req != req->next) ? req->next : NULL;
 463           }
 464 
 465           req->next = NULL;
 466         }
 467       }
 468    }
 469 
 470 
 471 
 472 
 473    devno = req->devno;
 474    req = ioqueue_physical_access_done(iodrivers[iodriverno]->devices[devno].queue, req);
 475 
 476    if (ctl) {
 477       ctl->numoutstanding--;
 478    }
 479 
 480    // special case for validate:
 481    if (disksim->traceformat == VALIDATE) {
 482       tmp = (ioreq_event *) getfromextraq();
 483       io_validate_do_stats1();
 484       tmp = iotrace_validate_get_ioreq_event(disksim->iotracefile, tmp);
 485       if (tmp) {
 486          io_validate_do_stats2(tmp);
 487          tmp->type = IO_REQUEST_ARRIVE;
 488          addtointq((event *) tmp);
 489 
 490          disksim_exectrace("Request issue: simtime %f, devno %d, blkno %d, time %f\n", 
 491                            simtime, tmp->devno, tmp->blkno, tmp->time);
 492       } 
 493       else {
 494          disksim_simstop();
 495       }
 496   } 
 497    else if (disksim->closedios) {
 498       tmp = (ioreq_event *) io_get_next_external_event(disksim->iotracefile);
 499       if (tmp) {
 500          io_using_external_event ((event *)tmp);
 501          tmp->time = simtime + disksim->closedthinktime;
 502          tmp->type = IO_REQUEST_ARRIVE;
 503          addtointq((event *) tmp);
 504       } else {
 505          disksim_simstop();
 506       }
 507    }
 508 
 509  
 510 
 511    while (req) {
 512       tmp = req;
 513       req = req->next;
 514       tmp->next = NULL;
 515       update_iodriver_statistics();
 516       if ((numreqs = logorg_mapcomplete(sysorgs, numsysorgs, tmp)) == COMPLETE) {
 517 
 518         /* update up overall I/O system stats for this completed request */
 519         ioreq_event *temp = ioqueue_get_specific_request (overallqueue, tmp);
 520         ioreq_event *temp2 = ioqueue_physical_access_done (overallqueue, temp);
 521         ASSERT (temp2 != NULL);
 522         addtoextraq((event *)temp);
 523         temp = NULL;
 524         
 525         if (iodrivers[iodriverno]->type != STANDALONE) {
 526           iodriver_add_to_intrp_eventlist(intrp, 
 527                                           io_done_notify(tmp), 
 528                                           iodrivers[iodriverno]->scale);
 529          } else {
 530             io_done_notify (tmp);
 531          }
 532       } 
 533       else if (numreqs > 0) {
 534         for (i = 0; i < numreqs; i++) {
 535           del = tmp->next;
 536           tmp->next = del->next;
 537           del->next = NULL;
 538           del->type = IO_REQUEST_ARRIVE;
 539           del->flags |= MAPPED;
 540           skip |= (del->devno == devno);
 541 
 542           if (iodrivers[iodriverno]->type == STANDALONE) 
 543           {
 544             del->time += simtime + 0.0000000001; /* to affect an ordering */
 545             addtointq((event *) del);
 546           } 
 547           else {
 548             iodriver_add_to_intrp_eventlist(intrp, 
 549                                             (event *) del, 
 550                                             iodrivers[iodriverno]->scale);
 551           }
 552         }
 553       }
 554       addtoextraq((event *) tmp);
 555    }
 556 
 557    if ((iodrivers[iodriverno]->consttime == IODRIVER_TRACED_QUEUE_TIMES) 
 558        || (iodrivers[iodriverno]->consttime == IODRIVER_TRACED_BOTH_TIMES)) 
 559    {
 560      if (ioqueue_get_number_in_queue(iodrivers[iodriverno]->devices[devno].queue) > 0) 
 561      {
 562        iodrivers[iodriverno]->devices[devno].flag = 1;
 563        iodrivers[iodriverno]->devices[devno].lastevent = simtime;
 564      }
 565      return;
 566    }
 567    if (skip) {
 568      return;
 569    }
 570 
 571 
 572    // fprintf(outputfile, "iodriver_access_complete::  calling ioqueue_get_next_request\n");
 573 
 574 
 575    req = ioqueue_get_next_request(iodrivers[iodriverno]->devices[devno].queue);
 576 
 577    
 578    // fprintf (outputfile, "next scheduled: req %p, req->blkno %d, req->flags %x\n", req, ((req) ? req->blkno : 0), ((req) ? req->flags : 0));
 579 
 580 
 581 schedule_next:
 582    if (req) {
 583       req->type = IO_ACCESS_ARRIVE;
 584       req->next = NULL;
 585 
 586       if (ctl) {
 587          ctl->numoutstanding++;
 588       }
 589 
 590       if (iodrivers[iodriverno]->type == STANDALONE) {
 591          req->time = simtime;
 592          addtointq((event *) req);
 593       } 
 594       else {
 595          iodriver_add_to_intrp_eventlist(intrp, 
 596                                          (event *) req, 
 597                                          iodrivers[iodriverno]->scale);
 598       }
 599    }
 600 }

 603 void iodriver_respond_to_device (int iodriverno, intr_event *intrp)
 604 {
 605    ioreq_event *req = NULL;
 606    int devno;
 607    int cause;
 608 
 609    if ((iodrivers[iodriverno]->consttime != 0.0) && (iodrivers[iodriverno]->consttime != IODRIVER_TRACED_QUEUE_TIMES)) {
 610       if (iodrivers[iodriverno]->type == STANDALONE) {
 611          addtoextraq((event *) intrp->infoptr);
 612       }
 613       return;
 614    }
 615    req = (ioreq_event *) intrp->infoptr;
 616 /*
 617 fprintf (outputfile, "%f, Responding to device - cause = %d, blkno %d\n", simtime, req->cause, req->blkno);
 618 */
 619    req->type = IO_INTERRUPT_COMPLETE;
 620    devno = req->devno;
 621    cause = req->cause;
 622    switch (cause) {
 623       case COMPLETION:
 624                            if (iodrivers[iodriverno]->type != STANDALONE) {
 625                               req = ioreq_copy((ioreq_event *) intrp->infoptr);
 626                            }
 627       case DISCONNECT:
 628       case RECONNECT:
 629                            iodriver_send_event_down_path(req);
 630                            break;
 631       case READY_TO_TRANSFER:
 632                            addtoextraq((event *) req);
 633                            break;
 634       default:
 635                fprintf(stderr, "Unknown io_interrupt cause - %d\n", req->cause);
 636                exit(1);
 637    }
 638    iodriver_check_c700_based_status(iodrivers[iodriverno], devno, cause, IO_INTERRUPT_COMPLETE, 0);
 639 }

 642 void iodriver_interrupt_complete (int iodriverno, intr_event *intrp)
 643 {
 644 /*
 645 fprintf (outputfile, "%f, Interrupt completing - cause = %d, blkno %d\n", simtime, ((ioreq_event *) intrp->infoptr)->cause, ((ioreq_event *) intrp->infoptr)->blkno);
 646 */
 647    if (iodrivers[iodriverno]->type == STANDALONE) {
 648       if (((ioreq_event *) intrp->infoptr)->cause == COMPLETION) {
 649          iodriver_access_complete(iodriverno, intrp);
 650       }
 651       iodriver_respond_to_device(iodriverno, intrp);
 652    }
 653    addtoextraq((event *) intrp);

 657 static double iodriver_get_time_to_handle_interrupt (iodriver *curriodriver, int cause, int read)
 658 {
 659    double retval = 0.0;
 660    if (cause == DISCONNECT) {
 661       retval = (read) ? IODRIVER_READ_DISCONNECT_TIME : IODRIVER_WRITE_DISCONNECT_TIME;
 662    } else if (cause == RECONNECT) {
 663       retval = IODRIVER_RECONNECT_TIME;
 664    } else if (cause == COMPLETION) {
 665       retval = (IODRIVER_BASE_COMPLETION_TIME - IODRIVER_COMPLETION_RESPTODEV_TIME) * (double) 5;   /* because it gets divided by five later */
 666    } else if (cause != READY_TO_TRANSFER) {
 667       fprintf(stderr, "Unknown interrupt at time_to_handle_interrupt: %d\n", cause);
 668       exit(1);
 669    }
 670    return(curriodriver->scale * retval);
 671 }

 674 static double iodriver_get_time_to_respond_to_device (iodriver *curriodriver, int cause, double fifthoftotal)
 675 {
 676    double retval;
 677    if (cause == COMPLETION) {
 678       retval = IODRIVER_COMPLETION_RESPTODEV_TIME;
 679    } else {
 680                         /* arbitrarily chosen based on some observations */
 681       retval = fifthoftotal * (double) 4;
 682    }
 683    return(curriodriver->scale * retval);
 684 }

 687 void iodriver_interrupt_arrive (int iodriverno, intr_event *intrp)
 688 {
 689    event *tmp;
 690    ioreq_event *infoptr = (ioreq_event *) intrp->infoptr;
 691 /*
 692 fprintf (outputfile, "%f, Interrupt arriving - cause = %d, blkno %d\n", simtime, infoptr->cause, infoptr->blkno);
 693 */
 694    if ((iodrivers[iodriverno]->consttime == 0.0) || (iodrivers[iodriverno]->consttime == IODRIVER_TRACED_QUEUE_TIMES)) {
 695       intrp->time = iodriver_get_time_to_handle_interrupt(iodrivers[iodriverno], infoptr->cause, (infoptr->flags & READ));
 696       iodriver_check_c700_based_status(iodrivers[iodriverno], infoptr->devno, infoptr->cause, IO_INTERRUPT_ARRIVE, infoptr->blkno);
 697    } else {
 698       intrp->time = 0.0;
 699    }
 700    if (iodrivers[iodriverno]->type == STANDALONE) {
 701       intrp->time += simtime;
 702       intrp->type = IO_INTERRUPT_COMPLETE;
 703       addtointq((event *) intrp);
 704    } else {
 705       tmp = getfromextraq();
 706       intrp->time /= (double) 5;
 707       tmp->time = intrp->time;
 708       tmp->type = INTEND_EVENT;
 709       tmp->next = NULL;
 710       ((intr_event *)tmp)->vector = IO_INTERRUPT;
 711       intrp->eventlist = tmp;
 712       if (infoptr->cause == COMPLETION) {
 713          tmp = getfromextraq();
 714          tmp->time = 0.0;
 715          tmp->type = IO_ACCESS_COMPLETE;
 716          tmp->next = intrp->eventlist;
 717          ((ioreq_event *)tmp)->tempptr1 = intrp;
 718          intrp->eventlist = tmp;
 719       }
 720       tmp = getfromextraq();
 721       tmp->time = iodriver_get_time_to_respond_to_device(iodrivers[iodriverno], infoptr->cause, intrp->time);
 722       tmp->type = IO_RESPOND_TO_DEVICE;
 723       tmp->next = intrp->eventlist;
 724       ((ioreq_event *)tmp)->tempptr1 = intrp;
 725       intrp->eventlist = tmp;
 726    }
 727 }

 744 event * iodriver_request (int iodriverno, ioreq_event *curr)
 745 {
 746    ioreq_event *temp = NULL;
 747    ioreq_event *ret = NULL;
 748    ioreq_event *retlist = NULL;
 749    int numreqs;
 750 /*
 751    printf ("Entered iodriver_request - simtime %f, devno %d, blkno %d, cause %d\n", simtime, curr->devno, curr->blkno, curr->cause);
 752    fprintf (outputfile, "Entered iodriver_request - simtime %f, devno %d, blkno %d, cause %d\n", simtime, curr->devno, curr->blkno, curr->cause);
 753    fprintf (stderr, "Entered iodriver_request - simtime %f, devno %d, blkno %d, cause %d\n", simtime, curr->devno, curr->blkno, curr->cause);
 754 */
 755    if (outios) {
 756       fprintf(outios, "%.6f\t%d\t%d\t%d\t%x\n", simtime, curr->devno, curr->blkno, curr->bcount, curr->flags);
 757    }
 758 
 759 #if 0
 760    fprintf (stderr, "Entered iodriver_request - simtime %f, devno %d, blkno %d, cause %d\n", simtime, curr->devno, curr->blkno, curr->cause);
 761 #endif
 762 
 763    /* add to the overall queue to start tracking */
 764    ret = ioreq_copy (curr);
 765    ioqueue_add_new_request (overallqueue, ret);
 766    ret = NULL;
 767  
 768    disksim->totalreqs++;
 769    if ((disksim->checkpoint_iocnt > 0) && ((disksim->totalreqs % disksim->checkpoint_iocnt) == 0)) {
 770       disksim_register_checkpoint (simtime);
 771    }
 772    if (disksim->totalreqs == disksim->warmup_iocnt) {
 773       warmuptime = simtime;
 774       resetstats();
 775    }
 776    numreqs = logorg_maprequest(sysorgs, numsysorgs, curr);
 777    temp = curr->next;
 778    for (; numreqs>0; numreqs--) {
 779           /* Request list size must match numreqs */
 780       ASSERT(curr != NULL);
 781       curr->next = NULL;
 782       if ((iodrivers[iodriverno]->consttime == IODRIVER_TRACED_QUEUE_TIMES) || (iodrivers[iodriverno]->consttime == IODRIVER_TRACED_BOTH_TIMES)) {
 783          ret = ioreq_copy(curr);
 784          ret->time = simtime + (double) ret->tempint1 / (double) 1000;
 785          ret->type = IO_TRACE_REQUEST_START;
 786          addtointq((event *) ret);
 787          ret = NULL;
 788          if ((curr->slotno == 1) && (ioqueue_get_number_in_queue(iodrivers[iodriverno]->devices[(curr->devno)].queue) == 0)) {
 789             iodrivers[(iodriverno)]->devices[(curr->devno)].flag = 2;
 790             iodrivers[(iodriverno)]->devices[(curr->devno)].lastevent = simtime;
 791          }
 792       }
 793       ret = handle_new_request(iodrivers[iodriverno], curr);
 794 
 795       if ((ret) && (iodrivers[iodriverno]->type == STANDALONE) && (ret->time == 0.0)) {
 796          ret->type = IO_ACCESS_ARRIVE;
 797          ret->time = simtime;
 798          iodriver_schedule(iodriverno, ret);
 799       } else if (ret) {
 800          ret->type = IO_ACCESS_ARRIVE;
 801          ret->next = retlist;
 802          ret->prev = NULL;
 803          retlist = ret;
 804       }
 805       curr = temp;
 806       temp = (temp) ? temp->next : NULL;
 807    }
 808    if (iodrivers[iodriverno]->type == STANDALONE) {
 809       while (retlist) {
 810          ret = retlist;
 811          retlist = ret->next;
 812          ret->next = NULL;
 813          ret->time += simtime;
 814          addtointq((event *) ret);
 815       }
 816    }
 817 /*
 818 fprintf (outputfile, "leaving iodriver_request: retlist %p\n", retlist);
 819 */
 820    return((event *) retlist);
 821 }

 824 void iodriver_trace_request_start (int iodriverno, ioreq_event *curr)
 825 {
 826    ioreq_event *tmp;
 827    device *currdev = &iodrivers[iodriverno]->devices[(curr->devno)];
 828    double tdiff = simtime - currdev->lastevent;
 829 
 830    if (currdev->flag == 1) {
 831       stat_update(&initiatenextstats, tdiff);
 832    } else if (currdev->flag == 2) {
 833       stat_update(&emptyqueuestats, tdiff);
 834    }
 835    currdev->flag = 0;
 836 
 837    tmp = ioqueue_get_specific_request(currdev->queue, curr);
 838    addtoextraq((event *) curr);
 839    ASSERT(tmp != NULL);
 840 
 841    schedule_disk_access(iodrivers[iodriverno], tmp);
 842    tmp->time = simtime;
 843    tmp->type = IO_ACCESS_ARRIVE;
 844    tmp->slotno = 0;
 845    if (tmp->time == simtime) {
 846       iodriver_schedule(iodriverno, tmp);
 847    } else {
 848       addtointq((event *) tmp);
 849    }

1171   struct logorg *l = 0;
1172 
1173   newlen = disksim->iodriver_info->sysorgs_len ? 
1174     disksim->iodriver_info->sysorgs_len * 2 : 2;
1175   zeroOff = (newlen == 2) ? 0 : (newlen / 2);
1176 
1177 
1178   l = disksim_logorg_loadparams(b); 
1179   if(!l) return 0;
1180 
1181   for(c = 0; c < disksim->iodriver_info->sysorgs_len; c++) {
1182     if(!sysorgs[c]) {
1183       goto foundslot;
1184     }
1185   }
1186 
1187   sysorgs = realloc(sysorgs, newlen * sizeof(struct logorg *));
1188 
1189   bzero(sysorgs + zeroOff, 
1190         (zeroOff ? numsysorgs : 2) * sizeof(struct logorg *));
1191 
1192   disksim->iodriver_info->sysorgs_len = newlen;
1193 
1194  foundslot:
1195   sysorgs[c] = l;
1196   logorg_set_arraydisk(l, c);
1197   numsysorgs++;
1198 
1199   return 1;
1200 }
