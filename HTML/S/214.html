  71 /* This remaps device numbers amongst multiple types of devices (each */
  72 /* of which keeps internal arrays based on the remapped numbers). So, */
  73 /* the device implementations need to be careful to call back to get  */
  74 /* the remapping info...                                              */
  75 
  76 /* To avoid the corresponding confusion, we will simply remap devnos  */
  77 /* to themselves for now.  So, each device implementation should      */
  78 /* allocate MAXDEVICES structures and track which of them are used.   */
  79 
  80 
  81 /* private remapping #defines for variables from device_info_t */
  82 #define numdevices                  (disksim->deviceinfo->numdevices)
  83         /* per-devno device type */
  84 #define devicetypes                 (disksim->deviceinfo->devicetypes)
  85         /* per-devno device number (for when remapping them) */
  86 #define devicenos                   (disksim->deviceinfo->devicenos)
  87         /* number of devices of each type */
  88 #define maxdeviceno                 (disksim->deviceinfo->maxdeviceno)

  91 struct device_header *
  92 getdevbyname(char *name, 
  93              int *gdevnum, /* global device number */
  94              int *ldevnum, /* type-specific device number */
  95              int *type)    /* device type */
  96 {
  97   int c;
  98   for(c = 0; c < disksim->deviceinfo->devs_len; c++) {
  99     if(!disksim->deviceinfo->devicenames[c]) continue;
 100     if(!strcmp(name, disksim->deviceinfo->devicenames[c])) {
 101 
 102       if(gdevnum) *gdevnum = c;
 103       if(ldevnum) *ldevnum = devicenos[c];
 104       if(type)    *type = devicetypes[c];
 105       switch(devicetypes[c]) {
 106       case DEVICETYPE_DISK:
 107         return (struct device_header *)getdisk(devicenos[c]);
 108         break;
 109       case DEVICETYPE_SIMPLEDISK:
 110         return (struct device_header *)getsimpledisk(devicenos[c]);
 111         break;
 112       case DEVICETYPE_MEMS:
 113         return (struct device_header *)getmems(devicenos[c]);
 114         break;
 115       case DEVICETYPE_SSD: /* SSD: */
 116         return (struct device_header *)getssd(devicenos[c]);
 117         break;
 118       }
 119     }
 120   }
 121   return 0;
 122 }

 124 void device_initialize_deviceinfo (void)
 125 {
 126    if (disksim->deviceinfo == NULL) {
 127       disksim->deviceinfo = malloc (sizeof(device_info_t));
 128       bzero ((char *)disksim->deviceinfo, sizeof(device_info_t));
 129    }
 130 }

 132 void device_add(struct device_header *d, int ldevno) {
 133   int c, newlen;
 134   int zerocnt;
 135   device_initialize_deviceinfo();
 136   
 137   for(c = 0; c < disksim->deviceinfo->devs_len; c++) {
 138     if(!disksim->deviceinfo->devicenames[c]) {
 139       goto foundslot;
 140     }
 141   }
 142 
 143   /* note that numdisks must be equal to diskinfo->disks_len */
 144   newlen = numdevices ? (2 * numdevices) : 2;
 145   zerocnt = (newlen == 2) ? 2 : (newlen/2);
 146   disksim->deviceinfo->devicenames = 
 147     realloc(disksim->deviceinfo->devicenames, newlen * sizeof(char *));
 148   bzero(disksim->deviceinfo->devicenames + c, zerocnt * sizeof(char *));
 149 
 150   devicenos = realloc(devicenos, newlen*sizeof(int));
 151 #ifndef WIN32
 152   bzero(devicenos + c, zerocnt * sizeof(int));
 153 #else
 154   bzero(devicenos + c, zerocnt * sizeof(*(devicenos)));
 155 #endif
 156 
 157   devicetypes = realloc(devicetypes, newlen*sizeof(int));
 158 #ifndef WIN32
 159   bzero(devicetypes + c, zerocnt * sizeof(int));
 160 #else
 161   bzero(devicetypes + c, zerocnt * sizeof(*(devicetypes)));
 162 #endif
 163 
 164   disksim->deviceinfo->devices = realloc(disksim->deviceinfo->devices, 
 165                                          newlen*sizeof(int));
 166 #ifndef WIN32
 167   bzero(disksim->deviceinfo->devices + c, zerocnt * sizeof(int));
 168 #else
 169   bzero(disksim->deviceinfo->devices + c, zerocnt * sizeof(*(disksim->deviceinfo->devices)));
 170 #endif
 171 
 172   disksim->deviceinfo->devs_len = newlen;
 173 
 174  foundslot:
 175   disksim->deviceinfo->devicenames[c] = d->device_name;
 176   devicetypes[c] = d->device_type;
 177   devicenos[c] = ldevno;
 178   if(maxdeviceno[devicetypes[c]] < devicenos[c]) {
 179     maxdeviceno[devicetypes[c]] = devicenos[c];
 180   }
 181   disksim->deviceinfo->devices[c] = d;
 182 
 183   numdevices++;
 184 }

 246 void device_printsetstats (int *set, int setsize, char *sourcestr)
 247 {
 248    int i;
 249    int devicetype = devicetypes[set[0]];
 250 
 251    /* verify that all set members are of same device type */
 252    for (i=0; i<setsize; i++) {
 253       if (devicetypes[set[i]] != devicetype) {
 254          // fprintf(stderr, "Can't have mismatching devicetypes in device_printsetstats (%d != %d)\n", devicetypes[set[i]], devicetype);
 255          /* Might want to just return (and put this message in the output */
 256          /* file) rather than exiting...                                  */
 257          // exit(1);
 258          return;
 259       }
 260    }
 261 
 262    /* call the appropriate one */
 263    switch(devicetype) {
 264    case DEVICETYPE_DISK:
 265       disk_printsetstats (set, setsize, sourcestr);
 266       break;
 267    case DEVICETYPE_SIMPLEDISK:
 268       simpledisk_printsetstats (set, setsize, sourcestr);
 269       break;
 270    case DEVICETYPE_MEMS:
 271       mems_printsetstats (set, setsize, sourcestr);
 272       break;
 273    case DEVICETYPE_SSD:
 274       ssd_printsetstats (set, setsize, sourcestr);
 275       break;
 276    default:
 277       fprintf(stderr, "Unknown value for device type: devicetype %d\n", devicetype);
 278       assert(0);
 279    }
 280 }

 472 int disksim_syncset_loadparams(struct lp_block *b)
 473 {
 474 
 475   int c;
 476     
 477   unparse_block(b, outputfile);
 478   for(c = 0; c < b->params_len; c++) {
 479     if(!b->params[c]) continue;
 480     if(PTYPE(b->params[c]) != S) continue;
 481     if(strcmp(b->params[c]->name, "type")) continue;
 482     else {
 483       if(!strcmp(SVAL(b->params[c]), "simpledisk")) {
 484         fprintf(stderr, "*** warning: no simpledisk syncsets\n");
 485 /*      simpledisk_load_syncsets(b); */
 486       }
 487       else if(!strcmp(SVAL(b->params[c]), "disk")) {
 488         return disk_load_syncsets(b);
 489       }
 490       else if(!strcmp(SVAL(b->params[c]), "mems")) {
 491         fprintf(stderr, "*** warning: no mems syncsets\n");
 492 /*      mems_load_syncsets(b); */
 493       }
 494       else {
 495       }
 496     }
 497   }
 498   return 1;
 499 }
