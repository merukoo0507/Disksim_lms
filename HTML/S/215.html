 101 /***************************************************************************
 102   This is a fairly simple module for managing a device that is being used
 103   as a cache for another device.  It does not have many functionalities
 104   that a real device-caching-device would have, such as
 105         1. No actual "cache"-iness.  That is, it simply maps given locations
 106            to the same locations on the cache device.  This limits the max
 107            amount of real device space that can be cached, and misses
 108            opportunities for creating cache-device locality where none
 109            exists in the original workload.
 110         2. No prefetching.
 111         3. No optimization.  For example, see #1.  Also, no grouping of
 112            writes to cache-device.  Also, no going to original device when
 113            cache-device is busy.
 114         4. No locking.  Specifically, requests for the same space can be in
 115            progress in parallel.  This could cause inconsistencies, with
 116            unfortunate sequencing.
 117         5. No buffer space limitations.  Specifically, there is no limit on
 118            the amount of buffer space used at once for managing the cache
 119            and real devices.
 120 ***************************************************************************/

 147 static void 
 148 cachedev_add_ongoing_request (struct cache_dev *cache, 
 149                               void *crq)
 150 {
 151   struct cache_dev_event *cachereq = (struct cache_dev_event *)crq;
 152    cachereq->next = cache->ongoing_requests;
 153    cachereq->prev = NULL;
 154    if (cachereq->next) {
 155       cachereq->next->prev = cachereq;
 156    }
 157    cache->ongoing_requests = cachereq;
 158 }

 161 static void 
 162 cachedev_remove_ongoing_request (struct cache_dev *cache, 
 163                                  struct cache_dev_event *cachereq)
 164 {
 165    if (cachereq->next) {
 166       cachereq->next->prev = cachereq->prev;
 167    }
 168    if (cachereq->prev) {
 169       cachereq->prev->next = cachereq->next;
 170    }
 171    if (cache->ongoing_requests == cachereq) {
 172       cache->ongoing_requests = cachereq->next;
 173    }
 174 }

 190 /* Return a count of how many dirty blocks are on the cachedev. */
 191 static int cachedev_count_dirty_blocks (struct cache_dev *cache)
 192 {
 193    int dirtyblocks = 0;
 194    int i;
 195 
 196    for (i=0; i<cache->size; i++) {
 197       if (bit_test(cache->dirtymap,i)) {
 198          dirtyblocks++;
 199       }
 200    }
 201    return (dirtyblocks);
 202 }

 217 static int cachedev_isreadhit (struct cache_dev *cache, ioreq_event *req)
 218 {
 219    int lastblk = req->blkno + req->bcount;
 220    int i;
 221 
 222    if (lastblk >= cache->size) {
 223       return (0);
 224    }
 225 
 226    for (i=req->blkno; i<lastblk; i++) {
 227       if (bit_test(cache->validmap,i) == 0) {
 228          return (0);
 229       }
 230    }
 231 
 232    return (1);
 233 }

 236 static int cachedev_iswritehit (struct cache_dev *cache, ioreq_event *req)
 237 {
 238    if ((req->blkno+req->bcount) >= cache->size) {
 239       return (0);
 240    }
 241    return (1);
 242 }

 245 static int cachedev_find_dirty_cache_blocks (struct cache_dev *cache, int *blknoPtr, int *bcountPtr)
 246 {
 247    int blkno;
 248 
 249    bit_ffs (cache->dirtymap, cache->size, blknoPtr);
 250    if (*blknoPtr == -1) {
 251       return (0);
 252    }
 253 
 254    blkno = *blknoPtr;
 255    while (bit_test (cache->dirtymap, blkno) != 0) {
 256       blkno++;
 257    }
 258    *bcountPtr = blkno - *blknoPtr;
 259 
 260    return (1);
 261 }

 272 static void cachedev_idlework_callback (void *idleworkparam, int idledevno)
 273 {
 274 
 275    struct cache_dev *cache = idleworkparam;
 276    struct cache_dev_event *flushdesc;
 277    ioreq_event *flushreq;
 278    int blkno, bcount;
 279    struct ioq *queue;
 280 
 281    ASSERT (idledevno == cache->real_devno);
 282 
 283    queue = (*cache->queuefind)(cache->queuefindparam, cache->real_devno);
 284    if (ioqueue_get_number_in_queue (queue) != 0) {
 285       return;
 286    }
 287 
 288    queue = (*cache->queuefind)(cache->queuefindparam, cache->cache_devno);
 289    if (ioqueue_get_number_in_queue (queue) != 0) {
 290       return;
 291    }
 292 
 293    if (cachedev_find_dirty_cache_blocks (cache, &blkno, &bcount) == 0) {
 294       return;
 295    }
 296 
 297    /* Just assume that bufferspace is available */
 298    cache->bufferspace += bcount;
 299    if (cache->bufferspace > cache->stat.maxbufferspace) {
 300       cache->stat.maxbufferspace = cache->bufferspace;
 301    }
 302 
 303    flushdesc = (struct cache_dev_event *) getfromextraq();
 304    flushdesc->type = CACHE_EVENT_IDLEFLUSH_READ;
 305    flushreq = (ioreq_event *) getfromextraq();
 306    flushreq->buf = flushdesc;
 307    flushreq->devno = cache->cache_devno;
 308    flushreq->blkno = blkno;
 309    flushreq->bcount = bcount;
 310    flushreq->type = IO_ACCESS_ARRIVE;
 311    flushreq->flags = READ;
 312    (*cache->issuefunc)(cache->issueparam, flushreq);
 313    cache->stat.destagereads++;
 314    cache->stat.destagereadblocks += bcount;
 315 }

 320 static int 
 321 cachedev_get_block (struct cache_if *c, 
 322                     ioreq_event *req, 
 323                     void (**donefunc)(void *, ioreq_event *), 
 324                     void *doneparam)
 325 {
 326   struct cache_dev *cache = (struct cache_dev *)c;
 327    struct cache_dev_event *rwdesc = (struct cache_dev_event *) getfromextraq();
 328    ioreq_event *fillreq;
 329    int devno;
 330 
 331    // fprintf (outputfile, "totalreqs = %d\n", disksim->totalreqs);
 332    // fprintf (outputfile, "%.5f: Entered cache_get_block: rw %d, devno %d, blkno %d, size %d\n", simtime, (req->flags & READ), req->devno, req->blkno, req->bcount);
 333 
 334    if (req->devno != cache->real_devno) {
 335       fprintf (stderr, "cachedev_get_block trying to cache blocks for wrong device (%d should be %d)\n", req->devno, cache->real_devno);
 336       ASSERT(0);
 337       exit(1);
 338    }
 339 
 340    /* Ignore request overlap and locking issues for now.  */
 341    /* Also ignore buffer space limitation issues for now. */
 342    cache->bufferspace += req->bcount;
 343    if (cache->bufferspace > cache->stat.maxbufferspace) {
 344       cache->stat.maxbufferspace = cache->bufferspace;
 345    }
 346 
 347    rwdesc->type = (req->flags & READ) ? CACHE_EVENT_READ : CACHE_EVENT_WRITE;
 348    rwdesc->donefunc = donefunc;
 349    rwdesc->doneparam = doneparam;
 350    rwdesc->req = req;
 351    req->next = NULL;
 352    req->prev = NULL;
 353    rwdesc->flags = 0;
 354    cachedev_add_ongoing_request (cache, rwdesc);
 355 
 356    if (req->flags & READ) {
 357       cache->stat.reads++;
 358       cache->stat.readblocks += req->bcount;
 359 
 360       /* Send read straight to whichever device has it (preferably cachedev). */
 361       if (cachedev_isreadhit (cache, req)) {
 362          devno = cache->cache_devno;
 363          cache->stat.readhitsfull++;
 364       } else {
 365          devno = cache->real_devno;
 366          cache->stat.readmisses++;
 367       }
 368       /* For now, just assume both device's store bits at same LBNs */
 369       fillreq = ioreq_copy (req);
 370       fillreq->buf = rwdesc;
 371       fillreq->type = IO_ACCESS_ARRIVE;
 372       fillreq->devno = devno;
 373 
 374       // fprintf (outputfile, "fillreq: devno %d, blkno %d, buf %p\n", fillreq->devno, fillreq->blkno, fillreq->buf);
 375 
 376       (*cache->issuefunc)(cache->issueparam, fillreq);
 377       return (1);
 378 
 379    } else {
 380       /* Grab buffer space and let the controller fill in data to be written. */
 381       /* (for now, just assume that there is buffer space available)          */
 382 
 383       (*donefunc)(doneparam, req);
 384       return (0);
 385    }
 386 }

 392 static void 
 393 cachedev_free_block_clean (struct cache_if *c, 
 394                            ioreq_event *req)
 395 {
 396   struct cache_dev *cache = (struct cache_dev *)c;
 397    struct cache_dev_event *rwdesc;
 398 
 399    // fprintf (outputfile, "%.5f: Entered cache_free_block_clean: blkno %d, bcount %d, devno %d\n", simtime, req->blkno, req->bcount, req->devno);
 400 
 401    /* For now, just find relevant rwdesc and free it.                       */
 402    /* Later, write it to the cache device (and update the cache map thusly. */
 403 
 404    rwdesc = cachedev_find_ongoing_request (cache, req);
 405    ASSERT (rwdesc != NULL);
 406 
 407    if (rwdesc->type == CACHE_EVENT_READ) {
 408       cache->bufferspace -= req->bcount;
 409       cachedev_remove_ongoing_request (cache, rwdesc);
 410       addtoextraq ((event *) rwdesc);
 411    } else {
 412       ASSERT (rwdesc->type == CACHE_EVENT_POPULATE_ALSO);
 413       rwdesc->type = CACHE_EVENT_POPULATE_ONLY;
 414    }
 415 }

 421 static int 
 422 cachedev_free_block_dirty (struct cache_if *c, 
 423                            ioreq_event *req, 
 424                            void (**donefunc)(void *, ioreq_event *), 
 425                            void *doneparam)
 426 {
 427   struct cache_dev *cache = (struct cache_dev *)c;
 428    ioreq_event *flushreq;
 429    struct cache_dev_event *writedesc;
 430 
 431    // fprintf (outputfile, "%.5f, Entered cache_free_block_dirty: blkno %d, size %d\n", simtime, req->blkno, req->bcount);
 432 
 433    cache->stat.writes++;
 434    cache->stat.writeblocks += req->bcount;
 435 
 436    writedesc = cachedev_find_ongoing_request (cache, req);
 437    ASSERT (writedesc != NULL);
 438    ASSERT (writedesc->type == CACHE_EVENT_WRITE);
 439 
 440    writedesc->donefunc = donefunc;
 441    writedesc->doneparam = doneparam;
 442    writedesc->req = req;
 443    req->type = IO_REQUEST_ARRIVE;
 444    req->next = NULL;
 445    req->prev = NULL;
 446 
 447    /* For now, just assume both device's store bits at same LBNs */
 448    flushreq = ioreq_copy(req);
 449    flushreq->type = IO_ACCESS_ARRIVE;
 450    flushreq->buf = writedesc;
 451    if (cachedev_iswritehit (cache, req)) {
 452       flushreq->devno = cache->cache_devno;
 453       cache->stat.writehitsfull++;
 454    } else {
 455       cache->stat.writemisses++;
 456    }
 457 
 458    // fprintf (outputfile, "flushreq: devno %d, blkno %d, buf %p\n", flushreq->devno, flushreq->blkno, flushreq->buf);
 459 
 460    (*cache->issuefunc)(cache->issueparam, flushreq);
 461 
 462 #if 0
 463    if (cache->flush_idledelay >= 0.0) {
 464       ioqueue_reset_idledetecter((*cache->queuefind)(cache->queuefindparam, req->devno), 0);
 465    }
 466 #endif
 467 
 468    return(1);
 469 }

 478 static void *
 479 cachedev_disk_access_complete (struct cache_if *c,
 480                                ioreq_event *curr)
 481 {
 482   struct cache_dev *cache = (struct cache_dev *)c;
 483    struct cache_dev_event *rwdesc = curr->buf;
 484    struct cache_dev_event *tmp = NULL;
 485 
 486    // fprintf (outputfile, "Entered cache_disk_access_complete: blkno %d, bcount %d, devno %d, buf %p\n", curr->blkno, curr->bcount, curr->devno, curr->buf);
 487 
 488    switch(rwdesc->type) {
 489    case CACHE_EVENT_READ:
 490       /* Consider writing same buffer to cache_devno, in order to populate it.*/
 491       /* Not clear whether it is more appropriate to do it from here or from  */
 492       /* "free_block_clean" -- do it here for now to get more overlap.        */
 493       if (curr->devno == cache->real_devno) {
 494          ioreq_event *flushreq = ioreq_copy(rwdesc->req);
 495          flushreq->type = IO_ACCESS_ARRIVE;
 496          flushreq->buf = rwdesc;
 497          flushreq->flags = WRITE;
 498          flushreq->devno = cache->cache_devno;
 499          rwdesc->type = CACHE_EVENT_POPULATE_ALSO;
 500          (*cache->issuefunc)(cache->issueparam, flushreq);
 501          cache->stat.popwrites++;
 502          cache->stat.popwriteblocks += rwdesc->req->bcount;
 503       }
 504 
 505       /* Ongoing read request can now proceed, so call donefunc from get_block*/
 506       (*rwdesc->donefunc)(rwdesc->doneparam,rwdesc->req);
 507 
 508       break;
 509    case CACHE_EVENT_WRITE:
 510 
 511       /* finished writing to cache-device */
 512       if (curr->devno == cache->cache_devno) {
 513          cachedev_setbits (cache->validmap, curr);
 514          cachedev_setbits (cache->dirtymap, curr);
 515          if (cache->writescheme == CACHE_WRITE_THRU) {
 516             ioreq_event *flushreq = ioreq_copy(rwdesc->req);
 517             flushreq->type = IO_ACCESS_ARRIVE;
 518             flushreq->buf = rwdesc;
 519             flushreq->flags = WRITE;
 520             flushreq->devno = cache->real_devno;
 521             rwdesc->type = CACHE_EVENT_FLUSH;
 522             (*cache->issuefunc)(cache->issueparam, flushreq);
 523             cache->stat.destagewrites++;
 524             cache->stat.destagewriteblocks += rwdesc->req->bcount;
 525          }
 526       }
 527       (*rwdesc->donefunc)(rwdesc->doneparam,rwdesc->req);
 528 
 529       if (rwdesc->type != CACHE_EVENT_FLUSH) {
 530          cachedev_remove_ongoing_request (cache, rwdesc);
 531          addtoextraq ((event *) rwdesc);
 532          cache->bufferspace -= curr->bcount;
 533       }
 534 
 535       break;
 536 
 537    case CACHE_EVENT_POPULATE_ONLY:
 538      cachedev_setbits (cache->validmap, curr);
 539      cachedev_remove_ongoing_request (cache, rwdesc);
 540       addtoextraq ((event *) rwdesc);
 541       cache->bufferspace -= curr->bcount;
 542       break;
 543       
 544    case CACHE_EVENT_POPULATE_ALSO:
 545      cachedev_setbits (cache->validmap, curr);
 546      rwdesc->type = CACHE_EVENT_READ;
 547      break;
 548 
 549    case CACHE_EVENT_FLUSH:
 550       cachedev_clearbits (cache->dirtymap, curr);
 551       cachedev_remove_ongoing_request (cache, rwdesc);
 552       addtoextraq ((event *) rwdesc);
 553       cache->bufferspace -= curr->bcount;
 554       break;
 555 
 556    case CACHE_EVENT_IDLEFLUSH_READ:
 557      {
 558        ioreq_event *flushreq = ioreq_copy (curr);
 559        flushreq->type = IO_ACCESS_ARRIVE;
 560        flushreq->flags = WRITE;
 561        flushreq->devno = cache->real_devno;
 562        rwdesc->type = CACHE_EVENT_IDLEFLUSH_FLUSH;
 563        (*cache->issuefunc)(cache->issueparam, flushreq);
 564        cache->stat.destagewrites++;
 565        cache->stat.destagewriteblocks += curr->bcount;
 566      } break;
 567 
 568    case CACHE_EVENT_IDLEFLUSH_FLUSH:
 569      cachedev_clearbits (cache->dirtymap, curr);
 570      cachedev_remove_ongoing_request (cache, rwdesc);
 571      addtoextraq ((event *) rwdesc);
 572      cachedev_idlework_callback (cache, curr->devno);
 573      cache->bufferspace -= curr->bcount;
 574      break;
 575 
 576    default:
 577      ddbg_assert2(0, "Unknown cachedev event type");
 578      break;
 579    }
 580 
 581    addtoextraq((event *) curr);
 582 
 583    /* returned cacheevent will get forwarded to cachedev_wakeup_continue... */
 584    return(tmp);
 585 }

 650 static void 
 651 cachedev_initialize (struct cache_if *c, 
 652                      void (**issuefunc)(void *,ioreq_event *), 
 653                      void *issueparam, 
 654                      struct ioq * (**queuefind)(void *,int), 
 655                      void *queuefindparam, 
 656                      void (**wakeupfunc)(void *, struct cacheevent *), 
 657                      void *wakeupparam, 
 658                      int numdevs)
 659 {
 660   struct cache_dev *cache = (struct cache_dev *)c;
 661   StaticAssert (sizeof(struct cache_dev_event) <= DISKSIM_EVENT_SIZE);
 662 
 663    cache->issuefunc = issuefunc;
 664    cache->issueparam = issueparam;
 665    cache->queuefind = queuefind;
 666    cache->queuefindparam = queuefindparam;
 667    cache->wakeupfunc = wakeupfunc;
 668    cache->wakeupparam = wakeupparam;
 669    cache->bufferspace = 0;
 670    cache->ongoing_requests = NULL;
 671    bzero (cache->validmap, bitstr_size(cache->size));
 672    bzero (cache->dirtymap, bitstr_size(cache->size));
 673    cachedev_resetstats(c);
 674 
 675    if (cache->flush_idledelay) {
 676       struct ioq *queue = (*queuefind)(queuefindparam,cache->real_devno);
 677       ASSERT (queue != NULL);
 678       ioqueue_set_idlework_function (queue, 
 679                                      &disksim->idlework_cachedev, 
 680                                      cache, 
 681                                      cache->flush_idledelay);
 682    }
 683 
 684    if (device_get_number_of_blocks(cache->cache_devno) < cache->size) {
 685       fprintf (stderr, "Size of cachedev exceeds that of actual cache device (devno %d): %d > %d\n", cache->cache_devno, cache->size, device_get_number_of_blocks(cache->cache_devno));
 686       ddbg_assert(0);
 687    }
 688 }

 697 static void 
 698 cachedev_printstats (struct cache_if *c, char *prefix)
 699 {
 700   struct cache_dev *cache = (struct cache_dev *)c;
 701    int reqs = cache->stat.reads + cache->stat.writes;
 702    int blocks = cache->stat.readblocks + cache->stat.writeblocks;
 703 
 704    fprintf (outputfile, "%scache requests:             %6d\n", prefix, reqs);
 705    if (reqs == 0) {
 706       return;
 707    }
 708 
 709    fprintf (outputfile, "%scache read requests:        %6d  \t%6.4f\n", prefix, cache->stat.reads, ((double) cache->stat.reads / (double) reqs));
 710 
 711    if (cache->stat.reads) {
 712      fprintf(outputfile, "%scache blocks read:           %6d  \t%6.4f\n", prefix, cache->stat.readblocks, ((double) cache->stat.readblocks / (double) blocks));
 713      fprintf(outputfile, "%scache read misses:          %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.readmisses, ((double) cache->stat.readmisses / (double) reqs), ((double) cache->stat.readmisses / (double) cache->stat.reads));
 714     
 715  fprintf(outputfile, "%scache read full hits:       %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.readhitsfull, ((double) cache->stat.readhitsfull / (double) reqs), ((double) cache->stat.readhitsfull / (double) cache->stat.reads));
 716 
 717      fprintf(outputfile, "%scache population writes:         %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.popwrites, ((double) cache->stat.popwrites / (double) reqs), ((double) cache->stat.popwrites / (double) cache->stat.reads));
 718 
 719      fprintf(outputfile, "%scache block population writes:    %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.popwriteblocks, ((double) cache->stat.popwriteblocks / (double) blocks), ((double) cache->stat.popwriteblocks / (double) cache->stat.readblocks));
 720    }
 721 
 722    fprintf(outputfile, "%scache write requests:       %6d  \t%6.4f\n", prefix, cache->stat.writes, ((double) cache->stat.writes / (double) reqs));
 723 
 724    if (cache->stat.writes) {
 725      fprintf(outputfile, "%scache blocks written:        %6d  \t%6.4f\n", prefix, cache->stat.writeblocks, ((double) cache->stat.writeblocks / (double) blocks));
 726      fprintf(outputfile, "%scache write misses:         %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.writemisses, ((double) cache->stat.writemisses / (double) reqs), ((double) cache->stat.writemisses / (double) cache->stat.writes));
 727 
 728       fprintf(outputfile, "%scache full write hits:   %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.writehitsfull, ((double) cache->stat.writehitsfull / (double) reqs), ((double) cache->stat.writehitsfull / (double) cache->stat.writes));
 729 
 730       fprintf(outputfile, "%scache destage pre-reads:     %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.destagereads, ((double) cache->stat.destagereads / (double) reqs), ((double) cache->stat.destagereads / (double) cache->stat.writes));
 731       fprintf(outputfile, "%scache block destage pre-reads: %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.destagereadblocks, ((double) cache->stat.destagereadblocks / (double) blocks), ((double) cache->stat.destagereadblocks / (double) cache->stat.writeblocks));
 732 
 733       fprintf(outputfile, "%scache destages (write):     %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.destagewrites, ((double) cache->stat.destagewrites / (double) reqs), ((double) cache->stat.destagewrites / (double) cache->stat.writes));
 734 
 735       fprintf(outputfile, "%scache block destages (write): %6d  \t%6.4f  \t%6.4f\n", prefix, cache->stat.destagewriteblocks, ((double) cache->stat.destagewriteblocks / (double) blocks), ((double) cache->stat.destagewriteblocks / (double) cache->stat.writeblocks));
 736 
 737       fprintf(outputfile, "%scache end dirty blocks:      %6d  \t%6.4f\n", prefix, cachedev_count_dirty_blocks(cache), ((double) cachedev_count_dirty_blocks(cache) / (double) cache->stat.writeblocks));
 738    }
 739 
 740    fprintf (outputfile, "%scache bufferspace use end:             %6d\n", prefix, cache->bufferspace);
 741 
 742    fprintf (outputfile, "%scache bufferspace use max:             %6d\n", prefix, cache->stat.maxbufferspace);
 743 }
