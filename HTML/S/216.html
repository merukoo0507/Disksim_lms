 112 static double synthio_get_uniform (synthio_distr *fromdistr)
 113 {
 114    return(((fromdistr->var - fromdistr->base) * DISKSIM_drand48()) + fromdistr->base);
 115 }

 118 static double synthio_get_normal (synthio_distr *fromdistr)
 119 {
 120    double y1, y2;
 121    double y = 0;
 122 
 123    while (y <= 0.0) {
 124       y2 = - log((double) 1.0 - DISKSIM_drand48());
 125       y1 = - log((double) 1.0 - DISKSIM_drand48());
 126       y = y2 - ((y1 - (double) 1.0) * (y1 - (double) 1.0)) / 2;
 127    }
 128    if (DISKSIM_drand48() < 0.5) {
 129       y1 = -y1;
 130    }
 131    return((fromdistr->var * y1) + fromdistr->mean);
 132 }

 135 static double synthio_get_exponential (synthio_distr *fromdistr)
 136 {
 137    double dtmp;
 138 
 139    dtmp = log((double) 1.0 - DISKSIM_drand48());
 140    return((fromdistr->base - (fromdistr->mean * dtmp)));
 141 }

 144 static double synthio_get_poisson (synthio_distr *fromdistr)
 145 {
 146    double dtmp = 1.0;
 147    int count = 0;
 148    double stop;
 149 
 150    stop = exp(-fromdistr->mean);
 151    while (dtmp >= stop) {
 152       dtmp *= DISKSIM_drand48();
 153       count++;
 154    }
 155    count--;
 156    return((double) count + fromdistr->base);
 157 }

 160 static double synthio_get_twovalue (synthio_distr *fromdistr)
 161 {
 162    if (DISKSIM_drand48() < fromdistr->var) {
 163       return(fromdistr->mean);
 164    } else {
 165       return(fromdistr->base);
 166    }
 167 }

 170 static double synthio_getrand (synthio_distr *fromdistr)
 171 {
 172    switch (fromdistr->type) {
 173       case SYNTHIO_UNIFORM:
 174                         return(synthio_get_uniform(fromdistr));
 175       case SYNTHIO_NORMAL:
 176                         return(synthio_get_normal(fromdistr));
 177       case SYNTHIO_EXPONENTIAL:
 178                         return(synthio_get_exponential(fromdistr));
 179       case SYNTHIO_POISSON:
 180                         return(synthio_get_poisson(fromdistr));
 181       case SYNTHIO_TWOVALUE:
 182                         return(synthio_get_twovalue(fromdistr));
 183       default:
 184                         fprintf(stderr, "Unrecognized distribution type - %d\n", fromdistr->type);
 185                         exit(1);
 186    }
 187 }

 190 static void synthio_appendio (process *procp, ioreq_event *tmp)
 191 {
 192    synthio_generator *gen = (synthio_generator *) procp->space;
 193    sleep_event *limittmp = NULL;
 194    event *prev = NULL;
 195    sleep_event *newsleep = NULL;
 196    sleep_event *callsleep = NULL;
 197    sleep_event *retsleep = NULL;
 198    ioreq_event *new;
 199    int blocksize;
 200    int gennum;
 201 
 202    blocksize = gen->blocksize;
 203    gennum = gen->number;
 204    new = (ioreq_event *) getfromextraq();
 205    new->type = IOREQ_EVENT;
 206    new->time = tmp->time;
 207    new->devno = tmp->devno;
 208    new->blkno = tmp->blkno * blocksize;
 209    new->bcount = tmp->bcount * blocksize;
 210    new->flags = tmp->flags;
 211    new->cause = tmp->cause;
 212    new->opid = (synthio_endiocnt * gennum) + synthio_iocnt;
 213    /* this is being considered "ok" under the assumption that opid will */
 214    /* never exceed 2^32.....                                            */
 215    new->buf = (void *) new->opid;
 216    if(synthio_syscalls){
 217       retsleep = (sleep_event *) getfromextraq();
 218       retsleep->type = SLEEP_EVENT;
 219       retsleep->time = synthio_sysret_time;
 220       retsleep->chan = new->buf;
 221       retsleep->iosleep = 1;
 222       callsleep = (sleep_event *) getfromextraq();
 223       callsleep->type = SLEEP_EVENT;
 224       callsleep->time = synthio_syscall_time;
 225       callsleep->chan = new->buf;
 226       callsleep->iosleep = 1;
 227    }
 228    if (new->flags & (TIME_CRITICAL|TIME_LIMITED)) {
 229       newsleep = (sleep_event *) getfromextraq();
 230       newsleep->type = SLEEP_EVENT;
 231       newsleep->time = 0.0;
 232       newsleep->chan = new->buf;
 233       newsleep->iosleep = 1;
 234       if (new->flags & TIME_CRITICAL) {
 235         if(synthio_syscalls){
 236           retsleep->next = procp->eventlist;
 237           new->next = (ioreq_event *) retsleep;
 238           callsleep->next = (event *) new;
 239           new = (ioreq_event *) callsleep;
 240           newsleep = NULL;
 241         }else{
 242           newsleep->next = procp->eventlist;
 243           new->next = (ioreq_event *) newsleep;
 244         }
 245       } else {
 246         new->next = (ioreq_event *) procp->eventlist;
 247         newsleep->time = -1.0;
 248         while (newsleep->time < 0.0) {
 249           newsleep->time = synthio_getrand (&gen->tmlimit);
 250         }
 251         newsleep->time += new->time;
 252         limittmp = gen->limits;
 253         if ((limittmp == NULL) || (newsleep->time < limittmp->time)) {
 254           newsleep->next = (event *) gen->limits;
 255           gen->limits = (sleep_event *) newsleep;
 256           if (limittmp) {
 257             limittmp->time -= newsleep->time;
 258           }
 259         } else {
 260           newsleep->time -= limittmp->time;
 261           while ((limittmp->next) && (newsleep->time >= limittmp->next->time)) {
 262             limittmp = (sleep_event *) limittmp->next;
 263             newsleep->time -= limittmp->time;
 264           }
 265           newsleep->next = limittmp->next;
 266           limittmp->next = (event *) newsleep;
 267           if (newsleep->next) {
 268             newsleep->next->time -= newsleep->time;
 269           }
 270         }
 271         newsleep = NULL;
 272       }
 273    } else {
 274      new->next = (ioreq_event *) procp->eventlist;
 275    }
 276    
 277    procp->eventlist = (event *) new;
 278 
 279 synthio_check_genlimits:
 280    limittmp = gen->limits;
 281    while ((limittmp) && (limittmp->time < new->time)) {
 282       gen->limits = (sleep_event *) limittmp->next;
 283       new->time -= limittmp->time;
 284       limittmp->next = (event *) new;
 285       if (prev) {
 286          prev->next = (event *) limittmp;
 287       } else {
 288          procp->eventlist = (event *) limittmp;
 289       }
 290       prev = (event *) limittmp;
 291       limittmp = gen->limits;
 292    }
 293 
 294    if ((newsleep) && (gen->limits)) {
 295       prev = (event *) new;
 296       new = (ioreq_event *) newsleep;
 297       newsleep = NULL;
 298       goto synthio_check_genlimits;
 299    }
 300 
 301 
 302    //fprintf (stderr, "New request %d, time %f, devno %d, blkno %d, bcount %d, flags %x\n", synthio_iocnt, new->time, new->devno, new->blkno, new->bcount, new->flags);
 303 }

 306 void synthio_initialize_generator (process *procp)
 307 {
 308    ioreq_event *tmp;
 309    synthio_generator *gen;
 310    event *evptr;
 311    double reqclass;
 312 
 313    evptr = getfromextraq();
 314    evptr->time = 0.0;
 315    evptr->type = SYNTHIO_EVENT;
 316    evptr->next = NULL;
 317    procp->eventlist = evptr;
 318    gen = (synthio_generator *) procp->space;
 319    if (gen == NULL) {
 320       fprintf(stderr, "Process with no synthetic generator in synthio_initialize\n");
 321       exit(1);
 322    }
 323    tmp = (ioreq_event *) getfromextraq();
 324    tmp->time = -1.0;
 325    while (tmp->time < 0.0) {
 326       tmp->time = synthio_getrand(&gen->genintr);
 327    }
 328    tmp->flags = 0;
 329    tmp->cause = gen->number;
 330    tmp->devno = gen->devno[(int) (DISKSIM_drand48() * (double) gen->numdisks)];
 331    tmp->blkno = tmp->bcount = gen->blksperdisk;
 332    while (((tmp->blkno + tmp->bcount) >= gen->blksperdisk) || (tmp->bcount == 0)) {
 333       tmp->blkno = (int) (DISKSIM_drand48() * (double) gen->blksperdisk);
 334       tmp->bcount = ((int) synthio_getrand(&gen->sizedist) + gen->blocksize - 1) / gen->blocksize;
 335    }
 336    if (DISKSIM_drand48() < gen->probread) {
 337       tmp->flags |= READ;
 338    }
 339    reqclass = DISKSIM_drand48() - gen->probtmcrit;
 340    if (reqclass < 0.0) {
 341       tmp->flags |= TIME_CRITICAL;
 342    } else if (reqclass < gen->probtmlim) {
 343       tmp->flags |= TIME_LIMITED;
 344    }
 345    gen->pendio = tmp;
 346    synthio_appendio(procp, tmp);
 347 
 348 fprintf (outputfile, "Initialized synthio process #%d, first event at time %f\n", procp->pid, procp->eventlist->time);
 349 
 350 }

 353 static int synthio_generatenextio (synthio_generator *gen)
 354 {
 355    double type;
 356    double reqclass;
 357    int blkno;
 358    ioreq_event *tmp;
 359 
 360    if ((simtime >= synthio_endtime) || (synthio_iocnt >= synthio_endiocnt))
 361       return 0;
 362 
 363    tmp = gen->pendio;
 364    if (gen->tracefile) {
 365       gen->pendio = 
 366         (ioreq_event *)iotrace_get_ioreq_event(gen->tracefile, 
 367                                                disksim->traceformat, 
 368                                                tmp);
 369       if (gen->pendio) {
 370          tmp->cause = gen->number;
 371          /* tmp->time = 0; */
 372       } else {
 373          fprintf(stderr, "Returning NULL event in synthio_generatenextio\n");
 374          return 0;
 375       }
 376       return 1;
 377    }
 378 
 379    type = DISKSIM_drand48();
 380 
 381    if ((type < gen->probseq) && 
 382        ((tmp->blkno + 2*tmp->bcount) < gen->blksperdisk)) {
 383 
 384      tmp->time = -1.0;
 385 
 386      while (tmp->time < 0.0) {
 387        tmp->time = synthio_getrand(&gen->seqintr);
 388      }
 389 
 390      tmp->flags = SEQ | (tmp->flags & READ);
 391      tmp->cause = gen->number;
 392      tmp->blkno += tmp->bcount;
 393    } 
 394    else if ((type < (gen->probseq + gen->probloc)) && 
 395             (type >= gen->probseq)) {
 396 
 397      tmp->time = -1.0;
 398      while (tmp->time < 0.0) {
 399        tmp->time = synthio_getrand(&gen->locintr);
 400      }
 401      tmp->flags = LOCAL;
 402      tmp->cause = gen->number;
 403      blkno = gen->blksperdisk;
 404      while (((blkno + tmp->bcount) >= gen->blksperdisk) 
 405             || (blkno < 0) 
 406             || (tmp->bcount <= 0)) 
 407        {
 408          blkno = tmp->blkno + 
 409            (int)synthio_getrand(&gen->locdist) / gen->blocksize;
 410          tmp->bcount = ((int) synthio_getrand(&gen->sizedist) + 
 411                         gen->blocksize - 1) / gen->blocksize;
 412        }
 413      tmp->blkno = blkno;
 414      if (DISKSIM_drand48() < gen->probread) {
 415        tmp->flags |= READ;
 416      }
 417    } 
 418    else {
 419       tmp->time = -1.0;
 420       while (tmp->time < 0.0) {
 421         tmp->time = synthio_getrand(&gen->genintr);
 422       }
 423       tmp->flags = 0;
 424       tmp->cause = gen->number;
 425       tmp->devno = gen->devno[(int)(DISKSIM_drand48() * 
 426                                     (double)gen->numdisks)];
 427 
 428       tmp->blkno = tmp->bcount = gen->blksperdisk;
 429       while (((tmp->blkno + tmp->bcount) >= gen->blksperdisk) || 
 430              (tmp->bcount <= 0)) {
 431 
 432         tmp->blkno = (int) (DISKSIM_drand48() * (double)gen->blksperdisk);
 433         tmp->bcount = ((int) synthio_getrand(&gen->sizedist) + 
 434                        gen->blocksize - 1) / gen->blocksize;
 435       }
 436 
 437       if (DISKSIM_drand48() < gen->probread) {
 438         tmp->flags = READ;
 439       }
 440    }
 441    reqclass = DISKSIM_drand48() - gen->probtmcrit;
 442 
 443    if (reqclass < 0.0) {
 444      tmp->flags |= TIME_CRITICAL;
 445    } 
 446    else if (reqclass < gen->probtmlim) {
 447      tmp->flags |= TIME_LIMITED;
 448    }
 449    return 1;
 450 }

 487 static void synthio_read_uniform (FILE *parfile, synthio_distr *newdistr, double mult)
 488 {
 489    newdistr->type = SYNTHIO_UNIFORM;
 490    if (fscanf(parfile, "Minimum value: %lf\n", &newdistr->base) != 1) {
 491       fprintf(stderr, "Error reading 'minimum value'\n");
 492       exit(1);
 493    }
 494    fprintf (outputfile, "Minimum value: %f\n", newdistr->base);
 495    newdistr->base *= mult;
 496    if (fscanf(parfile, "Maximum value: %lf\n", &newdistr->var) != 1) {
 497       fprintf(stderr, "Error reading 'maximum value'\n");
 498       exit(1);
 499    }
 500    fprintf (outputfile, "Maximum value: %f\n", newdistr->var);
 501    newdistr->var *= mult;
 502    if (newdistr->var <= newdistr->base) {
 503       fprintf(stderr, "'maximum value' is not greater than 'minimum value' - %f <= %f\n", newdistr->var, newdistr->base);
 504       exit(1);
 505    }
 506 }

 509 static void synthio_read_normal (FILE *parfile, synthio_distr *newdistr, double mult)
 510 {
 511    newdistr->type = SYNTHIO_NORMAL;
 512    if (fscanf(parfile, "Mean: %lf\n", &newdistr->mean) != 1) {
 513       fprintf(stderr, "Error reading normal mean\n");
 514       exit(1);
 515    }
 516    fprintf (outputfile, "Mean: %f\n", newdistr->mean);
 517    newdistr->mean *= mult;
 518    if (fscanf(parfile, "Variance: %lf\n", &newdistr->var) != 1) {
 519       fprintf(stderr, "Error reading normal variance\n");
 520       exit(1);
 521    }
 522    fprintf (outputfile, "Variance: %f\n", newdistr->var);
 523    if (newdistr->var < 0) {
 524       fprintf(stderr, "Invalid value for 'variance' - %f\n", newdistr->var);
 525       exit(1);
 526    }
 527    newdistr->var = mult * sqrt(newdistr->var);
 528 }

 567 static void synthio_read_twovalue (FILE *parfile, synthio_distr *newdistr, double mult)
 568 {
 569    newdistr->type = SYNTHIO_TWOVALUE;
 570    if (fscanf(parfile, "Default value: %lf\n", &newdistr->base) != 1) {
 571       fprintf(stderr, "Error reading 'default value'\n");
 572       exit(1);
 573    }
 574    fprintf(outputfile, "Default value: %f\n", newdistr->base);
 575    newdistr->base *= mult;
 576    if (fscanf(parfile, "Second value: %lf\n", &newdistr->mean) != 1) {
 577       fprintf(stderr, "Error reading 'second value'\n");
 578       exit(1);
 579    }
 580    fprintf(outputfile, "Second value: %f\n", newdistr->mean);
 581    newdistr->mean *= mult;
 582    if (fscanf(parfile, "Probability of second value: %lf\n", &newdistr->var) != 1) {
 583       fprintf(stderr, "Error reading 'probability of second value'\n");
 584       exit(1);
 585    }
 586    if ((newdistr->var < 0) || (newdistr->var > 1)) {
 587       fprintf(stderr, "Invalid probability of second value - %f\n", newdistr->var);
 588       exit(1);

 594 static void synthio_read_distr (FILE *parfile, synthio_distr *newdistr, double mult)
 595 {
 596    char type[80];
 597 
 598    if (fscanf(parfile, "Type of distribution: %s\n", type) != 1) {
 599       fprintf(stderr, "Error reading 'type of distribution'\n");
 600       exit(1);
 601    }
 602    fprintf (outputfile, "Type of distribution: %s\n", type);
 603    switch (type[0]) {
 604       case 'u':
 605                 synthio_read_uniform(parfile, newdistr, mult);
 606                 break;
 607       case 'n':
 608                 synthio_read_normal(parfile, newdistr, mult);
 609                 break;
 610       case 'e':
 611                 synthio_read_exponential(parfile, newdistr, mult);
 612                 break;
 613       case 'p':
 614                 synthio_read_poisson(parfile, newdistr, mult);
 615                 break;
 616       case 't':
 617                 synthio_read_twovalue(parfile, newdistr, mult);
 618                 break;
 619       default:
 620                 fprintf(stderr, "Invalid value for 'type of distribution' - %s\n", type);
 621                 exit(1);
 622    }
 623 
 624 }
