 133 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 134  *
 135  *  mems_media_access_time()
 136  *
 137  *  returns the time it takes to access the media between
 138  *  two coordinates.  this is called to find out how long
 139  *  a read or write will take.  it assumes that the device
 140  *  can read and write at the same rate.
 141  *
 142  *  inputs:
 143  *    mems_sled *sled - a pointer to the sled structure
 144  *    coord_t *start - the starting coordinate
 145  *    coord_t *end - the destination coordinate
 146  *
 147  *  outputs:
 148  *    double - the amount of time in milliseconds to access
 149  *             data between the two points.
 150  *
 151  *  modifies:
 152  *    nothing
 153  *
 154  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 155 
 156 double
 157 mems_media_access_time (mems_sled_t *sled, coord_t *start, coord_t *end)
 158 {
 159   double time;          /* seconds */
 160   double velocity;      /* nanometers per second */
 161   double distance;      /* nanometers */
 162 
 163   velocity = (double)(sled->y_access_speed_bit_s * sled->bit_length_nm);
 164   distance = (fabs ((double)(start->y_pos - end->y_pos))) * sled->bit_length_nm;
 165 
 166   time = distance / velocity;
 167 
 168   /* convert to milliseconds */
 169   time *= 1000.0;
 170 
 171   /*
 172   fprintf (stderr, "access: time = %f ms, dist = %f nm, vel = %f nm/s\n", time, distance, velocity);
 173   */
 174 
 175   return (time);
 176 }

 261 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 262  *
 263  *  find_settle_time()
 264  *
 265  *  finds the settle time, which is given thusly:
 266  *
 267  *  settle time = num_time_constants * (1 / (2 pi * resonant frequency))
 268  *
 269  *  in this version of the model, the settle time is _not_
 270  *  dependent on the seek distance or the spring_factor
 271  *
 272  *  inputs:
 273  *    double num_time_constants - the number of time constants to
 274  *           add to a seek
 275  *    double resonant_frequency - resonant frequency of the sled
 276  *
 277  *  outputs:
 278  *    double - the settle time
 279  *
 280  *  modifies:
 281  *    nothing
 282  *
 283  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 284 
 285 double
 286 find_settle_time(double num_time_constants, double resonant_frequency) {
 287 
 288   return (num_time_constants * (1.0 / ((6.2831853) * (resonant_frequency))));
 289 
 290 }

 292 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 293  *
 294  *  find_turnaround_time()
 295  *
 296  *  finds the turnaround time.  this is a function of the sled velocity,
 297  *  the actuator acceleration and the spring forces.  that makes it also
 298  *  a function of position (offset).
 299  *
 300  *  inputs:
 301  *    double offset_nm - the offset at which the turnaround occurs
 302  *    double velocity_nm_s - the sled velocity
 303  *    double base_accel_nm_s_s - the actuator acceleration
 304  *    double spring_factor - the spring_factor
 305  *    double length_nm - the length of the square
 306  *
 307  *  outputs:
 308  *    double - the turnaround time
 309  *
 310  *  modifies:
 311  *    nothing
 312  *
 313  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 314 
 315 double
 316 find_turnaround_time(double offset_nm, double velocity_nm_s, double base_accel_nm_s_s,
 317                      double spring_factor, double length_nm) {
 318   
 319   double turnaround_time;
 320   double spring_accel;
 321   
 322   if (_VERBOSE_) {
 323     fprintf(__OUTPUTFILE__, "find_turnaround_time::  entry\n");
 324     
 325     fprintf(__OUTPUTFILE__, "find_turnaround_time::  offset_nm = %f, length_nm = %f\n",
 326             offset_nm, length_nm);
 327   }
 328 
 329   /*  scale the base acceleration by the spring_factor and the offset  */
 330   spring_accel = (base_accel_nm_s_s * (spring_factor * (fabs(offset_nm) / (length_nm / 2.0))));
 331   
 332   if ( ((offset_nm < 0.0) && (velocity_nm_s > 0.0)) ||
 333        ((offset_nm > 0.0) && (velocity_nm_s < 0.0))
 334        )
 335     {
 336       /*  the springs will hurt my acceleration  */
 337       spring_accel = -spring_accel;
 338     }
 339 
 340   /*  turnaround time is the time to stop the sled plus the time to accelerate it back to
 341       velocity.  these are equal, so it's just twice the time to stop.  */
 342   turnaround_time = 2.0 * (fabs(velocity_nm_s)) / (base_accel_nm_s_s + spring_accel);
 343 
 344   if (_VERBOSE_) {
 345     fprintf(__OUTPUTFILE__, "find_turnaround_time::  velocity_nm_s = %f, base_accel_nm_s_s = %f, spring_accel = %f\n",
 346             velocity_nm_s, base_accel_nm_s_s, spring_accel);
 347     
 348     fprintf(__OUTPUTFILE__, "find_turnaround_time::  turnaround_time = %f\n", turnaround_time);
 349   }
 350 
 351   return turnaround_time;
 352 }

 354 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 355  *
 356  *  mems_seek_time()
 357  *
 358  *  this function finds the seek time from point to point,
 359  *  taking into account all turnarounds that might have to
 360  *  take place.  it also returns some useful statistics
 361  *  that may be recorded by the calling function.
 362  *
 363  *  inputs:
 364  *    mems_sled *sled - a pointer to the sled structure
 365  *    coord_t *begin - the beginning coordinate
 366  *    coord_t *end - the end coordinate
 367  *    double *return_x_seek_time - a pointer into which to return the x seek time
 368  *    double *return_y_seek_time - a pointer into which to return the y seek time
 369  *    double *return_turnaround_time - a pointer into which to return the turnaround time
 370  *    double *return_turnaround_number - a pointer into which to return the number of turnarounds
 371  *
 372  *  outputs:
 373  *    double - the total seek time
 374  *
 375  *  modifies:
 376  *    return_x_seek_time
 377  *    return_y_seek_time
 378  *    return_turnaround_time
 379  *    return_turnaround_number
 380  *
 381  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 382 
 383 double
 384 mems_seek_time (mems_sled_t *sled,
 385                 coord_t *begin, coord_t *end,
 386                 double *return_x_seek_time, double *return_y_seek_time,
 387                 double *return_turnaround_time, int *return_turnaround_number)
 388 {
 389   double seek_time_x=0.0;       /* second */
 390   double seek_time_y=0.0;       /* second */
 391   double dist_x_nm=0.0;         
 392   double dist_y_nm=0.0;         
 393   int dist_y_equiv_nm=0.0;      /* used for equivalent distance calc */
 394   double x_accel=0.0;           /* nanometer/sec/sec */
 395   double y_accel=0.0;           /* nanometer/sec/sec */
 396   double bit_width=0.0;         /* nanometer */
 397   double turnaround_time=0.0;   /* second */
 398   double settling_time_x=0.0;   /* second */
 399   int y_access_speed_nm_s;
 400   double total_time;            /* second */
 401 
 402   int number_of_turns;
 403 
 404   double spring_factor = 0.0;
 405   int x_length_nm = 0;
 406   int y_length_nm = 0;
 407 
 408   /* sled = &dev->sled[sledno]; */
 409   
 410   /*
 411   fprintf(__OUTPUTFILE__, "\nmemsdevice_seek_time::  begin->x_pos = %d, begin->y_pos = %d, begin->y_vel = %d\n",
 412           begin->x_pos, begin->y_pos, begin->y_vel);
 413   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  end->x_pos   = %d, end->y_pos = %d, end->y_vel = %d\n",
 414           end->x_pos, end->y_pos, end->y_vel);
 415   */
 416 
 417   /*  first check to see if begin == end  */
 418   if (mems_equal_coords(begin, end)) {
 419     //    fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  begin == end!\n");
 420     // fprintf(stderr, "memsdevice_seek_time::  begin == end!\n");
 421 
 422     if (return_turnaround_time != NULL) {
 423       *return_turnaround_time = 0.0;
 424     }
 425     
 426     if (return_turnaround_number != NULL) {
 427       *return_turnaround_number = 0;
 428     }
 429     
 430     if (return_x_seek_time != NULL) {
 431       *return_x_seek_time = 0.0;
 432     }
 433     
 434     if (return_y_seek_time != NULL) {
 435       *return_y_seek_time = 0.0;
 436     }
 437 
 438     return 0.0;
 439   }
 440 
 441   /*  initialize some variables to values given in the sled structure  */
 442   
 443   x_accel = (double)sled->x_accel_nm_s2;
 444   y_accel = (double)sled->y_accel_nm_s2;
 445   bit_width = (double)sled->bit_length_nm;
 446   y_access_speed_nm_s = sled->y_access_speed_bit_s * sled->bit_length_nm;
 447   x_length_nm = sled->x_length_nm;
 448   y_length_nm = sled->y_length_nm;
 449   spring_factor = sled->spring_factor;
 450 
 451   /*
 452   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  begin->x_pos = %d, bit_width = %f, x_begin_nm = %f\n",
 453           begin->x_pos, bit_width, x_begin_nm);
 454   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  end->x_pos = %d, bit_width = %f, x_end_nm = %f\n",
 455           end->x_pos, bit_width, x_end_nm);
 456   */
 457 
 458   if (end->x_pos == begin->x_pos) {
 459     dist_x_nm = (double)0.0;
 460     seek_time_x = (double)0.0;
 461   } else {
 462 
 463     /*  find the x settle time  */
 464     settling_time_x = find_settle_time(sled->num_time_constants, sled->sled_resonant_freq_hz);
 465 
 466     /*
 467     fprintf(__OUTPUTFILE__, "settling_time_x = %f\n", settling_time_x);
 468     */
 469 
 470     /*
 471     fprintf(__OUTPUTFILE__, "memsdevice_seek_time:: find_seek_time_piecewise - x_begin_nm = %f, x_end_nm = %f\n",
 472             x_begin_nm, x_end_nm);
 473     */
 474 
 475     /*  find the base seek time in x  */
 476 
 477     if (sled->dev->seek_function == MEMS_SEEK_HONG) {
 478       seek_time_x = find_seek_time_hong_x((begin->x_pos * bit_width),
 479                                           (end->x_pos * bit_width),
 480                                           spring_factor, x_accel,
 481                                           x_length_nm)
 482         +
 483         settling_time_x;
 484     } else {
 485       if (sled->dev->precompute_seek_count > 0) {
 486         seek_time_x =
 487           mems_find_precomputed_seek_time(sled,
 488                                           (begin->x_pos * bit_width), (end->x_pos * bit_width),
 489                                           _X_SEEK_)
 490           +
 491           settling_time_x;
 492       } else {
 493         seek_time_x =
 494           find_seek_time_piecewise((begin->x_pos * bit_width), (end->x_pos * bit_width),
 495                                    spring_factor, x_accel,
 496                                    x_length_nm, 0.0)
 497           +
 498           settling_time_x;
 499       }
 500     }
 501   }
 502   /* Now calculate seek time in Y direction.
 503    * First, general values needed in the calculations.
 504    *
 505    * there are eight cases to account for:
 506    *
 507    * begin = current velocity; end = new velocity;
 508    * below = new position BELOW current position;
 509    * above = new position ABOVE current position
 510    *
 511    *    begin UP        end UP          below   (1) turn, seek, turn
 512    *                                    above   (2) seek
 513    *
 514    *                    end DOWN        below   (3) turn, seek
 515    *                                    above   (4) seek, turn
 516    *
 517    *    begin DOWN      end UP          below   (5) seek, turn
 518    *                                    above   (6) turn, seek
 519    *
 520    *                    end DOWN        below   (7) seek
 521    *                                    above   (8) turn, seek, turn
 522    */
 523 
 524   /*
 525   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  begin->y_vel = %d, end->y_vel = %d\n",
 526           begin->y_vel, end->y_vel);
 527 
 528   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  begin->y_pos = %d, end->y1_pos = %d\n",
 529           begin->y_pos, end->y1_pos);
 530 
 531   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  end->y1_pos = %d, end->y2_pos = %d\n",
 532           end->y1_pos, end->y2_pos);
 533   */
 534 
 535   turnaround_time = 0.0;
 536   number_of_turns = 0;
 537 
 538   if (begin->y_vel >= 0) {
 539     if (end->y_vel > 0) {       /* final direction == begining direction */
 540       if (begin->y_pos >= end->y_pos) {
 541         /* case (1) */
 542 
 543         turnaround_time = find_turnaround_time((begin->y_pos * bit_width),
 544                                                begin->y_vel,
 545                                                y_accel,
 546                                                spring_factor, y_length_nm);
 547         turnaround_time += find_turnaround_time((end->y_pos * bit_width),
 548                                                 end->y_vel,
 549                                                 y_accel,
 550                                                 spring_factor, y_length_nm);
 551         number_of_turns = 2;
 552 
 553         /*
 554         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 1 - turnaround_time = %f\n",
 555                 turnaround_time);
 556         */
 557       } else {                                  /* no need to turn around */
 558         /* case (2) */
 559 
 560         number_of_turns = 0;
 561 
 562         /*
 563         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 2 - turnaround_time = %f\n",
 564                 turnaround_time);
 565         */
 566       }
 567     } else {                    /* final direction != begining direction */
 568       if (begin->y_pos >= end->y_pos) {
 569         /* case (3) */
 570 
 571         turnaround_time = find_turnaround_time((begin->y_pos * bit_width),
 572                                                begin->y_vel,
 573                                                y_accel,
 574                                                spring_factor, y_length_nm);
 575         number_of_turns = 1;
 576 
 577         /*
 578         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 3 - turnaround_time = %f\n",
 579                 turnaround_time);
 580         */
 581       } else {
 582         /* case (4) */
 583 
 584         turnaround_time = find_turnaround_time((end->y_pos * bit_width),
 585                                                end->y_vel,
 586                                                y_accel,
 587                                                spring_factor, y_length_nm);
 588         number_of_turns = 1;
 589 
 590         /*
 591         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 4 - turnaround_time = %f\n",
 592                 turnaround_time);
 593         */
 594       }
 595     }
 596   } else {      /* begin.y_vel < 0.0 */
 597     if (end->y_vel > 0) {       /* final direction != begining direction */
 598       if (begin->y_pos >= end->y_pos) {
 599         /* case (5) */
 600 
 601         turnaround_time = find_turnaround_time((end->y_pos * bit_width),
 602                                                end->y_vel,
 603                                                y_accel,
 604                                                spring_factor, y_length_nm);
 605         number_of_turns = 1;
 606 
 607         /*
 608         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 5 - turnaround_time = %f\n",
 609                 turnaround_time);
 610         */
 611       } else {
 612         /* case (6) */
 613         turnaround_time = find_turnaround_time((begin->y_pos * bit_width),
 614                                                begin->y_vel,
 615                                                y_accel,
 616                                                spring_factor, y_length_nm);
 617         number_of_turns = 1;
 618 
 619         /*
 620         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 6 - turnaround_time = %f\n",
 621                 turnaround_time);
 622         */
 623       }
 624     } else {                    /* final direction == begining direction */
 625       if (begin->y_pos >= end->y_pos) { /* no need to turn around */
 626         /* case (7) */
 627 
 628         number_of_turns = 0;
 629 
 630         /*
 631         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 7 - turnaround_time = %f\n",
 632                 turnaround_time);
 633         */
 634         dist_y_equiv_nm = 0;
 635       } else {
 636         /* case (8) */
 637 
 638         turnaround_time = find_turnaround_time((begin->y_pos * bit_width),
 639                                                begin->y_vel,
 640                                                y_accel,
 641                                                spring_factor, y_length_nm);
 642         turnaround_time += find_turnaround_time((end->y_pos * bit_width),
 643                                                 end->y_vel,
 644                                                 y_accel,
 645                                                 spring_factor, y_length_nm);
 646         number_of_turns = 2;
 647 
 648         /*
 649         fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  finding turnarounds - case 8 - turnaround_time = %f\n",
 650                 turnaround_time);
 651         */
 652       }
 653     }
 654   }
 655   
 656   dist_y_nm = (double)abs(end->y_pos - begin->y_pos) * sled->bit_length_nm;
 657   
 658   /*
 659   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  begin->y_pos = %d, bit_width = %f, y_begin_nm = %f\n",
 660           begin->y_pos, bit_width, y_begin_nm);
 661   fprintf(__OUTPUTFILE__, "memsdevice_seek_time::  end->y1_pos = %d, bit_width = %f, y_end_nm = %f\n",
 662           end->y1_pos, bit_width, y_end_nm);
 663   
 664   fprintf(__OUTPUTFILE__, "memsdevice_seek_time:: find_seek_time_piecewise - y_begin_nm = %f, y_end_nm = %f\n",
 665           y_begin_nm, y_end_nm);
 666   */
 667 
 668   if (sled->dev->seek_function == MEMS_SEEK_HONG) {
 669     seek_time_y =
 670       find_seek_time_hong_y((begin->y_pos * bit_width),
 671                             (end->y_pos * bit_width),
 672                             spring_factor, y_accel,
 673                             y_length_nm, y_access_speed_nm_s)
 674       +
 675       turnaround_time;
 676   } else {
 677     if (sled->dev->precompute_seek_count > 0) {
 678       seek_time_y =
 679         mems_find_precomputed_seek_time(sled,
 680                                         (begin->y_pos * bit_width), (end->y_pos * bit_width),
 681                                         _Y_SEEK_)
 682         +
 683         turnaround_time;
 684     } else {
 685       seek_time_y =
 686         find_seek_time_piecewise((begin->y_pos * bit_width), (end->y_pos * bit_width),
 687                                  spring_factor, y_accel,
 688                                  y_length_nm, y_access_speed_nm_s)
 689         + turnaround_time;
 690     }
 691   }
 692 
 693   /*
 694   if (distance != NULL) {
 695     if (distance_direction == 1) {
 696       *distance = (int)sqrt(dist_x_nm * dist_x_nm + ((dist_y_nm + dist_y_equiv_nm) * (dist_y_nm + dist_y_equiv_nm)));
 697     } else {
 698       *distance = (int)sqrt(dist_x_nm * dist_x_nm + dist_y_nm * dist_y_nm);
 699     }
 700     // fprintf (__OUTPUTFILE__, "dist_x = %f, dist_y = %f; dist = %d\n", dist_x_nm, dist_y_nm, *distance);
 701   }
 702   */
 703 
 704   if (return_turnaround_time != NULL) {
 705     *return_turnaround_time = (turnaround_time) * 1000.0;
 706   }
 707 
 708   if (return_turnaround_number != NULL) {
 709     *return_turnaround_number = number_of_turns;
 710   }
 711 
 712   if (return_x_seek_time != NULL) {
 713     *return_x_seek_time = seek_time_x * 1000.0;
 714   }
 715 
 716   if (return_y_seek_time != NULL) {
 717     *return_y_seek_time = seek_time_y * 1000.0;
 718   }
 719 
 720   /*
 721   fprintf (stderr, "turnaround_time = %f, number_of_turns = %d\n",
 722            (turnaround_time * 1000.0), number_of_turns);
 723   */
 724 
 725   if (_VERBOSE_) {
 726     printf("dist_x = %f; dist_y = %f; seek_time_x = %f; seek_time_y = %f\n",
 727            dist_x_nm, dist_y_nm, seek_time_x, seek_time_y);
 728   }
 729 
 730   /*  total seek time is the max of the seek time in the two dimensions  */
 731   total_time = max(seek_time_x, seek_time_y);
 732 
 733   /*  convert to milliseconds  */
 734   return (total_time * (double)1000.0);
 735 }

 737 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 738  *  This implements the seek cache.
 739  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 740 
 741 double
 742 mems_seek_time_seekcache (mems_sled_t *sled,
 743                           coord_t *begin, 
 744                           coord_t *end,
 745                           double *return_x_seek_time,
 746                           double *return_y_seek_time,
 747                           double *return_turnaround_time,
 748                           int *return_turnaround_number)
 749 {
 750   struct mems_seekcache *s;
 751   int i;
 752 
 753   /* First check if value is cached */
 754   for (i = 0; i < MEMS_SEEKCACHE; i++) {
 755     s = &sled->seekcache[i];
 756     if (mems_equal_coords(begin, &s->begin) &&
 757         mems_equal_coords(end, &s->end)) {
 758       if (return_x_seek_time) *return_x_seek_time = s->x_seek_time;
 759       if (return_y_seek_time) *return_y_seek_time = s->y_seek_time;
 760       if (return_turnaround_time) *return_turnaround_time = s->turnaround_time;
 761       if (return_turnaround_number) *return_turnaround_number = s->turnaround_number;
 762       return s->time;
 763     }
 764   }
 765 
 766   s = &sled->seekcache[sled->seekcache_next];
 767   mems_coord_t_copy(begin, &s->begin);
 768   mems_coord_t_copy(end, &s->end);
 769   s->time = mems_seek_time(sled, &s->begin, &s->end, &s->x_seek_time, &s->y_seek_time, &s->turnaround_time, &s->turnaround_number);
 770   if (return_x_seek_time) *return_x_seek_time = s->x_seek_time;
 771   if (return_y_seek_time) *return_y_seek_time = s->y_seek_time;
 772   if (return_turnaround_time) *return_turnaround_time = s->turnaround_time;
 773   if (return_turnaround_number) *return_turnaround_number = s->turnaround_number;
 774   sled->seekcache_next++;
 775   if (sled->seekcache_next == MEMS_SEEKCACHE) sled->seekcache_next = 0;
 776   // fprintf(stderr, "mems_seek_time_seekcache::  s->time = %f\n", s->time);
 777   return s->time;
 778 }

 796 double
 797 mems_find_precomputed_seek_time(mems_sled_t *sled,
 798                                 double start_offset_nm, double end_offset_nm,
 799                                 int direction) {
 800 
 801   double seektime = 0.0;
 802   int dist_nm;
 803   int i;    
 804   mems_t *dev = sled->dev;
 805 
 806   dist_nm = (int)find_dist_nm(start_offset_nm, end_offset_nm);
 807 
 808   if (_VERBOSE_) {
 809     fprintf(__OUTPUTFILE__, "mems_find_precomputed_seek_time::  entry - start_offset_nm = %f, end_offset_nm = %f\n",
 810             start_offset_nm, end_offset_nm);
 811     fprintf(__OUTPUTFILE__, "mems_find_precomputed_seek_time::  dist_nm = %d\n",
 812             dist_nm);
 813   }
 814 
 815   // fprintf(stderr, "dist_nm = %d\n", dist_nm);
 816   
 817   // mems_print_precomputed_seek_times(dev);
 818 
 819   if (dist_nm) {
 820     for (i=0; i <= dev->precompute_seek_count; i++) {
 821       /*
 822       printf("i = %d, dist = %d, x_seek = %f, y_seek = %f\n",
 823              i, dev->precompute_seek_distances[i], dev->precompute_x_seek_times[i], dev->precompute_y_seek_times[i]);
 824       */
 825       if (dist_nm <= dev->precompute_seek_distances[i]) {
 826         if (dist_nm == dev->precompute_seek_distances[i]) {
 827           // fprintf(stderr, "found precomputed seek distance %d\n", dist_nm);
 828           seektime = (direction == _X_SEEK_) ?
 829             dev->precompute_x_seek_times[i] :
 830             dev->precompute_y_seek_times[i];
 831         } else {
 832           double ddiff =
 833             (double) (dist_nm - dev->precompute_seek_distances[(i-1)])
 834             /
 835             (double) (dev->precompute_seek_distances[i] - dev->precompute_seek_distances[(i-1)]);
 836           /*
 837           fprintf(stderr, "did not find precompute seek distance %d\n", dist_nm);
 838           fprintf(stderr, "ddiff = %f\n", ddiff);
 839           fprintf(stderr, "i = %d\n", i);
 840           */
 841           if (direction == _X_SEEK_) {
 842             seektime = dev->precompute_x_seek_times[(i-1)];
 843             // fprintf(stderr, "seektime = %f\n", seektime);        
 844             seektime += ddiff * (dev->precompute_x_seek_times[i] - dev->precompute_x_seek_times[(i-1)]);
 845             /*
 846             fprintf(stderr, "times[i] = %f, times[i-1] = %f\n",
 847                     dev->precompute_x_seek_times[i], dev->precompute_x_seek_times[(i-1)]);
 848             fprintf(stderr, "seektime = %f\n", seektime);
 849             */
 850           } else {
 851             seektime = dev->precompute_y_seek_times[(i-1)];
 852             // fprintf(stderr, "seektime = %f\n", seektime);
 853             seektime += ddiff * (dev->precompute_y_seek_times[i] - dev->precompute_y_seek_times[(i-1)]);
 854             /*
 855             fprintf(stderr, "times[i] = %f, times[i-1] = %f\n",
 856                     dev->precompute_y_seek_times[i], dev->precompute_y_seek_times[(i-1)]);
 857             fprintf(stderr, "seektime = %f\n", seektime);
 858             */
 859           }
 860         }
 861         break;
 862       }
 863     }
 864     if (seektime == 0.0) {
 865       fprintf(stderr, "Seek distance exceeds precompute seek curve: %d\n", dist_nm);
 866       exit(0);
 867     }
 868   }
 869 
 870   if (_VERBOSE_) {
 871     fprintf(__OUTPUTFILE__, "mems_find_precompute_seek_time::  finished - seektime = %f\n",
 872             seektime);
 873   }
 874 
 875   // fprintf(stderr, "seektime / 1000.0 = %f\n", (seektime / 1000.0));
 876 
 877   return(seektime / 1000.0);
 878 }

 880 void
 881 mems_precompute_seek_curve(mems_t *dev) {
 882 
 883   coord_t begin, end;
 884 
 885   double x_seek_time, y_seek_time;
 886   int sled_length_bits;
 887   int offset;
 888   int bit_step;
 889   int nm_step;
 890   int i;
 891   double num_time_constants = dev->sled[0].num_time_constants;
 892   int precompute_seek_count = dev->precompute_seek_count;
 893 
 894   dev->sled[0].num_time_constants = 0.0;
 895   dev->precompute_seek_count = 0;
 896 
 897   sled_length_bits = (int) (dev->sled[0].x_length_nm / dev->sled[0].bit_length_nm);
 898   bit_step = sled_length_bits / precompute_seek_count;
 899   nm_step = (sled_length_bits * dev->sled[0].bit_length_nm) / precompute_seek_count;
 900 
 901   dev->precompute_seek_distances = (int *)malloc((precompute_seek_count + 1) * sizeof(int));
 902   dev->precompute_x_seek_times = (double *)malloc((precompute_seek_count + 1) * sizeof(double));
 903   dev->precompute_y_seek_times = (double *)malloc((precompute_seek_count + 1) * sizeof(double));
 904 
 905   /*
 906   printf("precompute_seek_count = %d, bit_step = %d, nm_step = %d\n",
 907          precompute_seek_count, bit_step, nm_step);
 908   */
 909 
 910   //  for(offset = 0; offset <= sled_length_bits; offset += bit_step) {
 911   for(i = 0; i < precompute_seek_count; i++) {
 912     
 913     offset = i * bit_step;
 914 
 915     // if (verbose) fprintf(stdout, "main:  starting a new seek (monkey)\n");

 918     begin.y_pos = 0;
 919     begin.y_vel = 0;

 921     end.x_pos = offset;

 927            begin.y_vel, end.y_vel);

 929 
 930     x_seek_time = mems_seek_time(&(dev->sled[0]),

 936     begin.y_pos = 0;
 937     begin.y_vel = dev->sled[0].y_access_speed_bit_s;
 938     
 939     end.x_pos = 0;
 940     end.y_pos = offset;
 941     end.y_vel = dev->sled[0].y_access_speed_bit_s;

 943     /*
 944     printf("begin = %d, end = %d\n",
 945            begin.y_pos, end.y_pos);

 949                                   &begin, &end,

 951                                   NULL, NULL);
 952 
 953     /*
 954     fprintf(stdout, "%d, %d, %d, %f, %f\n",

 960     dev->precompute_y_seek_times[i]   = y_seek_time;

 964             i, (offset * dev->sled[0].bit_length_nm),
 965             dev->precompute_x_seek_times[i],
 966             dev->precompute_y_seek_times[i]);
 967     */
 968 
 969   }
 970 
 971   begin.x_pos = 0;
 972   begin.y_pos = 0;
 973   begin.y_vel = 0;
 974   
 975   end.x_pos = sled_length_bits;
 976   end.y_pos = 0;
 977   end.y_vel = 0;
 978 
 979   x_seek_time = mems_seek_time(&(dev->sled[0]),
 980                                 &begin, &end,
 981                                 NULL, NULL,
 982                                 NULL, NULL);
 983   
 984   begin.x_pos = 0;
 985   begin.y_pos = 0;
 986   begin.y_vel = dev->sled[0].y_access_speed_bit_s;
 987   
 988   end.x_pos = 0;
 989   end.y_pos = sled_length_bits;
 990   end.y_vel = dev->sled[0].y_access_speed_bit_s;
 991   
 992   y_seek_time = mems_seek_time(&(dev->sled[0]),
 993                                 &begin, &end,
 994                                 NULL, NULL,
 995                                 NULL, NULL);
 996   
 997   dev->precompute_seek_distances[precompute_seek_count] = sled_length_bits * dev->sled[0].bit_length_nm;
 998   dev->precompute_x_seek_times[precompute_seek_count]   = x_seek_time;
 999   dev->precompute_y_seek_times[precompute_seek_count]   = y_seek_time;
1000 
1001   /*
1002   fprintf(stdout, "%d, %d, %d, %f, %f\n",
1003           precompute_seek_count, (precompute_seek_count * nm_step), sled_length_bits * dev->sled[0].bit_length_nm, x_seek_time, y_seek_time);
1004   */
1005 
1006   dev->sled[0].num_time_constants = num_time_constants;
1007   dev->precompute_seek_count = precompute_seek_count;
1008 
1009 }
