 113 static void controller_53c700_reset_complete (controller *currctlr, ioreq_event *curr)
 114 {
 115    ioreq_event *tmp;
 116 
 117    switch (currctlr->state) {
 118       case COMPLETION_PENDING:
 119       case DISCONNECT_PENDING:
 120          break;
 121       default:
 122          fprintf(stderr, "Controller not in appropriate state for reset completion\n");
 123          exit(1);
 124    }
 125    currctlr->state = FREE;
 126    addtoextraq((event *) curr);
 127    if (currctlr->connections != NULL) {
 128 /*
 129       fprintf (outputfile, "Pending reconnection for controller %d\n", ctlno);
 130 */
 131       tmp = currctlr->connections;
 132       currctlr->connections = tmp->next;
 133       currctlr->state = RECONNECTING;
 134       controller_send_event_up_path(currctlr, tmp, currctlr->ovrhd_reset);
 135    }
 136 }

 139 static void controller_53c700_reconnection_complete (controller *currctlr, ioreq_event *curr)
 140 {
 141    int read = curr->flags & READ;
 142 
 143    ASSERT(currctlr->state == RECONNECTING);
 144    currctlr->outbusowned = controller_get_downward_busno(currctlr, curr, NULL);
 145    controller_send_event_down_path(currctlr, curr, currctlr->ovrhd_disk_reconnect);
 146    currctlr->outbusowned = -1;
 147    currctlr->state = (read) ? READ_DATA_TRANSFER : WRITE_DATA_TRANSFER;
 148 }

 151 static void controller_53c700_reconnect_to_transfer (controller *currctlr, ioreq_event *curr)
 152 {
 153    switch (currctlr->state) {
 154 
 155       case COMPLETION_PENDING:
 156       case DISCONNECT_PENDING:
 157          curr->next = currctlr->connections;
 158          currctlr->connections = curr;
 159          break;
 160 
 161       case REQUEST_PENDING:
 162          controller_remove_type_from_buswait(currctlr, IO_ACCESS_ARRIVE);
 163 
 164       case FREE:
 165          currctlr->state = RECONNECTING;
 166          controller_send_event_up_path(currctlr, curr, currctlr->ovrhd_reconnect);
 167          break;
 168 
 169       default:
 170          fprintf(stderr, "Received reconnection request in invalid state - ctlno %d, state %d\n", currctlr->ctlno, currctlr->state);
 171          exit(1);
 172    }
 173 }

 176 static void controller_53c700_request_complete (controller *currctlr, ioreq_event *curr)
 177 {
 178    ioreq_event *ret;
 179 
 180    switch (currctlr->state) {
 181       case READ_DATA_TRANSFER:
 182       case WRITE_DATA_TRANSFER:
 183          break;
 184       default:
 185          fprintf(stderr, "Completing request not transfering in controller_53c700_request_complete\n");
 186          exit(1);
 187    }
 188    ret = ioreq_copy(curr);
 189    ret->type = IO_INTERRUPT_COMPLETE;
 190    currctlr->outbusowned = controller_get_downward_busno(currctlr, curr, NULL);
 191    controller_send_event_down_path(currctlr, ret, currctlr->ovrhd_disk_complete);
 192    currctlr->outbusowned = -1;
 193    currctlr->state = COMPLETION_PENDING;
 194    controller_send_event_up_path(currctlr, curr, currctlr->ovrhd_complete);
 195 }

 199 {

 204       case WRITE_DATA_TRANSFER:
 205       case REQUEST_PENDING:

 207       default:

 211    ret = ioreq_copy(curr);
 212    ret->type = IO_INTERRUPT_COMPLETE;
 213    currctlr->outbusowned = controller_get_downward_busno(currctlr, curr, NULL);
 214    controller_send_event_down_path(currctlr, ret, currctlr->ovrhd_disk_disconnect);
 215    currctlr->outbusowned = -1;
 216    currctlr->state = DISCONNECT_PENDING;
 217    controller_send_event_up_path(currctlr, curr, currctlr->ovrhd_disconnect);
 218 }

 221 static void controller_53c700_ready_to_transfer (controller *currctlr, ioreq_event *curr)
 222 {
 223    ASSERT(currctlr->state == REQUEST_PENDING);
 224    currctlr->state = (curr->flags & READ) ? READ_DATA_TRANSFER : WRITE_DATA_TRANSFER;
 225    curr->type = IO_INTERRUPT_COMPLETE;
 226    curr->cause = RECONNECT;
 227    currctlr->outbusowned = controller_get_downward_busno(currctlr, curr, NULL);
 228    controller_send_event_down_path(currctlr, curr, currctlr->ovrhd_disk_ready);
 229    /* shouldn't we also be sending a message up the path?? ovrhd_ready?? */
 230    currctlr->outbusowned = -1;
 231 }

 234 static void controller_53c700_request_arrive (controller *currctlr, ioreq_event *curr)
 235 {
 236 /*
 237 fprintf (outputfile, "Entered 53c700_request_arrive - devno %d, blkno %d\n", curr->devno, curr->blkno);
 238 */
 239    switch (currctlr->state) {
 240       case FREE:
 241                            break;
 242       case RECONNECTING:
 243                            addtoextraq((event *) curr);
 244                            return;
 245       default:
 246                    fprintf(stderr, "Request arriving at non-FREE controller\n");
 247                    exit(1);
 248    }
 249    controller_send_event_down_path(currctlr, curr, currctlr->ovrhd_disk_request);
 250    currctlr->state = REQUEST_PENDING;
 251 }

 254 static void controller_53c700_interrupt_arrive (controller *currctlr, ioreq_event *curr)
 255 {
 256    switch (curr->cause) {
 257 
 258       case COMPLETION:
 259          controller_53c700_request_complete(currctlr, curr);
 260          break;
 261 
 262       case RECONNECT:
 263          controller_53c700_reconnect_to_transfer(currctlr, curr);
 264          break;
 265 
 266       case DISCONNECT:
 267          controller_53c700_disconnect(currctlr, curr);
 268          break;
 269 
 270       case READY_TO_TRANSFER:
 271          controller_53c700_ready_to_transfer(currctlr, curr);
 272          break;
 273 
 274       default:
 275          fprintf(stderr, "Unknown interrupt cause in 53c700_interrupt_arrive\n");
 276          exit(1);
 277    }
 278 }

 301 static void controller_53c700_data_transfer (controller *currctlr, ioreq_event *curr)
 302 {
 303    ioreq_event *tmp;
 304 /*
 305 fprintf (outputfile, "Entered controller_53c700_data_transfer - ctlno %d, devno %d, blkno %d, bcount %d\n", ctlno, curr->devno, curr->blkno, curr->bcount);
 306 */
 307    if (curr->type == DEVICE_DATA_TRANSFER_COMPLETE) {
 308       curr->time = max(device_get_blktranstime(curr), currctlr->blktranstime);
 309    } else if (curr->type == CONTROLLER_DATA_TRANSFER_COMPLETE) {
 310       fprintf(stderr, "Not yet equiped to handle controller to controller transfers\n");
 311       exit(1);
 312    } else {
 313       fprintf(stderr, "Illegal event type at controller_53c700_data_transfer\n");
 314       exit(1);
 315    }
 316    curr->next = currctlr->datatransfers;
 317    curr->prev = NULL;
 318    if (curr->next) {
 319       curr->next->prev = curr;
 320    }
 321    currctlr->datatransfers = curr;
 322    tmp = (ioreq_event *) getfromextraq();
 323    tmp->time = simtime + (curr->time * (double) curr->bcount);
 324    tmp->type = CONTROLLER_DATA_TRANSFER_COMPLETE;
 325    tmp->blkno = curr->blkno;
 326    tmp->bcount = curr->bcount;
 327    tmp->tempint2 = currctlr->ctlno;
 328    tmp->tempptr1 = curr;
 329    curr->buf = tmp;
 330    addtointq((event *) tmp);
 331 }

 334 static void controller_53c700_data_transfer_complete (controller *currctlr, ioreq_event *curr)
 335 {
 336    ioreq_event *tmp;
 337 
 338    tmp = (ioreq_event *) curr->tempptr1;
 339    tmp->bcount -= curr->bcount;
 340    addtoextraq((event *) curr);
 341    if (tmp->bcount < 0) {
 342       fprintf(stderr, "Transfered more than requested at controller_data_transfer_done\n");
 343       exit(1);
 344    } else if (tmp->bcount == 0) {
 345       if (tmp->next) {
 346          tmp->next->prev = tmp->prev;
 347       }
 348       if (tmp->prev) {
 349          tmp->prev->next = tmp->next;
 350       } else {
 351          currctlr->datatransfers = tmp->next;
 352       }
 353       tmp->time = simtime;
 354       addtointq((event *) tmp);
 355    } else {
 356       fprintf(stderr, "Haven't required less than all out transfer at controller_data_transfer_done\n");
 357       exit(1);
 358    }
 359 }

 362 void controller_53c700_event_arrive (controller *currctlr, ioreq_event *curr)
 363 {
 364    int busno;
 365    int slotno;
 366 
 367    switch (curr->type) {
 368 
 369       case IO_ACCESS_ARRIVE:
 370          controller_53c700_request_arrive(currctlr, curr);
 371          break;
 372 
 373       case IO_INTERRUPT_ARRIVE:
 374          controller_53c700_interrupt_arrive(currctlr, curr);
 375          break;
 376 
 377       case IO_INTERRUPT_COMPLETE:
 378          controller_53c700_interrupt_complete(currctlr, curr);
 379          break;
 380 
 381       case DEVICE_DATA_TRANSFER_COMPLETE:
 382          controller_53c700_data_transfer(currctlr, curr);
 383          break;
 384 
 385       case CONTROLLER_DATA_TRANSFER_COMPLETE:
 386          controller_53c700_data_transfer_complete(currctlr, curr);
 387          break;
 388 
 389       case IO_QLEN_MAXCHECK:
 390          busno = controller_get_downward_busno(currctlr, curr, &slotno);
 391          bus_deliver_event(busno, slotno, curr);
 392          break;
 393 
 394       default:
 395          fprintf(stderr, "Unknown event type arriving at 53c700 controller: %d\n", curr->type);
 396          exit(1);
 397    }
 398 }

 401 void controller_passthru_event_arrive (controller *currctlr, ioreq_event *curr)
 402 {
 403    int busno;
 404    int slotno;
 405 
 406    switch (curr->type) {
 407 
 408       case IO_INTERRUPT_COMPLETE:
 409          currctlr->outbusowned = controller_get_downward_busno(currctlr, curr, NULL);
 410          controller_send_event_down_path(currctlr, curr, PASSTHRU_DELAY);
 411          currctlr->outbusowned = -1;
 412          break;
 413 
 414       case IO_ACCESS_ARRIVE:
 415 /*
 416 fprintf (outputfile, "Following passthru_event path, and passing access down\n");
 417 */
 418          controller_send_event_down_path(currctlr, curr, PASSTHRU_DELAY);
 419          break;
 420 
 421       case IO_INTERRUPT_ARRIVE:
 422          if (curr->cause == READY_TO_TRANSFER) {
 423             curr->cause = RECONNECT;
 424          }
 425          controller_send_event_up_path(currctlr, curr, PASSTHRU_DELAY);
 426          break;
 427 
 428       case DEVICE_DATA_TRANSFER_COMPLETE:
 429          controller_send_event_up_path(currctlr, curr, (double) 0.0);
 430          break;
 431 
 432       case IO_QLEN_MAXCHECK:
 433          busno = controller_get_downward_busno(currctlr, curr, &slotno);
 434          bus_deliver_event(busno, slotno, curr);
 435          break;
 436 
 437       default:
 438          fprintf(stderr, "Unknown event type arriving at passthru controller: %d\n", curr->type);
 439          exit(1);
 440    }
 441 }
