  65 typedef enum {
  66   DEVICETYPE_DISK       = 1,
  67   DEVICETYPE_SIMPLEDISK = 2,
  68   DEVICETYPE_MEMS      = 3,
  69   DEVICETYPE_SSD       = 4,
  70 
  71   // this is something of a hack so that things like synthgen can
  72   // refer to a logorg by name.  All we care here is that it has a
  73   // name and an underlying device number.
  74 } device_type_t;
  75 #define DEVICETYPE_MIN DEVICETYPE_DISK
  76 #define DEVICETYPE_MAX DEVICETYPE_MEMS

  79 typedef struct device_info {
  80    int numdevices;
  81 
  82    int *devicetypes;    /* actual type of device;
  83                          * this will go away with new device iface 
  84                          */
  85 
  86    int *devicenos;      /* number among devices of that device type */
  87   
  88    int maxdeviceno[DEVICETYPE_MAX];     /* maximum deviceno of each
  89                                          * devicetype -- may already be
  90                                          * defunct */
  91 
  92    char **devicenames;  /* pointers to name fields in actual
  93                          * device objects */
  94 
  95    struct device_header **devices;
  96 
  97    int devs_len;        /* allocated size of above arrays */
  98 
  99    int device_printqueuestats;
 100    int device_printcritstats;
 101    int device_printidlestats;
 102    int device_printintarrstats;
 103    int device_printsizestats;
 104    int device_printseekstats;
 105    int device_printlatencystats;
 106    int device_printxferstats;
 107    int device_printacctimestats;
 108    int device_printinterferestats;
 109    int device_printbufferstats;

 111 } device_info_t;
