  53 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
  54  *
  55  *  find_dist_nm()
  56  *
  57  *  returns the distance in nanometers from start_offset
  58  *  to end_offset.  remember that these are _offsets_ from
  59  *  the center of the sled.
  60  *
  61  *  inputs:
  62  *    mems_sled_t *sled - a pointer to the sled structure
  63  *    coord_t *start - the starting coordinate
  64  *    coord_t *end - the destination coordinate
  65  *
  66  *  outputs:
  67  *    double - the distance in nm from start_offset to
  68  *             end_offset
  69  *
  70  *  modifies:
  71  *    nothing
  72  *
  73  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
  74 
  75 double
  76 find_dist_nm(double start_offset, double end_offset) {
  77 
  78   double dist_nm;
  79 
  80   if (((start_offset > 0.0) && (end_offset > 0.0)) ||
  81       ((start_offset < 0.0) && (end_offset < 0.0))) {
  82     /*  we're on the same side of the sled  */
  83     dist_nm = fabs(end_offset - start_offset);
  84   } else {
  85     dist_nm = fabs(start_offset) + fabs(end_offset);
  86   }
  87 
  88   if (_VERBOSE_) {
  89     fprintf(__OUTPUTFILE__, "find_dist_nm::  start_offset = %f, end_offset = %f, dist_nm = %f\n",
  90             start_offset, end_offset, dist_nm);
  91   }
  92 
  93   return dist_nm;
  94 }

  96 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
  97  *
  98  *  find_local_actuator_direction()
  99  *
 100  *  returns the direction which the actuators are pulling
 101  *  the sled, INWARD or OUTWARD.  this is used to determine
 102  *  if the actuators are helping the seek or hurting the
 103  *  seek.  see below for more.
 104  *
 105  *  inputs:
 106  *    double start_offset_nm - the start offset of the seek
 107  *    double end_offset_nm - the end offset of the seek
 108  *    int overall_actuator_direction - the direction the
 109  *                         actuators are acting.
 110  *
 111  *  outputs:
 112  *    int - INWARD if the actuators are pulling the sled
 113  *          toward the center, OUTWARD if the actuators are
 114  *          pulling toward the edge.
 115  *
 116  *  modifies:
 117  *    nothing
 118  *
 119  *  what's really going on here:
 120  *
 121  *    okay, what's the difference between INWARD, OUTWARD, OVERALL_POSITIVE,
 122  *  OVERALL_NEGATIVE, etc, and what is this function for anyway?  second question
 123  *  first:  given start and end offsets, this fuction tells you if the actuators
 124  *  are pulling the sled toward the rest position (INWARD) or away from the rest
 125  *  position (OUTWARD).  for example, if the overall_actuator_direction is
 126  *  OVERALL_POSITIVE, and the seek is entirely on the positive half of the square,
 127  *  then the actuators are pulling the sled OUTWARD.  when the seek crosses the
 128  *  center line, then the function decides based the larger portion of the seek.
 129  *  i think that answers the second question, too.  it's used to determine when
 130  *  the springs will help the seek and when they'll hurt.
 131  *
 132  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 133 
 134 int find_local_actuator_direction(double start_offset_nm, double end_offset_nm,
 135                                   int overall_actuator_direction) {
 136 
 137   if (_VERBOSE_) {
 138     fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  start_offset_nm = %f, end_offset_nm = %f\n",
 139             start_offset_nm, end_offset_nm);
 140     
 141     if (overall_actuator_direction == OVERALL_POSITIVE)
 142       fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  overall_actuator_direction = OVERALL_POSITIVE\n");
 143     else
 144       fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  overall_actuator_direction = OVERALL_NEGATIVE\n");
 145   }
 146 
 147   if ((start_offset_nm <= 0.0) && (end_offset_nm <= 0.0)) {
 148     /*  if the seek is on the negative side  */
 149     if (overall_actuator_direction == OVERALL_POSITIVE) {
 150       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is INWARD (1)\n");
 151       return INWARD;
 152     } else {
 153       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is OUTWARD (2)\n");
 154       return OUTWARD;
 155     }
 156   } else if ((start_offset_nm >= 0.0) && (end_offset_nm >= 0.0)) {
 157     /*  if the seek is on the positive side  */
 158     if (overall_actuator_direction == OVERALL_POSITIVE) {
 159       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is OUTWARD (3)\n");
 160       return OUTWARD;
 161     } else {
 162       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is OUTWARD (4)\n");
 163       return INWARD;
 164     }
 165   } else {
 166     /*  the seek crosses the centerline  */
 167     if (fabs(start_offset_nm) > fabs(end_offset_nm)) {
 168       /*  i move more toward the center than away  */
 169       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is INWARD (5)\n");
 170       return INWARD;
 171     } else {
 172       if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_local_actuator_direction::  local_actuator_direction is OUTWARD (6)\n");
 173       return OUTWARD;
 174     }
 175   }
 176 }

 178 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 179  *
 180  *  adjust_accel()
 181  *
 182  *  returns the acceleration in the region given by the
 183  *  start and end points.  the acceleration at the two points
 184  *  is the sum of the actuator acceleration and the spring
 185  *  acceleration, which is given by
 186  *
 187  *      [ actuator_accel *  offset  ]
 188  *      [                  -------- ]  *  spring_factor
 189  *      [                  length/2 ] 
 190  *
 191  *  the springs' restoring force is then a function of the
 192  *  displacement (offset) and a fraction (spring_factor) of
 193  *  the actuator force.  put another way, if the spring_factor
 194  *  is 0.10, then at the maximum offset the force from the springs
 195  *  will be equal to 10% of the force from the actuators.
 196  *
 197  *  the returned adjusted acceleration is the average of the
 198  *  sums at the start and end points.
 199  *
 200  *  inputs:
 201  *    double accel - the base acceleration from the actuators
 202  *    double spring_factor - the spring factor
 203  *    double offset_1 - the offset of the starting point
 204  *    double offset_2 - the offset of the ending point
 205  *    double length - the length of the sled
 206  *    int overall_actuator_direction - INWARD if the actuators
 207  *          are pulling the sled toward the center, OUTWARD otherwise.
 208  *
 209  *  outputs:
 210  *    double - the adjusted acceleration for the region between
 211  *             the start and end offsets, given the base actuator
 212  *             acceleration, the spring_factor, the length of the
 213  *             sled, and the overall actuator direction
 214  *
 215  *  modifies:
 216  *    nothing
 217  *
 218  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 219 
 220 double adjust_accel(double accel, double spring_factor,
 221                     double offset_1, double offset_2,
 222                     double length, int overall_actuator_direction) {
 223 
 224   double adjusted_accel;
 225 
 226   if (find_local_actuator_direction(offset_1, offset_2, overall_actuator_direction) == INWARD) {
 227     /*  springs help  */
 228     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "adjust_accel::  springs helped!\n");
 229   } else {
 230     /*  springs hurt  */
 231     spring_factor = -spring_factor;
 232     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "adjust_accel::  springs hurt!\n");
 233   }
 234 
 235   adjusted_accel =
 236     (
 237      (accel
 238       +
 239       (accel * (spring_factor * (fabs(offset_1) / (length / 2))))
 240       )
 241      +
 242      (accel
 243       +
 244       (accel * (spring_factor * (fabs(offset_2) / (length / 2))))
 245       )
 246      ) / 2.0;
 247 
 248   if (_VERBOSE_) {
 249     fprintf(__OUTPUTFILE__, "adjust_accel::  offset_1 = %f, offset_2 = %f, accel = %f\n",
 250             offset_1, offset_2, accel);
 251     fprintf(__OUTPUTFILE__, "adjust_accel::  spring_factor = %f, length = %f\n",
 252             spring_factor, length);
 253     fprintf(__OUTPUTFILE__, "adjust_accel::  adjusted_accel = %f, diff = %f\n",
 254             adjusted_accel, ((adjusted_accel / accel) * 100.0) - 100.0);
 255   }
 256 
 257   return adjusted_accel;
 258 }

 260 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 261  *
 262  *  find_switch_offset_from_dist()
 263  *
 264  *  computes the switchpoint offset between start_offset_nm
 265  *  and end_offset_nm.  this finds the offset which is the
 266  *  midpoint between the two points.
 267  *
 268  *  inputs:
 269  *    double start_offset_nm - the starting offset
 270  *    double end_offset_nm - the end offset
 271  *    double dist_nm - the distance in nm between the two points
 272  *    int direction - the direction of the seek
 273  *
 274  *  outputs:
 275  *    double - the offset of the midpoint in nm
 276  *
 277  *  modifies:
 278  *    nothing
 279  *
 280  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 281 
 282 double
 283 find_switch_offset_from_dist (double start_offset_nm, double end_offset_nm,
 284                               double dist_nm, int direction) {
 285 
 286   double switch_offset_nm;
 287 
 288   if (direction == OVERALL_POSITIVE) {
 289     switch_offset_nm = start_offset_nm + (dist_nm / 2);
 290   } else {
 291     switch_offset_nm = start_offset_nm - (dist_nm / 2);
 292   }
 293 
 294   return switch_offset_nm;
 295 }

 328 /*-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
 329  *
 330  *  find_seek_time_piecewise()
 331  *
 332  *  this is the guts of the seek code.  find_seek_time_piecewise()
 333  *  returns the time to seek from start_offset_nm to end_offset_nm
 334  *  given the various parameters.  it assumes that the
 335  *  initial velocity is equal to the final velocity
 336  *  (i.e. there are no turnarounds necessary).  it uses an
 337  *  iterative algorithm to find the optimal switchpoint,
 338  *  attempting to balance the energy in the acceleration
 339  *  phase with the energy in the deceleration phase.
 340  *
 341  *  please see below for a complete explanation of how it works.
 342  *
 343  *  inputs:
 344  *    double start_offset_nm - the starting offset
 345  *    double end_offset_nm - the end offset
 346  *    double spring_factor - spring factor of the system
 347  *    double acceleration_nm_s_s - actuator acceleration
 348  *    double length_nm - the length of the sled
 349  *    double velocity_nm_s - the initial and final velocity
 350  *
 351  *  outputs:
 352  *    double - the seek time from start to end
 353  *
 354  *  modifies:
 355  *    nothing
 356  *
 357  *-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-*/
 358 
 359 double
 360 find_seek_time_piecewise(double start_offset_nm, double end_offset_nm,
 361                          double spring_factor, double acceleration_nm_s_s,
 362                          double length_nm, double velocity_nm_s) {
 363 
 364   double dist_nm;                      /*  distance of this seek  */
 365   double phase1_dist_nm;               /*  distance of the first phase  */
 366   double phase2_dist_nm;               /*  distance of the second phase  */
 367   double phase1_chunk_length_nm;       /*  length of chunks in the first phase  */
 368   double phase2_chunk_length_nm;       /*  length of chunks in the second phase  */
 369   double phase1_acceleration_nm_s_s;   /*  actuator acceleration in the first phase  */
 370   double phase2_acceleration_nm_s_s;   /*  actuator acceleration in the second phase  */
 371   double x[2 * NUM_CHUNKS + 1];        /*  position values for each chunk of the seek  */
 372   double v[2 * NUM_CHUNKS + 1];        /*  velocity values for each chunk of the seek  */
 373   double a[2 * NUM_CHUNKS + 1];        /*  adjusted acceleration values for each chunk  */
 374   double t[2 * NUM_CHUNKS + 1];        /*  times at each x point  */
 375 
 376   double x_switch, v_switch, a_switch, t_switch;  /*  position, velocity, acceleration, and time values for the switch point  */
 377   double v_diff, x_delta;
 378 
 379   double min_v_diff;
 380   double min_v_diff_time = 0.0;
 381 
 382   double total_seek_time;
 383 
 384   int i, jj;
 385   int overall_seek_direction;
 386   int phase1_actuator_direction, phase2_actuator_direction;
 387 
 388   min_v_diff = 666.0;  /*  this should be bigger than 100 percent  */
 389   
 390   if (_VERBOSE_) {
 391     fprintf(__OUTPUTFILE__, "find_seek_time_piecewise:: ** entry - start_offset_nm = %f, end_offset_nm = %f\n",
 392             start_offset_nm, end_offset_nm);
 393   }
 394 
 395   /*  find seek distance and overall direction  */
 396   dist_nm = find_dist_nm(start_offset_nm, end_offset_nm);
 397   overall_seek_direction = find_overall_seek_direction(start_offset_nm, end_offset_nm);
 398 
 399   /*  set initial x position and set switch point to be midpoint of seek  */
 400   x[0] = start_offset_nm;
 401   x[SWITCH_POINT] = find_switch_offset_from_dist(start_offset_nm, end_offset_nm,
 402                                                  dist_nm, overall_seek_direction);
 403 
 404   /*  this is the loop which iteratively tries to find the optimal switch point.  during each iteration
 405       the loop does the following:
 406 
 407       find the x values for each chunk.
 408       find the adjusted acceleration at each point.
 409       start at the start point and find the v and t values
 410          for each point in the first phase, including the switch point.
 411       start at the end point and find the v and t values
 412          for each point in the second phase, again including the switch point.
 413       
 414       if the switch point we picked initially were the optimal one, then the
 415       velocity at the switch point from each phase should be equal, by conservation
 416       of energy.  if they're not, then we need to adjust the switch point toward the
 417       end that had more energy, i.e. had the highest velocity at the switch point.
 418       the loop below iterates a fixed number of times, adjusting the switch point toward
 419       the start or end point by some percentage of the seek distance.
 420 
 421       at compile time, you could lower the number of MEMS_SWITCH_ITERATIONS to reduce the runtime, at the
 422       cost of accuracy.  if the phases of the calculation are unbalanced, then the sled will not be at the
 423       correct velocity when it arrives at the destination.  you could also reduce the number of chunks each
 424       phase is broken into.  this leads to discontinuities in the seek curve.  this is due to the way it
 425       decides when the springs switch from helping the seek to hurting the seek.  when a chunk straddles the
 426       middle of the square, the springs will be changing from helping to hurting.  however, we can only choose
 427       one or the other for each chunk, so for part of it the spring factor will be incorrectly determined.
 428       the smaller the chunks are, the less significant this inaccuracy becomes.
 429   */
 430   
 431   for (jj = 1; jj < MEMS_SWITCH_ITERATIONS; jj++) {
 432     
 433     /*  find the length of the two phases  */
 434     phase1_dist_nm = find_dist_nm(start_offset_nm, x[SWITCH_POINT]);
 435     phase2_dist_nm = find_dist_nm(x[SWITCH_POINT], end_offset_nm);
 436 
 437     /*  find the length of the chunks  */
 438     phase1_chunk_length_nm = phase1_dist_nm / NUM_CHUNKS;
 439     phase2_chunk_length_nm = phase2_dist_nm / NUM_CHUNKS;
 440 
 441     if (_VERBOSE_) {
 442       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  phase1_dist_nm = %f, phase1_chunk_length_nm = %f\n",
 443               phase1_dist_nm, phase1_chunk_length_nm);
 444       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  phase2_dist_nm = %f, phase2_chunk_length_nm = %f\n",
 445               phase2_dist_nm, phase2_chunk_length_nm);
 446     }
 447 
 448     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** figuring out phase 1 x values\n");
 449 
 450     /*  find the endpoints of all the chunks of the first phase  */
 451     for (i = 1; i < SWITCH_POINT; i++) {
 452       if (overall_seek_direction == OVERALL_POSITIVE)
 453         x[i] = x[i-1] + phase1_chunk_length_nm;
 454       else
 455         x[i] = x[i-1] - phase1_chunk_length_nm;
 456     }
 457 
 458     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** figuring out phase 2 x values\n");
 459 
 460     /*  find the endpoints of all the chunks of the second phase  */
 461     for (i = SWITCH_POINT + 1; i <= 2*NUM_CHUNKS; i++) {
 462       if (overall_seek_direction == OVERALL_POSITIVE)
 463         x[i] = x[i-1] + phase2_chunk_length_nm;
 464       else
 465         x[i] = x[i-1] - phase2_chunk_length_nm;
 466     }  
 467 
 468     /*  the absolute value of the actuator acceleration is equal for both phases...  */
 469     phase1_acceleration_nm_s_s = acceleration_nm_s_s;
 470     phase2_acceleration_nm_s_s = acceleration_nm_s_s;
 471 
 472     /*  ...but the directions will be different  */
 473     if (overall_seek_direction == OVERALL_POSITIVE) {
 474       phase1_actuator_direction = OVERALL_POSITIVE;
 475       phase2_actuator_direction = OVERALL_NEGATIVE;
 476     } else {
 477       phase1_actuator_direction = OVERALL_NEGATIVE;
 478       phase2_actuator_direction = OVERALL_POSITIVE;
 479     }
 480 
 481     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** adjusting phase 1 accelerations\n");
 482 
 483     /*  find the acceleration in each chunk of the first phase.  this takes into account the adjustment
 484         for the spring forces.  */
 485     for (i = 1; i <= SWITCH_POINT; i++) {
 486       if (_VERBOSE_) {
 487         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  i = %d, x[i-1] = %f, x[i] = %f\n",
 488                 i, x[i-1], x[i]);
 489       }
 490       a[i] = adjust_accel(phase1_acceleration_nm_s_s, spring_factor,
 491                           x[i-1], x[i],
 492                           length_nm, phase1_actuator_direction);
 493     }
 494 
 495     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** adjusting phase 2 accelerations\n");
 496 
 497     /*  find the acceleration in each chunk of the second phase.  this takes into account the adjustment
 498         for the spring forces.  */
 499     for (i = SWITCH_POINT + 1; i <= 2*NUM_CHUNKS; i++) {
 500       if (_VERBOSE_) {
 501         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  i = %d, x[i-1] = %f, x[i] = %f\n",
 502                 i, x[i-1], x[i]);
 503       }
 504       a[i] = adjust_accel(phase2_acceleration_nm_s_s, spring_factor,
 505                           x[i-1], x[i],
 506                           length_nm, phase2_actuator_direction);
 507     }
 508 
 509     /*  start at time 0.0 and with the correct initial velocity  */
 510     t[0] = 0.0;
 511     v[0] = velocity_nm_s;
 512 
 513     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** finding phase 1 t's and v's\n");
 514 
 515     /*  find the time and velocity at each point in the first phase  */
 516     for (i = 1; i <= SWITCH_POINT; i++) {
 517       if (_VERBOSE_) {
 518         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  v[i-1] = %f, a[i] = %f, t[i-1] = %f, x[i-1] = %f, x[i] = %f\n",
 519                 v[i-1], a[i], t[i-1], x[i-1], x[i]);
 520       }
 521       t[i] = (
 522               -(v[i-1] - (a[i] * t[i-1]))
 523               + sqrt(
 524                      (v[i-1] * v[i-1]) + ((2.0 * a[i]) * fabs((x[i] - x[i-1])))
 525                      )
 526               ) / (a[i]);
 527     
 528       if (_VERBOSE_) {
 529         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  t[%d] = %f\n", i, t[i]);
 530       }
 531 
 532       v[i] = (
 533               v[i-1] + (a[i] * (t[i] - t[i-1]))
 534               );
 535     
 536       if (_VERBOSE_) {
 537         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  v[%d] = %f\n", i, v[i]);    
 538       }
 539     }
 540 
 541     /*  now we've found half of the seek time  */
 542     total_seek_time = t[SWITCH_POINT];
 543 
 544     /*  save the values at the switch point since they're about to be over-written by
 545         the calculation for the second phase  */
 546     x_switch = x[SWITCH_POINT];
 547     v_switch = v[SWITCH_POINT];
 548     a_switch = a[SWITCH_POINT];
 549     t_switch = t[SWITCH_POINT];
 550 
 551     /*  start the second phase at time 0.0 and at the initial velocity  */
 552     t[2*NUM_CHUNKS] = 0.0;
 553     v[2*NUM_CHUNKS] = velocity_nm_s;
 554 
 555     if (_VERBOSE_) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  ** finding phase 2 t's and v's\n");
 556 
 557     /*  find the time and velocity at each point in the second phase  */
 558     for (i = 2*NUM_CHUNKS - 1; i >= SWITCH_POINT; i--) {
 559       if (_VERBOSE_) {
 560         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  v[i+1] = %f, a[i] = %f, t[i+1] = %f, x[i+1] = %f, x[i] = %f\n",
 561                 v[i+1], a[i], t[i+1], x[i+1], x[i]);
 562       }
 563       t[i] = (
 564               -(v[i+1] - (a[i] * t[i+1]))
 565               + sqrt(
 566                      (v[i+1] * v[i+1]) + ((2.0 * a[i]) * fabs((x[i+1] - x[i])))
 567                      )
 568               ) / (a[i]);
 569     
 570       if (_VERBOSE_) {
 571         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  t[%d] = %f\n", i, t[i]);
 572       }
 573 
 574       v[i] = (
 575               v[i+1] + (a[i] * (t[i] - t[i+1]))
 576               );
 577 
 578       if (_VERBOSE_) {
 579         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  v[%d] = %f\n", i, v[i]);
 580       }
 581     }
 582 
 583     /*  now we've found the total seek time  */
 584     total_seek_time += t[SWITCH_POINT];    
 585 
 586     /*  print out all of the x, t, a, and v values that we've found  */
 587     if (_VERBOSE_) {
 588       for (i = 0; i <= 2 * NUM_CHUNKS; i++) {
 589         if (i == SWITCH_POINT) {
 590           fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  x[%d] = %f, a[%d] = %f, v[%d] = %f, t[%d] = %f  ** SWITCH_POINT **\n",
 591                   i, x[i], i, a[i], i, v[i], i, t[i]);
 592         } else {
 593           fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  x[%d] = %f, a[%d] = %f, v[%d] = %f, t[%d] = %f\n",
 594                   i, x[i], i, a[i], i, v[i], i, t[i]);
 595         }
 596       }
 597 
 598       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  total_seek_time = %f\n", fabs(total_seek_time));
 599     
 600       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  x_switch = %f, x[SWITCH_POINT] = %f, diff = %f\n",
 601               x_switch, x[SWITCH_POINT], ((x_switch / x[SWITCH_POINT]) * 100.0) - 100.0);
 602       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  v_switch = %f, v[SWITCH_POINT] = %f, diff = %f\n",
 603               v_switch, v[SWITCH_POINT], ((v_switch / v[SWITCH_POINT]) * 100.0) - 100.0);
 604       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  a_switch = %f, a[SWITCH_POINT] = %f, diff = %f\n",
 605               a_switch, a[SWITCH_POINT], ((a_switch / a[SWITCH_POINT]) * 100.0) - 100.0);
 606       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  t_switch = %f, t[SWITCH_POINT] = %f, diff = %f\n",
 607               t_switch, t[SWITCH_POINT], ((t_switch / t[SWITCH_POINT]) * 100.0) - 100.0);
 608       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  trying to find new switch point.  jj = %d\n", jj);
 609     } 
 610 
 611     /*  look at the difference in the velocities calculated in the two phases  */
 612     v_diff = ((v_switch / v[SWITCH_POINT]) * 100.0) - 100.0;
 613 
 614     /*  if we're not taking the springs into account, then we don't need to iterate anymore.  */
 615     if (spring_factor == 0.0) {
 616       min_v_diff_time = total_seek_time;
 617       break;
 618     } else {
 619       /*  otherwise, come up with the amount by which we want to shift the switchpoint - a percentage
 620           of the seek distance minus a diminishing function of the iteration number  */
 621       x_delta = (MEMS_SHIFT_PERCENTAGE - (1.0 / ((MEMS_SWITCH_ITERATIONS - jj) * 100.0)));
 622     }
 623 
 624     if (v_switch > v[SWITCH_POINT]) {
 625       /*  the first phase ended up with more energy  */
 626       if (overall_seek_direction == OVERALL_POSITIVE) {
 627         x_delta = -x_delta;
 628       }      
 629     } else {
 630       if (overall_seek_direction == OVERALL_NEGATIVE) {
 631         x_delta = -x_delta;
 632       }      
 633     }
 634 
 635     /*  adjust the switch point  */
 636     x[SWITCH_POINT] += x_delta * dist_nm;
 637   
 638     if (_VERBOSE_) {
 639       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  shifting x_switch. x_delta = %f, x_delta_nm = %f, v_diff = %f\n",
 640               x_delta, (x_delta * dist_nm), v_diff);
 641     
 642       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  x_switch = %f, x[SWITCH_POINT] = %f, diff = %f\n",
 643               x_switch, x[SWITCH_POINT], ((x_switch / x[SWITCH_POINT]) * 100.0) - 100.0);
 644     }
 645 
 646     x[0] = start_offset_nm;
 647 
 648     if (_VERBOSE_) {
 649       fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  min_v_diff = %f, v_diff = %f\n",
 650               min_v_diff, v_diff);
 651     }
 652 
 653     /*  save the minimum velocity difference and the seek time which got me that v_diff  */
 654     if (fabs(v_diff) < min_v_diff) {
 655       min_v_diff = fabs(v_diff);
 656       min_v_diff_time = total_seek_time;
 657       if (_VERBOSE_) {
 658         fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  updating min_v_diff\n");
 659         if (min_v_diff == 0.0) fprintf(__OUTPUTFILE__, "find_seek_time_piecewise::  augh!!!  min_v_diff is zero!\n");
 660       }
 661     }
 662   }
 663 
 664   /*  return the seek time from the iteration which gave me the smallest v_diff  */
 665   return fabs(min_v_diff_time);
 666 }
