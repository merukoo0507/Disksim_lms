 134 /*
 135  * mems_seek_time()
 136  *
 137  * returns the seek time from start to end.  also returns some statistics
 138  *
 139  * inputs:  msled_t *sled - a pointer to the sled structure
 140  *          coord_t *start, *end - pointers to the start and end coordinates
 141  *          double *x_seek_time, *y_seek_time - pointers to statistics values to update
 142  *          double *turnaround_time - pointer into which to return the total turnaround time
 143  *          double *turnaround_number - pointer into which to return the number of turnarounds
 144  *
 145  * outputs: double - the total seek time
 146  * 
 147  * modifies: x_seek_time - if not NULL, returns the seek time in the X dimension
 148  *           y_seek_time - if not NULL, returns the seek time in the Y dimension
 149  *           turnaround_time - if not NULL, returns the turnaround time
 150  *           turnaround_number - if not NULL, returns the number of turnarounds
 151  *
 152  */
 153 
 154 double mems_seek_time(mems_sled_t *sled,
 155                       coord_t *begin, coord_t *end,
 156                       double *return_x_seek_time, double *return_y_seek_time,
 157                       double *return_turnaround_time, int *return_turnaround_number);
 158
