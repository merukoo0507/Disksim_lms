 135 /* HPLabs Trace Record Types */
 136 
 137 #define HPL_UNKNOWN             0
 138 #define HPL_SHORTIO             1
 139 #define HPL_LONGIO              2
 140 #define HPL_TEXTIO              3
 141 #define HPL_SUSPECTIO           4
 142 #define HPL_BREADFS             5
 143 #define HPL_BREADAFS            6
 144 #define HPL_BREADAAFS           7
 145 #define HPL_BDWRITEFS           8
 146 #define HPL_BWRITEFS            9
 147 #define HPL_CREATE_OPENFS       10
 148 
 149 /*
 150 File tr_buf.h from HPL TraceAccess code
 151 // These buffer types are taken from HP-UX vers 8.0.  In future versions, the 
 152 // actual bit positions of these types may change (as did in 9.0 when B_SYNC
 153 // became 0x00800000.  So these are what the SRT files use as the flags.
 154 // The only trick comes in converting KI files to srt files and knowing that 
 155 // the B_SYNC in the particular version of HP-UX that was traced needs to map
 156 // to this value of B_SYNC.
 157 
 158 // The HPL values match one for one with the values in HP-UX 8.0.  Those B_
 159 // values that have a 9 after them are the changes from 8.0 to 9.0 values,
 160 // so these need to be converted back to 8.0 values before processing ki
 161 // traces.
 162 */
 163 
 164 #define HPL_WRITE         0x00000000    /* non-read pseudo-flag */
 165 #define HPL_READ          0x00000001    /* read when I/O occurs */
 166 #define HPL_DONE          0x00000002    /* transaction finished */
 167 #define HPL_ERROR         0x00000004    /* transaction aborted */
 168 #define HPL_BUSY          0x00000008    /* not on av_forw/back list */
 169 
 170 #define HPL_PHYS          0x00000010    /* physical IO */
 171 #define HPL_END_OF_DATA   0x00000020
 172 #define HPL_WANTED        0x00000040    /* issue wakeup when BUSY goes off */
 173 #define HPL_NDELAY        0x00000080    /* don't retry on failures */
 174 
 175 #define HPL_ASYNC         0x00000100    /* don't wait for I/O completion */
 176 #define HPL_DELWRITE      0x00000200    /* write at exit of avail list */
 177 #define HPL_ORDWRI        0x00000400    /* flags ordered asynchronous writes */
 178 #define HPL_REWRITE       0x00000800    /* re-writing buffer - call bdwrite */
 179 
 180 #define HPL_PRIVATE       0x00001000    /* private, not part of buffers - LVM */
 181 #define HPL_WRITEV        0x00002000    /* verification of writes  - LVM */
 182 #define HPL_PFTIMEOUT     0x00004000    /* power failure time out - LVM */
 183 #define HPL_CACHE         0x00008000    /* did bread find us in the cache ? */
 184 
 185 #define HPL_INVAL         0x00010000    /* does not contain valid info  */
 186 #define HPL_LOCKED        0x00020000    /* locked in core (not reusable) */
 187 #define HPL_HEAD          0x00040000    /* a buffer header, not a buffer */
 188 #define HPL_FSYSIO        0x00080000    /* buffer from b{read,write}    */
 189 
 190 #define HPL_BAD           0x00100000    /* bad block revectoring in progress */
 191 #define HPL_CALL          0x00200000    /* call b_iodone from iodone */
 192 #define HPL_DUX_REM_REQ   0x00400000    /* Buffer is for a remote dux reqest */
 193 #define HPL_NOCACHE       0x00800000    /* don't cache block when released */
 194 
 195 #define HPL_SYNC          0x01000000    /* buffer write is synchronous */
 196 #define HPL_RAW           0x02000000    /* raw interface (LVM/disc3/autoch) */
 197 #define HPL_NETBUF        0x04000000    /* buffer in network pool */
 198 
 199 typedef struct {
 200    u_int32_t size;    /* record size */
 201    u_int32_t id;      /* record type */
 202    u_int32_t sec;     /* time of     */
 203    u_int32_t usec;    /*       event */
 204 } hpl_record_id;
 205 
 206 typedef struct {
 207    hpl_record_id   rec_id;
 208    u_int32_t       start;      /* Start time of I/O in usec from enqueue time */
 209    u_int32_t       stop;       /* Stop time of I/O in usec from enqueue time */
 210    u_int32_t       size;       /* size in bytes */
 211    u_int32_t       sectno;     /* sector number on disk addressed */
 212    u_int32_t       devno;      /* device addressed */
 213    u_int32_t       drivertype; /* type of driver used by this disk */
 214    u_int32_t       cylno;      /* cylinder number of the access */
 215    u_int32_t       ioflags;    /* I/O flags for the device */
 216    u_int32_t       info;       /* information about where this I/O originated */
 217    u_int32_t       queuelen;   /* length of the queue */
 218    u_int32_t       susflags;   /* flags indicating error (for suspectIO) */
 219    char space[4];
 220 } hpl_short_iorec_ki;
 221 
 222 #endif /* HPTRACE_H */
