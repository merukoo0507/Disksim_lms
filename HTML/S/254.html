 293 int ioqueue_get_dist (ioqueue *queue, int blkno)
 294 {
 295    int lastblkno;
 296 
 297    if (queue->lastsubqueue == IOQUEUE_BASE) {
 298       lastblkno = queue->base.lastblkno;
 299    } else if (queue->lastsubqueue == IOQUEUE_TIMEOUT) {
 300       lastblkno = queue->timeout.lastblkno;
 301    } else if (queue->lastsubqueue == IOQUEUE_PRIORITY) {
 302       lastblkno = queue->priority.lastblkno;
 303    } else {
 304       fprintf(stderr, "Unknown queue identification at ioqueue_get_dist - %d\n", queue->lastsubqueue);
 305       exit(1);
 306    }
 307    return(abs(blkno - lastblkno));
 308 }

 401 void ioqueue_reset_idledetecter (ioqueue *queue, int timechange)
 402 {
 403    timer_event *idledetect = queue->idledetect;
 404    int listlen = queue->base.listlen + queue->timeout.listlen + queue->priority.listlen;
 405 
 406    if ((listlen) || ((idledetect) && (!timechange))) {
 407       return;
 408    }

 411          fprintf(stderr, "existing idledetect event not on intq in ioqueue_reset_idledetecter\n");
 412          exit(1);
 413       }
 414    } else {
 415       idledetect = (timer_event *) getfromextraq();
 416       idledetect->type = TIMER_EXPIRED;
 417       idledetect->func = &disksim->timerfunc_ioqueue;
 418       idledetect->ptr = queue;
 419       queue->idledetect = idledetect;
 420    }
 421    idledetect->time = simtime + queue->idledelay;
 422    addtointq((event *)idledetect);
 423 }

 426 static void ioqueue_update_arrival_stats (ioqueue *queue, ioreq_event *curr)
 427 {
 428    double tdiff;
 429 
 430    if (queue->printintarrstats) {
 431       tdiff = simtime - queue->lastarr;
 432       queue->lastarr = simtime;
 433       stat_update(&queue->intarrstats, tdiff);
 434       if (curr->flags & READ) {
 435          tdiff = simtime - queue->lastread;
 436          queue->lastread = simtime;
 437          stat_update(&queue->readintarrstats, tdiff);
 438       } else {
 439          tdiff = simtime - queue->lastwrite;
 440          queue->lastwrite = simtime;
 441          stat_update(&queue->writeintarrstats, tdiff);
 442       }
 443    }
 444    if (queue->printsizestats) {
 445       stat_update(&queue->reqsizestats, (double) curr->bcount);
 446       if (curr->flags & READ) {
 447          stat_update(&queue->readsizestats, (double) curr->bcount);
 448       } else {
 449          stat_update(&queue->writesizestats, (double) curr->bcount);
 450       }
 451    }
 452    if (curr->blkno == queue->seqblkno) {
 453       if ((curr->flags & READ) && (queue->seqflags & READ)) {
 454          queue->seqreads++;
 455       } else if (!(curr->flags & READ) && !(queue->seqflags & READ)) {
 456          queue->seqwrites++;
 457       }
 458    }
 459    queue->seqblkno = curr->blkno + curr->bcount;
 460    queue->seqflags = curr->flags;
 461 }

 464 static void ioqueue_update_subqueue_statistics (subqueue *queue)
 465 {
 466    double tdiff = simtime - queue->lastalt;
 467 
 468    queue->runlistlen += tdiff * queue->listlen;
 469    queue->runreadlen += tdiff * queue->readlen;
 470    if (queue->readlen > queue->maxreadlen) {
 471       queue->maxreadlen = queue->readlen;
 472    }
 473    if ((queue->listlen - queue->readlen) > queue->maxwritelen) {
 474       queue->maxwritelen = queue->listlen - queue->readlen;
 475    }
 476    queue->runoutstanding += tdiff * queue->numoutstanding;
 477    if (queue->numoutstanding > queue->maxoutstanding) {
 478       queue->maxoutstanding = queue->numoutstanding;
 479    }
 480    if (queue->listlen > queue->maxlistlen) {
 481       queue->maxlistlen = queue->listlen;
 482    }
 483    if ((queue->listlen - queue->numoutstanding) > queue->maxqlen) {
 484       queue->maxqlen = queue->listlen - queue->numoutstanding;
 485    }
 486    queue->lastalt = simtime;
 487 }

 544     new_iobuf->next = queue->list->next;
 545     new_iobuf->prev = queue->list;
 546     new_iobuf->next->prev = new_iobuf;
 547     new_iobuf->prev->next = new_iobuf;
 548     queue->list = new_iobuf;
 549 
 550     new_iobuf->batch_complete = TRUE;
 551 
 552     // printf("ioqueue_insert_batch_fcfs_to_queue:: queue contents after\n");
 553     // print_batch_fcfs_queue(queue);
 554 
 555     return;
 556   }
 557 
 558   // if it is part of a batch, walk the entire queue and find where to insert it.
 559   // the new iobuf should go after the last request in its batch, or at the end of the queue
 560   // if this is the first request in the batch.
 561 
 562   tmp = queue->list;
 563 
 564   for (i = 0; i < (queue->iobufcnt - 1); i++) {
 565     if (tmp->batchno == new_iobuf->batchno) {
 566       // chain the new request onto this batch and free new_iobuf
 567       event_ptr = tmp->batch_list;
 568 
 569       while (event_ptr->batch_next != NULL) {
 570         event_ptr = event_ptr->batch_next;
 571       }
 572       
 573       event_ptr->batch_next = new_iobuf->iolist;
 574       event_ptr->batch_next->batch_prev = event_ptr;
 575       event_ptr->batch_next->batch_next = NULL;
 576 
 577       if (new_iobuf->iolist->batch_complete == TRUE) {
 578         tmp->batch_complete = TRUE;
 579       } else {
 580         tmp->batch_complete = FALSE;
 581       }
 582 
 583       tmp->batch_size++;
 584       tmp->reqcnt++;
 585 
 586       // queue->listlen--;
 587       // because we didn't actually add an iobuf to the queue after all
 588       queue->iobufcnt--;
 589       addtoextraq((event *)new_iobuf);
 590 
 591       // printf("ioqueue_insert_batch_fcfs_to_queue:: queue contents after\n");
 592       // print_batch_fcfs_queue(queue);
 593 
 594       return;
 595     } else {
 596       tmp = tmp->next;
 597     }
 598   }
 599 
 600   // this batch doesn't exist in the queue yet, so insert new_iobuf at the end of the queue
 601   new_iobuf->next = queue->list->next;
 602   new_iobuf->prev = queue->list;
 603   new_iobuf->next->prev = new_iobuf;
 604   new_iobuf->prev->next = new_iobuf;
 605   queue->list = new_iobuf;
 606 
 607   new_iobuf->batch_size = 1;
 608   
 609   // printf("ioqueue_insert_batch_fcfs_to_queue:: queue contents after\n");
 610   // print_batch_fcfs_queue(queue);
 611 
 612   return;
 613 }

 677 static int ioqueue_check_concat (subqueue *queue, iobuf *req1, iobuf *req2, int concatmax)
 678 {
 679    ioreq_event *tmp;
 680    int seqscheme = queue->bigqueue->seqscheme;
 681    int (**concatok)(void *,int,int,int,int) = queue->bigqueue->concatok;
 682    void *concatokparam = queue->bigqueue->concatokparam;
 683 
 684    if (req1->next != req2) {
 685       fprintf(stderr, "Non-adjacent requests sent to ioqueue_check_concat\n");
 686       exit(1);
 687    }
 688    if (req1->flags & READ) {
 689       if (!(seqscheme & IOQUEUE_CONCAT_READS) || !(req2->flags & READ)) {
 690          return(0);
 691       }
 692    } else {
 693       if (!(seqscheme & IOQUEUE_CONCAT_WRITES) || (req2->flags & READ)) {
 694          return(0);
 695       }
 696    }
 697    if ((req1 == queue->list) || (req1 == req2) || !READY_TO_GO(req1,queue) || !READY_TO_GO(req2,queue) || ((req1->blkno + req1->totalsize) != req2->blkno) || ((req1->totalsize + req2->totalsize) > concatmax)) {
 698       return(0);
 699    }
 700    if (concatok) {
 701       if ((*concatok)(concatokparam, req1->blkno, req1->totalsize, req2->blkno, req2->totalsize) == 0) {
 702          return(0);
 703       }
 704    }
 705    tmp = req1->iolist;
 706    if (tmp == NULL) {
 707       req1->iolist = req2->iolist;
 708    } else {
 709       while (tmp->next) {
 710          tmp = tmp->next;
 711       }
 712       tmp->next = req2->iolist;
 713    }
 714    req1->next = req2->next;
 715    req1->next->prev = req1;
 716    req1->reqcnt += req2->reqcnt;
 717    req1->totalsize += req2->totalsize;
 718    if (queue->list == req2) {
 719       queue->list = req1;
 720    }
 721    queue->iobufcnt--;
 722    addtoextraq((event *) req2);
 723    return(1);
 724 }

 774 /* enabled).  Otherwise returns 1.                                           */

 777 static int ioqueue_seqstream_head (ioqueue *queue, iobuf *listhead, iobuf *current)
 778 {
 779    int idiff;
 780    int ret = 0;
 781 
 782    if (!(queue->seqscheme & (IOQUEUE_SEQSTREAM_READS|IOQUEUE_SEQSTREAM_WRITES|IOQUEUE_SEQSTREAM_EITHER))) {
 783       return(1);
 784    }
 785    idiff = queue->seqstreamdiff;
 786    if ((current == listhead) || !READY_TO_GO(current->prev,(&queue->base))) {
 787       ret = 1;
 788    }
 789    if (ret == 0) {
 790       if (current->flags & READ) {
 791          if (current->prev->flags & READ) {
 792             if (!(queue->seqscheme & (IOQUEUE_SEQSTREAM_READS|IOQUEUE_SEQSTREAM_EITHER))) {
 793                ret = 1;
 794             }
 795          } else {
 796             if (!(queue->seqscheme & IOQUEUE_SEQSTREAM_EITHER)) {
 797                ret = 1;
 798             }
 799          }
 800          if ((current->prev->blkno != current->blkno) && (((current->prev->blkno + current->prev->totalsize + idiff) < current->blkno))) {
 801             ret = 1;
 802          }
 803       } else {
 804          if (current->prev->flags & READ) {
 805             if (!(queue->seqscheme & IOQUEUE_SEQSTREAM_EITHER)) {
 806                ret = 1;
 807             }
 808          } else {
 809             if (!(queue->seqscheme & (IOQUEUE_SEQSTREAM_WRITES|IOQUEUE_SEQSTREAM_EITHER))) {
 810                ret = 1;
 811             }
 812          }
 813          if ((current->prev->blkno != current->blkno) && ((current->prev->blkno + current->prev->totalsize) != current->blkno)) {
 814             ret = 1;
 815          }
 816       }
 817    }
 818    return(ret);

 884          ret = 1;
 885       }
 886    }

 892 {

 894    int priority_factor;
 895    int age_factor;
 896    iobuf *tmp;
 897    int curr_effpri = 0, best_effpri = 0;
 898    iobuf *ret = NULL;
 899    int i;
 900 
 901    schedalg = queue->bigqueue->timeout.sched_alg;
 902    priority_factor = queue->bigqueue->priority.sched_alg;
 903    age_factor = queue->bigqueue->to_time;
 904    if (schedalg == ELEVATOR_LBN) {
 905       vscan_value = numlbns;
 906    } else if (schedalg == SSTF_LBN) {
 907       vscan_value = 0;
 908    }

 910    for (i = 0; i < queue->iobufcnt; i++) {
 911       if (READY_TO_GO(tmp,queue)) {
 912          curr_effpri = tmp->blkno - queue->lastblkno;
 913          if (schedalg == CYCLE_LBN) {
 914             if (curr_effpri < 0) {
 915                curr_effpri += numlbns;
 916             }
 917          } else if (curr_effpri) {
 918             if (curr_effpri < 0) {
 919                curr_effpri = -curr_effpri;
 920                if (queue->dir == ASC) {
 921                   curr_effpri += vscan_value;
 922                }
 923             } else {
 924                if (queue->dir == DESC) {
 925                   curr_effpri += vscan_value;

 927             }
 928          }

 931          curr_effpri += (age_factor * (int) (simtime - tmp->starttime)) / 1280;

 933             ret = tmp;
 934             best_effpri = curr_effpri;
 935          }
 936       }
 937       tmp = tmp->next;
 938    }
 939    return(ret);

 991          bottom = bottom->prev;

 996          bestbottom = bottom;
 997          while ((bottom != head) && ((bottom->prev->blkno == bestbottom->blkno) || (!(ioqueue_seqstream_head(queue->bigqueue, head, bottom))))) {
 998             bottom = bottom->prev;
 999             bestbottom = (READY_TO_GO(bottom,queue)) ? bottom : bestbottom;
1000          }

1002       }
1003    }
1004    if (temp->next != head) {
1005       if (temp->blkno >= queue->lastblkno) {
1006          top = temp;
1007       } else {
1008          top = temp->next;
1009       }

1020       diff2 = diff(queue->lastblkno, bottom->blkno);

1028       } else if ((diff1 > diff2) || (queue->sstfupdown)) {
1029          temp = bottom;
1030       } else {
1031          temp = top;
1032       }
1033       if (diff1 == diff2) {
1034          queue->sstfupdown = (queue->sstfupdown) ? 0 : 1;
1035          queue->sstfupdowncnt++;
1036       }
1037    }
1038    if (temp->blkno != queue->lastblkno) {
1039       queue->dir = (temp->blkno > queue->lastblkno) ? ASC : DESC;
1040    }
1041    top = (temp == top) ? bottom : top;
1042    if ((top) && READY_TO_GO(top,queue)) {
1043       int cylno1, cylno2;
1044       int surface1, surface2;
1045

1049       diff2 = abs(cylno2 - queue->optcylno);

1051          if (diff2) {

1053          }
1054       } else if (diff1) {

1056             diff1 += (int) (queue->vscan_value * (double) device_get_numcyls(temp->iolist->devno));
1057          }
1058       }
1059       if (diff1 > diff2) {
1060          stat_update(&queue->infopenalty, (double) (diff1 - diff2));
1061 /*
1062 fprintf (outputfile, "diff1 %d, diff2 %d, diffdiff %d\n", diff1, diff2, (diff1 - diff2));
1063 */
1064       } else if ((diff1 == diff2) && (diff1 == 0)) {
1065          if ((queue->optsurface == surface2) && (surface1 != surface2)) {
1066             stat_update(&queue->infopenalty, (double) 0);
1067          }
1068       }
1069    }
1070    return(temp);
1071 }

1076 /* Returns pointer to "best" request on queue->lastcylno, or first request
1077    on next available forward cylinder, or head (in priority order). */
1078 
1079 static iobuf *ioqueue_identify_request_on_cylinder (subqueue *queue)
1080 {
1081    iobuf *temp;
1082    iobuf *head;
1083    iobuf *cylstop;
1084    iobuf *surfstop;
1085 
1086    int lastcylno;
1087    int lastsurface;
1088    int lastblkno;
1089 
1090    lastcylno = queue->lastcylno;
1091    lastsurface = queue->lastsurface;
1092    lastblkno = queue->lastblkno;
1093 
1094    head = queue->list->next;
1095    temp = head;
1096 
1097    if ((head->cylinder < lastcylno) && (lastcylno <= queue->list->cylinder)) {
1098       while ((temp->next != head) && (lastcylno > temp->next->cylinder)) {
1099          temp = temp->next;
1100       }
1101       temp = temp->next;
1102    }
1103 
1104    if (temp->cylinder != lastcylno) {
1105       return(temp);
1106    }
1107 
1108    cylstop = temp;      /* first req on cylinder */
1109 
1110    while ((temp->next != head) && (lastcylno == temp->next->cylinder) && (lastsurface > temp->surface)) {
1111       temp = temp->next;
1112    }
1113 
1114    if (lastsurface > temp->surface) {
1115       temp = cylstop;   /* drop out and try from first req on cylinder */
1116 
1117    } else if (!queue->surfoverforw) {
1118       while ((temp->next != head) && (lastcylno == temp->next->cylinder) && (!READY_TO_GO(temp,queue) || (temp->blkno < lastblkno))) {
1119          temp = temp->next;
1120       }
1121       if (READY_TO_GO(temp,queue) && (temp->blkno >= lastblkno)) {
1122         return(temp);
1123       } else {
1124          temp = cylstop;   /* drop out and try from first req on cylinder */
1125       }
1126 
1127    } else {    /* surfoverforw */
1128 
1129       surfstop = temp;     /* first req on surface */
1130 
1131       while ((temp->next != head) && (lastcylno == temp->next->cylinder) && (lastsurface == temp->next->surface) && (!READY_TO_GO(temp,queue) || (temp->blkno < lastblkno))) {
1132          temp = temp->next;
1133       }
1134       if (READY_TO_GO(temp,queue) && (temp->blkno >= lastblkno) && (temp->surface == lastsurface)) {
1135          return(temp);
1136       }
1137 
1138       temp = surfstop;
1139 
1140       while ((temp->next != head) && (lastcylno == temp->next->cylinder) && !READY_TO_GO(temp,queue)) {
1141          temp = temp->next;
1142       }
1143       if (READY_TO_GO(temp,queue)) {
1144          return(temp);
1145       }
1146 
1147       temp = cylstop;   /* drop out and try from first req on cylinder */
1148    }
1149 
1150    /* drop out recovery point */
1151 
1152    if (!READY_TO_GO(temp,queue)) {
1153       while (!READY_TO_GO(temp,queue) && (temp->next != head) && (temp->next->cylinder == lastcylno)) {
1154          temp = temp->next;
1155       }
1156       temp = temp->next;
1157    }
1158    return(temp);
1159 }
