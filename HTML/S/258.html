 193 typedef struct seg {
 194    double       time;
 195    disk_buffer_state_t state;
 196    struct seg  *next;
 197    struct seg  *prev;
 198    int          startblkno;
 199    int          endblkno;
 200    disk_buffer_outstate_t outstate;
 201    int          outbcount;
 202    int          minreadaheadblkno;      /* min prefetch blkno + 1 */
 203    int          maxreadaheadblkno;      /* max prefetch blkno + 1 */
 204 
 205    struct diskreq_t *diskreqlist;       /* sorted by ascendingly first blkno */
 206    int          size;
 207                                         /* diskreqlist normally starts
 208                                          * as a single request, more
 209                                          * can be added by read hits
 210                                          * on the segment and write
 211                                          * combining */
 212 
 213    ioreq_event *access;                 /* copy of the active ioreq
 214                                          * using this seg -rcohen */
 215    int          hold_blkno;             /* used for prepending */
 216    int          hold_bcount;            /* sequential writes   */
 217    struct diskreq_t *recyclereq;        /* diskreq to recycle this seg */
 218 } segment;
