 109 {
 110    disksim->iotrace_info = DISKSIM_malloc (sizeof(iotrace_info_t));
 111    bzero ((char *)disksim->iotrace_info, sizeof(iotrace_info_t));
 112

 114    firstio = TRUE;
 115    lasttime = 0.0;
 116    basebigtime = -1;

 126 void iotrace_set_format (char *formatname)
 127 {
 128    if (disksim->iotrace_info == NULL) {
 129       iotrace_initialize_iotrace_info();
 130    }
 131 
 132    disksim->traceendian = _LITTLE_ENDIAN;
 133    disksim->traceheader = TRUE;
 134    if (strcmp(formatname, "0") == 0) {
 135       disksim->traceformat = DEFAULT;
 136    } else if (strcmp(formatname, "ascii") == 0) {
 137         /* default ascii trace format */
 138       disksim->traceformat = ASCII;
 139    } else if (strcmp(formatname, "raw") == 0) {
 140         /* format of traces collected at Michigan */
 141       disksim->traceformat = RAW;
 142    } else if (strcmp(formatname, "validate") == 0) {
 143         /* format of disk validation traces */
 144       disksim->traceformat = VALIDATE;
 145    } else if (strcmp(formatname, "hpl") == 0) {
 146         /* format of traces provided by HPLabs for research purposes */
 147       disksim->traceformat = HPL;
 148       disksim->traceendian = _BIG_ENDIAN;
 149    } else if (strcmp(formatname, "hpl2") == 0) {
 150         /* format of traces provided by HPLabs for research purposes,     */
 151         /* after they have been modified/combined by the `hplcomb' program */
 152       disksim->traceformat = HPL;
 153       disksim->traceendian = _BIG_ENDIAN;
 154       disksim->traceheader = FALSE;
 155    } else if (strcmp(formatname, "dec") == 0) {
 156         /* format of some traces provided by dec for research purposes */
 157       disksim->traceformat = DEC;
 158    } else if (strcmp(formatname, "emcsymm") == 0) {
 159         /* format of Symmetrix traces provided by EMC for research purposes */
 160       disksim->traceformat = EMCSYMM;
 161    } else if (strcmp(formatname, "emcbackend") == 0) {
 162         /* format of Symmetrix backend traces that have been converted to map
 163            logical devices to physical devices  */
 164       disksim->traceformat = EMCBACKEND;
 165    } else if (strcmp(formatname, "batch") == 0) {
 166         /* ascii traces with added batch information */
 167       disksim->traceformat = BATCH;
 168    } else {
 169       fprintf(stderr, "Unknown trace format - %s\n", formatname);
 170       exit(1);
 171    }
 172 }

 194 static int iotrace_read_short (FILE *tracefile, short *shortptr)
 195 {
 196    StaticAssert (sizeof(short) == 2);
 197    if (fread(shortptr, sizeof(short), 1, tracefile) != 1) {
 198       return(-1);
 199    }
 200    if (disksim->endian != disksim->traceendian) {
 201       *shortptr = ((*shortptr) << 8) + (((*shortptr) >> 8) & 0xFF);
 202    }
 203    return(0);
 204 }

 207 static int iotrace_read_int32 (FILE *tracefile, int32_t *intP)
 208 {
 209    int i;
 210    intchar swapval;
 211    intchar intcharval;
 212 
 213    StaticAssert (sizeof(int) == 4);
 214    if (fread(&intcharval.value, sizeof(int), 1, tracefile) != 1) {
 215       return(-1);
 216    }
 217    if (disksim->endian != disksim->traceendian) {
 218       for (i=0; i<sizeof(int); i++) {
 219          swapval.byte[i] = intcharval.byte[(sizeof(int) - i - 1)];
 220       }
 221 /*
 222       fprintf (outputfile, "intptr.value %x, swapval.value %x\n", intcharval.value, swapval.value);
 223 */
 224       intcharval.value = swapval.value;
 225    }
 226    *intP = intcharval.value;
 227    return(0);
 228 }

 234 ioreq_event * iotrace_validate_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 235 {
 236    char line[201];
 237    char rw;
 238    double servtime;
 239 
 240    if (fgets(line, 200, tracefile) == NULL) {
 241       addtoextraq((event *) new);
 242       return(NULL);
 243    }
 244    new->time = simtime + (validate_nextinter / (double) 1000);
 245    if (sscanf(line, "%c %s %d %d %lf %lf\n", 
 246               &rw, 
 247               validate_buffaction, 
 248               &new->blkno, 
 249               &new->bcount, 
 250               &servtime, 
 251               &validate_nextinter) != 6) 
 252    {
 253       fprintf(stderr, "Wrong number of arguments for I/O trace event type\n");
 254       ddbg_assert(0);
 255    }
 256    validate_lastserv = servtime / (double) 1000;
 257    if (rw == 'R') {
 258       new->flags = READ;
 259    } else if (rw == 'W') {
 260       new->flags = WRITE;
 261    } else {
 262       fprintf(stderr, "Invalid R/W value: %c\n", rw);
 263       exit(1);
 264    }
 265    new->devno = 0;
 266    new->buf = 0;
 267    new->opid = 0;
 268    new->cause = 0;
 269    new->busno = 0;
 270    new->tempint2 = 0;
 271    new->tempint1 = 0;
 272    validate_lastblkno = new->blkno;
 273    validate_lastbcount = new->bcount;
 274    validate_lastread = new->flags & READ;
 275    return(new);
 276 }

 287 static void iotrace_hpl_srt_convert_flags (ioreq_event *curr)
 288 {
 289    int flags;
 290 
 291    flags = curr->flags;
 292    curr->flags = 0;
 293    if (flags & HPL_READ) {
 294       curr->flags |= READ;
 295       hpreads++;
 296    } else {
 297       hpwrites++;
 298    }
 299    if (!(flags & HPL_ASYNC)) {
 300       curr->flags |= TIME_CRITICAL;
 301       if (curr->flags & READ) {
 302          syncreads++;
 303       } else {
 304          syncwrites++;
 305       }
 306    }
 307    if (flags & HPL_ASYNC) {
 308       if (curr->flags & READ) {
 309          curr->flags |= TIME_LIMITED;
 310          asyncreads++;
 311       } else {
 312          asyncwrites++;
 313       }
 314    }
 315 }

 318 static ioreq_event * iotrace_hpl_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 319 {
 320    int32_t size;
 321    int32_t id;
 322    int32_t sec;
 323    int32_t usec;
 324    int32_t val;
 325    int32_t junkint;
 326    unsigned int failure = 0;
 327 
 328    while (TRUE) {
 329       failure |= iotrace_read_int32(tracefile, &size);
 330       failure |= iotrace_read_int32(tracefile, &id);
 331       failure |= iotrace_read_int32(tracefile, &sec);
 332       failure |= iotrace_read_int32(tracefile, &usec);
 333       if (failure) {
 334          addtoextraq((event *) new);
 335          return(NULL);
 336       }
 337       if (((id >> 16) < 1) || ((id >> 16) > 4)) {
 338          fprintf(stderr, "Error in trace format - id %x\n", id);
 339          exit(1);
 340       }
 341       if (((id & 0xFFFF) != HPL_SHORTIO) && ((id & 0xFFFF) != HPL_SUSPECTIO)) {
 342          fprintf(stderr, "Unexpected record type - %x\n", id);
 343          exit(1);
 344       }
 345       new->time = (double) sec * (double) MILLI;
 346       new->time += (double) usec / (double) MILLI;
 347 
 348       if ((disksim->traceheader == FALSE) && (new->time == 0.0)) {
 349          tracebasetime = simtime;
 350       }
 351 
 352       failure |= iotrace_read_int32(tracefile, &val);    /* traced request start time */
 353       new->tempint1 = val;
 354       failure |= iotrace_read_int32(tracefile, &val);    /* traced request stop time */
 355       new->tempint2 = val;
 356       new->tempint2 -= new->tempint1;
 357       failure |= iotrace_read_int32(tracefile, &val);
 358       new->bcount = val;
 359       if (new->bcount & 0x000001FF) {
 360          fprintf(stderr, "HPL request for non-512B multiple size: %d\n", new->bcount);
 361          exit(1);
 362       }
 363       new->bcount = new->bcount >> 9;
 364       failure |= iotrace_read_int32(tracefile, &val);
 365       new->blkno = val;
 366       failure |= iotrace_read_int32(tracefile, &val);
 367       new->devno = (val >> 8) & 0xFF;
 368       failure |= iotrace_read_int32(tracefile, &val);       /* drivertype */
 369       failure |= iotrace_read_int32(tracefile, &val);       /* cylno */
 370       /* for convenience and historical reasons, this cast is being allowed */
 371       /* (the value is certain to be less than 32 sig bits, and will not be */
 372       /* used as a pointer).                                                */
 373       new->buf = (void *) val;
 374       failure |= iotrace_read_int32(tracefile, &val);
 375       new->flags = val;
 376       iotrace_hpl_srt_convert_flags(new);
 377       failure |= iotrace_read_int32(tracefile, &junkint);           /* info */
 378       size -= 13 * sizeof(int32_t);
 379       if ((id >> 16) == 4) {
 380          failure |= iotrace_read_int32(tracefile, &val);  /* queuelen */
 381          new->slotno = val;
 382          size -= sizeof(int32_t);
 383       }
 384       if ((id & 0xFFFF) == HPL_SUSPECTIO) {
 385          failure |= iotrace_read_int32(tracefile, &junkint);    /* susflags */
 386          size -= sizeof(int32_t);
 387       }
 388       if (failure) {
 389          addtoextraq((event *) new);
 390          return(NULL);
 391       }
 392       if (size) {
 393          fprintf(stderr, "Unmatched size for record - %d\n", size);
 394          exit(1);
 395       }
 396       new->cause = 0;
 397       new->opid = 0;
 398       new->busno = 0;
 399       if ((id & 0xFFFF) == HPL_SHORTIO) {
 400          return(new);
 401       }
 402    }
 403 }

 406 static int iotrace_month_convert (char *monthstr, int year)
 407 {
 408    if (strcmp(monthstr, "Jan") == 0) {
 409       return(0);
 410    } else if (strcmp(monthstr, "Feb") == 0) {
 411       return(31);
 412    } else if (strcmp(monthstr, "Mar") == 0) {
 413       return((year % 4) ? 59 : 60);
 414    } else if (strcmp(monthstr, "Apr") == 0) {
 415       return((year % 4) ? 90 : 91);
 416    } else if (strcmp(monthstr, "May") == 0) {
 417       return((year % 4) ? 120 : 121);
 418    } else if (strcmp(monthstr, "Jun") == 0) {
 419       return((year % 4) ? 151 : 152);
 420    } else if (strcmp(monthstr, "Jul") == 0) {
 421       return((year % 4) ? 181 : 182);
 422    } else if (strcmp(monthstr, "Aug") == 0) {
 423       return((year % 4) ? 212 : 213);
 424    } else if (strcmp(monthstr, "Sep") == 0) {
 425       return((year % 4) ? 243 : 244);
 426    } else if (strcmp(monthstr, "Oct") == 0) {
 427       return((year % 4) ? 273 : 274);
 428    } else if (strcmp(monthstr, "Nov") == 0) {
 429       return((year % 4) ? 304 : 305);
 430    } else if (strcmp(monthstr, "Dec") == 0) {
 431       return((year % 4) ? 334 : 335);
 432    }
 433    assert(0);
 434    return(-1);
 435 }

 438 static double iotrace_raw_get_hirestime (int bigtime, int smalltime)
 439 {
 440    unsigned int loresticks;
 441    int small, turnovers;
 442    int smallticks;
 443 
 444    if (basebigtime == -1) {
 445       basebigtime = bigtime;
 446       basesmalltime = smalltime;
 447       basesimtime = 0.0;
 448    } else {
 449       small = (basesmalltime - smalltime) & 0xFFFF;
 450       loresticks = (bigtime - basebigtime) * 11932 - small;
 451       turnovers = (int) (((double) loresticks / (double) 65536) + (double) 0.5);
 452       smallticks = turnovers * 65536 + small;
 453       basebigtime = bigtime;
 454       basesmalltime = smalltime;
 455       basesimtime += (double) smallticks * (double) 0.000838574;
 456    }
 457    return(basesimtime);
 458 }

 463 static ioreq_event * iotrace_raw_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 464 {
 465    int bigtime;
 466    short small;
 467    int smalltime;
 468    int failure = 0;
 469    char order, crit;
 470    double schedtime, donetime;
 471    int32_t val;
 472 
 473    failure |= iotrace_read_int32(tracefile, &val);
 474    bigtime = val;
 475    failure |= iotrace_read_short(tracefile, &small);
 476    smalltime = ((int) small) & 0xFFFF;
 477    new->time = iotrace_raw_get_hirestime(bigtime, smalltime);
 478    failure |= iotrace_read_short(tracefile, &small);
 479    failure |= iotrace_read_int32(tracefile, &val);
 480    bigtime = val;
 481    smalltime = ((int) small) & 0xFFFF;
 482    schedtime = iotrace_raw_get_hirestime(bigtime, smalltime);
 483    failure |= iotrace_read_int32(tracefile, &val);
 484    bigtime = val;
 485    failure |= iotrace_read_short(tracefile, &small);
 486    smalltime = ((int) small) & 0xFFFF;
 487    donetime = iotrace_raw_get_hirestime(bigtime, smalltime);
 488    failure |= iotrace_read_char(tracefile, &order);
 489    failure |= iotrace_read_char(tracefile, &crit);
 490    if (crit) {
 491       new->flags |= TIME_CRITICAL;
 492    }
 493    failure |= iotrace_read_int32(tracefile, &val);
 494    new->bcount = val >> 9;
 495    failure |= iotrace_read_int32(tracefile, &val);
 496    new->blkno = val;
 497    failure |= iotrace_read_int32(tracefile, &val);
 498    new->devno = val;
 499    failure |= iotrace_read_int32(tracefile, &val);
 500    new->flags = val & READ;
 501    new->cause = 0;
 502    new->buf = 0;
 503    new->opid = 0;
 504    new->busno = 0;
 505    new->tempint1 = (int)((schedtime - new->time) * (double) 1000);
 506    new->tempint2 = (int)((donetime - schedtime) * (double) 1000);
 507    if (failure) {
 508       addtoextraq((event *) new);
 509       new = NULL;
 510    }
 511    return(new);
 512 }

 515 static ioreq_event * iotrace_emcsymm_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 516 {
 517    char line[201];
 518    char operation[15];
 519    unsigned int director;
 520 
 521    if (fgets(line, 200, tracefile) == NULL) {
 522       addtoextraq((event *) new);
 523       return(NULL);
 524    }
 525    if (sscanf(line, "%lf %s %x %x %d %d\n", &new->time, operation, &director, &new->devno, &new->blkno, &new->bcount) != 6) {
 526       fprintf(stderr, "Wrong number of arguments for I/O trace event type\n");
 527       fprintf(stderr, "line: %s", line);
 528       ddbg_assert(0);
 529    }
 530    if (!strcasecmp(operation,"Read")) {
 531       new->flags = READ;
 532    } else if (!strcasecmp(operation,"Write")) {
 533       new->flags = WRITE;
 534    } else {
 535       fprintf(stderr, "Unknown operation: %s in iotrace event\n",operation);
 536       fprintf(stderr, "line: %s", line);
 537       exit(1);
 538    }
 539    new->buf = 0;
 540    new->opid = 0;
 541    new->busno = 0;
 542    new->cause = 0;
 543    return(new);
 544 }

 546 static ioreq_event * iotrace_emcbackend_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 547 {
 548    char line[201];
 549    char operation[15];
 550    unsigned int director;
 551    char bus[2];
 552    unsigned int disk, hyper;
 553 
 554    if (fgets(line, 200, tracefile) == NULL) {
 555       addtoextraq((event *) new);
 556       return(NULL);
 557    }
 558 
 559    if (sscanf(line, "%lf %s %x %x %d %d %s %d %d\n", 
 560               &new->time, operation, &director, &hyper, &new->blkno, &new->bcount, bus, &disk, &new->devno) != 9) {
 561       fprintf(stderr, "Wrong number of arguments for I/O trace event type\n");
 562       fprintf(stderr, "line: %s", line);
 563       exit(0);
 564    }
 565    if (!strcasecmp(operation,"Read")) {
 566       new->flags = READ;
 567    } else if (!strcasecmp(operation,"Write")) {
 568       new->flags = WRITE;
 569    } else {
 570       fprintf(stderr, "Unknown operation: %s in iotrace event\n",operation);
 571       fprintf(stderr, "line: %s", line);
 572       exit(0);
 573    }
 574 
 575    new->time *= 1000.0;  /* emc trace times are in seconds!!  */
 576 
 577    new->buf = 0;
 578    new->opid = 0;
 579    new->busno = 0;
 580    new->cause = 0;
 581    return(new);
 582 }

 585 static ioreq_event * iotrace_ascii_get_ioreq_event (FILE *tracefile, ioreq_event *new)
 586 {
 587    char line[201];
 588 
 589    if (fgets(line, 200, tracefile) == NULL) {
 590       addtoextraq((event *) new);
 591       return(NULL);
 592    }
 593    if (sscanf(line, "%lf %d %d %d %x\n", &new->time, &new->devno, &new->blkno, &new->bcount, &new->flags) != 5) {
 594       fprintf(stderr, "Wrong number of arguments for I/O trace event type\n");
 595       fprintf(stderr, "line: %s", line);
 596       ddbg_assert(0);
 597    }
 598    if (new->flags & ASYNCHRONOUS) {
 599       new->flags |= (new->flags & READ) ? TIME_LIMITED : 0;
 600    } else if (new->flags & SYNCHRONOUS) {
 601       new->flags |= TIME_CRITICAL;
 602    }
 603 
 604    new->buf = 0;
 605    new->opid = 0;
 606    new->busno = 0;
 607    new->cause = 0;
 608    return(new);
 609 }

 688 static void iotrace_hpl_srt_tracefile_start (char *tracedate)
 689 {
 690    char crap[40];
 691    char monthstr[40];
 692    int day;
 693    int hour;
 694    int minute;
 695    int second;
 696    int year;
 697 
 698    if (sscanf(tracedate, "%s\t= \"%s %s %d %d:%d:%d %d\";\n", crap, crap, monthstr, &day, &hour, &minute, &second, &year) != 8) {
 699       fprintf(stderr, "Format problem with 'tracedate' line in HPL trace - %s\n", tracedate);
 700       exit(1);
 701    }
 702    if (baseyear == 0) {
 703       baseyear = year;
 704    }
 705    day = day + iotrace_month_convert(monthstr, year);
 706    if (year != baseyear) {
 707       day += (baseyear % 4) ? 365 : 366;
 708    }
 709    if (baseday == 0) {
 710       baseday = day;
 711    }
 712    second = second + (60 * minute) + (3600 * hour) + (86400 * (day - baseday));
 713    if (basesecond == 0) {
 714       basesecond = second;
 715    }
 716    second -= basesecond;
 717    tracebasetime += (double) 1000 * (double) second;
 718 }

 721 static void iotrace_hpl_initialize_file (FILE *tracefile, int print_tracefile_header)
 722 {
 723    char letter = '0';
 724    char line[201];
 725    char linetype[40];
 726 
 727    if (disksim->traceheader == FALSE) {
 728       return;
 729    }
 730    while (1) {
 731       if (fgets(line, 200, tracefile) == NULL) {
 732          fprintf(stderr, "No 'tracedate' line in HPL trace\n");
 733          exit(1);
 734       }
 735       sscanf(line, "%s", linetype);
 736       if (strcmp(linetype, "tracedate") == 0) {
 737          break;
 738       }
 739    }
 740    iotrace_hpl_srt_tracefile_start(line);
 741    while (letter != 0x0C) {
 742       if (fscanf(tracefile, "%c", &letter) != 1) {
 743          fprintf(stderr, "End of header information never found - end of file\n");
 744          exit(1);
 745       }
 746       if ((print_tracefile_header) && (letter != 0x0C)) {
 747          printf("%c", letter);
 748       }
 749    }
 750 }

 763    if (disksim->iotrace_info == NULL) {
 764       return;

 769       fprintf(outfile, "Total writes:   \t%d\t%5.2f\n", hpwrites, ((double) hpwrites / (double) (hpreads + hpwrites)));
 770       fprintf(outfile, "Sync Reads:  \t%d\t%5.2f\t%5.2f\n", syncreads, ((double) syncreads / (double) (hpreads + hpwrites)), ((double) syncreads / (double) hpreads));
 771       fprintf(outfile, "Sync Writes: \t%d\t%5.2f\t%5.2f\n", syncwrites, ((double) syncwrites / (double) (hpreads + hpwrites)), ((double) syncwrites / (double) hpwrites));
 772       fprintf(outfile, "Async Reads: \t%d\t%5.2f\t%5.2f\n", asyncreads, ((double) asyncreads / (double) (hpreads + hpwrites)), ((double) asyncreads / (double) hpreads));
 773       fprintf(outfile, "Async Writes:\t%d\t%5.2f\t%5.2f\n", asyncwrites, ((double) asyncwrites / (double) (hpreads + hpwrites)), ((double) asyncwrites / (double) hpwrites));
 774    }
