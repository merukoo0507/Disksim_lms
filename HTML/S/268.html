 127 static void controller_smart_issue_access (void *issuefuncparam, ioreq_event *curr)
 128 {
 129    controller *currctlr = issuefuncparam;
 130    struct ioq *queue = currctlr->devices[curr->devno].queue;
 131    int numout = ioqueue_get_reqoutstanding(queue);
 132 
 133    /* in case the cache changes to which device the request is sent */
 134 //fprintf (stderr, "busno %x, buspath %x, slotno %x, slotpath %x\n", curr->busno, currctlr->devices[curr->devno].buspath.value, curr->slotno, currctlr->devices[curr->devno].slotpath.value);
 135    curr->busno = currctlr->devices[curr->devno].buspath.value;
 136    curr->slotno = currctlr->devices[curr->devno].slotpath.value;
 137 
 138    ioqueue_add_new_request(queue, curr);
 139    if (numout < currctlr->devices[curr->devno].maxoutstanding) {
 140       ioreq_event *sched = ioqueue_get_next_request(queue);
 141       controller_send_event_down_path(currctlr, sched, currctlr->ovrhd_disk_request);
 142    }
 143 }

 146 static void controller_smart_disk_data_transfer (controller *currctlr, ioreq_event *curr)
 147 {
 148    ioreq_event *tmp = (ioreq_event *) getfromextraq();
 149 /*
 150 fprintf (outputfile, "%f: controller_smart_disk_data_transfer: devno %d, bcount %d\n", simtime, curr->devno, curr->bcount);
 151 */
 152    curr->time = max(device_get_blktranstime(curr), currctlr->blktranstime);
 153    tmp->time = simtime + ((double) curr->bcount * curr->time);
 154    tmp->type = CONTROLLER_DATA_TRANSFER_COMPLETE;
 155    tmp->devno = curr->devno;
 156    tmp->blkno = curr->blkno;
 157    tmp->bcount = curr->bcount;
 158    tmp->tempint2 = currctlr->ctlno;
 159    tmp->tempptr1 = curr;
 160 
 161    /* want to use the tempptr1 value for something else! */
 162    curr->tempptr1 = tmp;
 163    curr->next = currctlr->datatransfers;
 164    curr->prev = NULL;
 165    if (curr->next) {
 166       curr->next->prev = curr;
 167    }
 168    currctlr->datatransfers = curr;
 169 
 170    addtointq((event *) tmp);
 171 }

 174 static void controller_smart_disk_data_transfer_complete (controller *currctlr, ioreq_event *curr)
 175 {
 176    ioreq_event *tmp = (ioreq_event *) curr->tempptr1;
 177 
 178    tmp->bcount -= curr->bcount;
 179    addtoextraq((event *) curr);
 180    ASSERT(tmp->bcount >= 0);
 181    if (tmp->bcount == 0) {
 182       if (tmp->next) {
 183          tmp->next->prev = tmp->prev;
 184       }
 185       if (tmp->prev) {
 186          tmp->prev->next = tmp->next;
 187       } else {
 188          currctlr->datatransfers = tmp->next;
 189       }
 190       tmp->time = simtime;
 191       addtointq((event *) tmp);
 192    } else {
 193       fprintf(stderr, "Haven't required less than all out transfer at controller_smart_disk_data_transfer_complete\n");
 194       exit(1);
 195    }
 196 }

 212 static void controller_smart_host_data_transfer_complete (controller *currctlr, ioreq_event *curr)
 213 {
 214    /* DMA to/from host complete */
 215 
 216    if (curr->flags & READ) {
 217       currctlr->cache->cache_free_block_clean(currctlr->cache, curr);
 218       controller_smart_request_complete(currctlr, curr);
 219    } else {
 220       /* cache will call "done" function if request doesn't block */
 221       currctlr->cache->cache_free_block_dirty(currctlr->cache, curr, &disksim->donefunc_ctlrsmart_write, currctlr);
 222    }
 223    if (currctlr->hostwaiters) {
 224       curr = currctlr->hostwaiters->next;
 225       if (curr->next == curr) {
 226          currctlr->hostwaiters = NULL;
 227       } else {
 228          currctlr->hostwaiters->next = curr->next;
 229       }
 230       curr->next = NULL;
 231                                        /* Time for DMA */
 232       curr->time = simtime + ((double) curr->bcount * currctlr->blktranstime);
 233       addtointq((event *) curr);
 234    } else {
 235       currctlr->hosttransfer = FALSE;
 236    }
 237 }

 240 static void controller_smart_host_data_transfer (void *donefuncparam, ioreq_event *curr)
 241 {
 242    controller *currctlr = donefuncparam;
 243 
 244    /* DMA data to/from host */
 245 
 246    curr->type = CONTROLLER_DATA_TRANSFER_COMPLETE;
 247    curr->tempint2 = currctlr->ctlno;
 248    curr->tempptr1 = NULL;
 249 
 250    if (currctlr->hosttransfer) {
 251       if (currctlr->hostwaiters) {
 252          curr->next = currctlr->hostwaiters->next;
 253          currctlr->hostwaiters->next = curr;
 254       } else {
 255          curr->next = curr;
 256       }
 257       currctlr->hostwaiters = curr;
 258       return;
 259    }
 260                                        /* Time for DMA */
 261    curr->time = simtime + ((double) curr->bcount * currctlr->blktranstime);
 262    currctlr->hosttransfer = TRUE;
 263    addtointq((event *) curr);
 264 }

 279 static void controller_smart_access_complete (controller *currctlr, ioreq_event *curr)
 280 {
 281    struct ioq *queue = currctlr->devices[curr->devno].queue;
 282    ioreq_event *done = ioreq_copy(curr);
 283    int devno = curr->devno;
 284    int numout;
 285 
 286    /* Responds to completion interrupt */
 287 
 288    done->type = IO_INTERRUPT_COMPLETE;
 289    currctlr->outbusowned = controller_get_downward_busno(currctlr, done, NULL);
 290    controller_send_event_down_path(currctlr, done, currctlr->ovrhd_disk_complete);
 291    currctlr->outbusowned = -1;
 292 
 293    /* Handles request completion, including call-backs into cache */
 294 
 295    curr = ioqueue_physical_access_done(queue, curr);
 296    while ((done = curr)) {
 297       curr = curr->next;
 298       /* call back into cache with completion -- let it do request_complete */
 299       controller_smart_wakeup(currctlr, currctlr->cache->cache_disk_access_complete(currctlr->cache, done));
 300    }
 301 
 302    /* Initiate another request, if any pending */
 303 
 304    numout = ioqueue_get_reqoutstanding(queue);
 305    if ((numout < currctlr->devices[devno].maxoutstanding) && (curr = ioqueue_get_next_request(queue))) {
 306       controller_send_event_down_path(currctlr, curr, currctlr->ovrhd_disk_request);
 307    }
 308 }

 339 static void controller_smart_interrupt_arrive (controller *currctlr, ioreq_event *curr)
 340 {
 341    switch (curr->cause) {
 342 
 343       case COMPLETION:
 344          controller_smart_access_complete(currctlr, curr);
 345          break;
 346 
 347       case RECONNECT:
 348          controller_smart_reconnect_to_transfer(currctlr, curr);
 349          break;
 350 
 351       case DISCONNECT:
 352          controller_smart_disconnect(currctlr, curr);
 353          break;
 354 
 355       case READY_TO_TRANSFER:
 356          controller_smart_ready_to_transfer(currctlr, curr);
 357          break;
 358 
 359       default:
 360          fprintf(stderr, "Unknown interrupt cause in smart_interrupt_arrive: %d\n", curr->cause);
 361          exit(1);
 362    }
 363 }

 381 void controller_smart_event_arrive (controller *currctlr, ioreq_event *curr)
 382 {
 383    switch (curr->type) {
 384 
 385       case IO_ACCESS_ARRIVE:
 386          controller_smart_request_arrive(currctlr, curr);
 387          break;
 388 
 389       case IO_INTERRUPT_ARRIVE:
 390          controller_smart_interrupt_arrive(currctlr, curr);
 391          break;
 392 
 393       case IO_INTERRUPT_COMPLETE:
 394          controller_smart_interrupt_complete(currctlr, curr);
 395          break;
 396 
 397       case DEVICE_DATA_TRANSFER_COMPLETE:
 398          controller_smart_disk_data_transfer(currctlr, curr);
 399          break;
 400 
 401       case CONTROLLER_DATA_TRANSFER_COMPLETE:
 402          if (curr->tempptr1) {
 403             controller_smart_disk_data_transfer_complete(currctlr, curr);
 404          } else {
 405             controller_smart_host_data_transfer_complete(currctlr, curr);
 406          }
 407          break;
 408 
 409       case IO_QLEN_MAXCHECK:
 410          curr->tempint1 = currctlr->ctlno;
 411          curr->tempint2 = currctlr->maxoutstanding;
 412          curr->bcount = currctlr->cache->cache_get_maxreqsize(currctlr->cache);
 413          break;
 414 
 415       default:
 416          fprintf(stderr, "Unknown event type arriving at 53c700 controller: %d\n", curr->type);
 417          exit(1);
 418    }
 419 }

 434 void controller_smart_initialize (controller *currctlr)
 435 {
 436    int numdevs = device_get_numdevices();
 437    device *currdev;
 438    int i;
 439 
 440    controller_smart_setcallbacks ();
 441    currctlr->numdevices = numdevs;
 442    currctlr->devices = (device *) DISKSIM_malloc(numdevs * sizeof(device));
 443    ASSERT(currctlr->devices != NULL);
 444    for (i=0; i<numdevs; i++) {
 445       currdev = &currctlr->devices[i];
 446       currdev->devno = i;
 447       currdev->busy = FALSE;
 448       currdev->flag = 0;
 449       currdev->queue = ioqueue_copy(currctlr->queue);
 450       ioqueue_initialize(currdev->queue, i);
 451       iosim_get_path_to_device (0, i, &currdev->buspath, &currdev->slotpath);
 452       currdev->maxoutstanding = currctlr->maxdiskqsize;
 453    }
 454    currctlr->cache->cache_initialize(currctlr->cache, &disksim->issuefunc_ctlrsmart, currctlr, &disksim->queuefind_ctlrsmart, currctlr, &disksim->wakeupfunc_ctlrsmart, currctlr, numdevs);
 455 }

 472 void controller_smart_printstats (controller *currctlr, char *prefix)
 473 {
 474    ctlrinfo_t *ctlrinfo = disksim->ctlrinfo;
 475    int devno;
 476    char devprefix[181];
 477    struct ioq **queueset = DISKSIM_malloc (currctlr->numdevices * sizeof(void *));
 478 
 479    if (ctlrinfo->ctl_printcachestats) {
 480       currctlr->cache->cache_printstats(currctlr->cache, prefix);
 481    }
 482 
 483    for (devno=0; devno<currctlr->numdevices; devno++) {
 484       queueset[devno] = currctlr->devices[devno].queue;
 485    }
 486    sprintf (devprefix, "%sdevices ", prefix);
 487    ioqueue_printstats (queueset, currctlr->numdevices, devprefix);
 488 
 489    if ((ctlrinfo->ctl_printperdiskstats) && (currctlr->numdevices > 1)) {
 490       for (devno=0; devno<currctlr->numdevices; devno++) {
 491          struct ioq *queue = currctlr->devices[devno].queue;
 492          if (ioqueue_get_number_of_requests(queue)) {
 493             sprintf(devprefix, "%sdevice #%d ", prefix, devno);
 494             ioqueue_printstats(&queue, 1, devprefix);
 495          }
 496       }
 497    }
 498 }
