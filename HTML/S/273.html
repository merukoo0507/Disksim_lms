 133 int disk_add(struct disk *d) {
 134   int c;
 135 
 136   if(!disksim->diskinfo) disk_initialize_diskinfo();
 137   
 138   for(c = 0; c < disksim->diskinfo->disks_len; c++) {
 139     if(!disksim->diskinfo->disks[c]) {
 140       disksim->diskinfo->disks[c] = d;
 141      NUMDISKS++;
 142       return c;
 143     }
 144   }
 145 
 146   /* note that numdisks must be equal to diskinfo->disks_len */
 147   disksim->diskinfo->disks = realloc(disksim->diskinfo->disks, 
 148                            2 * NUMDISKS * sizeof(disk *));
 149   bzero(disksim->diskinfo->disks + NUMDISKS, NUMDISKS);
 150   disksim->diskinfo->disks[c] = d;
 151   NUMDISKS++;
 152   disksim->diskinfo->disks_len *= 2;
 153   return c;
 154 }

 262 static void disk_postpass_perdisk (disk *currdisk)
 263 {
 264   if ((currdisk->contread) && (currdisk->enablecache == FALSE)) {
 265     fprintf(stderr, "Makes no sense to use read-ahead but not use caching\n");
 266     ddbg_assert(0);;
 267   }
 268 
 269   if ((currdisk->contread == BUFFER_NO_READ_AHEAD) &&
 270       (currdisk->minreadahead > 0)) {
 271     // fprintf(stderr, "'Minimum read-ahead (blks)' forced to zero due to 'Buffer continuous read' value\n");
 272     currdisk->minreadahead = 0;
 273   }
 274 
 275   if ((currdisk->maxreadahead < currdisk->minreadahead) ||
 276       ((currdisk->contread == BUFFER_NO_READ_AHEAD) &&
 277        (currdisk->maxreadahead > 0))) {
 278     // fprintf(stderr, "'Maximum read-ahead (blks)' forced to zero due to 'Buffer continuous read' value\n");
 279     currdisk->maxreadahead = currdisk->minreadahead;
 280   }
 281 
 282   if (currdisk->keeprequestdata >= 0) {
 283     currdisk->keeprequestdata = abs(currdisk->keeprequestdata - 1);
 284   }
 285 
 286   if ((currdisk->readanyfreeblocks != FALSE) &&
 287       (currdisk->enablecache == FALSE)) {
 288     fprintf(stderr, "Makes no sense to read blocks with caching disabled\n");
 289     ddbg_assert(0);;
 290   }
 291 
 292   if ((currdisk->fastwrites) && (currdisk->enablecache == FALSE)) {
 293     fprintf(stderr, "Can't use fast write if not employing caching\n");
 294     ddbg_assert(0);;
 295   }
 296 
 297   /* ripped from the override function */
 298 
 299   if ((currdisk->readaheadifidle != 0) && (currdisk->readaheadifidle != 1)) {
 300     fprintf(stderr, "Invalid value for readaheadifidle in disk_postpass_perdisk: %d\n", currdisk->readaheadifidle);
 301     ddbg_assert(0);;
 302   }
 303 
 304   if ((currdisk->writecomb != 0) && (currdisk->writecomb != 1)) {
 305     fprintf(stderr, "Invalid value for writecomb in disk_postpass_perdisk: %d\n", currdisk->writecomb);
 306     ddbg_assert(0);;
 307   }
 308 
 309   if (currdisk->maxqlen < 0) {
 310     fprintf(stderr, "Invalid value for maxqlen in disk_postpass_perdisk: %d\n", currdisk->maxqlen);
 311     ddbg_assert(0);;
 312   }
 313 
 314   if ((currdisk->hold_bus_for_whole_read_xfer != 0) &&
 315       (currdisk->hold_bus_for_whole_read_xfer != 1)) {
 316     fprintf(stderr, "Invalid value for hold_bus_for_whole_read_xfer in disk_postpass_perdisk: %d\n", currdisk->hold_bus_for_whole_read_xfer);
 317     ddbg_assert(0);;
 318   }
 319 
 320   if ((currdisk->hold_bus_for_whole_write_xfer != 0) &&
 321       (currdisk->hold_bus_for_whole_write_xfer != 1)) {
 322     fprintf(stderr, "Invalid value for hold_bus_for_whole_write_xfer in disk_postpass_perdisk: %d\n", currdisk->hold_bus_for_whole_write_xfer);
 323     ddbg_assert(0);;
 324   }
 325   if ((currdisk->hold_bus_for_whole_read_xfer == 1) &&
 326       ((currdisk->sneakyfullreadhits == 1) ||
 327        (currdisk->sneakypartialreadhits == 1) ||
 328        (currdisk->sneakyintermediatereadhits == 1))) {
 329     fprintf(stderr, "hold_bus_for_whole_read_xfer and one or more sneakyreadhits detected disk_postpass_perdisk\n");
 330     ddbg_assert(0);;
 331   }
 332 
 333   if ((currdisk->almostreadhits != 0) && (currdisk->almostreadhits != 1)) {
 334     fprintf(stderr, "Invalid value for almostreadhits in disk_postpass_perdisk: %d\n", currdisk->almostreadhits);
 335     ddbg_assert(0);;
 336   }
 337 
 338   if ((currdisk->sneakyfullreadhits != 0) &&
 339       (currdisk->sneakyfullreadhits != 1)) {
 340     fprintf(stderr, "Invalid value for sneakyfullreadhits in disk_postpass_perdisk: %d\n", currdisk->sneakyfullreadhits);
 341     ddbg_assert(0);;
 342   }
 343   if ((currdisk->sneakyfullreadhits == 1) &&
 344       (currdisk->hold_bus_for_whole_read_xfer == 1)) {
 345     fprintf(stderr, "hold_bus_for_whole_read_xfer and sneakyfullreadhits detected disk_postpass_perdisk\n");
 346     ddbg_assert(0);;
 347   }
 348 
 349   if ((currdisk->sneakypartialreadhits != 0) &&
 350       (currdisk->sneakypartialreadhits != 1)) {
 351     fprintf(stderr, "Invalid value for sneakypartialreadhits in disk_postpass_perdisk: %d\n", currdisk->sneakypartialreadhits);
 352     ddbg_assert(0);;
 353   }
 354   if ((currdisk->sneakypartialreadhits == 1) &&
 355       (currdisk->hold_bus_for_whole_read_xfer == 1)) {
 356     fprintf(stderr, "hold_bus_for_whole_read_xfer and sneakypartialreadhits detected disk_postpass_perdisk\n");
 357     ddbg_assert(0);;
 358   }
 359 
 360   if ((currdisk->sneakyintermediatereadhits != 0) &&
 361       (currdisk->sneakyintermediatereadhits != 1)) {
 362     fprintf(stderr, "Invalid value for sneakyintermediatereadhits in disk_postpass_perdisk: %d\n", currdisk->sneakyintermediatereadhits);
 363     ddbg_assert(0);;
 364   }
 365   if ((currdisk->sneakyintermediatereadhits == 1) &&
 366       (currdisk->hold_bus_for_whole_read_xfer == 1)) {
 367     fprintf(stderr, "hold_bus_for_whole_read_xfer and sneakyintermediatereadhits detected disk_postpass_perdisk\n");
 368     ddbg_assert(0);;
 369   }
 370 
 371   if ((currdisk->readhitsonwritedata != 0) &&
 372       (currdisk->readhitsonwritedata != 1)) {
 373     fprintf(stderr, "Invalid value for readhitsonwritedata in disk_postpass_perdisk: %d\n", currdisk->readhitsonwritedata);
 374     ddbg_assert(0);;
 375   }
 376 
 377   if ((currdisk->writeprebuffering != 0) &&
 378       (currdisk->writeprebuffering != 1)) {
 379     fprintf(stderr, "Invalid value for writeprebuffering in disk_postpass_perdisk: %d\n", currdisk->writeprebuffering);
 380     ddbg_assert(0);;
 381   }
 382 
 383   if ((currdisk->preseeking != 0) && (currdisk->preseeking != 1) &&
 384       (currdisk->preseeking != 2)) {
 385     fprintf(stderr, "Invalid value for preseeking in disk_postpass_perdisk: %d\n", currdisk->preseeking);
 386     ddbg_assert(0);;
 387   }
 388 
 389   if ((currdisk->neverdisconnect != 0) && (currdisk->neverdisconnect != 1)) {
 390     fprintf(stderr, "Invalid value for neverdisconnect in disk_postpass_perdisk: %d\n", currdisk->neverdisconnect);
 391     ddbg_assert(0);;
 392   }
 393 
 394   if (currdisk->numsegs < 1) {
 395     fprintf(stderr, "Invalid value for numsegs in disk_postpass_perdisk: %d\n", currdisk->numsegs);
 396     ddbg_assert(0);;
 397   }
 398 
 399   if ((currdisk->segsize < 1) ||
 400       (currdisk->segsize > currdisk->model->dm_sectors)) {
 401     fprintf(stderr, "Invalid value for segsize in disk_postpass_perdisk: %d\n", currdisk->model->dm_sectors);
 402     ddbg_assert(0);;
 403   }
 404 
 405   if ((currdisk->numwritesegs > currdisk->numsegs) ||
 406       (currdisk->numwritesegs < 1)) {
 407     fprintf(stderr, "Invalid value for numwritesegs in disk_postpass_perdisk: %d\n", currdisk->numwritesegs);
 408     ddbg_assert(0);;
 409   }
 410 
 411   if ((currdisk->numsegs <= 1) && (currdisk->dedicatedwriteseg)) {
 412     fprintf(stderr, "Must have more segments than dedicated write segments\n");
 413     ddbg_assert(0);;
 414   }
 415   if (((int)currdisk->dedicatedwriteseg != 0) &&
 416       ((int)currdisk->dedicatedwriteseg != 1)) {
 417     fprintf(stderr, "Invalid value for dedicatedwriteseg in disk_postpass_perdisk: %d\n", (int)currdisk->dedicatedwriteseg);
 418     ddbg_assert(0);;
 419   }
 420 
 421   if ((currdisk->fastwrites != 0) && (currdisk->fastwrites != 1) &&
 422       (currdisk->fastwrites != 2)) {
 423     fprintf(stderr, "Invalid value for fastwrites in disk_postpass_perdisk: %d\n", currdisk->fastwrites);
 424     ddbg_assert(0);;
 425   }
 426   if ((currdisk->fastwrites != 0) && (currdisk->enablecache == FALSE)) {
 427     fprintf(stderr, "disk_postpass_perdisk:  Can't use fast write if not employing caching\n");
 428     ddbg_assert(0);;
 429   }
 430 
 431   if (currdisk->model->dm_surfaces < 1) {
 432     fprintf(stderr, "Invalid value for numsurfaces in disk_postpass_perdisk: %d\n", currdisk->model->dm_surfaces);
 433     ddbg_assert(0);;
 434   }
 435 
 436   /* This is probably too difficult to be worthwhile  -rcohen
 437      if (currdisk->model->dm_cyls < 1) {
 438      fprintf(stderr, "Invalid value for numcyls in disk_postpass_perdisk: %d\n", currdisk->model->dm_cyls);
 439      ddbg_assert(0);;
 440      }
 441   */
 442 }

 445 static void disk_postpass (void)
 446 {
 447   int i;
 448 
 449   if (NUMDISKS == 0) {
 450     return;
 451   }
 452 
 453   if (reading_buffer_whole_servtime < 0.0) {
 454     fprintf(stderr, "Invalid value for reading_buffer_whole_servtime in disk_postpass: %f\n", reading_buffer_whole_servtime);
 455     ddbg_assert(0);;
 456   }
 457 
 458 
 459   ddbg_assert3(buffer_whole_servtime >= 0.0, ("Invalid value for buffer_whole_servtime in disk_postpass: %f\n", buffer_whole_servtime));
 460 
 461 
 462   if (reading_buffer_partial_servtime < 0.0) {
 463     fprintf(stderr, "Invalid value for reading_buffer_partial_servtime in disk_postpass: %f\n", reading_buffer_partial_servtime);
 464     ddbg_assert(0);;
 465   }
 466 
 467   if (buffer_partial_servtime < 0.0) {
 468     fprintf(stderr, "Invalid value for buffer_partial_servtime in disk_postpass: %f\n", buffer_partial_servtime);
 469     ddbg_assert(0);
 470   }
 471 
 472   for (i = 0; i < MAXDEVICES; i++) {
 473     disk *currdisk = getdisk (i);
 474     if (currdisk) {
 475       disk_postpass_perdisk(currdisk);
 476     }
 477   }
 478 
 479   /* syncset stuff, perhaps should be in its own function */
 480   /* unnecessary for now anyway */
 481 
 482   /*
 483     for (i = 0; i < MAXDEVICES; i++) {
 484     disk *currdisk = getdisk (i);
 485     if (currdisk) {
 486     currdisk->syncset = 0;
 487     }
 488     }
 489   */
 490 
 491 }

 538 void disk_set_syncset (int setstart, int setend)
 539 {
 540   if ((setstart < 0) || (setend >= MAXDEVICES) || (setend < setstart)) {
 541     fprintf (stderr, "Bad range passed to disk_set_syncset (%d - %d)\n", setstart, setend);
 542     exit (0);
 543   }
 544 
 545   numsyncsets++;
 546   for (; setstart <= setend; setstart++) {
 547     disk *currdisk = getdisk (setstart);
 548     if (currdisk->syncset != 0) {
 549       fprintf (stderr, "Overlapping synchronized disk sets (%d and %d)\n", numsyncsets, currdisk->syncset);
 550       exit (0);
 551     }
 552     currdisk->syncset = numsyncsets;
 553   }
 554 }

 576 static void disk_buffer_printstats (int *set, int setsize, char *prefix)
 577 {
 578   int writecombs = 0;
 579   int readmisses = 0;
 580   int writemisses = 0;
 581   int fullreadhits = 0;
 582   int prependhits = 0;
 583   int appendhits = 0;
 584   int readinghits = 0;
 585   double runreadingsize = 0.0;
 586   double remreadingsize = 0.0;
 587   int parthits = 0;
 588   double runpartsize = 0.0;
 589   double rempartsize = 0.0;
 590   double waitingforbus = 0.0;
 591   int numbuswaits = 0;
 592   int hits;
 593   int misses;
 594   int total;
 595   int reads;
 596   int writes;
 597   int i;
 598 
 599   if (device_printbufferstats == FALSE)
 600     return;
 601 
 602   for (i=0; i<setsize; i++) {
 603     disk *currdisk = getdisk (set[i]);
 604     writecombs += currdisk->stat.writecombs;
 605     readmisses += currdisk->stat.readmisses;
 606     writemisses += currdisk->stat.writemisses;
 607     fullreadhits += currdisk->stat.fullreadhits;
 608     prependhits += currdisk->stat.prependhits;
 609     appendhits += currdisk->stat.appendhits;
 610     readinghits += currdisk->stat.readinghits;
 611     runreadingsize += currdisk->stat.runreadingsize;
 612     remreadingsize += currdisk->stat.remreadingsize;
 613     parthits += currdisk->stat.parthits;
 614     runpartsize += currdisk->stat.runpartsize;
 615     rempartsize += currdisk->stat.rempartsize;
 616     waitingforbus += currdisk->stat.waitingforbus;
 617     numbuswaits += currdisk->stat.numbuswaits;
 618   }
 619   hits = fullreadhits + appendhits + prependhits + parthits + readinghits;
 620   misses = readmisses + writemisses;
 621   reads = fullreadhits + readmisses + readinghits + parthits;
 622   total = hits + misses;
 623   writes = total - reads;
 624   if (total == 0) {
 625     return;
 626   }
 627   fprintf(outputfile, "%sNumber of buffer accesses:    %d\n", prefix, total);
 628   fprintf(outputfile, "%sBuffer hit ratio:        %6d \t%f\n", prefix, hits, ((double) hits / (double) total));
 629   fprintf(outputfile, "%sBuffer miss ratio:            %6d \t%f\n", prefix, misses, ((double) misses / (double) total));
 630   fprintf(outputfile, "%sBuffer read hit ratio:        %6d \t%f \t%f\n", prefix, fullreadhits, ((double) fullreadhits / (double) max(1,reads)), ((double) fullreadhits / (double) total));
 631   fprintf(outputfile, "%sBuffer prepend hit ratio:       %6d \t%f\n", prefix, prependhits, ((double) prependhits / (double) max(1,writes)));
 632   fprintf(outputfile, "%sBuffer append hit ratio:       %6d \t%f\n", prefix, appendhits, ((double) appendhits / (double) max(1,writes)));
 633   fprintf(outputfile, "%sWrite combinations:           %6d \t%f\n", prefix, writecombs, ((double) writecombs / (double) max(1,writes)));
 634   fprintf(outputfile, "%sOngoing read-ahead hit ratio: %6d \t%f \t%f\n", prefix, readinghits, ((double) readinghits / (double) max(1,reads)), ((double) readinghits / (double) total));
 635   fprintf(outputfile, "%sAverage read-ahead hit size:  %f\n", prefix, (runreadingsize / (double) max(1,readinghits)));
 636   fprintf(outputfile, "%sAverage remaining read-ahead: %f\n", prefix, (remreadingsize / (double) max(1,readinghits)));
 637   fprintf(outputfile, "%sPartial read hit ratio: %6d \t%f \t%f\n", prefix, parthits, ((double) parthits / (double) max(1,reads)), ((double) parthits / (double) total));
 638   fprintf(outputfile, "%sAverage partial hit size:     %f\n", prefix, (runpartsize / (double) max(1,parthits)));
 639   fprintf(outputfile, "%sAverage remaining partial:    %f\n", prefix, (rempartsize / (double) max(1,parthits)));
 640   fprintf(outputfile, "%sTotal disk bus wait time: %f\n", prefix, waitingforbus);
 641   fprintf(outputfile, "%sNumber of disk bus waits: %d\n", prefix, numbuswaits);
 642 }

 770 void disk_printstats (void)
 771 {
 772   struct ioq * queueset[MAXDEVICES];
 773   int set[MAXDEVICES];
 774   int i;
 775   int reqcnt = 0;
 776   char prefix[80];
 777   int diskcnt;
 778 
 779   fprintf(outputfile, "\nDISK STATISTICS\n");
 780   fprintf(outputfile, "---------------\n\n");
 781 
 782   sprintf(prefix, "Disk ");
 783 
 784   diskcnt = 0;
 785   for (i=0; i<MAXDEVICES; i++) {
 786     disk *currdisk = getdisk (i);
 787     if (currdisk) {
 788       queueset[diskcnt] = currdisk->queue;
 789       reqcnt += ioqueue_get_number_of_requests(currdisk->queue);
 790       diskcnt++;
 791     }
 792   }
 793   ddbg_assert (diskcnt == NUMDISKS);
 794 
 795   if (reqcnt == 0) {
 796     fprintf(outputfile, "No disk requests encountered\n");
 797     return;
 798   }
 799   /*
 800     fprintf(outputfile, "Number of extra write disconnects:   %5d  \t%f\n", extra_write_disconnects, ((double) extra_write_disconnects / (double) reqcnt));
 801   */
 802   ioqueue_printstats(queueset, NUMDISKS, prefix);
 803 
 804   diskcnt = 0;
 805   for (i=0; i<MAXDEVICES; i++) {
 806     disk *currdisk = getdisk (i);
 807     if (currdisk) {
 808       set[diskcnt] = i;
 809       diskcnt++;
 810     }
 811   }
 812   ddbg_assert (diskcnt == NUMDISKS);
 813 
 814   disk_seek_printstats(set, NUMDISKS, prefix);
 815   disk_latency_printstats(set, NUMDISKS, prefix);
 816   disk_transfer_printstats(set, NUMDISKS, prefix);
 817   disk_acctime_printstats(set, NUMDISKS, prefix);
 818   disk_interfere_printstats(set, NUMDISKS, prefix);
 819   disk_buffer_printstats(set, NUMDISKS, prefix);
 820   fprintf (outputfile, "\n\n");
 821 
 822   if (NUMDISKS <= 1) {
 823     return;
 824   }
 825 
 826   for (i=0; i<NUMDISKS; i++) {
 827     disk *currdisk = getdisk (set[i]); 
 828     if (currdisk->printstats == FALSE) {
 829       continue;
 830     }
 831     if (ioqueue_get_number_of_requests(currdisk->queue) == 0) {
 832       fprintf(outputfile, "No requests for disk #%d\n\n\n", set[i]);
 833       continue;
 834     }
 835     fprintf(outputfile, "Disk #%d:\n\n", set[i]);
 836     fprintf(outputfile, "Disk #%d highest block number requested: %d\n", set[i], currdisk->stat.highblkno);
 837     sprintf(prefix, "Disk #%d ", set[i]);
 838     ioqueue_printstats(&currdisk->queue, 1, prefix);
 839     disk_seek_printstats(&set[i], 1, prefix);
 840     disk_latency_printstats(&set[i], 1, prefix);
 841     disk_transfer_printstats(&set[i], 1, prefix);
 842     disk_acctime_printstats(&set[i], 1, prefix);
 843     disk_interfere_printstats(&set[i], 1, prefix);
 844     disk_buffer_printstats(&set[i], 1, prefix);
 845     fprintf (outputfile, "\n\n");
 846   }
 847 }

 908 int disk_load_syncsets(struct lp_block *b) {
 909   int c, d;
 910   int type, devnum;
 911   struct lp_list *l;
 912   disk *currdisk;
 913 
 914   ++numsyncsets;
 915 
 916   for(c = 0; c < b->params_len; c++) {
 917     if(!b->params[c]) continue;
 918     else if(strcmp(b->params[c]->name, "devices")) continue;
 919     l = LVAL(b->params[c]);
 920     for(d = 0; d < l->values_len; d++) {
 921       if(!l->values[c]) continue;
 922       currdisk = (disk*)getdevbyname(l->values[c]->v.s, 0, &devnum, &type);
 923       if(!currdisk || (type != DEVICETYPE_DISK)) {
 924         fprintf(stderr, "*** error: bad syncset spec: no such device %s or %s is of the wrong type.\n", l->values[c]->v.s, l->values[c]->v.s);
 925         return 0;
 926       }
 927 
 928       currdisk->syncset = numsyncsets;
 929 
 930     }
 931     break;
 932   }
 933 
 934 
 935   return 1;
 936 }

 940 
 941 void disk_initialize (void)
 942 {
 943    int i, j;
 944    diskreq *tmpdiskreq;
 945    segment *tmpseg;
 946 
 947    // fprintf (outputfile, "Entered disk_initialize - numdisks %d\n", NUMDISKS);
 948 
 949    StaticAssert (sizeof(segment) <= DISKSIM_EVENT_SIZE);
 950    StaticAssert (sizeof(diskreq) <= DISKSIM_EVENT_SIZE);
 951 
 952    if (disksim->diskinfo == NULL) {
 953       disk_initialize_diskinfo();
 954    }
 955 
 956    disk_setcallbacks();
 957    disk_postpass();
 958    disk_syncset_init();
 959 
 960    for (i = 0; i < MAXDEVICES; i++) {
 961      disk *currdisk = getdisk (i); if(!currdisk) continue;
 962      /*        if (currdisk->inited != 0) { */
 963      /*           continue; */
 964      /*        } */
 965      ioqueue_initialize(currdisk->queue, i);
 966      ioqueue_set_enablement_function (currdisk->queue, &disksim->enablement_disk);
 967      addlisttoextraq((event **) &currdisk->outwait);
 968      addlisttoextraq((event **) &currdisk->buswait);
 969      
 970      if (currdisk->currentbus) {
 971        if (currdisk->currentbus == currdisk->effectivebus) {
 972          currdisk->effectivebus = NULL;
 973        }
 974        tmpdiskreq = currdisk->currentbus;
 975        if (tmpdiskreq->seg) {
 976          disk_buffer_remove_from_seg(tmpdiskreq);
 977        }
 978        addlisttoextraq((event **) &tmpdiskreq->ioreqlist);
 979        currdisk->currentbus = NULL;
 980        addtoextraq((event *) tmpdiskreq);
 981      }
 982 
 983      if (currdisk->effectivebus) {
 984        tmpdiskreq = currdisk->effectivebus;
 985        if (tmpdiskreq->seg) {
 986          disk_buffer_remove_from_seg(tmpdiskreq);
 987        }
 988        addlisttoextraq((event **) &tmpdiskreq->ioreqlist);
 989        currdisk->effectivebus = NULL;
 990        addtoextraq((event *) tmpdiskreq);
 991      }
 992 
 993      if (currdisk->currenthda) {
 994        if (currdisk->currenthda == currdisk->effectivehda) {
 995          currdisk->effectivehda = NULL;
 996        }
 997        tmpdiskreq = currdisk->currenthda;
 998        if (tmpdiskreq->seg) {
 999          disk_buffer_remove_from_seg(tmpdiskreq);
1000        }
1001        addlisttoextraq((event **) &tmpdiskreq->ioreqlist);
1002        currdisk->currenthda = NULL;
1003        addtoextraq((event *) tmpdiskreq);
1004      }
1005 
1006      if (currdisk->effectivehda != NULL) {
1007        tmpdiskreq = currdisk->effectivehda;
1008        if (tmpdiskreq->seg) {
1009          disk_buffer_remove_from_seg(tmpdiskreq);
1010        }
1011        addlisttoextraq((event **) &tmpdiskreq->ioreqlist);
1012        currdisk->effectivehda = NULL;
1013        addtoextraq((event *) tmpdiskreq);
1014      }
1015 
1016      while (currdisk->pendxfer) {
1017        tmpdiskreq = currdisk->pendxfer;
1018        if (tmpdiskreq->seg) {
1019          disk_buffer_remove_from_seg(tmpdiskreq);
1020        }
1021        addlisttoextraq((event **) &tmpdiskreq->ioreqlist);
1022        currdisk->pendxfer = tmpdiskreq->bus_next;
1023        addtoextraq((event *) tmpdiskreq);
1024      }
1025 
1026      currdisk->outstate = DISK_IDLE;
1027      currdisk->busy = FALSE;
1028      currdisk->prev_readahead_min = -1;
1029      currdisk->extradisc_diskreq = NULL;
1030 
1031      currdisk->currtime = 0.0;
1032      currdisk->lastflags = READ;
1033 
1034      currdisk->lastgen = -1;
1035      currdisk->busowned = -1;
1036      currdisk->numdirty = 0;
1037      if (currdisk->seglist == NULL) {
1038        currdisk->seglist = (segment *) DISKSIM_malloc(sizeof(segment));
1039        currdisk->seglist->next = NULL;
1040        currdisk->seglist->prev = NULL;
1041        currdisk->seglist->diskreqlist = NULL;
1042        currdisk->seglist->recyclereq = NULL;
1043        currdisk->seglist->access = NULL;
1044        for (j = 1; j < currdisk->numsegs; j++) {
1045          tmpseg = (segment *) DISKSIM_malloc(sizeof(segment));
1046          tmpseg->next = currdisk->seglist;
1047          currdisk->seglist = tmpseg;
1048          tmpseg->next->prev = tmpseg;
1049          tmpseg->prev = NULL;
1050          tmpseg->diskreqlist = NULL;
1051          tmpseg->recyclereq = NULL;
1052          tmpseg->access = NULL;
1053        }
1054        if (currdisk->dedicatedwriteseg) {
1055          currdisk->dedicatedwriteseg = currdisk->seglist;
1056        }
1057      }
1058      tmpseg = currdisk->seglist;
1059 
1060       // initialize cache buffers
1061      for (j = 0; j < currdisk->numsegs; j++) {
1062        tmpseg->outstate = BUFFER_IDLE;
1063        tmpseg->state = BUFFER_EMPTY;
1064        tmpseg->size = currdisk->segsize;
1065        while (tmpseg->diskreqlist) {
1066          addlisttoextraq((event **) &tmpseg->diskreqlist->ioreqlist);
1067          tmpdiskreq = tmpseg->diskreqlist;
1068          tmpseg->diskreqlist = tmpdiskreq->seg_next;
1069          addtoextraq((event *) tmpdiskreq);
1070        }
1071        /* recyclereq should have already been "recycled" :) by the
1072           effectivehda or currenthda recycling above */
1073        tmpseg->recyclereq = NULL;
1074        addlisttoextraq((event **) &tmpseg->access);
1075        tmpseg = tmpseg->next;
1076      }
1077 
1078 
1079      diskstatinit(i, TRUE);
1080    }
1081 }

1085 void disk_acctimestats (disk *currdisk, int distance, double seektime,
1086                         double latency, double xfertime, double acctime)
1087 {
1088    int dist;
1089 
1090    if (device_printseekstats == TRUE) {
1091       dist = abs(distance);
1092       if (dist == 0) {
1093          currdisk->stat.zeroseeks++;
1094       }
1095       stat_update(&currdisk->stat.seekdiststats, (double) dist);
1096       stat_update(&currdisk->stat.seektimestats, seektime);
1097    }
1098    if (device_printlatencystats == TRUE) {
1099       if (latency == (double) 0.0) {
1100          currdisk->stat.zerolatency++;
1101       }
1102       stat_update(&currdisk->stat.rotlatstats, latency);
1103    }
1104    if (device_printxferstats == TRUE) {
1105       stat_update(&currdisk->stat.xfertimestats, xfertime);
1106    }
1107    if (device_printacctimestats == TRUE) {
1108       stat_update(&currdisk->stat.postimestats, (seektime+latency));
1109       stat_update(&currdisk->stat.acctimestats, acctime);
1110    }
1111 }
