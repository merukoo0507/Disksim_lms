 109 static int logorg_modulus_update (int inc, int val, int maxval)
 110 {
 111    int ret;
 112 
 113    ret = val + inc;
 114    if (ret < 0) {
 115       ret += maxval;
 116    } else if (ret >= maxval) {
 117       ret -= maxval;
 118    }
 119    return(ret);
 120 }

 123 int logorg_tabular_rottype (int maptype, int reduntype, int rottype, int stripeunit)
 124 {
 125    if (reduntype != PARITY_ROTATED) {
 126       return(FALSE);
 127    }
 128    if (maptype != STRIPED) {
 129       return(FALSE);
 130    }
 131    if (stripeunit == 0) {
 132       return(FALSE);
 133    }
 134    switch (rottype) {
 135       case PARITY_LEFT_SYM:
 136       case PARITY_LEFT_ASYM:
 137       case PARITY_RIGHT_SYM:
 138       case PARITY_RIGHT_ASYM:
 139       return(TRUE);
 140    }
 141    return(FALSE);
 142 }

 168 static int logorg_shadowed_get_short_dist (logorg *currlogorg, ioreq_event *curr, int numtocheck, int *checklist)
 169 {
 170    int i, j;
 171    int dist;
 172    int shortdev = -1;
 173    int shortdist = 999999999;
 174    int ties[MAXCOPIES];
 175    int no;
 176 
 177    j = -1;
 178    for (i = 0; i < numtocheck; i++) {
 179       no = checklist[i];
 180       dist = ioqueue_get_dist(currlogorg->devs[(curr->devno + (no * currlogorg->numdisks))].queue, (curr->blkno + currlogorg->devs[(curr->devno + (no * currlogorg->numdisks))].startblkno));
 181       if (dist >= 999999999) {
 182          fprintf(stderr, "Haven't allowed for large enough 'dist's in logorg_shadowed_get_short_dist - %d\n", dist);
 183          exit(1);
 184       }
 185       if (dist == shortdist) {
 186          j++;
 187          if (j >= MAXCOPIES) {
 188             fprintf(stderr, "Haven't allowed for enough ties in logorg_shadowed_get_short_dist - %d\n", j);
 189             exit(1);
 190          }
 191          ties[j] = no;
 192       }
 193       if (dist < shortdist) {
 194          shortdist = dist;
 195          shortdev = no;
 196          j = -1;
 197       }
 198    }
 199    if (shortdev == -1) {
 200       fprintf(stderr, "Illegal condition in logorg_shadowed_get_short_dist\n");
 201       exit(1);
 202    } else if (j != -1) {
 203       i = (int) (DISKSIM_drand48() * (double) (j+2));
 204       if (i != 0) {
 205          shortdev = ties[(i-1)];
 206       }
 207    }
 208    return(shortdev);
 209 }

 212 static int logorg_shadowed_get_short_queue (logorg *currlogorg, ioreq_event *curr, int def)
 213 {
 214    int i, j;
 215    int len;
 216    int shortdev = -1;
 217    int shortlen = 999999999;
 218    int ties[MAXCOPIES];
 219 
 220    j = -1;
 221    for (i = 0; i < currlogorg->copies; i++) {
 222       len = ioqueue_get_number_in_queue(currlogorg->devs[(curr->devno + (i * currlogorg->numdisks))].queue);
 223       if (len >= 999999999) {
 224          fprintf(stderr, "Haven't allowed for large enough 'dist's in logorg_shadowed_get_short_queue - %d\n", len);
 225          exit(1);
 226       }
 227       if (len == shortlen) {
 228          j++;
 229          if (j >= MAXCOPIES) {
 230             fprintf(stderr, "Haven't allowed for enough ties in logorg_shadowed_get_short_queue - %d\n", j);
 231             exit(1);
 232          }
 233          ties[j] = i;
 234       }
 235       if (len < shortlen) {
 236          shortlen = len;
 237          shortdev = i;
 238          j = -1;
 239       }
 240    }
 241    if (shortdev == -1) {
 242       fprintf(stderr, "Illegal condition in logorg_shadowed_get_short_queue\n");
 243       exit(1);
 244    } else if (j != -1) {
 245       if (def == 1) {
 246          i = (int) (DISKSIM_drand48() * (double) (j+2));
 247          if (i != 0) {
 248             shortdev = ties[(i-1)];
 249          }
 250       } else if (def == 2) {
 251          j++;
 252          ties[j] = shortdev;
 253          shortdev = logorg_shadowed_get_short_dist(currlogorg, curr, (j+1), ties);
 254       } else {
 255          fprintf(stderr, "Unknown default tie breaker in logorg_shadowed_get_short_queue - %d\n", def);
 256          exit(1);
 257       }
 258    }
 259    return(shortdev);
 260 }

 263 int logorg_shadowed (logorg *currlogorg, ioreq_event *curr, int numreqs)
 264 {
 265    int i, j;
 266    ioreq_event *temp;
 267    ioreq_event *newreq;
 268    int checklist[MAXCOPIES];
 269 
 270    temp = curr;
 271    if (curr->flags & READ) {
 272       for (i = 0; i < numreqs; i++) {
 273          switch (currlogorg->copychoice) {
 274             case SHADOW_PRIMARY:
 275 /* May want to move head on secondary device */
 276                           break;
 277             case SHADOW_RANDOM:
 278                           temp->devno += currlogorg->numdisks * (int) (DISKSIM_drand48() * (double) currlogorg->copies);
 279                           break;
 280             case SHADOW_ROUNDROBIN:
 281                           temp->devno += currlogorg->numdisks * currlogorg->reduntoggle;
 282                           currlogorg->reduntoggle++;
 283                           currlogorg->reduntoggle %= currlogorg->copies;
 284                           break;
 285             case SHADOW_SHORTDIST:
 286                           for (j = 0; j < currlogorg->copies; j++)
 287                              checklist[j] = j;
 288                           temp->devno += currlogorg->numdisks * logorg_shadowed_get_short_dist(currlogorg, temp, currlogorg->copies, checklist);
 289                           break;
 290             case SHADOW_SHORTQUEUE:
 291                           temp->devno += currlogorg->numdisks * logorg_shadowed_get_short_queue(currlogorg, temp, 1);
 292                           break;
 293             case SHADOW_SHORTQUEUE2:
 294                           temp->devno += currlogorg->numdisks * logorg_shadowed_get_short_queue(currlogorg, temp, 2);
 295                           break;
 296             default:
 297                     fprintf(stderr, "Unknown shadow choice type at logorg_shadowed\n");
 298                     exit(1);
 299          }
 300          temp = temp->next;
 301       }
 302       return(numreqs);
 303    } else {
 304       for (i = 0; i < numreqs; i++) {
 305          for (j = 1; j < currlogorg->copies; j++) {
 306             newreq = ioreq_copy(temp);
 307             temp->next = newreq;
 308             newreq->devno += currlogorg->numdisks;
 309             temp = newreq;
 310          }
 311          temp = temp->next;
 312       }
 313       return(currlogorg->copies * numreqs);
 314    }
 315 }

 318 int logorg_parity_disk (logorg *currlogorg, ioreq_event *curr, int numreqs)
 319 {
 320    depends *depend;
 321 
 322    if (curr->flags & READ) {
 323       return(numreqs);
 324    }
 325    if ((currlogorg->maptype == STRIPED) && (currlogorg->stripeunit == 0)) {
 326       curr->next = ioreq_copy(curr);
 327       curr->next->devno = currlogorg->numdisks;
 328       return(numreqs+1);
 329    }
 330    if (numreqs != 1) {
 331       fprintf(stderr, "Too many reqs at logorg_parity_disk - %d\n", numreqs);
 332       exit(1);
 333    }
 334    depend = (depends *) getfromextraq();
 335    depend->blkno = curr->blkno;
 336    depend->devno = curr->devno;
 337    depend->numdeps = 2;
 338    depend->deps[0] = ioreq_copy(curr);
 339    depend->deps[1] = ioreq_copy(curr);
 340    depend->deps[0]->opid = 2;
 341    depend->deps[0]->devno = currlogorg->numdisks;
 342    depend->next = (depends *) getfromextraq();
 343    depend->next->next = NULL;
 344    depend->next->blkno = curr->blkno;
 345    depend->next->devno = currlogorg->numdisks;
 346    depend->next->deps[0] = depend->deps[0];
 347    depend->next->deps[1] = depend->deps[1];
 348    if (currlogorg->writesync == TRUE) {
 349       depend->deps[1]->opid = 2;
 350       depend->next->numdeps = 2;
 351    } else {
 352       depend->deps[1]->opid = 1;
 353       depend->next->numdeps = 1;
 354    }
 355    curr->flags |= READ;
 356    curr->next = ioreq_copy(curr);
 357    curr->next->devno = currlogorg->numdisks;
 358    curr->prev = (ioreq_event *) depend;
 359    return(4);
 360 }

 363 static int logorg_parity_rotate_updates (logorg *currlogorg, ioreq_event *curr, int inc)
 364 {
 365    int reqs = 2;
 366    int parityno;
 367    int blksleft;
 368    int bcount;
 369    ioreq_event *newreq;
 370    int parityunit;
 371    int numdisks;
 372    int devno;
 373    int parityblock;
 374 
 375    parityunit = currlogorg->parityunit;
 376    numdisks = currlogorg->numdisks;
 377 
 378    parityblock = curr->blkno / (parityunit * (numdisks - 1));
 379    parityno = (curr->blkno % (parityunit * (numdisks - 1))) / parityunit;
 380    curr->blkno += parityblock * parityunit;
 381    if (inc == -1) {
 382       devno = numdisks - (parityno % numdisks) - 1;
 383    } else {
 384       devno = parityno % numdisks;
 385    }
 386    if (((inc == 1) && (curr->devno <= devno)) || ((inc == -1) && (curr->devno >=devno))) {
 387       curr->blkno += parityunit;
 388       devno = logorg_modulus_update(inc, devno, numdisks);
 389       parityno++;
 390    }
 391    blksleft = parityunit - (curr->blkno % currlogorg->parityunit);
 392    bcount = curr->bcount;
 393    if ((parityunit != currlogorg->parityunit) && ((bcount > blksleft) || (parityunit > currlogorg->parityunit))) {
 394       fprintf(stderr, "Failed integrity check in logorg_parity_rotate_left\n");
 395       exit(1);
 396    }
 397    while (bcount > blksleft) {
 398       if (!(curr->flags & READ)) {
 399          newreq = ioreq_copy(curr);
 400          newreq->devno = devno;
 401          newreq->bcount = blksleft;
 402          newreq->blkno = curr->blkno + curr->bcount - bcount;
 403          newreq->next = curr->next;
 404          curr->next = newreq;
 405          reqs++;
 406       }
 407       bcount -= blksleft;
 408       devno = logorg_modulus_update(inc, devno, numdisks);
 409       if (devno == curr->devno) {
 410          curr->bcount += parityunit;
 411          devno = logorg_modulus_update(inc, devno, numdisks);
 412          parityno++;
 413       }
 414       parityno++;
 415       blksleft = parityunit;
 416    }
 417    if (curr->flags & READ) {
 418       return(1);
 419    }
 420    newreq = ioreq_copy(curr);
 421    newreq->devno = devno;
 422    newreq->blkno = curr->blkno + curr->bcount - bcount;
 423    newreq->bcount = bcount;
 424    newreq->next = curr->next;
 425    curr->next = newreq;
 426    if (parityno > currlogorg->numfull) {
 427       fprintf(stderr, "Integrity check failure in logorg_parity_rotate_left\n");
 428       exit(1);
 429    }
 430    return(reqs);
 431 }

 434 static void logorg_parity_read_old_sync (logorg *currlogorg, ioreq_event *curr, int numreqs)
 435 {
 436    int i, j;
 437    depends *depend;
 438    depends *tmpdep;
 439    ioreq_event *temp;
 440    int devno = 1;
 441 
 442    depend = (depends *) getfromextraq();
 443    depend->blkno = curr->blkno;
 444    depend->devno = curr->devno;
 445    depend->deps[0] = ioreq_copy(curr);
 446    depend->deps[0]->opid = numreqs;
 447    curr->prev = (ioreq_event *) depend;
 448    depend->numdeps = numreqs;
 449    temp = curr->next;
 450    for (i=1; i<numreqs; i++) {
 451       if (devno == 0) {
 452          depend->cont = (depends *) getfromextraq();
 453          depend = depend->cont;
 454       }
 455       depend->deps[devno] = ioreq_copy(temp);
 456       depend->deps[devno]->opid = numreqs;
 457       devno = logorg_modulus_update(1, devno, 10);
 458       temp = temp->next;
 459    }
 460    temp = curr->next;
 461    curr->flags |= READ;
 462    for (i=1; i<numreqs; i++) {
 463       depend = (depends *) ioreq_copy(curr->prev);
 464       depend->next = (depends *) curr->prev;
 465       curr->prev = (ioreq_event *) depend;
 466       depend->blkno = temp->blkno;
 467       depend->devno = temp->devno;
 468       tmpdep = depend->cont;
 469       for (j=0; j<(numreqs/10); j++) {
 470          depend->cont = (depends *) ioreq_copy((ioreq_event *) tmpdep);
 471          tmpdep = tmpdep->cont;
 472          depend = depend->cont;
 473       }
 474       temp->flags |= READ;
 475       temp = temp->next;
 476    }
 477 }

 519 int logorg_parity_rotate (logorg *currlogorg, ioreq_event *curr, int numreqs)
 520 {
 521    int reqs = 2;
 522 
 523    if (currlogorg->maptype == IDEAL) {
 524       logorg_parity_rotate_ideal(currlogorg, curr);
 525    } else if (currlogorg->maptype == RANDOM) {
 526       logorg_parity_rotate_random(currlogorg, curr);
 527    } else if (currlogorg->maptype == ASIS) {
 528       switch (currlogorg->rottype) {
 529 
 530          case PARITY_LEFT_SYM:
 531          case PARITY_LEFT_ASYM:
 532             reqs = logorg_parity_rotate_updates(currlogorg, curr, -1);
 533             break;
 534 
 535          case PARITY_RIGHT_SYM:
 536          case PARITY_RIGHT_ASYM:
 537             reqs = logorg_parity_rotate_updates(currlogorg, curr, 1);
 538             break;
 539 
 540          default:
 541             fprintf(stderr, "Unknown rottype at logorg_parity_rotate - %d\n", currlogorg->rottype);
 542             exit(1);
 543       }
 544    } else {
 545       fprintf(stderr, "Unknown maptype at logorg_parity_rotate - %d\n", currlogorg->maptype);
 546       exit(1);
 547    }
 548    if (curr->flags & READ) {
 549       return(1);
 550    }
 551 
 552    if (currlogorg->writesync == TRUE) {
 553       logorg_parity_read_old_sync(currlogorg, curr, reqs);
 554    } else {
 555       logorg_parity_read_old_nosync(currlogorg, curr, reqs);
 556    }
 557 
 558    return(2*reqs);
 559 }

 563 {
 564    ioreq_event *temp;
 565    ioreq_event *del;
 566    int numreqs = 0;
 567    int distance;
 568 
 569    temp = reqlist;
 570    if (temp) {
 571       while (temp->next) {

 575 */
 576          if (distance < 0) {

 578             exit(1);
 579          }
 580          if (((temp->flags & READ) == (temp->next->flags & READ)) && (distance <= seqgive)) {
 581             del = temp->next;
 582             temp->next = del->next;
 583             temp->bcount += del->bcount + distance;
 584             temp->opid |= del->opid;
 585             addtoextraq((event *) del);
 586          } else {
 587             del = temp;
 588             temp = del->next;
 589             del->next = curr->next;
 590             curr->next = del;
 591             del->time = curr->time;
 592             del->buf = curr->buf;
 593             numreqs++;
 594          }
 595       }
 596       numreqs++;
 597       temp->next = curr->next;
 598       curr->next = temp;
 599       temp->time = curr->time;
 600       temp->buf = curr->buf;
 601    }
 602    return(numreqs);
 603 }

 606 static void logorg_parity_table_insert (ioreq_event **head, ioreq_event *curr)
 607 {
 608    ioreq_event *temp;
 609 
 610    if (((*head) == NULL) || (curr->blkno < (*head)->blkno)) {
 611       curr->next = (*head);
 612       (*head) = curr;
 613    } else {
 614       temp = (*head);
 615       while ((temp->next) && (curr->blkno >= temp->next->blkno)) {
 616          temp = temp->next;
 617       }
 618       curr->next = temp->next;
 619       temp->next = curr;
 620    }
 621 }

 624 static void logorg_parity_table_read_old (logorg *currlogorg, ioreq_event *rowhead, ioreq_event **reqlist, int opid)
 625 {
 626    ioreq_event *temp;
 627    ioreq_event *newreq;
 628    ioreq_event *prev = NULL;
 629 
 630    temp = rowhead;
 631    while (temp) {
 632       newreq = (ioreq_event *) getfromextraq();
 633       newreq->blkno = temp->blkno;
 634       newreq->devno = temp->devno;
 635       newreq->bcount = temp->bcount;
 636       newreq->flags = temp->flags | READ;
 637       newreq->opid = opid;
 638       logorg_parity_table_insert(&reqlist[newreq->devno], newreq);
 639       newreq->prev = prev;
 640       prev = newreq;
 641         /* only the last write (the parity update) must depend on the reads */
 642       if (temp->prev == NULL) {
 643          temp->opid = opid;
 644       }
 645       temp = temp->prev;
 646    }
 647 }

 650 static void logorg_parity_table_recon (logorg *currlogorg, ioreq_event *rowhead, ioreq_event **reqlist, int stripeno, int unitno, int tableadd, int opid)
 651 {
 652    ioreq_event *temp;
 653    ioreq_event *newreq;
 654    int entryno;
 655    int minblkno;
 656    int maxblkno;
 657    int lastentry;
 658    int i;
 659    int blkno;
 660    int offset;
 661 
 662    temp = rowhead;
 663    if (temp == NULL) {
 664       fprintf(stderr, "Can't have NULL rowhead at logorg_parity_table_recon\n");
 665       exit(1);
 666    }
 667    lastentry = (stripeno + 1) * currlogorg->partsperstripe + stripeno;
 668    while (temp->prev) {
 669       temp = temp->prev;
 670    }
 671    temp->opid = opid;
 672    minblkno = temp->blkno - currlogorg->table[lastentry].blkno - tableadd;
 673    maxblkno = minblkno + temp->bcount;
 674    entryno = stripeno * currlogorg->partsperstripe + stripeno;
 675    for (i = 0; i < unitno; i++) {
 676       newreq = (ioreq_event *) getfromextraq();
 677       newreq->devno = currlogorg->table[entryno].devno;
 678       newreq->blkno = tableadd + minblkno + currlogorg->table[entryno].blkno;
 679       newreq->bcount = maxblkno - minblkno;
 680       newreq->flags = rowhead->flags | READ;
 681       newreq->opid = opid;
 682       logorg_parity_table_insert(&reqlist[newreq->devno], newreq);
 683       entryno++;
 684    }
 685    temp = rowhead;
 686    while (temp->prev) {
 687       blkno = currlogorg->table[entryno].blkno;
 688       offset = temp->blkno - tableadd - blkno;
 689       if (offset > minblkno) {
 690          newreq = (ioreq_event *) getfromextraq();
 691          newreq->devno = temp->devno;
 692          newreq->blkno = tableadd + blkno + minblkno;
 693          newreq->bcount = offset - minblkno;
 694          newreq->flags = temp->flags | READ;
 695          newreq->opid = opid;
 696          logorg_parity_table_insert(&reqlist[newreq->devno], newreq);
 697       }
 698       if ((offset + temp->bcount) < maxblkno) {
 699          newreq = (ioreq_event *) getfromextraq();
 700          newreq->devno = temp->devno;
 701          newreq->blkno = temp->blkno + temp->bcount;
 702          newreq->bcount = maxblkno - offset - temp->bcount;
 703          newreq->flags = temp->flags | READ;
 704          newreq->opid = opid;
 705          logorg_parity_table_insert(&reqlist[newreq->devno], newreq);
 706       }
 707       entryno++;
 708       temp = temp->prev;
 709    }
 710    for (; entryno < lastentry; entryno++) {
 711       newreq = (ioreq_event *) getfromextraq();
 712       newreq->devno = currlogorg->table[entryno].devno;
 713       newreq->blkno = tableadd + minblkno + currlogorg->table[entryno].blkno;
 714       newreq->bcount = maxblkno - minblkno;
 715       newreq->flags = rowhead->flags | READ;
 716       newreq->opid = opid;
 717       logorg_parity_table_insert(&reqlist[newreq->devno], newreq);
 718    }
 719 }

 722 static void logorg_parity_table_dodeps_sync (logorg *currlogorg, ioreq_event *curr, ioreq_event **redunlist, ioreq_event **reqlist)
 723 {
 724    int i, j;
 725    depends *depend;
 726    depends *tmpdep;
 727    int opid = 0;
 728    int numdeps = 0;
 729    ioreq_event *temp;
 730    ioreq_event *deplist = NULL;
 731    int devno = 1;
 732 
 733    for (i=0; i<currlogorg->actualnumdisks; i++) {
 734       temp = redunlist[i];
 735       while (temp) {
 736          temp->opid = curr->opid;
 737          opid++;
 738          if (temp->next == NULL) {
 739             temp->next = curr->next;
 740             curr->next = redunlist[i];
 741             break;
 742          }
 743          temp = temp->next;
 744       }
 745       temp = reqlist[i];
 746       while (temp) {
 747          numdeps++;
 748          if (temp->next == NULL) {
 749             temp->next = deplist;
 750             deplist = reqlist[i];
 751             break;
 752          }
 753          temp = temp->next;
 754       }
 755    }
 756    if (numdeps <= 0) {
 757       fprintf(stderr, "Can't have zero requests at logorg_parity_table_dodeps_sync\n");
 758       exit(1);
 759    }
 760    if (opid == 0) {
 761       curr->prev = NULL;
 762       curr->next = deplist;
 763       return;
 764    }
 765    depend = (depends *) getfromextraq();
 766    depend->blkno = curr->next->blkno;
 767    depend->devno = curr->next->devno;
 768    depend->numdeps = numdeps;
 769    curr->prev = (ioreq_event *) depend;
 770    depend->deps[0] = deplist;
 771    deplist->opid = opid;
 772    temp = deplist->next;
 773    for (i=1; i<numdeps; i++) {
 774       if (devno == 0) {
 775          depend->cont = (depends *) getfromextraq();
 776          depend = depend->cont;
 777       }
 778       depend->deps[devno] = temp;
 779       temp->opid = opid;
 780       devno = logorg_modulus_update(1, devno, 10);
 781       temp = temp->next;
 782    }
 783    temp = curr->next->next;
 784    for (i=1; i<opid; i++) {
 785       depend = (depends *) ioreq_copy(curr->prev);
 786       depend->next = (depends *) curr->prev;
 787       curr->prev = (ioreq_event *) depend;
 788       depend->blkno = temp->blkno;
 789       depend->devno = temp->devno;
 790       tmpdep = depend->cont;
 791       for (j=0; j<(numdeps/10); j++) {
 792          depend->cont = (depends *) ioreq_copy((ioreq_event *) tmpdep);
 793          tmpdep = tmpdep->cont;
 794          depend = depend->cont;
 795       }
 796       temp = temp->next;
 797    }
 798 }

 801 static void logorg_parity_table_dodeps_nosync (logorg *currlogorg, ioreq_event *curr, ioreq_event **redunlist, ioreq_event **reqlist)
 802 {
 803    int i;
 804    depends *depend;
 805    depends *tmpdep;
 806    ioreq_event *temp;
 807    ioreq_event *del;
 808    ioreq_event *tmpread;
 809    ioreq_event *readlist = NULL;
 810    ioreq_event *deplist = NULL;
 811    int holddep;
 812 
 813    for (i=0; i<currlogorg->actualnumdisks; i++) {
 814       temp = redunlist[i];
 815       while (temp) {
 816          depend = (depends *) getfromextraq();
 817          depend->blkno = temp->blkno;
 818          depend->devno = temp->devno;
 819          depend->numdeps = 0;
 820          depend->next = NULL;
 821          temp->prev = (ioreq_event *) depend;
 822          if (temp->next == NULL) {
 823             temp->next = readlist;
 824             readlist = redunlist[i];
 825             break;
 826          }
 827          temp = temp->next;
 828       }
 829       temp = reqlist[i];
 830       while (temp) {
 831          del = temp;
 832          temp = temp->next;
 833          if ((redunlist[i] == NULL) && (del->opid == 0)) {
 834             del->next = curr->next;
 835             curr->next = del;
 836          } else {
 837             del->next = deplist;
 838             deplist = del;
 839          }
 840       }
 841    }
 842    if (readlist == NULL) {
 843       if (deplist) {
 844          fprintf(stderr, "Unacceptable condition in logorg_parity_table_dodeps_nosync\n");
 845          exit(1);
 846       }
 847       curr->prev = NULL;
 848       return;
 849    }
 850    temp = deplist;
 851    while (temp) {
 852       holddep = temp->opid;
 853       temp->opid = 0;
 854       tmpread = readlist;
 855       while (tmpread) {
 856          if ((tmpread->devno == temp->devno) | (holddep & tmpread->opid)) {
 857             tmpdep = (depends *) tmpread->prev;
 858             ASSERT (tmpdep != NULL);
 859             if (tmpdep->numdeps > 0) {
 860                for (i=0; i<((tmpdep->numdeps-1)/10); i++) {
 861                   tmpdep = tmpdep->cont;
 862                }
 863                if ((tmpdep->numdeps % 10) == 0) {
 864                   tmpdep->cont = (depends *) getfromextraq();
 865                   tmpdep = tmpdep->cont;
 866                }
 867             }
 868             tmpdep->deps[(tmpdep->numdeps % 10)] = temp;
 869             tmpdep->numdeps++;
 870             temp->opid++;
 871          }
 872          tmpread = tmpread->next;
 873       }
 874       temp = temp->next;
 875    }
 876    temp = readlist;
 877    curr->prev = NULL;
 878    while (temp) {
 879       depend = (depends *) temp->prev;
 880       temp->prev = NULL;
 881       depend->next = (depends *) curr->prev;
 882       curr->prev = (ioreq_event *) depend;
 883       if (temp->next == NULL) {
 884          temp->next = curr->next;
 885          curr->next = readlist;
 886          break;
 887       }
 888       temp = temp->next;
 889    }
 890 }

 892 
 893 int logorg_parity_table (logorg *currlogorg, ioreq_event *curr, int numreqs)
 894 {
 895    int i;
 896    ioreq_event *reqs[MAXDEVICES];
 897    ioreq_event *redunreqs[MAXDEVICES];
 898    int reqcnt = 0;
 899    ioreq_event *lastrow = NULL;
 900    int stripeunit;
 901    ioreq_event *temp;
 902    ioreq_event *newreq;
 903    int unitno;
 904    int stripeno;
 905    int entryno;
 906    int blkno;
 907    int blksinpart;
 908    int reqsize;
 909    int partsperstripe;
 910    int rowcnt = 1;
 911    int firstrow = 0;
 912    int opid = 0x1;
 913    int blkscovered;
 914    tableentry *table;
 915    int tablestart;
 916    int preventryno;
 917 /*
 918 fprintf (outputfile, "Entered logorg_parity_table - devno %d, blkno %d, bcount %d, read %d\n", curr->devno, curr->blkno, curr->bcount, (curr->flags & READ));
 919 */
 920    if (numreqs != 1) {
 921       fprintf(stderr, "Multiple numreqs at logorg_parity_table is not acceptable - %d\n", numreqs);
 922       exit(1);
 923    }
 924 
 925    stripeunit = currlogorg->stripeunit;
 926    partsperstripe = currlogorg->partsperstripe;
 927    table = currlogorg->table;
 928    if (currlogorg->addrbyparts) {
 929       curr->blkno += curr->devno * currlogorg->blksperpart;
 930    }
 931    for (i=0; i<currlogorg->actualnumdisks; i++) {
 932       reqs[i] = NULL;
 933    }
 934    unitno = curr->blkno / stripeunit;
 935    stripeno = unitno / partsperstripe;
 936    tablestart = (stripeno / currlogorg->tablestripes) * currlogorg->tablesize;
 937    stripeno = stripeno % currlogorg->tablestripes;
 938    blkno = tablestart + table[(stripeno*(partsperstripe+1))].blkno;
 939    if (blkno == currlogorg->numfull) {
 940       stripeunit = currlogorg->actualblksperpart - blkno;
 941       curr->blkno -= blkno;
 942       unitno = curr->blkno / stripeunit;
 943    }
 944    blksinpart = stripeunit;
 945    unitno = unitno % partsperstripe;
 946    curr->blkno = curr->blkno % stripeunit;
 947    reqsize = curr->bcount;
 948    entryno = stripeno * partsperstripe + stripeno + unitno;
 949    blkno = tablestart + table[entryno].blkno;
 950    blksinpart -= curr->blkno;
 951    temp = ioreq_copy(curr);
 952    curr->next = temp;
 953    temp->blkno = blkno + curr->blkno;
 954    temp->devno = table[entryno].devno;
 955    temp->opid = 0;
 956    blkscovered = curr->bcount;
 957    curr->blkno = stripeno;
 958    curr->devno = unitno;
 959    curr->bcount = tablestart;
 960    reqs[temp->devno] = curr->next;
 961    temp->next = NULL;
 962    temp->prev = NULL;
 963    if (reqsize > blksinpart) {
 964       temp->bcount = blksinpart;
 965       blkscovered = blksinpart;
 966       reqsize -= blksinpart;
 967       unitno++;
 968       if (unitno == partsperstripe) {
 969          if (!(curr->flags & READ)) {
 970             newreq = (ioreq_event *) getfromextraq();
 971             newreq->devno = table[(entryno+1)].devno;
 972             newreq->blkno = table[(entryno+1)].blkno;
 973             newreq->blkno += tablestart + temp->blkno - blkno;
 974             newreq->bcount = blkscovered;
 975             newreq->flags = curr->flags;
 976             newreq->opid = 0;
 977             reqs[newreq->devno] = newreq;
 978             newreq->next = NULL;
 979             temp->prev = newreq;
 980             newreq->prev = NULL;
 981          }
 982          blkscovered = 0;
 983          unitno = 0;
 984          if (firstrow == 0) {
 985             firstrow = 1;
 986          }
 987          rowcnt = 0;
 988          stripeno++;
 989          temp = NULL;
 990          if (stripeno == currlogorg->tablestripes) {
 991             stripeno = 0;
 992             tablestart += currlogorg->tablesize;
 993          }
 994       }
 995       entryno = (stripeno * partsperstripe) + stripeno + unitno;
 996       blkno = tablestart + table[entryno].blkno;
 997       blksinpart = (blkno != currlogorg->numfull) ? stripeunit : currlogorg->actualblksperpart - blkno;
 998       while (reqsize > blksinpart) {
 999          rowcnt++;
1000          newreq = (ioreq_event *) getfromextraq();
1001          newreq->blkno = blkno;
1002          newreq->devno = table[entryno].devno;
1003          newreq->bcount = blksinpart;
1004          blkscovered = max(blkscovered, blksinpart);
1005          newreq->flags = curr->flags;
1006          newreq->opid = 0;
1007          newreq->prev = NULL;
1008          if (temp) {
1009             temp->prev = newreq;
1010          } else {
1011             lastrow = newreq;
1012          }
1013          temp = newreq;
1014          logorg_parity_table_insert(&reqs[temp->devno], temp);
1015          reqsize -= blksinpart;
1016          unitno++;
1017          if (unitno == partsperstripe) {
1018             if (!(curr->flags & READ)) {
1019                newreq = (ioreq_event *) getfromextraq();
1020                newreq->devno = table[(entryno+1)].devno;
1021                newreq->blkno = table[(entryno+1)].blkno + tablestart;
1022                newreq->bcount = blkscovered;
1023                newreq->flags = curr->flags;
1024                newreq->opid = 0;
1025                temp->prev = newreq;
1026                newreq->prev = NULL;
1027                logorg_parity_table_insert(&reqs[newreq->devno], newreq);
1028             }
1029             blkscovered = 0;
1030             unitno = 0;
1031             if (firstrow == 0) {
1032                firstrow = rowcnt;
1033             }
1034             rowcnt = 0;
1035             temp = NULL;
1036             stripeno++;
1037             if (stripeno == currlogorg->tablestripes) {
1038                stripeno = 0;
1039                tablestart += currlogorg->tablesize;
1040             }
1041          }
1042          entryno = (stripeno * partsperstripe) + stripeno + unitno;
1043          blkno = tablestart + table[entryno].blkno;
1044          blksinpart = (blkno != currlogorg->numfull) ? stripeunit : currlogorg->actualblksperpart - blkno;
1045       }
1046       newreq = (ioreq_event *) getfromextraq();
1047       newreq->blkno = blkno;
1048       newreq->devno = table[entryno].devno;
1049       newreq->bcount = reqsize;
1050       rowcnt++;
1051       blkscovered = max(blkscovered, blksinpart);
1052       newreq->flags = curr->flags;
1053       newreq->opid = 0;
1054       newreq->prev = NULL;
1055       if (temp) {
1056          temp->prev = newreq;
1057       } else {
1058          lastrow = newreq;
1059       }
1060       temp = newreq;
1061       logorg_parity_table_insert(&reqs[newreq->devno], newreq);
1062    }
1063    if (curr->flags & READ) {
1064       curr->next = curr;
1065       for (i=0; i<currlogorg->actualnumdisks; i++) {
1066          reqcnt += logorg_join_seqreqs(reqs[i], curr, LOGORG_PARITY_SEQGIVE);
1067       }
1068    } else {
1069       preventryno = entryno;
1070       entryno = (stripeno * partsperstripe) + stripeno + partsperstripe;
1071       newreq = (ioreq_event *) getfromextraq();
1072       newreq->devno = table[entryno].devno;
1073       newreq->blkno = table[entryno].blkno + temp->blkno - table[preventryno].blkno;
1074       newreq->bcount = (rowcnt == 1) ? temp->bcount : blkscovered;
1075       newreq->flags = curr->flags;
1076       newreq->opid = 0;
1077       temp->prev = newreq;
1078       newreq->prev = NULL;
1079       logorg_parity_table_insert(&reqs[newreq->devno], newreq);
1080       if (firstrow == 0) {
1081          if ((rowcnt == 2) && ((curr->next->blkno - temp->blkno - temp->bcount) > 0)) {
1082             newreq->bcount = temp->bcount;
1083             newreq = (ioreq_event *) getfromextraq();
1084             newreq->devno = temp->prev->devno;
1085             lastrow = temp;
1086             temp = curr->next;
1087             newreq->blkno = table[entryno].blkno + temp->blkno - table[(preventryno-1)].blkno;
1088             newreq->bcount = temp->bcount;
1089             newreq->flags = curr->flags;
1090             newreq->opid = 0;
1091             temp->prev = newreq;
1092             newreq->prev = NULL;
1093             logorg_parity_table_insert(&reqs[newreq->devno], newreq);
1094             firstrow = 1;
1095             rowcnt = 1;
1096          } else {
1097             firstrow = rowcnt;
1098             rowcnt = 0;
1099          }
1100       }
1101       for (i=0; i<currlogorg->actualnumdisks; i++) {
1102          redunreqs[i] = NULL;
1103       }
1104       if (firstrow < partsperstripe) {
1105          if (firstrow < currlogorg->rmwpoint) {
1106             logorg_parity_table_read_old(currlogorg, curr->next, redunreqs, opid);
1107          } else {
1108             logorg_parity_table_recon(currlogorg, curr->next, redunreqs, curr->blkno, curr->devno, curr->bcount, opid);
1109          }
1110          opid = opid << 1;
1111       }
1112       if ((rowcnt) && (rowcnt != partsperstripe)) {
1113          if (rowcnt < currlogorg->rmwpoint) {
1114             logorg_parity_table_read_old(currlogorg, lastrow, redunreqs, opid);
1115          } else {
1116             logorg_parity_table_recon(currlogorg, lastrow, redunreqs, stripeno, 0, tablestart, opid);
1117          }
1118       }
1119       for (i=0; i<currlogorg->actualnumdisks; i++) {
1120          curr->next = NULL;
1121          reqcnt += logorg_join_seqreqs(redunreqs[i], curr, LOGORG_PARITY_SEQGIVE);
1122          redunreqs[i] = curr->next;
1123          curr->next = NULL;
1124          reqcnt += logorg_join_seqreqs(reqs[i], curr, LOGORG_PARITY_SEQGIVE);
1125          reqs[i] = curr->next;
1126       }
1127       curr->next = curr;
1128       if (currlogorg->writesync) {
1129          logorg_parity_table_dodeps_sync(currlogorg, curr, redunreqs, reqs);
1130       } else {
1131          logorg_parity_table_dodeps_nosync(currlogorg, curr, redunreqs, reqs);
1132       }
1133    }
1134    if (curr->next) {
1135       temp = curr->next;
1136       curr->blkno = temp->blkno;
1137       curr->devno = temp->devno;
1138       curr->bcount = temp->bcount;
1139       curr->flags = temp->flags;
1140       curr->next = temp->next;
1141       addtoextraq((event *) temp);
1142    } else {
1143       fprintf(stderr, "Seem to have no requests when leaving logorg_parity_table\n");
1144       exit(1);
1145    }
1146 /*
1147 fprintf (outputfile, "Exiting logorg_parity_table - reqcnt %d\n", reqcnt);
1148 */
1149    return(reqcnt);
1150 }

1153 static void logorg_create_table_left_sym (logorg *currlogorg)
1154 {
1155    int i, j;
1156    int numdisks;
1157    int stripeno = 0;
1158    int devno = 0;
1159    int blkno = 0;
1160    int parityno;
1161 
1162    numdisks = currlogorg->numdisks;

1236 int logorg_check_dependencies (logorg *currlogorg, outstand *req, ioreq_event *curr)
1237 {
1238    int i;
1239    depends *tmpdep;
1240    depends *del = NULL;
1241    int numreqs = 0;
1242    int devno = 0;
1243    int numdeps;
1244    ioreq_event *temp;
1245 
1246    tmpdep = req->depend;
1247    if (tmpdep == NULL) {
1248       return(0);
1249    }
1250    if ((tmpdep->devno == curr->devno) && (tmpdep->blkno == curr->blkno)) {
1251       req->depend = tmpdep->next;
1252    } else {
1253       while (tmpdep->next) {
1254          if ((tmpdep->next->devno == curr->devno) && (tmpdep->next->blkno == curr->blkno)) {
1255             del = tmpdep->next;
1256             tmpdep->next = del->next;
1257             tmpdep = del;
1258             break;
1259          }
1260          tmpdep = tmpdep->next;
1261       }
1262       if (del == NULL) {
1263          return(0);
1264       }
1265    }
1266    curr->next = curr;
1267    numdeps = tmpdep->numdeps;
1268    i = 0;
1269    while (i < numdeps) {
1270       temp = tmpdep->deps[devno];
1271       temp->opid--;
1272       if (temp->opid == 0) {
1273          numreqs++;
1274          temp->next = curr->next;
1275          curr->next = temp;
1276          temp->opid = req->opid;
1277          temp->time = 0.0;
1278          if (req->flags & TIME_CRITICAL) {
1279             temp->flags |= TIME_CRITICAL;
1280          }
1281       }
1282       if ((i++) >= numdeps) {
1283          break;
1284       }
1285       devno = logorg_modulus_update(1, devno, 10);
1286       if (devno == 0) {
1287          del = tmpdep;
1288          tmpdep = tmpdep->cont;
1289          addtoextraq((event *) del);
1290       }
1291    }
1292    addtoextraq((event *) tmpdep);
1293    return(numreqs);
1294 }
