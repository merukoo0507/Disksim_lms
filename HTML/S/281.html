 111 /* Disk sparing schemes (for replacing bad media) */
 112 
 113 #define NO_SPARING                            0
 114 #define TRACK_SPARING                         1
 115 #define SECTPERCYL_SPARING                    2
 116 #define SECTPERTRACK_SPARING                  3
 117 #define SECTPERCYL_SPARING_SLIPTOEND          4
 118 #define SECTPERCYL_SPARING_ATFRONT            5
 119 #define SECTPERCYL_SPARING_ATFRONT_SLIPTOEND  6
 120 #define SECTATEND_SPARING                     7
 121 #define SECTPERRANGE_SPARING                  8
 122 #define SECTSPERZONE_SPARING                  9
 123 #define SECTSPERZONE_SPARING_SLIPTOEND       10
 124 #define MAXSPARESCHEME                       10
 125 
 126 
 127 /* Disk layout algorithms */
 128 
 129 #define LAYOUT_NORMAL           0
 130 #define LAYOUT_CYLSWITCHONSURF1 1   /* cyl. switches stay on same surface  */
 131                                     /* -- first cyl of each zone is normal */
 132 #define LAYOUT_CYLSWITCHONSURF2 2   /* cyl. switches stay on same surface  */
 133                                     /* -- all even-#d cyls are normal      */
 134 #define LAYOUT_SERPENTINE       3   /* planning for the future */
 135 #define LAYOUT_RAWFILE          4   /* the layout is in block-to-block file*/
 136 #define LAYOUT_MAX              4
 137 
 138 
 139 /* convenient macros */
 140 
 141 #define isspareatfront(currdisk) \
 142                 (((currdisk)->sparescheme == SECTPERCYL_SPARING_ATFRONT) || \
 143                  ((currdisk)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND))
 144 
 145 #define issliptoend(currdisk) \
 146                 (((currdisk)->sparescheme == SECTPERCYL_SPARING_SLIPTOEND) || \
 147                  ((currdisk)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND) || \
 148                  ((currdisk)->sparescheme == SECTATEND_SPARING))
 149 
 150 #define issectpercyl(currdisk) \
 151                 (((currdisk)->sparescheme == SECTPERCYL_SPARING) || \
 152                  ((currdisk)->sparescheme == SECTPERCYL_SPARING_SLIPTOEND) || \
 153                  ((currdisk)->sparescheme == SECTPERCYL_SPARING_ATFRONT) || \
 154                  ((currdisk)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND))
