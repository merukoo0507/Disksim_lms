 130 void stat_update (statgen *statptr, double value)
 131 {
 132    int  i = 0;
 133    int  start = statptr->distbrks[0];
 134    int  step = statptr->distbrks[1];
 135    int  grow = statptr->distbrks[2];
 136    int  equals = statptr->equals;
 137    int  buckets = statptr->distbrks[(DISTSIZE-1)];
 138    int  intval = (int) (value * (double) statptr->scale);
 139 
 140    statptr->count++;
 141    if (statptr->maxval < value) {
 142       statptr->maxval = value;
 143    }
 144    statptr->runval += value;
 145    statptr->runsquares += (value*value);
 146    if (buckets > DISTSIZE) {
 147       if (intval < start) {
 148       } else if (!grow) {
 149          if ((equals) && (intval <= (start - step + (step * equals))) && (((intval - start) % step) == 0)) {
 150             i = intval / step;
 151          } else {
 152             i = (intval + step - start) / step;
 153             if (i >= buckets) {
 154                i = buckets - 1;
 155             }
 156          }
 157       } else {
 158          int top = buckets - 1;
 159 
 160          while (i < equals) {
 161             if (intval == statptr->largediststarts[i]) {
 162                break;
 163             }
 164             i++;
 165          }
 166          if ((i >= equals) && (i < buckets)) {
 167             top--;
 168             if (intval < statptr->largediststarts[top]) {
 169                int bottom = i;
 170                while (top != bottom) {
 171                   i = bottom + ((top - bottom) / 2);
 172                   if (intval < statptr->largediststarts[(i+1)]) {
 173                      top = i;
 174                   } else {
 175                      bottom = i+1;
 176                   }
 177                }
 178             }
 179             i = top + 1;
 180          }
 181       }
 182       statptr->largedistvals[i]++;
 183    } else {
 184       i = 0;
 185       while (i<(DISTSIZE-1)) {
 186          if ((i < equals) && (value == (double) statptr->distbrks[i])) {
 187             break;
 188          } else if ((i >= equals) && (intval < statptr->distbrks[i])) {
 189             break;
 190          }
 191          i++;
 192       }
 193       statptr->smalldistvals[i]++;
 194    }
 195 }

 198 static void stat_print_large_dist (statgen **statset, int statcnt, int count, char *identstr)
 199 {
 200    int  i, j;
 201    statgen *statptr = statset[0];
 202    int  step = statptr->distbrks[1];
 203    int  grow = statptr->distbrks[2];
 204    int  buckets = statptr->distbrks[(DISTSIZE-1)];
 205    int  bucketno = statptr->distbrks[0];
 206    double scale = (double) statptr->scale;
 207    int scaled = (scale == 1.0) ? FALSE : TRUE;
 208    double dist = 0.0;
 209    int  bucketcnt;
 210    double density;
 211 
 212    if (count == 0) {
 213       return;
 214    }
 215    fprintf (outputfile, "%s%s distribution\n", identstr, statptr->statdesc);
 216    for (i=0; i<buckets; i++) {
 217       if (statcnt > 1) {
 218          bucketcnt = 0;
 219          for (j=0; j<statcnt; j++) {
 220             statptr = statset[j];
 221             bucketcnt += statptr->largedistvals[i];
 222          }
 223       } else {
 224          bucketcnt = statptr->largedistvals[i];
 225       }
 226       density = (double) bucketcnt / (double) count;
 227       dist += density;
 228       if (scaled) {
 229          fprintf (outputfile, "%6.3f  \t%7d  \t%f  \t%f\n", ((double) bucketno / scale), bucketcnt, density, dist);
 230       } else {
 231          fprintf (outputfile, "%4d  \t%7d  \t%f  \t%f\n", bucketno, bucketcnt, density, dist);
 232       }
 233       bucketno += step + (int)((double) (abs(bucketno) * grow) / (double) 100);
 234    }
 235 }

 244 void stat_print_file (statgen *statptr, char *identstr, FILE *outfile){
 245    int i;
 246    int count = 0;
 247    int buckets = statptr->distbrks[(DISTSIZE-1)];
 248    double scale = (double) statptr->scale;
 249    int intval = statptr->distbrks[(DISTSIZE-2)];
 250    char *statdesc = statptr->statdesc;
 251    char distchar = '=';
 252    double runsquares = 0.0;
 253    double avg = 0.0;
 254 
 255    if (statptr->count > 0) {
 256       avg = statptr->runval / (double) statptr->count;
 257       runsquares = statptr->runsquares / (double) statptr->count - (avg*avg);
 258       runsquares = (runsquares > 0.0) ? sqrt(runsquares) : 0.0;
 259    }
 260    fprintf(outfile, "%s%s average: \t%f\n", identstr, statdesc, avg);
 261    fprintf(outfile, "%s%s std.dev.:\t%f\n", identstr, statdesc, runsquares);
 262    if (statptr->maxval == (double) ((int)statptr->maxval)) {
 263       fprintf(outfile, "%s%s maximum:\t%d\n", identstr, statdesc, ((int)statptr->maxval));
 264    } else {
 265       fprintf(outfile, "%s%s maximum:\t%f\n", identstr, statdesc, statptr->maxval);
 266    }
 267    if (buckets > DISTSIZE) {
 268       stat_print_large_dist(&statptr, 1, statptr->count, identstr);
 269       return;
 270    }
 271    fprintf(outfile, "%s%s distribution\n", identstr, statdesc);
 272    for (i=(DISTSIZE-buckets); i<(DISTSIZE-1); i++) {
 273       if (i >= statptr->equals) {
 274          distchar = '<';
 275       }
 276       if (statptr->scale == 1) {
 277          fprintf(outfile, "   %c%3d ", distchar, statptr->distbrks[i]);
 278       } else {
 279          fprintf(outfile, "  %c%4.1f ", distchar, ((double) statptr->distbrks[i] / scale));
 280       }
 281    }
 282    if (statptr->equals == (DISTSIZE-1)) {
 283       intval++;
 284    }
 285    fprintf(outfile, "   %3d+\n", intval);
 286    for (i=(DISTSIZE-buckets); i<DISTSIZE; i++) {
 287       count += statptr->smalldistvals[i];
 288       fprintf(outfile, " %6d ", statptr->smalldistvals[i]);
 289    }
 290    fprintf(outfile, "\n");
 291    ASSERT(count == statptr->count);
 292 }

 295 void stat_print_set (statgen **statset, int statcnt, char *identstr)
 296 {
 297    int i, j;
 298    int count = 0;
 299    statgen *statptr = statset[0];
 300    char *statdesc = statptr->statdesc;
 301    int buckets = statptr->distbrks[(DISTSIZE-1)];
 302    int intval = statptr->distbrks[(DISTSIZE-2)];
 303    double runval = 0.0;
 304    double runsquares = 0.0;
 305    double avg = 0.0;
 306    double maxval = statptr->maxval;
 307    double scale = (double) statptr->scale;
 308    int runcount = 0;
 309    int smalldistvals[DISTSIZE];
 310    char distchar = '=';
 311 
 312    for (i=0; i<statcnt; i++) {
 313       statptr = statset[i];
 314       if (statptr->count > 0) {
 315          runcount += statptr->count;
 316          runval += statptr->runval;
 317          runsquares += statptr->runsquares;
 318          if (maxval < statptr->maxval) {
 319             maxval = statptr->maxval;
 320          }
 321       }
 322    }
 323    if (runcount > 0) {
 324       avg = runval / (double) runcount;
 325       runsquares = (runsquares / (double) runcount) - (avg*avg);
 326       runsquares = (runsquares > 0.0) ? sqrt(runsquares) : 0.0;
 327    } else {
 328       runsquares = 0.0;
 329    }
 330    fprintf(outputfile, "%s%s average: \t%f\n", identstr, statdesc, avg);
 331    fprintf(outputfile, "%s%s std.dev.:\t%f\n", identstr, statdesc, runsquares);
 332    if (maxval == (double) ((int)maxval)) {
 333       fprintf(outputfile, "%s%s maximum:\t%d\n", identstr, statdesc, (int)maxval);
 334    } else {
 335       fprintf(outputfile, "%s%s maximum:\t%f\n", identstr, statdesc, maxval);
 336    }
 337    if (buckets > DISTSIZE) {
 338       stat_print_large_dist(statset, statcnt, runcount, identstr);
 339       return;
 340    }
 341    fprintf(outputfile, "%s%s distribution\n", identstr, statdesc);
 342    statptr = statset[0];
 343    for (i=(DISTSIZE-buckets); i<(DISTSIZE-1); i++) {
 344       if (i >= statptr->equals) {
 345          distchar = '<';
 346       }
 347       if (statptr->scale == 1) {
 348          fprintf(outputfile, "   %c%3d ", distchar, statptr->distbrks[i]);
 349       } else {
 350          fprintf(outputfile, "  %c%4.1f ", distchar, ((double) statptr->distbrks[i] / (double) scale));
 351       }
 352    }
 353    if (statptr->equals == (DISTSIZE-1)) {
 354       intval++;
 355    }
 356    fprintf(outputfile, "   %3d+\n", intval);
 357    for (i=0; i<DISTSIZE; i++) {
 358       smalldistvals[i] = 0;
 359    }
 360    for (i=0; i<statcnt; i++) {
 361       statptr = statset[i];
 362       for (j=0; j<DISTSIZE; j++) {
 363          smalldistvals[j] += statptr->smalldistvals[j];
 364       }
 365    }
 366    for (i=(DISTSIZE-buckets); i<DISTSIZE; i++) {
 367       count += smalldistvals[i];
 368       fprintf(outputfile, " %6d ", smalldistvals[i]);
 369    }
 370    fprintf(outputfile, "\n");
 371    ASSERT2(count == runcount, "count", count, "runcount", runcount);
 372 }

 375 void stat_reset (statgen *statptr)
 376 {
 377    int buckets = statptr->distbrks[(DISTSIZE-1)];
 378    int i;
 379 
 380    statptr->count = 0;
 381    statptr->runval = 0.0;
 382    statptr->runsquares = 0.0;
 383    statptr->maxval = 0.0;
 384    if (buckets > DISTSIZE) {
 385       for (i=0; i<buckets; i++) {
 386          statptr->largedistvals[i] = 0;
 387       }
 388    } else {
 389       for (i=0; i<DISTSIZE; i++) {
 390          statptr->smalldistvals[i] = 0;
 391       }
 392    }
 393 }

 396 static void stat_get_large_dist (FILE *statdef_file, statgen *statptr, int buckets)
 397 {
 398    int i;
 399    char line[201];
 400 
 401    statptr->largedistvals = (int *) DISKSIM_malloc(buckets*sizeof(int));
 402    ASSERT(statptr->largedistvals != NULL);
 403 
 404    if (fgets(line, 200, statdef_file) == NULL) {
 405       fprintf(stderr, "Can't get line describing large dist\n");
 406       exit(1);
 407    }
 408    if (sscanf(line, "Start %d  step %d  grow %d", &statptr->distbrks[0], &statptr->distbrks[1], &statptr->distbrks[2]) != 3) {
 409       fprintf(stderr, "Bad format for line describing large dist - %s\n", line);
 410       exit(1);
 411    }
 412    for (i=0; i<buckets; i++) {
 413       statptr->largedistvals[i] = 0;
 414    }
 415    if (statptr->distbrks[2]) {
 416       int grow = statptr->distbrks[2];
 417       int step = statptr->distbrks[1];
 418       int bucketval = statptr->distbrks[0];
 419       int i;
 420 
 421       statptr->largediststarts = (int *) DISKSIM_malloc(buckets*sizeof(int));
 422       ASSERT(statptr->largediststarts != NULL);
 423 
 424       for (i=0; i < buckets; i++) {
 425          statptr->largediststarts[i] = bucketval;
 426          bucketval += step + (int)((double) (abs(bucketval) * grow) / (double) 100);
 427       }
 428    }
 429 }

 432 void stat_initialize (FILE *statdef_file, char *statdesc, statgen *statptr)
 433 {
 434    int i;
 435    int buckets;
 436    char line[201];
 437    char line2[201];
 438 
 439    if (fseek(statdef_file, 0L, 0)) {
 440       fprintf(stderr, "Can't rewind the statdef_file\n");
 441       exit(1);
 442    }
 443    sprintf(line2, "%s\n", statdesc);
 444    while (1) {
 445       if (fgets(line, 200, statdef_file) == NULL) {
 446          fprintf(stderr, "No configuration for stat %s given in statdefs file\n", statdesc);
 447          exit(1);
 448       }
 449       if (strcmp(line, line2) == 0) {
 450          break;
 451       }
 452    }
 453 
 454    if (fgets(line, 200, statdef_file) == NULL) {
 455       fprintf(stderr, "Stat definition only partially found: %s\n", statdesc);
 456       exit(1);
 457    }
 458 
 459    if (sscanf(line, "Distribution size: %d\n", &buckets) != 1) {
 460       fprintf(stderr, "Invalid format for 'distribution size': %s\n", statdesc);
 461       exit(1);
 462    }
 463 
 464    if (fgets(line, 200, statdef_file) == NULL) {
 465       fprintf(stderr, "Stat definition only partially found: %s\n", statdesc);
 466       exit(1);
 467    }
 468 
 469    if (sscanf(line, "Scale/Equals: %d/%d\n", &statptr->scale, &statptr->equals) != 2) {
 470       fprintf(stderr, "Invalid format for 'scale/equals': %s\n", statdesc);
 471       exit(1);
 472    }
 473 
 474    statptr->count = 0;
 475    statptr->runval = 0.0;
 476    statptr->runsquares = 0.0;
 477    statptr->maxval = 0.0;
 478    statptr->statdesc = DISKSIM_malloc (strlen(statdesc)+1);
 479    strcpy (statptr->statdesc, statdesc);
 480    for (i=0; i<DISTSIZE; i++) {
 481       statptr->distbrks[i] = 0;
 482       statptr->smalldistvals[i] = 0;
 483    }
 484    statptr->distbrks[(DISTSIZE-1)] = buckets;
 485    statptr->largedistvals = NULL;
 486    statptr->largediststarts = NULL;
 487 
 488    if (buckets > DISTSIZE) {
 489       stat_get_large_dist(statdef_file, statptr, buckets);
 490    } 
 491    else {
 492      for (i = 0; i < (DISTSIZE-1); i++) {
 493        if (fscanf(statdef_file, "%d", &statptr->distbrks[i]) != 1) {
 494          fprintf(stderr, "Distribution buckets only partially specified: %s %d\n", statdesc, i);
 495          exit(1);
 496        }
 497      }
 498    }
 499 }
