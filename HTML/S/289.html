  39 // public interface to libdisksim

  41 #include "disksim_reqflags.h"

  43 /* System level request */
  44 struct disksim_request {
  45   int flags;
  46   short devno;
  47   unsigned long blkno;
  48   int bytecount;
  49   double start;
  50   int batchno;
  51   int batch_complete;
  52   void *reqctx;         /* context passed with the request */
  53 };
  54 
  55 struct disksim_interface;
  56 
  57 // callback when requests complete
  58 typedef void(*disksim_interface_complete_t)(double time,
  59                                             struct disksim_request *,
  60                                             void *ctx);
  61 
  62 // scheduled callback function (below)
  63 typedef void(*disksim_interface_callback_t)(struct disksim_interface *,
  64                                             double t,
  65                                             void *ctx);
  66 
  67 // callback (disksim -> client) to register a callback (client ->
  68 // disksim); when simulated time reaches t, client code should invoke
  69 // fn.
  70 typedef void(*disksim_interface_sched_t)(disksim_interface_callback_t fn,
  71                                          double,
  72                                          void *ctx);
  73 
  74 // cancel the pending callback set for t
  75 typedef void(*disksim_interface_desched_t)(double t, void *ctx);
  76 
  77 // Allocate and initialize a disksim_interface handle.
  78 // pfile is the name of disksim's parameter file.
  79 // ofile is the name of disksim's output file.
  80 // ctx is an arbitrary pointer which will be passed back to the callback
  81 // functions when they are invoked.
  82 struct disksim_interface * 
  83 disksim_interface_initialize (const char *pfile, 
  84                               const char *ofile,
  85                               disksim_interface_complete_t,
  86                               disksim_interface_sched_t,
  87                               disksim_interface_desched_t,
  88                               void *ctx,
  89                               int argc,
  90                               char **argv);
  91 
  92 
  93 void 
  94 disksim_interface_shutdown (struct disksim_interface *,
  95                             double syssimtime);
  96 void 
  97 disksim_interface_dump_stats (struct disksim_interface *,
  98                               double syssimtime);
  99 
 100 // pump disksim's event loop
