  63 static void
  64 mems_send_event_up_path (ioreq_event * curr,
  65                          double delay)
  66 {
  67   /* Acquire the bus (if not already acquired), then use bus_delay
  68    * to send the event up the path.
  69    */
  70 
  71   mems_t *dev   = getmems(curr->devno);
  72   int     busno  = mems_get_busno(curr);
  73   int     slotno = mems_get_slotno(curr->devno);
  74 
  75   /* Put new request at head of buswait queue */
  76   curr->next     = dev->buswait;
  77   dev->buswait   = curr;
  78   curr->tempint1 = busno;
  79   curr->time     = delay;
  80 
  81   if (dev->busowned == -1)
  82     {
  83       /* Must get ownership of bus first */
  84       if (bus_ownership_get(busno, slotno, curr) == FALSE) {
  85         /* FIXME: dev->stat.requestedbus = simtime */
  86       }
  87       else
  88         {
  89           bus_delay(busno, DEVICE, curr->devno, delay, curr);
  90         }
  91     }
  92   else if (dev->busowned == busno)
  93     {
  94       /* Already own the bus, so send it on up! */
  95       bus_delay(busno, DEVICE, curr->devno, delay, curr);
  96     }
  97   else
  98     {
  99       fprintf(stderr, "Wrong bus owned for transfer desired!\n");
 100       exit(1);
 101     }
 102 }

 138 mems_bus_delay_complete (int devno,
 139                          ioreq_event *curr,
 140                          int sentbusno)
 141 {
 142   mems_t *dev = getmems(devno);
 143   intchar slotno;
 144   intchar busno;
 145   int     depth;
 146 
 147 #ifdef VERBOSE_EVENTLOOP
 148   printf("Hello! I'm in mems_bus_delay_complete: %f\n", simtime);
 149   fflush(stdout);
 150 #endif
 151 
 152   busno.value  = curr->busno;
 153   slotno.value = curr->slotno;
 154   depth = mems_get_depth(curr->devno);
 155   slotno.byte[depth] = slotno.byte[depth] >> 4;
 156 
 157   curr->time = (double)0.0;
 158 
 159   if (curr == dev->buswait)
 160     {
 161       dev->buswait = curr->next;
 162     }
 163   else
 164     {
 165       ioreq_event *tmp = dev->buswait;
 166       while ((tmp->next != NULL) && (tmp->next != curr)) tmp = tmp->next;
 167       if (tmp->next != curr)
 168         {
 169           fprintf(stderr,
 170                   "Bus delay complete for unknown request -- devno %d, busno %d\n",
 171                   devno, busno.value);
 172           exit(1);
 173         }
 174     }
 175 
 176   if (depth == 0)
 177     {
 178       intr_request ((event *)curr);
 179     }
 180   else
 181     {
 182       bus_deliver_event (busno.byte[depth], slotno.byte[depth], curr);
 183     }

 295 static mems_reqinfo_t *
 296 mems_get_new_reqinfo (ioreq_event *curr)
 297 {
 298   ioreq_event *event_ptr;
 299   mems_extent_t *extent_ptr;
 300 
 301   // mems_reqinfo_t *r = (mems_reqinfo_t *)getfromextraq();
 302   mems_reqinfo_t *r = (mems_reqinfo_t *)malloc(sizeof(mems_reqinfo_t));
 303   mems_reqinfo_mallocs++;
 304 
 305   extent_ptr = r->extents;
 306 
 307   r->extents = mems_get_new_extent(curr);
 308   r->num_extents = 1;
 309 
 310   extent_ptr = r->extents;
 311   event_ptr = curr;
 312   // if there are more in the batch  (there shouldn't be)
 313   while (event_ptr->batch_next != NULL)
 314     {
 315       event_ptr = event_ptr->batch_next;
 316       extent_ptr->next = mems_get_new_extent(event_ptr);
 317       r->num_extents++;
 318     }
 319 
 320   r->bus_pending = FALSE;
 321   r->bus_done    = FALSE;
 322   r->media_done  = FALSE;
 323 
 324   r->bus_extent = NULL;
 325 
 326   r->request_energy_uj = 0.0;
 327   r->firstseek            = 1;  /* TRUE: first seek hasn't yet completed */
 328   r->subtrack_access_num  = 0;
 329   r->subtrack_access_tips = 0;
 330 
 331   return r;

 617 #ifdef VERBOSE_EVENTLOOP
 618   printf("\nIO_ACCESS_ARRIVE: %f -- %s for %d(%d), batchno = %d, batch_complete = %d\n",
 619          simtime, (curr->flags & READ) ? "R" : "W", curr->blkno, curr->bcount, curr->batchno, curr->batch_complete);
 620   fflush(stdout);
 621 #endif
 622  
 623   /* A new request!
 624    * - Initialize the per-request mems info structure.
 625    * - Map the request to the appropriate sled. */
 626 
 627   curr->batch_next = NULL;
 628   curr->batch_prev = NULL;
 629   curr->batch_size = 0;
 630 
 631   curr->mems_sled = mems_lbn_to_sled(dev, curr->blkno);
 632   curr->mems_reqinfo = mems_get_new_reqinfo(curr);
 633   
 634   dev->busowned = mems_get_busno(curr);
 635   
 636   curr->time = simtime + dev->overhead;
 637   curr->type = DEVICE_OVERHEAD_COMPLETE;
 638   addtointq((event *)curr);

 772           mems_energy_update_device_overhead_complete_inactive(sled);
 773         }
 774       else if (sled->active == MEMS_SLED_IDLE)
 775         {
 776           /*  the sled is idle, but not spun down  */
 777           coord_t *update_coord;
 778           double elapsed_time;
 779           double sweep_time;
 780           double leftover_time;
 781           double extra_sweeps;
 782           int extra_distance;
 783           int sled_length_bits;
 784         
 785           ioreq_event *r = (ioreq_event *)getfromextraq();
 786           // printf("GETFROMEXTRAQ - second\n");
 787           r->devno = curr->devno;
 788           /* FIXME: Crashes without above line. Why? Use ioreq_copy? */

 795         
 796           /*  figure out where the sled should be at this point  */
 797           update_coord = mems_get_current_position(sled);
 798         
 799           elapsed_time = simtime - sled->lastreq_comptime;
 800           sweep_time =

 802                                  sled->y_access_speed_bit_s * sled->bit_length_nm,
 803                                  sled->x_accel_nm_s2,
 804                                  sled->spring_factor,
 805                                  sled->y_length_nm);

 807             sled->y_length_nm / sled->y_access_speed_bit_s * sled->bit_length_nm;
 808         
 809           leftover_time = fmod(elapsed_time, sweep_time);
 810           extra_sweeps = (double)(leftover_time / sweep_time);
 811         
 812           sled_length_bits = sled->y_length_nm / sled->bit_length_nm;
 813         
 814           extra_distance =
 815             (int)(extra_sweeps * (double)(sled->y_length_nm)) / sled->bit_length_nm;
 816         
 817           if (update_coord->y_vel > 0)
 818             {
 819               /*  if we're moving up  */
 820               update_coord->y_pos += extra_distance;
 821             
 822               if (abs(update_coord->y_pos) > (sled_length_bits / 2))
 823                 {
 824                   /*  if we've gone too far...  */
 825                   update_coord->y_pos =
 826                     ((sled_length_bits / 2)
 827                      -
 828                      (update_coord->y_pos - (sled_length_bits / 2)));
 829                   update_coord->y_vel = -update_coord->y_vel;
 830                 }
 831             
 832             }
 833           else
 834             {
 835               /*  we're moving down  */
 836               update_coord->y_pos -= extra_distance;
 837             
 838               if (abs(update_coord->y_pos) > (sled_length_bits / 2))
 839                 {
 840                   /*  if we've gone too far...  */
 841                   update_coord->y_pos =
 842                     ((-sled_length_bits / 2)
 843                      +
 844                      (abs(update_coord->y_pos) - (sled_length_bits / 2)));
 845                   update_coord->y_vel = -update_coord->y_vel;
 846                 }
 847             }
 848           mems_commit_move(sled, update_coord);
 849         
 850           r->mems_sled = sled;
 851           addtointq((event *)r);
 852           sled->active = MEMS_SLED_ACTIVE;
 853         
 854           mems_energy_update_device_overhead_complete_active(sled,
 855                                                              elapsed_time);
 856         }
 857     }

 903     {
 904       /*  the sled is idle, but now there's a new request  */
 905       // printf("Sled was idle, but there's a new request coming in.\n");
 906       sled->active = MEMS_SLED_ACTIVE;

 911

 914     {
 915       sled->active_request = ioqueue_get_next_request(sled->queue);

 925           // returns the ioreq_event of the first request in the batch.  so if that
 926           // request finishes before the others in the batch, it will hose the entire
 927           // batch and all of its extents.  i need to make a copy of the ioreq_event
 928           // and use that to execute this batch.  and make sure i free it when all of
 929           // the extents are complete.
 930           //
 931           // i think...
 932

 965             }
 966         }
 967     }
 968   
 969   /* Now, if there's an active request, figure out which position to seek
 970    * to, and which blocks to access once we're there. */
 971   if (sled->active_request)
 972     {
 973       int extent;
 974       int block;
 975       int lastblock;
 976       mems_reqinfo_t *reqinfo;
 977       int started_mapping = FALSE;
 978       tipsector_coord_set_t tmp_coordset_up, tmp_coordset_dn;
 979       struct tipset tmp_tipset;
 980       tipsector_coord_set_t first_coordset_up, first_coordset_dn;
 981       struct tipset first_tipset;
 982       mems_extent_t *extent_ptr;
 983       
 984       reqinfo = sled->active_request->mems_reqinfo;

 986 #ifdef VERBOSE_EVENTLOOP

 990

 992       extent_ptr = reqinfo->extents;
 993

 995         {
 996           if (extent_ptr->media_done == TRUE)
 997             {
 998               extent_ptr = extent_ptr->next;
 999               continue;
1000             }
1001 
1002           if (extent_ptr->completed_block_media == -1)
1003             {
1004               extent_ptr->next_block_start = extent_ptr->firstblock;
1005             }
1006           else
1007             {
1008               extent_ptr->next_block_start = extent_ptr->completed_block_media + 1;
1009             }
1010           
1011           // set the first lbn's position - this is what we compare to the

1013           if (started_mapping == FALSE)

1024             }
1025 
1026           // find the position for this extent's next block
1027           mems_lbn_to_position(extent_ptr->next_block_start,
1028                                sled,
1029                                &extent_ptr->coordset_up,
1030                                &extent_ptr->coordset_dn,
1031                                &extent_ptr->tipset,
1032                                NULL, NULL, NULL);
1033

1039               extent_ptr = extent_ptr->next;

1041             }
1042           
1043           if (sled->tipset.num_tips + extent_ptr->tipset.num_tips
1044               >
1045               sled->tips_simultaneous)
1046             {
1047               extent_ptr->next_block_start = -1;
1048               extent_ptr->next_block_end = -1;
1049               extent_ptr = extent_ptr->next;
1050               continue;
1051             }
1052           
1053           sled->tipset.num_tips += extent_ptr->tipset.num_tips;
1054           extent_ptr->next_block_end = extent_ptr->next_block_start;
1055           
1056           // go through this extent's blocks and find the last that can be
1057           // accessed in parallel
1058           for (block = extent_ptr->next_block_end + 1;
1059                block <= extent_ptr->lastblock;
1060                block++)
1061             {
1062               // if this is a write and we haven't transferred this block over the bus,
1063               // it can't be written yet.
1064               if (!(sled->active_request->flags & READ) &&
1065                   (block > extent_ptr->completed_block_bus))
1066                 {
1067                   break;
1068                 }
1069               
1070               mems_lbn_to_position(block,
1071                                    sled,
1072                                    &tmp_coordset_up,
1073                                    &tmp_coordset_dn,
1074                                    &tmp_tipset,
1075                                    NULL, NULL, NULL);
1076               
1077               // if this block's coordset is not equal to first_coordset,
1078               // then they are in different positions and can't be accessed
1079               // in parallel and we're done
1080               if (mems_equal_coord_sets(&sled->coordset_up,
1081                                         &tmp_coordset_up) == FALSE)
1082                 {
1083                   break;
1084                 }
1085               
1086               // if adding this block exceeds the number of available tips,
1087               // then break.
1088               if (sled->tipset.num_tips + tmp_tipset.num_tips
1089                   >
1090                   sled->tips_simultaneous)
1091                 {
1092                   break;
1093                 }
1094         
1095               sled->tipset.num_tips += tmp_tipset.num_tips;
1096               extent_ptr->next_block_end = block;
1097             }
1098           extent_ptr = extent_ptr->next;
1099         }
1100       /* If we've started prefetching, update prefetch info */
1101       /*
1102         if (reqinfo->next_block_end > reqinfo->lastblock) {
1103         sled->prefetch_info->next_block_end = reqinfo->next_block_end;
1104         sled->prefetch_info->next_block_start = reqinfo->lastblock + 1;
1105         reqinfo->next_block_end = reqinfo->lastblock;
1106         }
1107       */
1108     }
1109   else
1110     {
1111       /* No active request, no prefetch information, nothing to do, so
1112          set the sled to be idle and schedule an event for when it
1113          should actually spin down. */
1114       if (curr->time < sled->lastreq_comptime + sled->inactive_delay_ms)
1115         {
1116           // printf("I think it's idle, but not spun down\n");
1117           sled->lastreq_comptime = curr->time;
1118           sled->active = MEMS_SLED_IDLE;
1119           sled->active_request = curr;   /* keep a pointer to this event in case */
1120           
1121           curr->type = MEMS_SLED_SCHEDULE;
1122           curr->time = simtime + sled->inactive_delay_ms;
1123           addtointq((event *)curr);
1124           return;
1125         }
1126       else if (curr->time
1127                >=
1128                sled->lastreq_comptime + sled->inactive_delay_ms)
1129         {
1130           /* If enough time has passed, switch sled to inactive state */
1131           // printf("Now I've spun down.\n");
1132           sled->lastreq_comptime = curr->time;
1133           // sled->lastreq_lbn = mems_centermost_lbn(sled);
1134           
1135           /* Initialize sled position, velocity at centermost LBN */
1136           mems_lbn_to_position(mems_centermost_lbn(sled), sled, &sled->coordset_up, 
1137                                &sled->coordset_dn, &sled->tipset, 
1138                                NULL, NULL, NULL);
1139           mems_coord_t_copy(&sled->coordset_up.servo_start, &sled->pos);
1140           sled->active = MEMS_SLED_INACTIVE;
1141           addtoextraq((event *)curr);
1142           
1143           /* Spindown/spinup statistics */
1144           sled->stat.num_spindowns++;
1145           sled->dev->stat.num_spindowns++;
1146           
1147           return;
1148         }
1149     }
1150   
1151   // printf("I DID THIS!!\n");
1152 #ifdef VERBOSE_EVENTLOOP
1153   printf("MEMS_SLED_SCHEDULE:: after\n");
1154   print_extent_info(sled->active_request->mems_reqinfo);
1155 #endif
1156 
1157   curr->type = MEMS_SLED_SEEK;
1158   addtointq((event *)curr);
1159 }

1227   enum direction_enum { UP, DOWN } direction;
1228   double time_up, x_up, y_up, to_up;
1229   double time_dn = 0.0;
1230   double x_dn, y_dn, to_dn;
1231   int num_up, num_dn;
1232   double timedelta;
1233 
1234 #ifdef VERBOSE_EVENTLOOP
1235   printf("MEMS_SLED_SEEK: %f\n", simtime);
1236   fflush(stdout);
1237 #endif
1238 
1239   time_up = mems_seek_time_seekcache(sled, &sled->pos, 
1240                                      &sled->coordset_up.servo_start, 
1241                                      &x_up, &y_up, &to_up, &num_up);
1242   /*
1243     time_up = mems_seek_time(sled, &sled->pos, 
1244     &sled->coordset_up.servo_start, 
1245     &x_up, &y_up, &to_up, &num_up);
1246   */
1247   /* This call can be expensive, so don't do it if unnecessary */
1248   if (sled->bidirectional_access)

1250       time_dn = mems_seek_time_seekcache(sled, &sled->pos, 
1251                                          &sled->coordset_dn.servo_start, 
1252                                          &x_dn, &y_dn, &to_dn, &num_dn);
1253     }
1254   
1255   if (!sled->bidirectional_access)
1256     {
1257       direction = UP;
1258     }
1259   else if (time_up < time_dn)
1260     {
1261       direction = UP;
1262     }
1263   else

1266     }

1268   /* FIXME: If the request spans multiple adjacent sectors,
1269    * we want zero seek time between the adjacent sectors.  To ensure
1270    * this, we want the _initial_ seek to go to the position farthest
1271    * from the _second_ sector, so when it finishes accessing the
1272    * first sector it can just keep on trucking in the same
1273    * direction.  (Am I making sense here?) */
1274   
1275   if (direction == UP)
1276     {
1277       timedelta = time_up;
1278       mems_tipsector_coord_set_t_copy(&sled->coordset_up, &sled->coordset);

1286     }
1287 
1288   /*
1289     fprintf(stderr, "MEMS_SLED_SEEK - timedelta = %f\n", timedelta);
1290   */
1291   curr->type = MEMS_SLED_SERVO;
1292   curr->time += timedelta;
1293   addtointq((event *)curr);
1294   mems_commit_move(sled, &sled->coordset.servo_start);
1295   
1296   /* Energy statistics */
1297   mems_energy_update_sled_seek(sled, timedelta);
1298 }

1300 void
1301 mems_sled_servo(ioreq_event *curr)
1302 {
1303   mems_sled_t *sled = curr->mems_sled;
1304   double timedelta;
1305 
1306 #ifdef VERBOSE_EVENTLOOP
1307   printf("MEMS_SLED_SERVO: %f\n", simtime);
1308   fflush(stdout);
1309   /*
1310     fprintf(stderr, "x_pos = %d, y_pos = %d\n",
1311     sled->pos.x_pos, sled->pos.y_pos);
1312   
1313     fprintf(stderr, "tip_start = %d, tip_end = %d\n",
1314     sled->tipset.tip_start, sled->tipset.tip_end);
1315   */
1316 #endif
1317 
1318   /* Calculate servo access time */
1319   timedelta = mems_media_access_time(sled, &sled->pos, 
1320                                      &sled->coordset.tipsector_start);
1321   
1322   /* Only access data if there is an active request or prefetching */
1323   if (sled->active_request || sled->prefetch_info)
1324     {
1325       curr->type = MEMS_SLED_DATA;
1326     }
1327   else
1328     {
1329       curr->type = MEMS_SLED_SCHEDULE;
1330     }
1331 
1332   /* Weird special case:  If this is a write request, and we
1333    * haven't busxfered ANY of the blocks we're trying to write, then
1334    * we should return to MEMS_SLED_SCHEDULE immediately.  It
1335    * was easier to put in this small hack than do a lot of
1336    * hacking above. 
1337    *
1338    * Note I don't expect this to ever actually happen, but
1339    * we're building a quality simulator here. */
1340   if (sled->active_request)
1341     {
1342       mems_reqinfo_t *reqinfo = sled->active_request->mems_reqinfo;
1343       mems_extent_t *extent_ptr = reqinfo->extents;
1344       int i;
1345       int data_to_write = TRUE;
1346       
1347       for (i = 0; i < reqinfo->num_extents; i++)
1348         {
1349           if (!(extent_ptr->request->flags & READ) &&
1350               (extent_ptr->next_block_start > extent_ptr->completed_block_bus))
1351             {
1352               data_to_write = FALSE;
1353             }
1354           else
1355             {
1356               data_to_write = TRUE;
1357             }
1358           extent_ptr = extent_ptr->next;
1359         }
1360       if (data_to_write == FALSE)
1361         {
1362           curr->type = MEMS_SLED_SCHEDULE;
1363           /* Turn only one tip on during the access (minimize energy) */
1364           // sled->tipset.tip_end = sled->tipset.tip_start;
1365           sled->tipset.num_tips = 1;
1366           
1367           // fprintf(stderr,"Woah! The weird special case is here!\n");
1368         }
1369     }
1370   
1371   curr->time += timedelta;
1372   addtointq((event *)curr);
1373   mems_commit_move(sled, &sled->coordset.tipsector_start);
1374   
1375   /* Energy statistics */
1376   mems_energy_update_sled_servo(sled, timedelta);
1377 }

1409 void
1410 mems_sled_update(ioreq_event *curr)
1411 {
1412   mems_sled_t *sled = curr->mems_sled;
1413 
1414 #ifdef VERBOSE_EVENTLOOP
1415   printf("MEMS_SLED_UPDATE: %f\n", simtime);
1416   fflush(stdout);
1417 #endif
1418   
1419   if (sled->active_request)
1420     {
1421       mems_reqinfo_t *reqinfo = sled->active_request->mems_reqinfo;
1422       int extent;
1423       mems_extent_t *extent_ptr;
1424       mems_extent_t *next_extent;
1425       int all_media_done = TRUE;
1426       int num_initial_extents = reqinfo->num_extents;
1427 
1428 #ifdef VERBOSE_EVENTLOOP
1429       printf("MEMS_SLED_UPDATE:: before\n");
1430       print_extent_info(reqinfo);
1431 #endif
1432       
1433       extent_ptr = reqinfo->extents;
1434       
1435       for (extent = 0; extent < num_initial_extents; extent++)
1436         {
1437           double latency = 0.0;
1438           next_extent = extent_ptr->next;
1439           extent_ptr->completed_block_media = extent_ptr->next_block_end;
1440           /* BUFFER - insert the blocks into the buffer */
1441           if (extent_ptr->request->flags & READ) {
1442             mems_buffer_insert(extent_ptr->next_block_start, 
1443                                extent_ptr->next_block_end, sled->dev);      
1444           }
1445           
1446           // if this extent is a read and the bus is not pending,
1447           // then initiate a bus access - but only one!
1448           if ((extent_ptr->request->flags & READ) &&
1449               (reqinfo->bus_pending == FALSE))
1450             {
1451               ioreq_event *busreq = ioreq_copy(sled->active_request);
1452               busreq->type = MEMS_BUS_INITIATE;
1453               busreq->time = simtime;
1454               addtointq((event *)busreq);
1455               reqinfo->bus_pending = TRUE;
1456               assert(extent_ptr->bus_done == FALSE);
1457             }
1458           
1459           if (extent_ptr->completed_block_media >= extent_ptr->lastblock)
1460             {
1461               extent_ptr->media_done = TRUE;
1462               
1463               // if the bus transfer is done as well, then pull this extent out and
1464               // respond to its request
1465               if (extent_ptr->bus_done == TRUE)
1466                 {
1467                   if (extent_ptr->prev)
1468                     {
1469                       extent_ptr->prev->next = extent_ptr->next;
1470                     }
1471                   else
1472                     {
1473                       reqinfo->extents = extent_ptr->next;
1474                     }
1475                   if (extent_ptr->next)
1476                     {
1477                       extent_ptr->next->prev = extent_ptr->prev;
1478                     }
1479                   reqinfo->num_extents--;
1480                   if (reqinfo->num_extents == 0)
1481                     {
1482                       // printf("completing batch! simtime = %f, arrival time = %f, diff = %f\n",
1483                       // simtime, reqinfo->batch_arrival_time, (simtime - reqinfo->batch_arrival_time));
1484                       stat_update(&sled->dev->stat.batch_response_time,
1485                                   (simtime - reqinfo->batch_arrival_time));
1486                       stat_update(&sled->stat.batch_response_time,
1487                                   (simtime - reqinfo->batch_arrival_time));
1488                     }
1489                   // SCHLOS - this is an ugly hack to make the write requests not complete
1490                   //  simultaneously.  If they do, then the Atropos threads break!
1491                   // latency += 0.000;
1492                   latency += 0.001;
1493                   mems_request_complete(extent_ptr->request, latency);
1494                   addtoextraq((event *)extent_ptr->request);
1495                   free(extent_ptr);
1496                   mems_extent_frees++;
1497                   // addtoextraq(extent_ptr);
1498                 }
1499             }
1500           else
1501             {
1502               all_media_done = FALSE;
1503             }
1504           
1505 #ifdef VERBOSE_EVENTLOOP
1506           printf("media completed = %d, last = %d\n",
1507                  extent_ptr->completed_block_media, extent_ptr->lastblock);
1508           fflush(stdout);
1509 #endif
1510           
1511           extent_ptr = next_extent;
1512         }
1513 
1514 #ifdef VERBOSE_EVENTLOOP
1515       printf("MEMS_SLED_UPDATE:: after\n");
1516       print_extent_info(reqinfo);
1517 #endif
1518 
1519       // if all of the extents are done with media transfers, then we can free 
1520       // the active request and the bus unit will clean everything else up
1521       if (all_media_done == TRUE)
1522         {
1523           addtoextraq((event *)sled->active_request);
1524           sled->lastreq_comptime = curr->time;
1525           sled->active_request   = NULL;
1526         }
1527     }
1528   // ELSE??
1529   
1530   curr->type = MEMS_SLED_SCHEDULE;
1531   addtointq((event *)curr);
1532 }

1593 {

1596   mems_reqinfo_t *reqinfo = curr->mems_reqinfo;
1597   mems_extent_t *extent_ptr;
1598   int num_extents;
1599   int i;
1600 
1601 #ifdef VERBOSE_EVENTLOOP
1602   printf("MEMS_BUS_INITIATE: %f\n", simtime);
1603   fflush(stdout);
1604 #endif
1605 
1606   // choose an extent to start bxfering
1607   if (reqinfo->bus_extent == NULL)
1608     {
1609       extent_ptr = reqinfo->extents;
1610       num_extents = reqinfo->num_extents;
1611       
1612       for (i = 0; i < num_extents; i++)
1613         {
1614           if (extent_ptr->completed_block_bus == -1)
1615             {
1616               reqinfo->bus_extent = extent_ptr;
1617               // printf("bus_initiate chose this extent - %d\n",
1618               //        extent_ptr->firstblock);
1619               break;
1620             }
1621           extent_ptr = extent_ptr->next;
1622         }
1623     }
1624   /* else {
1625      printf("bus initiate stuck with this extent - %d\n",
1626      reqinfo->bus_extent->firstblock);
1627      }
1628   */
1629 
1630   if (!dev->dataxfer_req)
1631     {
1632       /* This request has the bus: grab it and start transferring */
1633       dev->dataxfer_req = curr;
1634       send_reconnect(curr, 0.0);
1635     }
1636   else
1637     {
1638       /* Another request is waiting to transfer, so add to tail of
1639        * dataxfer queue */
1640       ioreq_event *tmp = dev->dataxfer_queue;
1641       if (!dev->dataxfer_queue)
1642         { /* If list is empty... */
1643           dev->dataxfer_queue = curr;
1644           curr->prev = NULL;
1645           curr->next = NULL;
1646         }
1647       else
1648         {
1649           while (tmp->next) tmp = tmp->next;    /* Find last node in list */
1650           tmp->next  = curr;    /* Make curr the new last node in list */
1651           curr->prev = tmp;
1652           curr->next = NULL;
1653         }
1654     }
1655 }

1673 void
1674 mems_bus_update(ioreq_event *curr)
1675 {
1676   mems_sled_t *sled = curr->mems_sled;
1677   mems_t *dev = getmems(curr->devno);
1678   mems_reqinfo_t *reqinfo = curr->mems_reqinfo;
1679   mems_extent_t *extent_ptr;
1680   mems_extent_t *bus_extent;
1681   int extent;
1682   int num_extents;
1683   int isread;
1684 
1685 #ifdef VERBOSE_EVENTLOOP
1686   printf("MEMS_BUS_UPDATE: %f\n", simtime);
1687   fflush(stdout);
1688 #endif
1689 
1690   bus_extent = reqinfo->bus_extent;
1691   isread = (bus_extent->request->flags & READ);
1692 
1693   /* Update last block transferred */
1694   if (bus_extent->completed_block_bus == -1)
1695     {
1696       bus_extent->completed_block_bus = bus_extent->firstblock;
1697     }
1698   else
1699     {
1700       bus_extent->completed_block_bus++;
1701     }
1702 
1703   /* BUFFER - insert blocks into buffer */
1704   if (!isread) {
1705     mems_buffer_insert(bus_extent->completed_block_bus,
1706                        bus_extent->completed_block_bus, dev);
1707   }
1708 
1709 #ifdef VERBOSE_EVENTLOOP
1710   printf("bus completed = %d, last = %d\n",
1711          bus_extent->completed_block_bus,
1712          bus_extent->lastblock);
1713   fflush(stdout);
1714 #endif
1715   
1716   /* Check if bus portion of request is complete */
1717   if (bus_extent->completed_block_bus >= bus_extent->lastblock)
1718     {
1719       bus_extent->bus_done = TRUE;
1720       reqinfo->bus_extent = NULL;
1721       
1722 #ifdef VERBOSE_EVENTLOOP
1723       printf("set bus_extent->bus_done to be TRUE\n");
1724       fflush(stdout);
1725 #endif
1726       
1727       /* If media access is also complete, do final request processing
1728        * Else free(req) -- sled will do final request processing */
1729       if (bus_extent->media_done == TRUE)
1730         {
1731           if (bus_extent->prev)
1732             {
1733               bus_extent->prev->next = bus_extent->next;
1734             }
1735           else
1736             {
1737               reqinfo->extents = bus_extent->next;
1738             }
1739           if (bus_extent->next)
1740             {
1741               bus_extent->next->prev = bus_extent->prev;
1742             }
1743           reqinfo->num_extents--;
1744           if (reqinfo->num_extents == 0)
1745             {
1746               // printf("completing batch! simtime = %f, arrival time = %f, diff = %f\n",
1747               // simtime, reqinfo->batch_arrival_time, (simtime - reqinfo->batch_arrival_time));
1748               stat_update(&sled->dev->stat.batch_response_time,
1749                           (simtime - reqinfo->batch_arrival_time));
1750               stat_update(&sled->stat.batch_response_time,
1751                           (simtime - reqinfo->batch_arrival_time));
1752             }
1753           mems_request_complete(bus_extent->request, 0.0);
1754           addtoextraq((event *)bus_extent->request);  // schlos
1755           free(bus_extent);
1756           mems_extent_frees++;
1757           // addtoextraq(bus_extent);
1758         }
1759 
1760       // check if there are other extents that need bus transfers
1761 #ifdef VERBOSE_EVENTLOOP
1762           printf("reqinfo->num_extents = %d\n",
1763                  reqinfo->num_extents);
1764           fflush(stdout);
1765 #endif
1766       extent_ptr = reqinfo->extents;
1767 
1768       while (extent_ptr != NULL)
1769         {
1770 #ifdef VERBOSE_EVENTLOOP
1771           printf("firstblock = %d, completed_block_bus = %d, request_type = %s, completed_block_media = %d\n",
1772                  extent_ptr->firstblock,
1773                  extent_ptr->completed_block_bus,
1774                  (((extent_ptr->request->flags & 0x1) == DISKSIM_READ) ? ("R") : ("W")),
1775                  extent_ptr->completed_block_media);
1776           fflush(stdout);
1777 #endif    
1778           if (extent_ptr->completed_block_bus == -1)
1779             {
1780               if (((extent_ptr->request->flags & 0x1) == DISKSIM_READ)
1781                   &&
1782                   (extent_ptr->completed_block_media != -1))
1783                 {
1784                   break;
1785                 }
1786               else if ((extent_ptr->request->flags & 0x1) == DISKSIM_WRITE)
1787                 {
1788                   break;
1789                 } 
1790             }
1791           extent_ptr = extent_ptr->next;
1792         }
1793 
1794 #ifdef VERBOSE_EVENTLOOP
1795       if (extent_ptr)
1796         {
1797           printf("selected extent for block %d as the next bus transfer\n",
1798                  extent_ptr->firstblock);
1799           fflush(stdout);
1800         }
1801       else
1802         {
1803           printf("extent_ptr is NULL!\n");
1804           fflush(stdout);
1805         }
1806 #endif
1807       
1808       if (!extent_ptr)
1809         {
1810           dev->dataxfer_req = NULL;
1811           reqinfo->bus_pending = FALSE;
1812           if (isread && (reqinfo->extents == NULL))
1813             {
1814               free(reqinfo);
1815               mems_reqinfo_frees++;
1816               // addtoextraq((event *)reqinfo);  // schlos - I fixed a leak here, i think
1817             }
1818         }
1819       else
1820         {
1821           reqinfo->bus_extent = extent_ptr;
1822         }
1823     
1824       // chucks curr because the request is totally finished
1825       /* addtoextraq((event *)curr); // moved to below */
1826     }
1827   else if ((bus_extent->request->flags & READ) &&
1828            (bus_extent->completed_block_bus >= 
1829             bus_extent->completed_block_media))
1830     {
1831       /* If this is a read request but the media hasn't finished
1832        * enough blocks for us to transfer, give up the bus until
1833        * such time as we have enough blocks to transfer */
1834       dev->dataxfer_req = NULL;
1835       reqinfo->bus_pending = FALSE;
1836       // chucks curr because there is no bus transfer to be done
1837       /* addtoextraq((event *)curr); // moved to below */
1838     }
1839   
1840   /* If we set (dataxfer_req = NULL) above, check if other requests
1841    * are waiting to transfer & give them the bus */
1842   if (!dev->dataxfer_req)
1843     {
1844       if (dev->dataxfer_queue)
1845         {
1846           /* Pop first node off dataxfer_queue */
1847           dev->dataxfer_req = dev->dataxfer_queue;
1848           dev->dataxfer_queue = dev->dataxfer_queue->next;
1849           if (dev->dataxfer_queue) dev->dataxfer_queue->prev = NULL;
1850           dev->dataxfer_req->next = NULL;
1851       
1852           /* FIXME: The 0.000001 is to ensure the RECONNECT or DISCONNECT
1853            * interrupts occur after the COMPLETION (if any, above) --
1854            * for some reason they're not being sequenced correctly. */
1855           send_reconnect(dev->dataxfer_req, 0.000001);
1856         }
1857       else
1858         {
1859           send_disconnect(curr, 0.000001);
1860         }
1861       addtoextraq((event *)curr);  // SCHLOS - this is where it messes up the intq!  curr is already there.
1862       return;
1863     }
1864   curr->type = MEMS_BUS_TRANSFER;
1865   addtointq((event *)curr);
1866 }
