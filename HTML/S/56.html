  73 struct idx_ent {
  74 
  75   // Offset
  76 
  77   int lbn;      // absolute if top-level, relative otherwise
  78   int cyl;      // ditto
  79 
  80   // skew to the start of each inst relative to the start of the
  81   // enclosing pat.
  82   dm_angle_t off;
  83 
  84   // Child Length
  85 
  86   // if this is RLE'd, the offset of the next inst relative to the
  87   // start of this one.  I don't think we need arunlen because
  88   // we never need to know the number of radians that a thing covers...
  89   int len;      
  90   int cyllen;
  91   dm_angle_t alen; 
  92 
  93 
  94   // Run Length -- indicates RLE if runlen > len, etc
  95 
  96   int runlen;
  97   int cylrunlen;  
  98 
  99 
 100   g4_node_t childtype;
 101   union g4_node child;
 102   int head; // only for childtype==TRACK
 103 };
