  39 // offset of element p in array base of size s
  40 static int
  41 aoffset(void *base, void *p, int s) {
  42 #ifdef WIN32
  43   return ((char *)p - (char *)base) / s;
  44 #else
  45   return (p - base) / s;
  46 #endif
  47 }

  82 
  83   l->linelen = 2;
  84 
  85   if(slips[0].off == 0 && slips[0].count == 0) {
  86     i = 1;
  87   }
  88 
  89   for( ; i < n; i++) {
  90     struct lp_value *v;
  91     v = lp_new_intv(slips[i].off);
  92     lp_list_add(l,v);
  93     v = lp_new_intv(slips[i].count - slips[i-1].count);

 122 struct lp_list *
 123 marshal_g4_idx_ent(struct idx_ent *e, 
 124                    struct idx *idx, 
 125                    struct track *track,
 126                    struct lp_list *l) 
 127 {
 128   int i;
 129   struct lp_value *v;
 130   char *ctstr; // child type
 131   int coffset; // offset of child in its array
 132 
 133   printf("%s() lbn %d alen %f off %f\n", __func__, e->lbn,
 134          dm_angle_itod(e->alen), dm_angle_itod(e->off));
 135 
 136   v = lp_new_intv(e->lbn);
 137   lp_list_add(l, v);
 138   v = lp_new_intv(e->cyl);
 139   lp_list_add(l, v);
 140   v = lp_new_doublev(dm_angle_itod(e->off));
 141   lp_list_add(l, v);
 142 
 143   v = lp_new_intv(e->len);
 144   lp_list_add(l, v);
 145   v = lp_new_intv(e->cyllen);
 146   lp_list_add(l, v);
 147   v = lp_new_doublev(dm_angle_itod(e->alen));
 148   lp_list_add(l, v);
 149 
 150   v = lp_new_intv(e->runlen);
 151   lp_list_add(l, v);
 152   v = lp_new_intv(e->cylrunlen);
 153   lp_list_add(l, v);
 154 
 155   switch(e->childtype) {
 156   case IDX:
 157     ctstr = "IDX";
 158     coffset = aoffset(idx, e->child.i, sizeof(*e->child.i));
 159     break;
 160   case TRACK:
 161     ctstr = "TRACK";
 162     coffset = aoffset(track, e->child.t, sizeof(*e->child.t));
 163     break;
 164   default: ddbg_assert(0); break;
 165   }
 166 
 167   v = lp_new_stringv(ctstr);
 168   lp_list_add(l, v);
 169 
 170   v = lp_new_intv(coffset);
 171   lp_list_add(l, v);
 172 
 173   if(e->childtype == TRACK) {
 174     v = lp_new_intv(e->head);
 175     lp_list_add(l, v);
 176   }
 177 
 178   return l;
 179 }

 181 struct lp_list *
 182 marshal_g4_idx(struct idx *idx, int n, struct idx *idxen, struct track *track) {
 183   int i, j;
 184   struct lp_list *l = lp_new_list();
 185 
 186   l->linelen = 1;
 187 
 188   for(i = 0; i < n; i++) {
 189     struct idx *ii = &idx[i];
 190     struct lp_list *lent = lp_new_list();
 191     struct lp_value *vlent = lp_new_listv(lent);
 192 
 193     // this breaks if an index has a mixture of track and index
 194     // children
 195     if(idx[i].ents[0].childtype == TRACK) {
 196       lent->linelen = 11;
 197     }
 198     else {
 199       lent->linelen = 10;
 200     }
 201 
 202     for(j = 0; j < ii->ents_len; j++) {
 203       marshal_g4_idx_ent(&ii->ents[j],  idxen, track, lent);
 204     }
 205 
 206     lp_list_add(l, vlent);
 207   }
 208 
 209 
 210   return l;
 211 }

 213 struct lp_block *
 214 marshal_layout_g4(struct dm_layout_g4 *lin) {
 215 
 216   struct lp_block *b = lp_new_block();
 217   struct lp_list *l;
 218   struct lp_value *v;
 219   struct lp_param *p;
 220 
 221   b->source_file = 0;
 222   b->name = 0;
 223   lp_lookup_type("dm_layout_g4", &b->type);
 224 
 225   l = marshal_g4_tp(lin->track, lin->track_len);
 226   v = lp_new_listv(l);
 227   p = lp_new_param("TP", 0, v);
 228   lp_add_param(&b->params, &b->params_len, p);
 229

 231   v = lp_new_listv(l);
 232   p = lp_new_param("IDX", 0, v);
 233   lp_add_param(&b->params, &b->params_len, p);
 234 
 235   l = marshal_g4_slip(lin->slips, lin->slips_len);
 236   v = lp_new_listv(l);
 237   p = lp_new_param("Slips", 0, v);
 238   lp_add_param(&b->params, &b->params_len, p);
 239 
 240   l = marshal_g4_remaps(lin->remaps, lin->remaps_len);
 241   v = lp_new_listv(l);
 242   p = lp_new_param("Remaps", 0, v);
 243   lp_add_param(&b->params, &b->params_len, p);

 290 #define TP_FIELDS 3
 291 int 
 292 g4_load_tp(struct dm_layout_g4 *r,
 293            struct lp_list *l)
 294 {
 295   int i;
 296   
 297   ddbg_assert(r->track_len == 0);
 298 
 299   r->track_len = l->values_pop / TP_FIELDS;
 300 
 301   r->track = calloc(r->track_len, sizeof(*r->track));
 302 
 303   for(i = 0; i < r->track_len; i++) {
 304     r->track[i].low = l->values[TP_FIELDS*i]->v.i;
 305     r->track[i].high = l->values[TP_FIELDS*i+1]->v.i;
 306     r->track[i].spt = l->values[TP_FIELDS*i+2]->v.i;
 307     r->track[i].sw = dm_angle_dtoi(1.0 / r->track[i].spt);
 308   }
 309 
 310   return 0;
 311 }

 314 // this is now the minimum since they're variable lenth
 315 #define IDX_FIELDS 10
 316 int 
 317 g4_load_idx(struct dm_layout_g4 *r,
 318            struct lp_list *l)
 319 {
 320 
 321   int i, j;
 322   // offset in l
 323   int loff = 0;
 324   
 325   r->idx_len = l->values_pop;
 326 
 327   r->idx = calloc(r->idx_len, sizeof(*r->idx));
 328 
 329   for(i = 0; i < r->idx_len; i++) {
 330     int reallen = 0;
 331     struct idx *opi = &r->idx[i];
 332     struct lp_list *li = l->values[i]->v.l;
 333     struct idx_ent *entj;
 334 
 335     loff = 0;
 336     
 337     // this is a little sloppy yields an upper bound, not an
 338     // exact number
 339     opi->ents_len = li->values_pop / IDX_FIELDS;
 340     opi->ents = calloc(opi->ents_len, sizeof(*opi->ents));
 341 
 342     for(j = 0, entj = opi->ents; 
 343         j < opi->ents_len && loff < li->values_pop; 
 344         j++, entj++) 
 345     {
 346       int off;
 347 
 348       entj->lbn = li->values[loff++]->v.i;
 349       entj->cyl = li->values[loff++]->v.i;
 350       entj->off = dm_angle_dtoi(li->values[loff++]->v.d);
 351 
 352       entj->len = li->values[loff++]->v.i;
 353       entj->cyllen = li->values[loff++]->v.i;
 354       entj->alen = dm_angle_dtoi(li->values[loff++]->v.d);
 355 
 356       entj->runlen = li->values[loff++]->v.i;
 357       entj->cylrunlen = li->values[loff++]->v.i;
 358 
 359       if(!strcmp(li->values[loff]->v.s, "TRACK")) {
 360         loff++;
 361         entj->childtype = TRACK;
 362         off = li->values[loff++]->v.i;
 363         ddbg_assert(off < r->track_len);
 364         entj->child.t = &r->track[off];
 365         entj->head = li->values[loff++]->v.i;
 366       }
 367       else if(!strcmp(li->values[loff]->v.s, "IDX")) {
 368         loff++;
 369         entj->childtype = IDX;
 370         // XXX weird recursion
 371         off = li->values[loff++]->v.i;
 372         ddbg_assert(off < r->idx_len);
 373         entj->child.i = &r->idx[off];
 374       }
 375       else {
 376         ddbg_assert(0);
 377       }
 378 
 379       reallen++;
 380     }
 381 
 382     opi->ents_len = reallen;
 383   }
 384 
 385 
 386 
 387   return 0;
 388 }

 390 #define SLIP_FIELDS 2
 391 int 
 392 g4_load_slips(struct dm_layout_g4 *r,
 393               struct lp_list *l)
 394 {
 395   int i;
 396   int tot = 0;
 397   r->slips_len = l->values_pop / SLIP_FIELDS;
 398   r->slips_len; // add "0" entry
 399 
 400   // allocate one extra; see below
 401   r->slips = calloc(r->slips_len+1, sizeof(*r->slips));
 402 
 403   for(i = 0; i < r->slips_len; i++) {
 404     r->slips[i].off = l->values[SLIP_FIELDS * i]->v.i;
 405     tot += l->values[SLIP_FIELDS * i + 1]->v.i;
 406     r->slips[i].count = tot;
 407   }
 408 
 409   // 0th lbn could be slipped
 410   if(r->slips[0].off > 0) { 
 411     memmove(r->slips + 1, r->slips, r->slips_len * sizeof(*r->slips));
 412     r->slips[0].off = 0;
 413     r->slips[0].count = 0;
 414     r->slips_len++;
 415   }
 416 
 417   return 0;
 418 }

 436     r->remaps[i].spt = l->values[REMAP_FIELDS * i + 5]->v.i;
 437   }
 438 
 439 
 440   return 0;
