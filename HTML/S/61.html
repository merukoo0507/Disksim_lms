  33 #include "dm.h"
  34 
  35 struct dm_layout_g2 {
  36   struct dm_layout_if hdr;
  37 
  38   struct dm_layout_g2_node *ltop_map;
  39   struct dm_layout_g2_cyl *ptol_map;
  40   int ltop_map_len;
  41 
  42   struct dm_layout_g2_zone *zones;
  43   int zones_len;
  44 
  45   char *mapfile; // name of raw layout file
  46   dm_skew_unit_t skew_units;
  47 
  48 
  49   // list of defect locations
  50   // currently only used for dm_defect_count()
  51   struct dm_pbn *defects;
  52   int defects_len;
  53 };
  54 
  55 
  56 struct dm_layout_g2_node {
  57   int lbn;
  58   //  int cyl;
  59   //  int head;
  60   struct dm_pbn loc; // location of lbnlow
  61   int len;
  62 };
  63 
  64 struct dm_layout_g2_cyl {
  65   struct dm_layout_g2_surf *surfaces;
  66   int surfaces_len;
  67   dm_angle_t skew;
  68   int first_ltop_extent; // first entry in ltop_map for this cyl
  69 };
  70 
  71 struct dm_layout_g2_surf {
  72   int lbnlow;
  73   int lbnhigh;
  74   // pointers back to ltop map extents
  75   struct dm_layout_g2_node *extents;
  76   int extents_len;
  77 };
  78 
  79 
  80 struct dm_layout_g2_zone {
  81   int st;
  82   int cyllow;
  83   int cylhigh;
  84 
  85   int lbnlow;
  86   int lbnhigh;
  87 
  88   dm_angle_t csskew; // cyl switch
  89   dm_angle_t hsskew; // head switch
  90   dm_angle_t zskew;  // skew to beginning of zone ("first block offset")
  91 
  92   dm_angle_t sector_width;
  93 
  94   dm_skew_unit_t skew_units;
  95 };
