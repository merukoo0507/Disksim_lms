  46 // slides the lbn forward while the latter does not.

  48 // optimization: keep track of the points at which the running total
  49 // of the slip/spare list is 0.  Have a fast index of those and 
  50 // start at the highest one <= lbn.

  52 // Or:  keep running total and binsearch.

  62 static int
  63 slipcount_bins(struct dm_layout_g4 *l,
  64                int lbn,
  65                int low,
  66                int high)
  67 {
  68   ddbg_assert(low <= high);
  69   ddbg_assert(0 <= low);
  70   ddbg_assert(high <= l->slips_len);
  71 
  72   if(high == low+1 || high == low) {
  73     return low;
  74   }
  75   else {
  76     int midpt = low + ((high - low) / 2);
  77     if(l->slips[midpt].off > lbn) {
  78       high = midpt;
  79     }
  80     else {  // off <= lbn
  81       low = midpt;
  82     }
  83     return slipcount_bins(l,lbn,low,high);
  84   }
  85 }

  87 static int 
  88 slipcount(struct dm_layout_g4 *l,
  89           int lbn)
  90 {
  91   int i;
  92   int ct;
  93   int l2 = lbn;
  94 
  95   // the upper limit of l->slips_len is not in error
  96   i = slipcount_bins(l, lbn, 0, l->slips_len);
  97   ddbg_assert(0 <= i);
  98   ddbg_assert(i < l->slips_len);
  99 
 100   ct = l->slips[i].count;
 101 
 102   while(i < l->slips_len && l->slips[i].off <= l2) {
 103     ct = l->slips[i].count;
 104     l2 = lbn + ct;
 105     i++;
 106   }
 107 
 108   return ct;
 109 }

 111 static int 
 112 slipcount_rev(struct dm_layout_g4 *l,
 113               int lbn,
 114               int *slip_len) 
 115 {
 116   int i = 0;
 117 
 118   // the upper limit of l->slips_len is not in error
 119   i = slipcount_bins(l, lbn, 0, l->slips_len); 
 120   ddbg_assert(0 <= i);
 121   ddbg_assert(i < l->slips_len);
 122 
 123   if(l->slips[i].off <= lbn 
 124      && (i > 0 
 125          && lbn < l->slips[i].off 
 126          + (l->slips[i].count - l->slips[i-1].count))) 
 127   {
 128     if(slip_len) {
 129       ddbg_assert(0);      // XXX fixme
 130     }
 131 
 132     return -1;
 133   }
 134   else {
 135     return l->slips[i].count;
 136   }

 140 struct remap *
 141 remap_lbn(struct dm_layout_g4 *l,
 142           int lbn)
 143 {
 144   int i;
 145   struct remap *r, *result = 0;
 146 
 147   for(i = 0, r = &l->remaps[i]; i < l->remaps_len; i++, r++) {
 148     if(r->off <= lbn && lbn < (r->off + r->count)) {
 149       result = r;
 150       break;
 151     }
 152   }
 153 
 154   return result;
 155 }

 157 // is p the destination of a remap?
 158 struct remap *
 159 remap_pbn(struct dm_layout_g4 *l,
 160           struct dm_pbn *p)
 161 {
 162   int i;
 163   struct remap *r, *result = 0;
 164   for(i = 0, r = &l->remaps[i]; i < l->remaps_len; i++, r++) {
 165     if(r->dest.cyl == p->cyl 
 166        && r->dest.head == p->head
 167        && r->dest.sector <= p->sector 
 168        && p->sector < (r->dest.sector + r->count))
 169     {
 170       result = r;
 171       break;
 172     }
 173   }
 174 
 175   return result;
 176 }

 205 // For ptol, have to keep track of the size of the enclosing thing so
 206 // we don't map past the end of it.
 207 struct g4_path *
 208 g4_r(struct dm_layout_g4 *l, 
 209      union g4_node *n,
 210      g4_node_t t,
 211      int *lbn,
 212      int max,
 213      struct dm_pbn *p,
 214      struct g4_path *acc) // accumlator
 215 {
 216   int i;
 217   union g4_node *nn; // where we're going to recurse
 218   g4_node_t tt;
 219 
 220   int quot = 0;
 221   int resid = 1;
 222 
 223   ddbg_assert(lbn || p);
 224 
 225   switch(t) {
 226   case TRACK:
 227     if(lbn) {
 228       if(n->t->low <= *lbn && *lbn <= n->t->high) {
 229 
 230         // ?
 231         resid = *lbn - n->t->low;
 232 
 233         nn = 0;
 234         goto out;
 235       }
 236       else {
 237         goto out_err;
 238       }
 239     }
 240     else {
 241       if(n->t->low <= p->sector && p->sector <= n->t->high) {
 242         if(p->sector > max) {
 243           goto out_err;
 244         }
 245 
 246         // quot/resid?
 247         resid = p->sector - n->t->low;
 248          
 249         nn = 0;
 250         goto out;
 251       }
 252       else {
 253         goto out_err;
 254       }
 255     }
 256     break;
 257 
 258   case IDX:
 259   {
 260     struct idx_ent *e;
 261     for(i = 0, e = &n->i->ents[0]; i < n->i->ents_len; i++, e++) 
 262     {
 263       if(lbn) {
 264         if(e->lbn <= *lbn && *lbn < (e->lbn + e->runlen)) {
 265           *lbn -= e->lbn;
 266 
 267           quot = *lbn / e->len;
 268           // XXX
 269           quot = e->cylrunlen < 0 ? -quot : quot;
 270           resid = *lbn % e->len;
 271 
 272           nn = &e->child;
 273           tt = e->childtype;
 274 
 275           *lbn = resid;
 276           goto out;
 277         }       
 278       }
 279       else {
 280         int low = e->cylrunlen > 0 ? e->cyl : e->cyl + e->cylrunlen + 1;
 281         int high = e->cylrunlen < 0 ? e->cyl : e->cyl + e->cylrunlen - 1;
 282         if(low <= p->cyl && p->cyl <= high
 283            && (e->childtype != TRACK || p->head == e->head))
 284 
 285         { 
 286           p->cyl -= e->cyl;
 287 
 288           quot = p->cyl / e->cyllen;
 289           resid = p->cyl % e->cyllen;
 290 
 291           max = min(max, e->lbn + e->runlen - 1);
 292           max -= e->lbn;
 293 
 294           // cut down again for RLE
 295           quot = p->cyl / e->cyllen;
 296           // XXX
 297           quot = e->cylrunlen < 0 ? -quot : quot;
 298           max = min(max, (quot+1) * e->len - 1);
 299           max -= quot * e->len;
 300 
 301           resid = p->cyl % e->cyllen;
 302           
 303           p->cyl = resid;
 304          
 305           nn = &e->child;
 306           tt = e->childtype;
 307           goto out;
 308         }
 309       }
 310     }
 311     goto out_err;
 312   } break;
 313   default:
 314     ddbg_assert(0);
 315     break;
 316   }
 317 
 318  out:
 319   g4_path_append(acc, *n, t, i, quot, resid);
 320   if(nn) {
 321     return g4_r(l, nn, tt, lbn, max, p, acc);
 322   }
 323   else {
 324     return acc;
 325   }
 326 
 327  out_err:
 328   free(acc);
 329   return 0;
 330 }

 332 struct g4_path *
 333 g4_recurse(struct dm_layout_g4 *l, int *lbn, struct dm_pbn *p) 
 334 {
 335   int lbncopy;
 336   struct dm_pbn pbncopy;
 337   struct g4_path *acc = calloc(1, sizeof(*acc));
 338   union g4_node n;
 339   int max;
 340 
 341   if(lbn) {
 342     lbncopy = *lbn;
 343     lbn = &lbncopy;
 344   }
 345 
 346   if(p) {
 347     pbncopy = *p;
 348     p = &pbncopy;
 349   }
 350   
 351   n.i = l->root;
 352   g4_path_append(acc, n, IDX, 
 353                  0,  // i
 354                  0,  // quot
 355                  1); // resid
 356 
 357   max = l->parent->dm_sectors-1;
 358   max += slipcount_rev(l, max, 0);
 359   return g4_r(l, &n, IDX, lbn, max, p, acc);
 360 }

 363 dm_ptol_result_t
 364 ltop(struct dm_disk_if *d,
 365      int lbn, 
 366      dm_layout_maptype junk,
 367      struct dm_pbn *result,
 368      int *remapsector)
 369 {
 370   struct dm_layout_g4 *l = (struct dm_layout_g4 *)d->layout;
 371   struct remap *r;
 372 
 373   if(lbn < 0 || d->dm_sectors <= lbn) {
 374     return DM_NX;
 375   }
 376 
 377   if((r = remap_lbn(l, lbn))) {
 378     *result = r->dest;
 379     result->sector += (lbn - r->off);
 380     return DM_OK;
 381     // not always -- spare remaps
 382     // return DM_REMAPPED;
 383   }
 384   else {
 385     int i;
 386     struct g4_path *p;
 387     struct g4_path_node *n;
 388 
 389     // fiddle lbn according to slips/spares
 390     lbn += slipcount(l, lbn);
 391     
 392     p = g4_recurse(l, &lbn, 0);
 393     if(!p) {
 394       return DM_NX;
 395     }
 396 
 397     for(i = 0; i < p->length; ++i) {
 398       n = &p->path[i];
 399       switch(n->type) {
 400       case TRACK:
 401         result->sector = n->n.t->low + lbn;
 402         break;
 403         
 404       case IDX:
 405       {
 406         struct idx_ent *e = &n->n.i->ents[n->i];
 407         lbn -= e->lbn;
 408 
 409         if(i > 0) {
 410           result->cyl += e->cyl + (n->quot * e->cyllen);
 411         }
 412         else {
 413           result->cyl = e->cyl + (n->quot * e->cyllen);
 414         }
 415 
 416         if(e->childtype == TRACK) {
 417           result->head = e->head;
 418         }
 419 
 420         lbn = n->resid;
 421       } break;
 422 
 423       default: ddbg_assert(0); break;
 424       }
 425     }
 426     free(p);
 427   }
 428 
 429   return DM_OK;
 430 }

 433 dm_ptol_result_t
 434 ptol(struct dm_disk_if *d,
 435      struct dm_pbn *pbn,
 436      int *remapsector)
 437 {
 438   int result = 0;
 439   struct dm_pbn pbncopy = *pbn; // we modify this
 440   struct remap *r;
 441 
 442   struct dm_layout_g4 *l = (struct dm_layout_g4 *)d->layout;
 443 
 444   pbn = &pbncopy;
 445 
 446   if((r = remap_pbn(l, pbn))) {
 447     return r->off + pbn->sector - r->dest.sector;
 448   }
 449   else {
 450     int rv = 0;
 451     int i;
 452     struct g4_path *path;
 453     struct g4_path_node *n;
 454 
 455     path = g4_recurse(l, 0, pbn);
 456 
 457     if(!path) {
 458       return DM_NX;
 459     }
 460 
 461     for(i = 0; i < path->length; ++i) {
 462       n = &path->path[i];
 463       switch(n->type) {
 464       case TRACK:
 465         result += (pbn->sector - n->n.t->low);
 466         break;
 467       case IDX:
 468         {
 469           struct idx_ent *e = &n->n.i->ents[n->i];
 470 
 471           pbn->cyl -= e->cyl;
 472         
 473           if(i > 0) {
 474             result += e->lbn + (n->quot * e->len);
 475           }
 476           else {
 477             result = e->lbn + (n->quot * e->len);
 478           }
 479         
 480           pbn->cyl = n->resid;
 481           
 482         } break;
 483 
 484       default: ddbg_assert(0); break;
 485       }
 486     }
 487 
 488     // if result coincides with an entry in the slip list...
 489 
 490     rv = slipcount_rev(l, result, NULL); // &slipct
 491     if(rv == -1) {
 492       result = DM_NX;
 493     }
 494     else {
 495       result -= rv;
 496     }
 497 
 498     free(path);
 499   }
 500 
 501   return result;
 502 }

 571                struct dm_pbn *pbn,
 572                int *l0,

 574                int *remapsector)

 577   struct remap *r;
 578   struct dm_pbn pi;
 579   struct dm_layout_g4 *l = (struct dm_layout_g4 *)d->layout;
 580 
 581   int spt = g4_spt_pbn(d, pbn);
 582 
 583   /* Dushyanth: must set l0 and ln even if ptol() returns an error */
 584   lbn = ptol(d, pbn, 0);
 585   //if(lbn < 0) {
 586   //  return lbn;
 587   //}
 588 
 589   if(l0) {
 590     pi = *pbn;
 591     pi.sector = 0;
 592 
 593     while((*l0 = ptol(d, &pi, 0)) < 0) {
 594       pi.sector++;
 595     }
 596 
 597     ddbg_assert(*l0 < d->dm_sectors);
 598   }
 599   
 600   if(ln) {
 601     pi = *pbn;
 602     pi.sector = spt-1;
 603     
 604     while((*ln = ptol(d, &pi, 0)) < 0) {
 605       pi.sector--;
 606       ddbg_assert(pi.sector >= 0);
 607     }
 608 
 609     ddbg_assert(*ln < d->dm_sectors);
 610   }
 611 
 612   if (l0 && ln) {
 613     ddbg_assert(*l0 <= *ln);
 614   }
 615 
 616   /* Dushyanth: check error here; see comment above about setting l0, ln */
 617   if(lbn < 0) {
 618     return lbn;
 619   }
 620 
 621   return DM_OK;
 622 }

 659 // Compute the starting offset of a pbn relative to 0. 
 660 dm_angle_t
 661 g4_skew(struct dm_disk_if *d,
 662         struct dm_pbn *pbn)
 663 {
 664   struct dm_layout_g4 *l = (struct dm_layout_g4 *)d->layout;
 665   struct g4_path *p;
 666   struct g4_path_node *n;
 667   int i;
 668   dm_angle_t result;
 669 
 670   p = g4_recurse(l, 0, pbn);
 671   if(!p) { 
 672     // DM_NX
 673     return 0; 
 674   }
 675 
 676   for(i = 0; i < p->length; ++i) {
 677     n = &p->path[i];
 678     switch(n->type) {
 679     case TRACK:
 680       result += n->resid * n->n.t->sw;
 681       break;
 682 
 683     case IDX:
 684     {
 685       struct idx_ent *e = &n->n.i->ents[n->i];
 686 
 687       if(i > 0) {
 688         result += e->off;
 689       }
 690       else {
 691         result = e->off;
 692       }
 693 
 694       result += n->quot * e->alen;
 695 
 696     } break;
 697       
 698     default: ddbg_assert(0); break;
 699     }
 700   }
 701 
 702   free(p);
 703   return result;
 704 }

 752 int
 753 g4_get_zone(struct dm_disk_if *d, int n, struct dm_layout_zone *z)
 754 {
 755   struct dm_layout_g4 *l = (struct dm_layout_g4 *)d->layout;
 756   struct idx_ent *e;
 757 
 758 
 759   if(n >= l->root->ents_len) {
 760     return -1;
 761   }
 762 
 763   e = &l->root->ents[n];
