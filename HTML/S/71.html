 104 // Disk sparing schemes (for replacing bad media)
 105 typedef enum {
 106   NO_SPARING                           = 0,
 107   TRACK_SPARING                        = 1,
 108   SECTPERCYL_SPARING                   = 2,
 109   SECTPERTRACK_SPARING                 = 3,
 110   SECTPERCYL_SPARING_SLIPTOEND         = 4,
 111   SECTPERCYL_SPARING_ATFRONT           = 5,
 112   SECTPERCYL_SPARING_ATFRONT_SLIPTOEND = 6,
 113   SECTATEND_SPARING                    = 7,
 114   SECTPERRANGE_SPARING                 = 8,
 115   SECTSPERZONE_SPARING                 = 9,
 116   SECTSPERZONE_SPARING_SLIPTOEND       = 10
 117 } dm_layout_g1_spare_t;
 118 #define MAXSPARESCHEME SECTSPERZONE_SPARING_SLIPTOEND
 119 
 120 // layout types
 121 typedef enum {
 122   LAYOUT_NORMAL           = 0,
 123   LAYOUT_CYLSWITCHONSURF1 = 1,  /* cyl. switches stay on same surface  
 124                                  * -- first cyl of each zone is normal 
 125                                  */
 126 
 127   LAYOUT_CYLSWITCHONSURF2 = 2   /* cyl. switches stay on same surface  
 128                                  * -- all even-#d cyls are normal      
 129                                  */
 130 } dm_layout_g1_map_t;
 131 
 132 #define LAYOUT_MAX LAYOUT_CYLSWITCHONSURF2
 133

 136 // stupid macros
 137 #define isspareatfront(l) \
 138                 (((l)->sparescheme == SECTPERCYL_SPARING_ATFRONT) || \
 139                  ((l)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND))
 140 
 141 #define issliptoend(l) \
 142                 (((l)->sparescheme == SECTPERCYL_SPARING_SLIPTOEND) || \
 143                  ((l)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND) || \
 144                  ((l)->sparescheme == SECTATEND_SPARING))
 145 
 146 #define issectpercyl(l) \
 147                 (((l)->sparescheme == SECTPERCYL_SPARING) || \
 148                  ((l)->sparescheme == SECTPERCYL_SPARING_SLIPTOEND) || \
 149                  ((l)->sparescheme == SECTPERCYL_SPARING_ATFRONT) || \
 150                  ((l)->sparescheme == SECTPERCYL_SPARING_ATFRONT_SLIPTOEND))

 157 struct dm_layout_g1 {
 158   struct dm_layout_if hdr;
 159   struct dm_disk_if *disk; // back pointer
 160 
 161   dm_layout_g1_spare_t  sparescheme;
 162   int                   rangesize;
 163   dm_layout_g1_map_t    mapping;
 164 
 165 
 166   struct dm_layout_g1_band *bands;
 167   int bands_len;
 168   int         *band_blknos;    // first lbn per band indexed by band
 169   dm_skew_unit_t skew_units;
 170 };
 171 
 172 
 173 
 174 struct dm_layout_g1_band {
 175   int    startcyl;
 176   int    endcyl;
 177   int    blkspertrack;
 178   int    deadspace;
 179 
 180   dm_angle_t firstblkangle;
 181   dm_angle_t cylskew;
 182   dm_angle_t trackskew;
 183 
 184   int    blksinband;
 185   int    sparecnt;
 186   int    numslips;
 187   int    numdefects;
 188   int   *slip;
 189   int   *defect;
 190   int   *remap;
 191   int   num; /* number of this band */
 192   /*     int  *remap; */
 193 
 194   dm_angle_t sector_width;
 195 
 196   // dead space at the end of each track
 197   // blkspertrack * sector_width + trkspace should equal the number of 
 198   // ticks in a circle
 199   dm_angle_t trkspace;  
 200 
 201 
 202   dm_skew_unit_t skew_units;
 203 };
