 112 static dm_time_t 
 113 dm_seek_time_g1(struct dm_disk_if *d, 
 114                 struct dm_mech_state *start_track,
 115                 struct dm_mech_state *end_track,
 116                 int rw)
 117 {
 118   struct dm_mech_g1 *m = (struct dm_mech_g1 *)d->mech;
 119   dm_time_t seektime, hst, result = 0;
 120 
 121   seektime = m->seekfn(d,start_track,end_track,rw);
 122 
 123   //  seektime -= 700 * DM_TIME_USEC;
 124 
 125   hst = m->hdr.dm_headswitch_time(d,start_track->head,end_track->head);
 126 
 127   result = hst > seektime ? hst : seektime;
 128 
 129   if((result != 0) && (rw == 0)) {
 130     result += m->seekwritedelta;
 131   }
 132 
 133   return result;
 134 }

 136 // computes how long to reposition from the first head position to
 137 // the second
 138 // does not change the state of the disk
 139 static dm_time_t 
 140 dm_pos_time_g1(struct dm_disk_if *d,
 141                struct dm_mech_state *initial,
 142                struct dm_pbn *start,
 143                int len,
 144                int rw,
 145                int immed,
 146                struct dm_mech_acctimes *breakdown)
 147 {
 148   dm_angle_t seekrotate;
 149   struct dm_mech_state end_track, state = *initial;
 150   dm_time_t seektime = 0;
 151   dm_time_t result = 0;
 152   dm_time_t latency = 0;
 153   //  struct dm_mech_g1 *m = (struct dm_mech_g1 *)d->mech;
 154   
 155   end_track.head = start->head; end_track.cyl = start->cyl;
 156   seektime = d->mech->dm_seek_time(d, &state, &end_track, rw);
 157 
 158   state.head = start->head;
 159   state.cyl = start->cyl;
 160 
 161  
 162   // advance the angle by how far we rotated during the seek
 163   seekrotate = d->mech->dm_rotate(d, &seektime);
 164   state.theta += seekrotate;
 165 
 166   // initial rotational latency given our new position
 167   latency = d->mech->dm_latency(d, 
 168                                 &state, 
 169                                 start->sector, 
 170                                 len, 
 171                                 immed,  
 172                                 0);
 173 
 174   result = seektime + latency;
 175 
 176   if(breakdown) {
 177     breakdown->seektime = seektime;
 178     breakdown->initial_latency = latency;
 179     breakdown->initial_xfer = 0;
 180     breakdown->addl_latency = 0;
 181     breakdown->addl_xfer = 0;
 182   }
 183 
 184   return result;
 185 }

 188 // computes the access time for an access time on a single track
 189 static dm_time_t 
 190 dm_acctime_track(struct dm_disk_if *d, 
 191               struct dm_mech_state *istate,
 192               struct dm_pbn *start,
 193               int len,
 194               int rw,
 195               int immed,
 196               struct dm_mech_state *result_state,
 197               struct dm_mech_acctimes *breakdown)
 198 {
 199   //  struct dm_mech_state start_track;
 200   struct dm_mech_state temp;
 201   //  struct dm_pbn seekdest;
 202   struct dm_mech_state state = *istate;
 203   dm_time_t seektime, latency, xfertime, initxfer;
 204   dm_time_t addtolatency = 0, result = 0, *atlp;
 205 
 206   // start by seeking/headswitching/writesettling 
 207   temp.cyl = start->cyl; temp.head = start->head;
 208   seektime = d->mech->dm_seek_time(d, &state, &temp, rw);
 209 
 210   // we're on track now
 211   state.head = start->head;
 212   state.cyl = start->cyl;
 213 
 214   // advance the angle by how far we rotated during the seek
 215   temp.theta = d->mech->dm_rotate(d, &seektime);
 216   state.theta += temp.theta;
 217 
 218   // initial rotational latency given our new position
 219   atlp = immed ? &addtolatency : 0;
 220   latency = d->mech->dm_latency(d, 
 221                                 &state, 
 222                                 start->sector, 
 223                                 len, 
 224                                 immed,  
 225                                 atlp);
 226 
 227   // if zero-latency access, add intermediate rotational latency
 228   latency += immed ? *atlp : 0;
 229 
 230   // data transfer
 231   xfertime = d->mech->dm_xfertime(d, &state, len);
 232 
 233   result = seektime + latency + xfertime;
 234 
 235   if(result_state) {
 236     result_state->head = start->head;
 237     result_state->cyl = start->cyl;
 238     result_state->theta = istate->theta;
 239     result_state->theta += d->mech->dm_rotate(d, &result);
 240   }
 241 
 242 
 243   // little gross; need to break down the xfertime for zero-latency access
 244   if(immed) {
 245     int firstsect = d->mech->dm_access_block(d, 
 246                                              &state, 
 247                                              start->sector, 
 248                                              len, 
 249                                              immed);
 250 
 251     initxfer = d->mech->dm_xfertime(d, &state, len - firstsect + 1);
 252   }
 253   else {
 254     initxfer = xfertime;
 255   }
 256 
 257   if(breakdown) {
 258     breakdown->seektime = seektime;
 259     breakdown->initial_latency = latency - addtolatency;
 260     breakdown->initial_xfer = initxfer;
 261     breakdown->addl_latency = addtolatency;
 262     breakdown->addl_xfer = xfertime - initxfer;
 263   }
 264 
 265 
 266   return result;
 267 }

 269 // Note that this is an ESTIMATE and if there are defects,
 270 // the result here will be SMALLER than the correct value.
 271 // This works for accesses spanning multiple tracks.  In this case,
 272 // the access time breakdown is as follows: single track
 273 // accesses are normal.  Multi-track accesses
 274 // report the initial seek and rotation as the positioning time
 275 // before the first transfer begins.  All additional xfertime 
 276 // and postime is reported as initial transfer time.
 277 static dm_time_t 
 278 dm_acctime_g1(struct dm_disk_if *d, 
 279               struct dm_mech_state *istate,
 280               struct dm_pbn *start,
 281               int len,
 282               int rw,
 283               int immed,
 284               struct dm_mech_state *result_state,
 285               struct dm_mech_acctimes *breakdown)
 286 {
 287   struct dm_pbn pbn = *start;
 288   int lbnhigh; // highest lbn on the current track
 289 
 290   int l;       // first lbn to read
 291   int lastlbn; // last lbn to read
 292 
 293   struct dm_mech_state s = *istate;
 294   struct dm_mech_state s2;
 295 
 296   dm_time_t result = 0;
 297 
 298   if(breakdown) {
 299     breakdown->seektime = 0;
 300     breakdown->initial_latency = 0;
 301     breakdown->initial_xfer = 0;
 302     breakdown->addl_latency = 0;
 303     breakdown->addl_xfer = 0;
 304   }
 305 
 306   // back translate the pbn
 307   l = d->layout->dm_translate_ptol(d, &pbn, 0);
 308   // last lbn we're going to read
 309   lastlbn = l+len-1;
 310 
 311   do {
 312     struct dm_mech_acctimes ibreak; // intermediate
 313 
 314     //    l = d->layout->dm_translate_ptol(d, &pbn, 0);
 315 
 316     d->layout->dm_translate_ltop(d, l, MAP_FULL, &pbn, 0);
 317     d->layout->dm_get_track_boundaries(d, 
 318                                        &pbn, 
 319                                        0,        // first lbn on track
 320                                        &lbnhigh, 
 321                                        0);       // remapsector
 322 
 323     ddbg_assert((lbnhigh != DM_SLIPPED) && (lbnhigh != DM_REMAPPED));
 324 
 325     if(lbnhigh > lastlbn) { lbnhigh = lastlbn; }
 326 
 327     // redundant??
 328     //    d->layout->dm_translate_ltop(d, l, MAP_FULL, &pbn, 0);
 329 
 330     // access from l to lbnhigh at each step
 331 
 332     result += dm_acctime_track(d, 
 333                                &s, 
 334                                &pbn, 
 335                                lbnhigh - l + 1, 
 336                                rw,
 337                                immed,
 338                                &s2,
 339                                &ibreak);
 340 
 341     s = s2;
 342     
 343     l = lbnhigh + 1;
 344 
 345     if(breakdown) {
 346       if((pbn.cyl == start->cyl)
 347          && (pbn.head == start->head)
 348          && (pbn.sector == start->sector))
 349       {
 350         breakdown->seektime = ibreak.seektime;
 351         breakdown->initial_latency = ibreak.initial_latency;
 352         breakdown->initial_xfer = ibreak.initial_xfer;
 353         if(lbnhigh == lastlbn) {
 354           // single track access
 355           breakdown->addl_latency = ibreak.addl_latency;
 356           breakdown->addl_xfer = ibreak.addl_xfer;
 357         }
 358         else {
 359           breakdown->initial_xfer += ibreak.addl_latency;
 360           breakdown->initial_xfer += ibreak.addl_xfer;
 361         }
 362       }
 363       else {
 364 /*      breakdown->initial_xfer += ibreak.seektime; */
 365 /*      breakdown->initial_xfer += ibreak.initial_latency; */
 366 /*      breakdown->initial_xfer += ibreak.initial_xfer; */
 367 /*      breakdown->initial_xfer += ibreak.addl_latency; */
 368 /*      breakdown->initial_xfer += ibreak.addl_xfer; */
 369       }
 370     }      
 371       
 372   } while(lastlbn > lbnhigh);
 373 
 374   return result;
 375 }

 378 // compute the rotational latency incurred for the given
 379 // (possibly "zero-latency") access
 380 static dm_time_t 
 381 dm_latency_g1(struct dm_disk_if *d,
 382               struct dm_mech_state *begin,
 383               int start_sect,
 384               int len,
 385               int immed,
 386               dm_time_t *addtolatency)
 387 {
 388   struct dm_pbn tmppbn;
 389   int sectors;
 390   //  dm_angle_t trackstart;
 391   dm_angle_t start, end;
 392   dm_angle_t theta = begin->theta; // rotational position of the disk
 393   dm_time_t result;
 394   //  dm_time_t period = d->mech->dm_period(d);
 395 
 396   tmppbn.head = begin->head;
 397   tmppbn.cyl = begin->cyl;
 398   tmppbn.sector = 0; // XXX sector may make a difference to g4 layout!
 399   sectors = d->layout->dm_get_sectors_pbn(d, &tmppbn);
 400 
 401   ddbg_assert(len <= sectors);
 402 
 403   tmppbn.head = begin->head;
 404   tmppbn.cyl = begin->cyl;
 405   tmppbn.sector = start_sect;
 406   start = d->layout->dm_pbn_skew(d, &tmppbn);
 407 
 408   //  printf("%s() skew %f\n", __func__, dm_angle_itod(start));
 409 
 410   // non-zero-latency
 411   if(!immed) {
 412 
 413     //    printf("mech_g1::latency skew = %f distance = %f\n", 
 414     //     dm_angle_itod(start), dm_angle_itod(start - theta));
 415 
 416     // rottime does the Right Thing if theta > start
 417     result = d->mech->dm_rottime(d, theta, start);
 418       
 419     if(addtolatency) {
 420       *addtolatency = 0;
 421     }
 422     return result;
 423   }
 424   else {
 425     dm_angle_t extra = 0;
 426     tmppbn.sector = (start_sect + len - 1); // % sectors;
 427         
 428     if(tmppbn.sector >= sectors) {
 429       tmppbn.sector = sectors - 1;
 430     }
 431 
 432     //    d->layout->dm_convert_ptoa(d, &tmppbn, &end, 0);
 433     end = d->layout->dm_pbn_skew(d, &tmppbn);
 434     
 435     
 436     // head is outside of the access
 437     if(((theta <= start) && (start < end))
 438        || ( (end < theta) && (theta <= start) )
 439        || ( (start < end) && (end <= theta))
 440        || (tmppbn.sector == start_sect))
 441       
 442       {
 443         extra = 0;
 444       }
 445     // head is in the middle of the access
 446     else {
 447       tmppbn.head = begin->head;
 448       tmppbn.cyl = begin->cyl;
 449       tmppbn.sector = d->mech->dm_access_block(d, begin, start_sect, len, immed);
 450       start = d->layout->dm_pbn_skew(d, &tmppbn);
 451 
 452       if(len != sectors) {
 453         int extrasectors = sectors - len;
 454         struct dm_pbn tmppbn2;
 455 
 456         tmppbn2.head = begin->head;
 457         tmppbn2.cyl = begin->cyl;
 458         tmppbn2.sector = extrasectors;
 459         
 460         extra = d->layout->dm_get_sector_width(d, &tmppbn2, extrasectors);
 461         ddbg_assert(extra != 0);
 462       }
 463       else {
 464         extra = 0;
 465       }
 466     }
 467 
 468     result = d->mech->dm_rottime(d, theta, start);
 469 
 470     if(addtolatency && extra) {
 471       *addtolatency = d->mech->dm_rottime(d, 0, extra);
 472     }
 473 
 474     return result;
 475   }
 476 
 477   // UNREACHED
 478   return 0;
 479 }

 494 // initial->theta may be in unmapped space.  For the time being,
 495 // we assume a whole-track layout and return start_sect in that case.
 496 static int
 497 g1_access_block(struct dm_disk_if *d,
 498                 struct dm_mech_state *initial,
 499                 int start_sect,
 500                 int len,
 501                 int immed)
 502 {
 503   int sectors;
 504   dm_ptol_result_t rv;
 505   dm_angle_t start, end, skew;
 506   struct dm_pbn tmppbn;
 507   dm_angle_t theta = initial->theta;
 508 
 509 
 510   tmppbn.cyl = initial->cyl;
 511   tmppbn.head = initial->head;
 512   tmppbn.sector = start_sect;
 513   sectors = d->layout->dm_get_sectors_pbn(d, &tmppbn);
 514 
 515 
 516   ddbg_assert(len <= sectors);
 517 
 518   skew = d->layout->dm_pbn_skew(d, &tmppbn);
 519   start = skew;
 520 
 521   if(!immed) {
 522     return start_sect;
 523   }
 524   else {
 525     //    dm_angle_t extra = 0;
 526     tmppbn.sector = (start_sect + len - 1); // % sectors;
 527         
 528     if(tmppbn.sector >= sectors) {
 529       tmppbn.sector = sectors - 1;
 530     }
 531 
 532     //    d->layout->dm_convert_ptoa(d, &tmppbn, &end, 0);
 533     end = d->layout->dm_pbn_skew(d, &tmppbn);
 534     
 535     // head is between 0 and the start of the access
 536     if(((theta <= start) && (start < end))
 537        || ( (end < theta) && (theta <= start) )
 538        || ( (start < end) && (end <= theta))
 539        || (tmppbn.sector == start_sect)) // corner case!
 540       {
 541         ddbg_assert(start_sect < sectors);
 542         return start_sect;
 543       }
 544     // head is in the middle of the access
 545     else {
 546       // dm_angle_t skewed;
 547       dm_angle_t track_skew;
 548       struct dm_pbn track_endpbn;
 549       struct dm_mech_state skewed;
 550       struct dm_pbn result_pbn;
 551       dm_angle_t sectWidth = 
 552         d->layout->dm_get_sector_width(d, &tmppbn, 1);
 553       
 554       if(sectWidth == 0) {
 555         goto out_nx;
 556       }
 557 
 558       track_endpbn.cyl = initial->cyl;
 559       track_endpbn.head = initial->head;
 560 
 561       track_endpbn.sector = 0;
 562       track_skew = d->layout->dm_pbn_skew(d, &track_endpbn);
 563 
 564       skewed.cyl = initial->cyl;      
 565       skewed.head = initial->head;
 566       skewed.theta = theta - track_skew;
 567       
 568       if((skewed.theta % sectWidth) != 0) {
 569         skewed.theta += (sectWidth - (skewed.theta % sectWidth));
 570       }
 571 
 572 
 573       rv = d->layout->dm_convert_atop(d, &skewed, &result_pbn);
 574       if(rv < 0) {
 575         goto out_nx;
 576       }
 577 
 578       ddbg_assert(result_pbn.sector < sectors);
 579       return result_pbn.sector;
 580     }
 581   }
 582 
 583 
 584  out_nx:
 585   return start_sect;
 586 
 587 }

 618 static dm_time_t 
 619 dm_rottime_g1(struct dm_disk_if *d,
 620               dm_angle_t begin,
 621               dm_angle_t end)
 622 {
 623   struct dm_mech_g1 *m = (struct dm_mech_g1 *)d->mech;
 624   dm_angle_t diff;
 625   if(end == begin) {
 626     return 0;
 627   }
 628   else if(end > begin) {
 629     diff = end - begin;
 630   }
 631   else {
 632     // Ladies and Gentlemen, please disregard the man behind the curtain.
 633     diff = -(begin - end);
 634   }
 635 
 636   //printf("dm_rottime_g1: rotdistance = %f\n", dm_angle_itod(diff));
 637 
 638 
 639 
 640   // XXX this is immensly evil
 641   return ((((long long)diff << 20) >> DM_ANGLE_EXP) * m->rotatetime) >> 20;
 642 }

 697 // how far will the media rotate in the given amount of time
 698 // only the angle in the result is set
 699 static dm_angle_t
 700 dm_rotate_g1(struct dm_disk_if *d, 
 701              dm_time_t *time)
 702 {
 703   struct dm_mech_g1 *m = (struct dm_mech_g1 *)d->mech;
 704   if(*time == 0) {
 705     return 0;
 706   }
 707   else {
 708     // time / time/tics = tics
 709     //    uint64_t result1
 710     uint64_t result2;
 711 
 712     //    result1 = dm_angle_dtoi(dm_time_itod(*time) / dm_time_itod(m->rotatetime));
 713 
 714     // This is a little tricky; if we do time << DM_ANGLE_T, we may
 715     // run out of bits so we shift by a bit less, then divide and then
 716     // shift some more.  Consequently, this is somewhat lossy.
 717     // These shifts assume that DM_TIME_EXP > DM_ANGLE_EXP and that a
 718     // dm_time_t is 64 bits wide
 719 
 720     result2 =  ((*time << (64 - DM_TIME_EXP)) 
 721                 / m->rotatetime) 
 722                        << (DM_TIME_EXP - DM_ANGLE_EXP);
 723 
 724 
 725     return result2;
 726   }
 727 }

 857

 860   unmarshal_fns((void **)&m->hdr, 
 861                  sizeof(struct dm_mech_if) / sizeof(void *),
 862                  ptr,
 863                  DM_MECH_G1_TYP);
 864 
 865   {
 866     int *seekfnoffset = (int *)ptr + sizeof(struct dm_mech_if);
 867     m->seekfn = unmarshal_fn(seekfnoffset,
 868                               DM_MECH_G1_TYP);
 869   }
 870   ptr += sizeof(struct dm_mech_g1) + sizeof(void*);
 871 
 872   if(m->xseekcnt != 0) {
 873     int distsize = m->xseekcnt * sizeof(int);
 874     int timesize = m->xseekcnt * sizeof(dm_time_t);
 875     
 876     m->xseekdists = malloc(distsize);
 877     m->xseektimes = malloc(timesize);
 878     memcpy((char *)m->xseekdists, ptr, distsize);
 879     ptr += distsize;
 880     memcpy((char *)m->xseektimes, ptr, timesize);
 881     ptr += timesize;
 882   }
 883 
 884 
 885   m->disk = parent;
 886 
 887   *result = m;
 888   return ptr;
