 129 dm_mech_g1_loadparams(struct lp_block *b, int *junk) {
 130   
 131   struct dm_mech_g1 *result = malloc(sizeof(*result));

 143 void 
 144 dm_mech_g1_read_extracted_seek_curve (char *filename, 
 145                                       int *cntptr,
 146                                       int **distsptr, 
 147                                       dm_time_t **timesptr)
 148 {
 149    int rv, mat;
 150    int lineflag = 1;
 151    int count = 0, buflen = 128;
 152    int *dists;
 153    dm_time_t *times;
 154    FILE *seekfile;
 155    char linebuf[1024];
 156 
 157    char *pathname = lp_search_path(lp_cwd, filename);
 158 
 159    if(pathname) {
 160      seekfile = fopen(pathname, "r");
 161    }
 162    else {
 163      ddbg_assert2(0, "Seek file not found in path!");
 164    }
 165 
 166    ddbg_assert3(seekfile != 0, ("fopen seekfile (%s) failed: %s", 
 167                                 filename,
 168                                 strerror(errno)));
 169 
 170 
 171    rv = (fgets(linebuf, sizeof(linebuf), seekfile) != 0);
 172 
 173    mat = sscanf(linebuf, "Seek distances measured: %d\n", &count);
 174    if(mat == 1) {
 175      buflen = count;
 176      lineflag = 0;
 177    }
 178 
 179    dists = calloc(buflen, sizeof(*dists));
 180    times = calloc(buflen, sizeof(*times));
 181 
 182 
 183    do {
 184      double time, stdev;
 185      int dist;
 186    
 187      if(!lineflag) {
 188        rv = (fgets(linebuf, sizeof(linebuf), seekfile) != 0);
 189      }
 190      else {
 191        lineflag = 0;       
 192      }
 193      if(rv) {
 194        mat = sscanf(linebuf, "%d, %lf, %lf\n", &dist, &time, &stdev);
 195        
 196        if(mat == 2 || mat == 3) {
 197          if(count >= buflen-1) {
 198            buflen *= 2;
 199            dists = realloc(dists, buflen * sizeof(*dists));
 200            times = realloc(times, buflen * sizeof(*times));
 201          }
 202        
 203          dists[count] = dist;
 204          times[count] = dm_time_dtoi(time);
 205          count++;
 206        }
 207        else {
 208          fprintf(stderr, "*** bogus line in seek curve (%s:%d): %s\n", 
 209                  __FILE__, __LINE__, linebuf);
 210        }
 211      }
 212    } while(rv);
 213 
 214    fclose(seekfile);
 215    *cntptr = count;
 216    *distsptr = dists;
 217    *timesptr = times;
 218 }

 222 int do_1st10_seeks(struct dm_mech_g1 *result, 
 223                    struct lp_list *l) 
 224 {
 225   if(result->seektime != SEEK_1ST10_PLUS_HPL) {
 226     fprintf(stderr, "*** warning: ignoring First 10 seeks parameter for seek function other than First 10 plus hpl.\n");
 227     return 0;
 228   }
 229 
 230   if(l->values_len < 10) {
 231     fprintf(stderr, "*** error: Want 10 first seek times (got %d)\n", l->values_len);
 232     return -1;
 233   }
 234   else {
 235     int e;
 236     for(e = 0; e < 10; e++) {
 237       if(l->values[e]->t != D) {
 238         fprintf(stderr, "*** error: First 10 seeks must be floats.\n");
 239         return -1;
 240       } else if(l->values[e]->v.d < 0.0) {
 241         fprintf(stderr, "*** error: First 10 seeks must be nonnegative.\n");
 242         return -1;
 243       }
 244       result->first10seeks[e] = dm_time_dtoi(l->values[e]->v.d);
 245      
 246     }
 247   }
 248   return 0;
 249 }

 251 int do_hpl_seek(struct dm_mech_g1 *result, struct lp_list *l) {
 252 
 253   if((result->seektime != SEEK_HPL)
 254      && (result->seektime != SEEK_1ST10_PLUS_HPL)) 
 255     {
 256       fprintf(stderr, 
 257               "*** warning: ignoring hpl parameters for non-hpl seek type.\n");
 258       return 0;
 259     }
 260 
 261   if(l->values_len < 6) {
 262     fprintf(stderr, "*** error: Want 6 HPL seek equation parameters (got %d)\n", l->values_len);
 263     return -1;
 264   }
 265   else {
 266     int e;
 267 
 268     // first one is in cyls, rest are in milliseconds
 269     if(l->values[0]->t != I) {
 270       fprintf(stderr, "*** error: HPL v1 is in cylinders.\n");
 271       return -1;
 272     } 
 273     else if(l->values[0]->v.i < 0) {
 274       fprintf(stderr, "*** error: HPL v1 must be nonnegative.\n");
 275       return -1;
 276     }
 277     result->hpseek_v1 = l->values[0]->v.i;
 278 
 279     for(e = 1; e < 6; e++) {
 280       if(l->values[e]->t != D) {
 281         fprintf(stderr, "*** error: HPL seek eqn. values v2..v6 must  be floats.\n");
 282         return -1;
 283       } 
 284       else if((l->values[e]->v.d < 0.0) && (e != 5)) {
 285         fprintf(stderr, "*** error: HPL seek eqn. values v2..v5 must be nonnegative.\n");
 286         return -1;
 287       }
 288 
 289       if((l->values[e]->v.d == -1.0) && (e == 5)) {
 290         result->hpseek[e] = -1;
 291       }
 292       else {
 293         result->hpseek[e] = dm_time_dtoi(l->values[e]->v.d);
 294       }
 295     }
 296   }
 297 
 298   if(result->hpseek[5] != -1) {
 299     result->seekone = result->hpseek[5];
 300   }
 301 
 302   return 0;
 303 }
