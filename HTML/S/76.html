  43 void layout_test_simple(struct dm_disk_if *d) {
  44   int c, lbn, count = 0, runlbn = 0;
  45   struct dm_pbn pbn, trkpbn = {0,0,0};
  46   printf("got a dm_disk with %d sectors!\n", d->dm_sectors);
  47 
  48   for(c = 0; c < d->dm_sectors; c ++) {
  49     int lbn2;
  50     dm_angle_t skew, zerol;
  51     struct dm_mech_state track;
  52 
  53     d->layout->dm_translate_ltop(d, c, MAP_FULL, &pbn, 0);
  54     lbn2 = d->layout->dm_translate_ptol(d, &pbn, 0);
  55 
  56     if(c != lbn2) {
  57       printf("*** test_0_t: %8d -> (%8d, %3d, %4d) -> %8d\n", 
  58              c, pbn.cyl, pbn.head, pbn.sector, lbn2);
  59     }
  60     else {
  61       if((lbn > 0)
  62          && (trkpbn.head == pbn.head)
  63          && (trkpbn.cyl == pbn.cyl)
  64          && ((trkpbn.sector+count) == pbn.sector)) { count++;  }
  65       else {
  66         printf("l %d %d %d %d %d\n", runlbn, trkpbn.cyl, trkpbn.head, trkpbn.sector, count);
  67         runlbn = c;
  68         trkpbn = pbn;
  69         count = 1;
  70       }
  71 
  72 
  73     }    
  74   }
  75 }

 356 void mech_test_access_immed(struct dm_disk_if *d) {
 357   // zero latency access; read half the track starting with the head over
 358   // the middle of the access
 359   int c,h;
 360   int sectors;
 361   struct dm_mech_state m = {0,0,0}, tempm, resultstate = {0,0,0};
 362   dm_time_t result;
 363   struct dm_pbn startpbn, tmppbn;
 364   double time;
 365   int tracks, lbn = 0;
 366 
 367 
 368   //  tracks = d->dm_cyls * d->dm_surfaces;
 369   tracks = 10 * d->dm_surfaces;
 370 
 371   for(c = 0; c < tracks; c++) {
 372     int len;
 373     sectors = d->layout->dm_get_sectors_lbn(d, c);
 374     lbn = c * sectors;
 375     len = sectors / 2;
 376 
 377     d->layout->dm_translate_ltop(d, lbn, MAP_FULL, &startpbn, 0);
 378     
 379     m.head = startpbn.head;
 380     m.cyl = startpbn.cyl;
 381     
 382     m.theta = d->layout->dm_pbn_skew(d, &startpbn);
 383     m.theta += dm_angle_dtoi((double)(len / 2) / (double)sectors);
 384 
 385     
 386 
 387     result = d->mech->dm_acctime(d,         // disk
 388                                  &m,        // start state
 389                                  &startpbn, // first pbn to read
 390                                  len,      
 391                                  1,         // rw
 392                                  1,         // immed
 393                                  &resultstate,        // result state
 394                                  0);
 395     time = dm_time_itod(result);
 396 
 397     printf("access_immed (%d, %d) -- %f\n\n\n",
 398            m.head, m.cyl, time);
 399 
 400 
 401   }
 402 }
