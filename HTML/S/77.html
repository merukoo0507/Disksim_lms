 126 // What is the angle relative to?
 127 
 128 // There are three reasonable "zeroes" to speak of.  The first which
 129 // we'll call "literal" or "absolute" zero or just 0 is a fixed point
 130 // that is the same on every track on the disk.  The second is the
 131 // offset of the leading edge of the first sector lying entirely past
 132 // 0 on the track.  This will typically be the same for all of the
 133 // tracks on a given surface within a zone.  We refer to this as 0t
 134 // for "track."  (in the current implementation, 0t is probably the
 135 // same as 0).  The third zero is the offset of the leading edge of
 136 // the block whose lbn is numerically lowest of all of the blocks on
 137 // the track.  This will also probably be the same for all tracks per
 138 // surface,zone.  We refer to this as 0l for lbn or logical.
 139 
 140 // 0t is not affected by skews but is affected by 
 141 // slipping in some defect-management schemes.
 142 
 143 // 0l is affected by: track and cylinder switch skew as well as
 144 // slipping in some defect-management schemes.
 145 
 146 // We try to be precise about which 0 we mean in describing the
 147 // interface below.
 148 
 149 // Further complicating this is how to number the sectors on a given
 150 // track in pbns.  It would be reasonable to call either the one at 0t
 151 // sector zero or the one at 0l sector zero.
 152

 312 // Anatomy of a mech-model access and related terminology
 313 // (bucy 5/2002)
 314 
 315 // +-------------------------+------------+----------+---------+----------+
 316 // | seek                    | initial    |          | add.    |          |
 317 // | headswitch              | rotational | xfertime | rot.    | xfertime |
 318 // |            extra settle | latency    |          | latency |          |
 319 // +-------------------------+------------+----------+---------+----------+
 320 
 321 // |---------seektime--------|
 322 // |-----------positioning-time-----------|
 323 // |------------------------------access-time-----------------------------|
 324 
 325 //                                                             ^
 326 //                                                             |
 327 //                                                  bus transfer to host
 328 //                                                  begins now
 329 
 330 // What disksim calls "SERVTIME" is really access time minus xfertime;
 331 // we've agreed to call this NOXFERTIME in the future to reduce confusion.
 332 
 333 // For zero-latency accesses, disksim's disklatency() function returns
 334 // the initial rotational latency in this nomenclature; disksim's 
 335 // "addtolatency" is our additional rotational latency.
 336 
 337 struct dm_mech_acctimes {
 338    dm_time_t seektime;
 339    dm_time_t initial_latency;
 340    dm_time_t initial_xfer;
 341    dm_time_t addl_latency;
 342    dm_time_t addl_xfer;
 343 };
 344 
 345 struct dm_mech_if {
 346 
 347   // how long to seek from the first to the second track, possibly
 348   // including a head switch and additional write settling time.  This
 349   // is only track-to-track so the angles in the parameters are
 350   // ignored.
 351   dm_time_t(*dm_seek_time)(struct dm_disk_if *, 
 352                            struct dm_mech_state *start_track,
 353                            struct dm_mech_state *end_track,
 354                            int rw);
 355 
 356 
 357   // this function subsumes the functionality of the "trackstart"
 358   // global in disksim.  From the given inital condition and access,
 359   // it will return the first block on the track to be read.
 360   int(*dm_access_block)(struct dm_disk_if *,
 361                         struct dm_mech_state *initial,
 362                         int start,
 363                         int len,
 364                         int immed);
 365 
 366   // There are two rotational latency functions.  Read carefully to
 367   // make sure you're using the one you want.
 368 
 369 
 370   // This function implements the "disksim semantic: read *up to*
 371   // <len> blocks from the track starting from angle <initial> and
 372   // sector <start>.  This will access to the end of the track *but
 373   // not wrap around* e.g. for a sequential access that starts on the
 374   // given track and switches to another, after reaching the end of
 375   // the first.  The return value is the amount of time before the
 376   // media transfer begins.  Any additional rotational latency (as
 377   // defined above) will be filled into addtolatency if nonzero.
 378   // Initial angle is relative to 0!
 379   dm_time_t(*dm_latency)(struct dm_disk_if *,
 380                          struct dm_mech_state *initial, // initial state
 381                          int start,                     // start sector 
 382                          int len,                       // number of sectors
 383                          int immed,                     // zero-latency?
 384                          dm_time_t *addtolatency);
 385 
 386   // UNIMPLEMENTED
 387   // Compute the rotational latency incurred for the given (possibly
 388   // "zero-latency") access.  The access is of the form: starting from
 389   // the initial angle <initial>, read len blocks from the track
 390   // starting with start, wrapping around zero if the access is that
 391   // long.  The return value is only rotational latency before the
 392   // data transfer begins.  In the case of zero-latency accesses,
 393   // there may be an additional rotational latency.  This additional
 394   // amount is filled into the last parameter if nonzero.
 395   dm_time_t(*dm_latency_seq)(struct dm_disk_if *,
 396                              struct dm_mech_state *initial, // initial state
 397                              int start,                     // start sector 
 398                              int len,                    // number of sectors
 399                              int immed,                     // zero-latency?
 400                              dm_time_t *addtolatency);
 401 
 402 
 403   // Seek, any extra settling for write, initial rotational latency.
 404   // i.e. the amount of time before media transfer starts.
 405   // len must be >= 1
 406   dm_time_t(*dm_pos_time)(struct dm_disk_if *,
 407                           struct dm_mech_state *initial,
 408                           struct dm_pbn *start,
 409                           int len,
 410                           int rw,
 411                           int immed,
 412                           struct dm_mech_acctimes *breakdown);
 413 
 414   // access time is postime, xfertime and any additional rotational 
 415   // latency not included in postime, e.g. in the middle of a 
 416   // zero-latency access xfer.
 417   // if the last parameter is nonzero, it will be filled
 418   // in with the state of the disk as of the reported result time;
 419   // i.e. where the disk will be when the access completes.
 420 
 421   // Consider changing this to an lbn since most occurances in 
 422   // disksim seem to do ltop and pass the result here.
 423   // Simplifies g1 "multi-track" wrapper.
 424   // len must be >= 1
 425   dm_time_t(*dm_acctime)(struct dm_disk_if *, 
 426                          struct dm_mech_state *initial_state,
 427                          struct dm_pbn *start,
 428                          int len,
 429                          int rw,
 430                          int immed,
 431                          struct dm_mech_state *result_state,
 432                          struct dm_mech_acctimes *breakdown);
 433 
 434   // compute how long it will take the disk to rotate from the angle
 435   // in the first position to that in the second position
 436   dm_time_t(*dm_rottime)(struct dm_disk_if *,
 437                          dm_angle_t begin,
 438                          dm_angle_t end);
 439   
 440 
 441   // Amount of time to read len sectors from the track designated by
 442   // the 2nd argument.  This is a convience wrapper for 
 443   // rottime.
 444   dm_time_t(*dm_xfertime)(struct dm_disk_if *d,
 445                           struct dm_mech_state *,
 446                           int len);
 447 
 448 
 449   // This is probably constant for now but might not be in the future.
 450   dm_time_t(*dm_headswitch_time)(struct dm_disk_if *, 
 451                                  int h1, 
 452                                  int h2);
 453 
 454 
 455   // how far will the media rotate in the given amount of time
 456   // only the angle in the result is set
 457   dm_angle_t(*dm_rotate)(struct dm_disk_if *, 
 458                          dm_time_t *time);
 459 
 460   // how long does one revolution take
 461   dm_time_t(*dm_period)(struct dm_disk_if *);
 462 
 463 
 464   // interface to disksim for rpm randomization
 465   // 2nd argument is the amount of time one rotation takes
 466   void(*dm_randomize_rpm)(struct dm_disk_if *);
 467 
 468   // how big will this layout struct be when marshaled
 469   int(*dm_marshaled_len)(struct dm_disk_if *);
 470 
 471   // unmarshal this layout struct into the provided buffer
 472   // returns a pointer to the first byte in the buffer it didn't write to
 473   void *(*dm_marshal)(struct dm_disk_if *, char *);
 474 };
