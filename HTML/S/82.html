  38 int layout_test_trackbound(struct dm_disk_if *d) {
  39   int c;
  40   int badct = 0;
  41   int bad = 0;
  42   int lastbad = 0;
  43   int lastlbn; // last lbn on the previous track
  44 
  45   setlinebuf(stdout);
  46 
  47   for(c = 0; c < d->dm_sectors; ) {
  48     int l1 = 0, l2 = 0;
  49     struct dm_pbn pbn;
  50     int sptl, sptp;
  51     int remapsector = 0;
  52 
  53     d->layout->dm_translate_ltop(d, c, MAP_FULL, &pbn, 0);
  54 
  55     sptl = d->layout->dm_get_sectors_lbn(d, c);
  56     sptp = d->layout->dm_get_sectors_pbn(d, &pbn);
  57 
  58     if(sptl != sptp) {
  59       printf("spt (lbn) %d != %d (pbn)\n", sptl, sptp);
  60       bad = 1;
  61     }
  62 
  63     d->layout->dm_get_track_boundaries(d, &pbn, &l1, &l2, 0);
  64 
  65         //printf("%d -> (%d,%d) = %d (%d)\n", c, l1, l2, l2 - l1 + 1, sptl);
  66 
  67     if(l1 >= d->dm_sectors) {
  68       printf("l1 past end of disk! (%d >= %d)\n", l1, d->dm_sectors);
  69     }
  70     
  71     if(l2 >= d->dm_sectors) {
  72       printf("l2 past end of disk! (%d >= %d)\n", l2, d->dm_sectors);
  73     }
  74     
  75 
  76     // sanity checks
  77     if(l1 >= l2) {
  78       printf("test_trackbound: %d >= %d\n", l1, l2);
  79       bad = 1;
  80     }
  81 
  82     if(l2 - l1 + 1 > sptl) {
  83       printf("%d > %d (phys spt)\n", l2 - l1 + 1, sptl);
  84       bad = 1;
  85     }
  86 
  87     if(!((l1 <= c) && (c <= l2))) {
  88       printf("test_trackbound: %d (%d,%d,%d) -> (%d, %d) \n", 
  89              c, pbn.cyl, pbn.head, pbn.sector,
  90              l1, l2);
  91       bad = 1;
  92     }
  93 
  94     // "smoothness" test... the results should cover the lbn space
  95     if(!lastbad && (c>0) && (l1 != (lastlbn+1)) && (lastlbn != l2)) 
  96     {
  97       printf("test_trackbound (%d): (%d,%d) <> %d+1\n", c, l1, l2, lastlbn);
  98       bad = 1;
  99     }
 100     
 101     lastlbn = l2;
 102     if(bad) {
 103       printf("bad %d\n", c);
 104       badct++;
 105     }
 106 
 107     if(bad) {
 108       c++;
 109     }
 110     else {
 111       c = l2 + 1;
 112     }
 113 
 114     lastbad = bad;
 115     bad = 0;
 116   }
 117   return badct;
 118 }
