  38 // Try to reverse-map bogus PBNs -- they should all return DM_NX, not
  39 // some (wrong) lbn.
  40 
  41 int 
  42 layout_bogon(struct dm_disk_if *d) {
  43   int lbn = 0;
  44   int i, rv;
  45   struct dm_pbn p, p2;
  46   int l0, ln;
  47   int bad = 0;
  48 
  49   do {
  50     d->layout->dm_translate_ltop(d, lbn, MAP_FULL, &p, 0);
  51     d->layout->dm_get_track_boundaries(d, &p, &l0, &ln, 0);
  52     d->layout->dm_translate_ltop(d, ln, MAP_FULL, &p, 0);
  53     p2 = p;
  54     for(i = 0; i < 1; i++) {
  55       p2.sector++;
  56       rv = d->layout->dm_translate_ptol(d, &p2, 0);
  57       if(rv >= 0) {
  58         printf("(%d,%d,%d) -> %d  !!! max (%d,%d,%d)\n",
  59                p2.cyl, p2.head, p2.sector, rv,
  60                p.cyl, p.head, p.sector);
  61         bad++;
  62       }
  63     }
  64 
  65     lbn = ln + 1;
  66   } while(lbn < d->dm_sectors);
  67 
  68   return bad;
  69 }
