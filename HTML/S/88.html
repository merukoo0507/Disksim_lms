  41 layout_test_simple(struct dm_disk_if *d) {
  42   int i, lbn, count = 0, runlbn = 0;
  43   struct dm_pbn pbn, trkpbn = {0,0,0};
  44   int bad = 0;
  45 
  46   for(i = 0; i < d->dm_sectors; i++) {
  47     int lbn2;
  48     dm_angle_t skew, zerol;
  49     struct dm_mech_state track;
  50     dm_ptol_result_t rc;
  51 
  52     startClock(0);
  53     rc = d->layout->dm_translate_ltop(d, i, MAP_FULL, &pbn, 0);
  54     stopClock(0);
  55     if(rc == DM_NX) {
  56       printf("*** %s %d -> NX\n", __func__, i);
  57       bad++;
  58       continue;
  59     }
  60     
  61     startClock(1);
  62     lbn2 = d->layout->dm_translate_ptol(d, &pbn, 0);
  63     stopClock(1);
  64     if(lbn2 == DM_NX) {
  65       printf("*** layout_test_simple %d -> (%d,%d,%d) -> NX\n",
  66              i, pbn.cyl, pbn.head, pbn.sector);
  67       bad++;
  68       continue;
  69     }
  70 
  71     if(i != lbn2) {
  72       printf("*** layout_test_simple: %8d -> (%8d, %3d, %4d) -> %8d\n", 
  73              i, pbn.cyl, pbn.head, pbn.sector, lbn2);
  74       bad++;
  75     }
  76     else {
  77       //      printf("%8d <-> (%8d, %3d, %4d)\n", 
  78       //             c, pbn.cyl, pbn.head, pbn.sector);
  79       if((lbn2 > 0)
  80          && (trkpbn.head == pbn.head)
  81          && (trkpbn.cyl == pbn.cyl)
  82          && ((trkpbn.sector+count) == pbn.sector)) { count++;  }
  83       else {
  84         //      printf("l %d %d %d %d %d\n", runlbn, trkpbn.cyl, trkpbn.head, trkpbn.sector, count);
  85         runlbn = i;
  86         trkpbn = pbn;
  87         count = 1;
  88       }
  89     }    
  90 
  91     fflush(stdout);
  92   }
  93 
  94 
  95   return bad;
  96 }
