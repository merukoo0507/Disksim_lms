 107 // Compute start + mult*q and add x for the number of times it 
 108 // wraps around 0.  This can probably be optimized.
 109 #ifdef WIN32
 110 static __inline dm_angle_t 
 111 #else
 112 static inline dm_angle_t 
 113 #endif
 114 addmult(dm_angle_t start, int mult, dm_angle_t q, dm_angle_t x) 
 115 {
 116   //  int c;
 117   dm_angle_t result = start;
 118   long long tmp = (long long)start + mult * (long long)q;
 119   long long max = (long long)1 << 32;
 120   
 121 
 122 
 123 /*    for(c = 0; c < mult; c++) { */
 124 /*      tmp = result + q; */
 125 /*      if(tmp < result) { */
 126 /*        result = tmp + x; */
 127 /*      } */
 128 /*      else { */
 129 /*        result = tmp; */
 130 /*      } */
 131 /*    } */
 132   
 133   return result + mult * q + (tmp / max) * x;
 134 }

 137 static struct dm_layout_g1_band *
 138 find_band_lbn(struct dm_layout_g1 *l, int lbn)
 139 {
 140   struct dm_layout_g1_band *b = &l->bands[0];
 141   int bandstart = 0;
 142   int bandno = 0;
 143 
 144   while((lbn >= b->blksinband) || (lbn < 0)) {
 145     bandstart += b->blksinband;
 146     lbn -= b->blksinband;
 147     bandno++;
 148     b = &l->bands[bandno];
 149     ddbg_assert(bandno < l->bands_len);
 150     ddbg_assert(lbn >= 0);
 151   }
 152   
 153   return b;
 154 }

 164 static struct dm_layout_g1_band *
 165 find_band_pbn(struct dm_layout_g1 *l, struct dm_pbn *p)
 166 {
 167   int c;
 168   for(c = 0; c < l->bands_len; c++) {
 169     if((p->cyl >= l->bands[c].startcyl) &&
 170        (p->cyl <= l->bands[c].endcyl)) 
 171       {
 172         return &l->bands[c];
 173       }
 174   }
 175 
 176   ddbg_assert2(0, "band not found!");
 177   return 0;
 178 }

 197 static int g1_surfno_on_cyl(struct dm_layout_g1 *d,
 198                             struct dm_layout_g1_band *b, 
 199                             struct dm_pbn *p)
 200 {
 201 
 202   switch(d->mapping) {
 203   case LAYOUT_CYLSWITCHONSURF1:
 204     if((p->cyl - b->startcyl) % 2) {
 205       return (d->disk->dm_surfaces - p->head - 1);
 206     }  
 207     break;
 208 
 209   case LAYOUT_CYLSWITCHONSURF2:
 210     if(p->cyl % 2) {
 211       return (d->disk->dm_surfaces - p->head - 1);
 212     }
 213     break;
 214 
 215   default: break;
 216   }
 217   return p->head;
 218 }

 221 static dm_angle_t
 222 g1_map_pbn_skew(struct dm_disk_if *d, 
 223                 struct dm_pbn *p)
 224 {
 225   struct dm_layout_g1 *l = (struct dm_layout_g1 *)d->layout;
 226   struct dm_pbn pbn = *p;
 227   struct dm_layout_g1_band *b = find_band_pbn(l, p);
 228   int trackswitches;
 229 
 230   dm_angle_t result = b->firstblkangle;
 231 
 232   int slipoffs = 0;
 233 
 234 
 235 
 236   switch (l->mapping) {
 237   case LAYOUT_NORMAL:
 238   case LAYOUT_CYLSWITCHONSURF1:
 239   case LAYOUT_CYLSWITCHONSURF2:
 240     {
 241       //      int c;
 242       // relative to the start of the zone...
 243       int cylno = p->cyl - b->startcyl;
 244       pbn.cyl -= b->startcyl;
 245       pbn.head = g1_surfno_on_cyl(l, b, p);
 246       
 247       trackswitches = pbn.head + (d->dm_surfaces - 1) * pbn.cyl;
 248 
 249       result = addmult(result, trackswitches, b->trackskew, b->trkspace);
 250       result = addmult(result, pbn.cyl, b->cylskew, b->trkspace);
 251 
 252       /* The following assumes that slips also push skew forward in
 253        * some schemes 
 254        */
 255       if ((issectpercyl(l)) 
 256           || (l->sparescheme == SECTPERTRACK_SPARING)) {
 257         
 258         int tracks = cylno * d->dm_surfaces;
 259         int cutoff;
 260         int i;
 261         
 262         if (l->sparescheme == SECTPERTRACK_SPARING) {
 263           tracks += p->head;
 264         }
 265 
 266         // not quite accurate; the cutoff should be the lbn
 267         // corresponding to p
 268         cutoff = tracks * b->blkspertrack;
 269         for (i = 0; i < b->numslips; i++) {
 270           if (b->slip[i] < cutoff) {
 271             slipoffs++;
 272           }
 273         }
 274       }
 275       result = addmult(result, slipoffs, b->sector_width, b->trkspace);
 276       
 277     }
 278     break;
 279 
 280 
 281   default:
 282     ddbg_assert(0);
 283   }
 284 
 285   result = addmult(result, pbn.sector, b->sector_width, b->trkspace);
 286 
 287   return result;
 288 }

 319   struct dm_pbn pbn = *p;

 323

 325 
 326   p = &pbn;
 327 
 328   switch (l->mapping) {
 329   case LAYOUT_NORMAL:
 330   case LAYOUT_CYLSWITCHONSURF1:
 331   case LAYOUT_CYLSWITCHONSURF2:
 332     p->cyl -= b->startcyl;
 333 
 334     /* The following assumes that slips also push skew forward in
 335      * some schemes 
 336      */
 337     if ((issectpercyl(l)) 
 338         || (l->sparescheme == SECTPERTRACK_SPARING)) {
 339 
 340       int tracks = p->cyl * d->dm_surfaces;
 341       int cutoff;
 342       int i;
 343  
 344       if (l->sparescheme == SECTPERTRACK_SPARING) {
 345         tracks += p->head;
 346       }
 347       cutoff = tracks * b->blkspertrack;
 348       for (i = 0; i < b->numslips; i++) {
 349         if (b->slip[i] < cutoff) {
 350           slipoffs++;
 351         }

 364

 369 static dm_ptol_result_t
 370 g1_convert_atop(struct dm_disk_if *d,
 371                 struct dm_mech_state *a,
 372                 struct dm_pbn *result)
 373 {
 374   //  int trackswitches;
 375   struct dm_pbn track;
 376   struct dm_layout_g1_band *b;
 377   dm_angle_t angle;
 378 
 379   int slipoffs = 0;
 380   struct dm_layout_g1 *l = (struct dm_layout_g1 *)d->layout;
 381 
 382 
 383   track.cyl = a->cyl;
 384   track.head = a->head;
 385 
 386   b = find_band_pbn(l, &track);
 387 
 388   angle = a->theta - b->firstblkangle;
 389 
 390   switch (l->mapping) {
 391   case LAYOUT_NORMAL:
 392   case LAYOUT_CYLSWITCHONSURF1:
 393   case LAYOUT_CYLSWITCHONSURF2:
 394 
 395     /* The following assumes that slips also push skew forward in
 396      * some schemes 
 397      */
 398     if ((issectpercyl(l)) 
 399         || (l->sparescheme == SECTPERTRACK_SPARING)) {
 400 
 401       int tracks = track.cyl * d->dm_surfaces;
 402       int cutoff;
 403       int i;
 404  
 405       if (l->sparescheme == SECTPERTRACK_SPARING) {
 406         tracks -= track.head;
 407       }
 408       // count the slips
 409       cutoff = tracks * b->blkspertrack;
 410       for (i = 0; i < b->numslips; i++) {
 411         if (b->slip[i] < cutoff) {
 412           slipoffs++;
 413         }
 414       }
 415     }
 416     // subtract out the slips
 417     angle -= addmult(0, slipoffs, b->sector_width, b->trkspace);
 418     break;
 419 
 420 
 421   default:
 422     ddbg_assert(0);
 423   }
 424 
 425   result->cyl = a->cyl;
 426   result->head = a->head;
 427   // rounds angle down to lower sector boundary
 428   result->sector = (angle / (b->sector_width)) % b->blkspertrack;
 429 
 430   // if skews are not in whole sectors, we need to round up here
 431   if((b->trackskew % b->sector_width)
 432      || (b->cylskew % b->sector_width)) {
 433     if((angle % b->sector_width) > 0) {
 434       result->sector = ((result->sector+1) % b->blkspertrack);
 435     }
 436   }
 437 
 438   // XXX ... check that angle isn't a defect
 439   return DM_OK;
 440 }

 474 /*
 475  * The next several functions compute the lbn stored in a physical
 476  * media sector <cyl, surf, blk>, for different sparing/mapping
 477  * schemes.  The returned value is a proper lbn, or DM_REMAPPED if the
 478  * sector is a remapped defect, or DM_SLIPPED if the sector is a
 479  * slipped defect or an unused spare.
 480  */
 481 
 482 static dm_ptol_result_t
 483 g1_ptol_nosparing(struct dm_disk_if *d, 
 484                   struct dm_pbn *p,
 485                   int *remapsector)
 486 
 487 {
 488   struct dm_layout_g1 *l = (struct dm_layout_g1 *)d->layout;
 489   struct dm_layout_g1_band *b = find_band_pbn(l, p);
 490   int lbn = l->band_blknos[b->num];
 491   struct dm_pbn pbn = *p;
 492 
 493   p = &pbn;
 494 
 495 
 496   ddbg_assert(b);
 497   ddbg_assert(p->cyl >= 0);
 498   ddbg_assert(p->cyl >= b->startcyl);
 499   ddbg_assert(p->cyl < d->dm_cyls);
 500   ddbg_assert(p->head >= 0);
 501   ddbg_assert(p->head < d->dm_surfaces);
 502   ddbg_assert(p->sector >= 0);
 503   ddbg_assert(p->sector < b->blkspertrack);
 504 
 505 
 506   p->head = g1_surfno_on_cyl(l, b, p);
 507   
 508   lbn += (((p->cyl - b->startcyl) * d->dm_surfaces) + p->head) * b->blkspertrack;
 509   lbn += p->sector - b->deadspace;
 510 
 511   if(lbn < 0) {
 512     return DM_SLIPPED;
 513   }
 514   else {
 515     return lbn;
 516   }
 517 }

 521 static dm_ptol_result_t
 522 g1_ptol_sectpertrackspare(struct dm_disk_if *d, 
 523                           struct dm_pbn *p,
 524                           int *remapsector)
 525 {
 526   int i;
 527   int lbnspercyl;
 528   int firstblkoncyl;
 529   struct dm_layout_g1 *l = (struct dm_layout_g1 *)d->layout;
 530   struct dm_layout_g1_band *b = find_band_pbn(l, p);
 531   int lbn = l->band_blknos[b->num];
 532   struct dm_pbn pbn = *p;
 533   p = &pbn;
 534 
 535   ddbg_assert(b);
 536   ddbg_assert(p->cyl >= 0);
 537   ddbg_assert(p->cyl >= b->startcyl);
 538   ddbg_assert(p->cyl < d->dm_cyls);
 539   ddbg_assert(p->head >= 0);
 540   ddbg_assert(p->head < d->dm_surfaces);
 541   ddbg_assert(p->sector >= 0);
 542   ddbg_assert(p->sector < b->blkspertrack);
 543 
 544 
 545   p->head = g1_surfno_on_cyl(l, b, p);
 546   firstblkoncyl = (p->cyl - b->startcyl) * d->dm_surfaces * b->blkspertrack;
 547   p->sector += firstblkoncyl + (p->head * b->blkspertrack);
 548   for (i=(b->numdefects-1); i>=0; i--) {
 549     if (p->sector == b->defect[i]) {        /* Remapped bad block */
 550       return DM_REMAPPED;
 551     }
 552     if(p->sector == b->remap[i]) {
 553       if(remapsector) *remapsector = 1;
 554       p->sector = b->defect[i];
 555       break;
 556     }
 557   }
 558   for (i = (b->numslips-1); i >= 0; i--) {
 559     if (p->sector == b->slip[i]) {          /* Slipped bad block */
 560       return DM_SLIPPED;
 561     }
 562     if ((p->sector > b->slip[i]) 
 563         && ((b->slip[i] / (b->blkspertrack * d->dm_surfaces)) == (p->cyl - b->startcyl)) 
 564         && ((b->slip[i] % (b->blkspertrack * d->dm_surfaces)) == p->head)) 
 565       {
 566         p->sector--;
 567       }
 568   }
 569 
 570   if((p->sector % b->blkspertrack) >= (b->blkspertrack - b->sparecnt)) {   
 571     /* Unused spare block */
 572     return DM_SLIPPED;
 573   }
 574 
 575   /* recompute in case sector was remapped */
 576   p->cyl = p->sector / (b->blkspertrack * d->dm_surfaces);
 577   p->head = p->sector % (b->blkspertrack * d->dm_surfaces);
 578   p->head = p->head / b->blkspertrack;
 579   p->sector = p->sector % b->blkspertrack;
 580 
 581   lbnspercyl = (b->blkspertrack - b->sparecnt) * d->dm_surfaces;
 582   lbn += p->cyl * lbnspercyl;
 583   lbn += p->head * (b->blkspertrack - b->sparecnt);
 584   lbn += p->sector - b->deadspace;
 585 
 586 
 587   if(lbn < 0) {
 588     return DM_SLIPPED;
 589   }
 590   else {
 591     return lbn;
 592   }
 593 }

 669   p->sector = p->sector % b->blkspertrack;

 673   lbn += p->sector - b->deadspace;

 675 
 676 
 677   if(lbn < 0) {
 678     return DM_SLIPPED;
 679   }

1863 char *
1864 layout_g1_unmarshal(struct dm_marshal_hdr *hdr, 
1865                      void **result,
1866                      void *parent)
1867 {
1868   int c;
1869   struct dm_layout_g1 *l = malloc(sizeof(struct dm_layout_g1));
1870   char *ptr = (char *)hdr;
1871   
1872   ddbg_assert(hdr->type == DM_LAYOUT_G1_TYP);
1873   
1874   ptr += sizeof(struct dm_marshal_hdr);
1875 
1876   memcpy(l, ptr, sizeof(struct dm_layout_g1));
1877   unmarshal_fns((void **)&l->hdr,
1878                  sizeof(l->hdr) / sizeof(void*),
1879                  ptr,
1880                  DM_LAYOUT_G1_TYP);
1881 
1882   ptr += sizeof(struct dm_layout_g1);
1883   
1884   l->bands = malloc(l->bands_len * sizeof(struct dm_layout_g1_band));
1885 
1886   // now do the zones
1887   for(c = 0; c < l->bands_len; c++) {
1888     struct dm_layout_g1_band *b = &l->bands[c];
1889     memcpy(b, ptr, sizeof(struct dm_layout_g1_band));
1890     ptr += sizeof(struct dm_layout_g1_band);
1891 
1892     // slips
1893     if(b->numslips != 0) {
1894       b->slip = malloc(b->numslips * sizeof(int));
1895       memcpy(b->slip, ptr, b->numslips * sizeof(int));
1896       ptr += b->numslips * sizeof(int);
1897     }
1898 
1899     // defect locations
1900     if(b->numdefects != 0) {
1901       b->defect = malloc(b->numdefects * sizeof(int));
1902       memcpy(b->defect, ptr, b->numdefects * sizeof(int));
1903       ptr += b->numdefects * sizeof(int);
1904 
1905       // defect remaps
1906       b->remap = malloc(b->numdefects * sizeof(int));
1907       memcpy(b->remap, ptr, b->numdefects * sizeof(int));
1908       ptr += b->numdefects * sizeof(int);
1909     }
1910   }
1911 
1912   // do the band blknos array
1913   l->band_blknos = malloc(l->bands_len * sizeof(int));
1914   memcpy(l->band_blknos, ptr, l->bands_len * sizeof(int));
1915   ptr += l->bands_len * sizeof(int);
1916 
1917 
1918   l->disk = parent;
1919 
1920   *result = l;
1921   return ptr;
1922 }

1933 // Fetch the info about the nth zone and store it in the result
1934 // parameter z.  Returns 0 on success, -1 on error (bad n)
1935 // n should be within 0 and (dm_get_numzones() - 1)
1936 static int 
1937 g1_get_zone(struct dm_disk_if *d, 
1938             int n, 
1939             struct dm_layout_zone *result)
1940 {
1941   struct dm_layout_g1 *l = (struct dm_layout_g1 *)d->layout;
1942   struct dm_layout_g1_band *z;
1943 
1944   // check args  
1945   if(z == 0) { return -1; }
1946   if(n < 0 || n >= l->bands_len) { return -1; }
1947 
1948   z = &l->bands[n];
1949 
1950   result->spt = z->blkspertrack;
1951   result->lbn_low = l->band_blknos[n];
1952   result->lbn_high = l->band_blknos[n] + z->blksinband - 1;
1953 
1954   result->cyl_low = z->startcyl;
1955   result->cyl_high = z->endcyl;
1956 
1957   return 0;
1958 }
