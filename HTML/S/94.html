  98 #ifndef _DM_MECH_H
  99 #define _DM_MECH_H
 100 
 101 #include "dm.h"
 102 
 103 // Seek types 
 104 typedef enum {
 105   SEEK_CONST,                        
 106   SEEK_3PT_LINE,              
 107   SEEK_3PT_CURVE,                
 108   SEEK_HPL,
 109   SEEK_1ST10_PLUS_HPL,
 110   SEEK_EXTRACTED
 111 } disk_seek_t;
 112 
 113 #define DM_MECH_G1_MAX_SEEK SEEK_EXTRACTED
 114 
 115 
 116 // Latency types
 117 typedef enum {
 118   AVGROTATE               = -1,
 119   SKEWED_FOR_TRACK_SWITCH = -2,
 120   NON_SKEWED              = -3
 121 } disk_latency_t;
 122 
 123 
 124 typedef dm_time_t(*dm_mech_g1_seekfn)(struct dm_disk_if *d,
 125                                       struct dm_mech_state *begin,
 126                                       struct dm_mech_state *end,
 127                                       int rw);
 128 
 129 struct dm_mech_g1 {
 130   struct dm_mech_if hdr;
 131 
 132   // a seek time function
 133   dm_mech_g1_seekfn seekfn;
 134 
 135   struct dm_disk_if *disk; // back pointer to disk this goes with
 136 
 137 
 138   //
 139   // Seek Time related
 140   //
 141 
 142   // comes from "Constant access time" and "Access time type"
 143   dm_time_t       acctime;
 144 
 145   disk_seek_t  seektype;
 146   // "Constant seek time"
 147   dm_time_t       seektime;
 148 
 149   // single cylinder seek
 150   dm_time_t       seekone; // only used by non-extracted algs
 151 
 152   // average seek time
 153   dm_time_t       seekavg;
 154 
 155   // full stroke seek time
 156   dm_time_t       seekfull; // only used by non-extracted algs
 157 
 158   // extra settling time for writes
 159   dm_time_t       seekwritedelta;
 160 
 161   int             hpseek_v1; // v_1 from hpl seek equation (is in cyls)
 162   dm_time_t       hpseek[6]; // v_2..v_6 .. in 1..6 (0 not used)
 163   dm_time_t       first10seeks[10];
 164 
 165   // extracted seek curve
 166   int           xseekcnt;
 167   int          *xseekdists;
 168   dm_time_t    *xseektimes;
 169 
 170 
 171   dm_time_t       headswitch;
 172 
 173 
 174   // if positive, latency in msecs, otherwise 
 175   // values from latency_t
 176 
 177   disk_latency_t  latency;
 178   dm_time_t       rotatetime;
 179 
 180   uint64_t        rpmerr;
 181   uint64_t        rpm;
 182 
 183 
 184   // zero-latency io parameters
 185   int           immedread;
 186   int           immedwrite;
 187   int           immed;
 188 
 189 
 190   // "Bulk sector transfer time"
 191   dm_time_t     blktranstime;
 192 
 193 
 194 
 195 };
