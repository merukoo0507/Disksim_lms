  38 static struct dm_layout_g2_zone *
  39 find_zone_lbn(struct dm_disk_if *d,
  40               int lbn)
  41 {
  42   int c;
  43   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
  44   for(c = 0; c < l->zones_len; c++) {
  45     if((l->zones[c].lbnlow <= lbn) 
  46        && (lbn <= l->zones[c].lbnhigh)) 
  47       {
  48         return &l->zones[c];
  49       }
  50   }
  51   ddbg_assert(0);
  52 
  53   return 0;
  54 }

  76 // return 1 if they overlap, ..
  77 static int
  78 g2_nodecmp(const void *p1, const void *p2) 
  79 {
  80   struct dm_layout_g2_node *n1 = (struct dm_layout_g2_node *)p1;
  81   struct dm_layout_g2_node *n2 = (struct dm_layout_g2_node *)p2;
  82 
  83   if(n1->lbn >= (n2->lbn + n2->len)) {
  84     return 1;
  85   }
  86   else if(n2->lbn >= (n1->lbn + n1->len)) {
  87     return -1;
  88   }
  89   else return 0;
  90 }

  93 static dm_ptol_result_t
  94 ltop(struct dm_disk_if *d, 
  95      int lbn, 
  96      dm_layout_maptype mt,
  97      struct dm_pbn *result,
  98      int *remapsector)
  99 {
 100   struct dm_layout_g2_node keynode, *rn;
 101   struct dm_layout_g2_zone *z = find_zone_lbn(d,lbn);
 102   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 103 
 104   memset(&keynode, 0, sizeof(keynode));
 105   keynode.lbn = lbn;
 106   keynode.len = 1;
 107 
 108   rn = bsearch(&keynode, 
 109                l->ltop_map, 
 110                l->ltop_map_len, 
 111                sizeof(struct dm_layout_g2_node),
 112                g2_nodecmp);
 113 
 114   if(!rn) {
 115     return DM_NX;
 116   }
 117   else {
 118     *result = rn->loc;
 119     result->sector += (lbn - rn->lbn);
 120     return DM_OK;
 121   }
 122 }

 137 static dm_ptol_result_t
 138 ptol(struct dm_disk_if *d, 
 139      struct dm_pbn *p,
 140      int *remapsector)
 141 {
 142   int i;
 143   struct dm_layout_g2_node *curr;
 144 
 145   struct dm_layout_g2_surf *track;
 146   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 147 
 148   ddbg_assert(p->cyl < d->dm_cyls);
 149   ddbg_assert(p->head < d->dm_surfaces);
 150 
 151   curr = &l->ptol_map[p->cyl].surfaces[p->head].extents[0];
 152   for(i = 0; 
 153       i < l->ptol_map[p->cyl].surfaces[p->head].extents_len; 
 154       i++, curr++) 
 155   {
 156     if((curr->loc.cyl == p->cyl)
 157        && (curr->loc.head == p->head)
 158        && (p->sector >= curr->loc.sector)
 159        && (p->sector < (curr->loc.sector + curr->len)))
 160       
 161     {
 162       return curr->lbn + (p->sector - curr->loc.sector);
 163     }
 164   } 
 165 
 166   return DM_NX;
 167 }

 181 static int 
 182 trkcmp(const void *ptr1, const void *ptr2)
 183 {
 184   const struct dm_pbn *p1 = ptr1;
 185   const struct dm_pbn *p2 = ptr2;
 186 
 187   if(p1->cyl < p2->cyl) return -1;
 188   else if(p2->cyl < p1->cyl) return 1;
 189 
 190   if(p1->head < p2->head) return -1;
 191   else if(p2->head < p1->head) return 1;
 192 
 193   return 0;
 194 }

 197 static dm_ptol_result_t
 198 g2_defect_count(struct dm_disk_if *d,
 199                 struct dm_pbn *track,
 200                 int *result)
 201 {
 202   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 203   struct dm_pbn *p, *p1, *p2;
 204 
 205   if(!l->defects_len) {
 206     *result = 0;
 207     goto out;
 208   }
 209 
 210   p = bsearch(track,
 211               l->defects, 
 212               l->defects_len, 
 213               sizeof(struct dm_pbn), 
 214               trkcmp);
 215 
 216   if(!p) {
 217     *result = 0;
 218   }
 219   else {
 220     p1 = p2 = p;
 221     
 222     while(!trkcmp(p1, p)) p1--;  
 223     while(!trkcmp(p2, p)) p2++;  
 224     
 225     *result = p2 - p1 - 1;
 226   }
 227 
 228  out:
 229 
 230   return DM_OK;
 231 }

 242 static int
 243 st_pbn(struct dm_disk_if *d,
 244        struct dm_pbn *p)
 245 {
 246   struct dm_pbn p2 = *p;
 247   struct dm_layout_g2_zone *z;
 248 
 249   // return st for the nearest (lower) zone if this cyl is unmapped
 250   while(!(z = find_zone_pbn(d, &p2)) && p2.cyl >= 0) { p2.cyl--; }
 251   ddbg_assert(z);
 252 
 253   return z->st;
 254 }

 257 static void
 258 track_boundaries(struct dm_disk_if *d,
 259                  struct dm_pbn *p,
 260                  int *l1,
 261                  int *l2,
 262                  int *remapsector)
 263 {
 264   struct dm_pbn p1, p2;
 265   struct dm_layout_g2_zone *z = find_zone_pbn(d, p);
 266 
 267   p1 = *p;
 268   p2 = *p;
 269 
 270   p1.sector = 0;
 271   p2.sector = z->st;
 272   
 273   if(l1) {
 274     do { 
 275       *l1 = d->layout->dm_translate_ptol(d, &p1, remapsector);
 276       p1.sector++;
 277     } while((*l1 == DM_NX) && (p1.sector < z->st));
 278   }
 279 
 280   if(l2) {
 281     do {
 282       p2.sector--;
 283       *l2 = d->layout->dm_translate_ptol(d, &p2, remapsector);
 284     } while((*l2 == DM_NX) && p2.sector);
 285   }
 286 }

 299 static dm_angle_t
 300 pbn_skew_new(struct dm_disk_if *d,
 301          struct dm_pbn *p)
 302 {
 303   int i;
 304   int lasthead, lastcyl;
 305 
 306   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 307   struct dm_layout_g2_zone *z = find_zone_pbn(d, p);
 308   int fltop_ext = l->ptol_map[p->cyl].first_ltop_extent;
 309 
 310   dm_angle_t result = l->ptol_map[p->cyl].skew;
 311 
 312 
 313   lastcyl = l->ltop_map[fltop_ext].loc.cyl;
 314   lasthead = l->ltop_map[fltop_ext].loc.head;
 315 
 316   for(i = fltop_ext; i < l->ltop_map_len; i++) {
 317     if(l->ltop_map[i].loc.cyl != lastcyl) { 
 318       result += z->csskew;
 319       lastcyl = l->ltop_map[i].loc.cyl;
 320     }
 321     if(l->ltop_map[i].loc.head != lasthead) {
 322       result += z->hsskew;
 323       lasthead = l->ltop_map[i].loc.head;
 324     }
 325 
 326     if((l->ltop_map[i].loc.cyl == p->cyl)
 327        && (l->ltop_map[i].loc.head == p->head)) goto done;
 328   }
 329   ddbg_assert(0);
 330 
 331  done:
 332 
 333   result += p->sector * z->sector_width;
 334 
 335   return result;
 336 }

 339 static dm_angle_t
 340 pbn_skew(struct dm_disk_if *d,
 341          struct dm_pbn *p)
 342 {
 343   int i;
 344   int lastcyl, lasthead;
 345   dm_angle_t result = 0;
 346   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 347   struct dm_layout_g2_zone *z = find_zone_pbn(d, p);
 348 
 349   // count headswitch/cylswitch
 350   // find start of zone in ltop map
 351   // slooooow
 352   // precompute -- table of first index per zone
 353   for(i = 0; i < l->ltop_map_len; i++) {
 354     if(l->ltop_map[i].loc.cyl == z->cyllow) break;
 355   }
 356 
 357   if((l->ltop_map[i].loc.cyl == p->cyl)
 358      && (l->ltop_map[i].loc.head == p->head)) goto done;
 359 
 360   lastcyl = l->ltop_map[i].loc.cyl;
 361   lasthead = l->ltop_map[i].loc.head;
 362   
 363   // walk forward until we get to p
 364   // precompute: skew per cylinder
 365   for( ; i < l->ltop_map_len; i++) {
 366     if(l->ltop_map[i].loc.cyl != lastcyl) result += z->csskew;
 367     if(l->ltop_map[i].loc.head != lasthead) result += z->hsskew;
 368     if((l->ltop_map[i].loc.cyl == p->cyl)
 369        && (l->ltop_map[i].loc.head == p->head)) goto done;
 370   }
 371   ddbg_assert(0);
 372  done:
 373   result += p->sector * z->sector_width;
 374 
 375   return result;
 376 }

 378 static void
 379 ptoa(struct dm_disk_if *d,

 381      dm_angle_t *start,

 394 static dm_ptol_result_t
 395 atop(struct dm_disk_if *d,
 396      struct dm_mech_state *a,
 397      struct dm_pbn *result)
 398 {
 399   struct dm_pbn pbn;
 400   struct dm_layout_g2_zone *z;
 401   result->head = a->head;
 402   result->cyl = a->cyl;
 403 
 404   pbn.head = a->head;
 405   pbn.cyl = a->cyl;
 406 
 407   z = find_zone_pbn(d, &pbn);
 408 
 409   result->sector = (a->theta / z->sector_width) % z->st;
 410 
 411   return DM_OK;
 412 }

 415 static dm_ptol_result_t
 416 g2_seek_distance(struct dm_disk_if *d,
 417                  int start_lbn,
 418                  int end_lbn)
 419 {
 420   struct dm_pbn p1, p2;
 421   int rv;
 422 
 423   rv = d->layout->dm_translate_ltop(d, start_lbn, MAP_NONE, &p1, 0);
 424   if(rv != DM_OK) {
 425     return rv;
 426   }
 427 
 428   rv = d->layout->dm_translate_ltop(d, end_lbn, MAP_NONE, &p2, 0);
 429   if(rv != DM_OK) {
 430     return rv;
 431   }
 432 
 433   
 434   return abs(p2.cyl - p1.cyl);
 435 }

 446 // Fetch the info about the nth zone and store it in the result
 447 // parameter z.  Returns 0 on success, -1 on error (bad n)
 448 // n should be within 0 and (dm_get_numzones() - 1)
 449 static int 
 450 g2_get_zone(struct dm_disk_if *d, int n, struct dm_layout_zone *result)
 451 {
 452   struct dm_layout_g2 *l = (struct dm_layout_g2 *)d->layout;
 453   struct dm_layout_g2_zone *z;
 454 
 455   if(result == 0) { return -1; }
 456   if(n < 0 || n >= l->zones_len) { return -1; }
 457   
 458   z = &l->zones[n];
 459 
 460   result->spt = z->st;
 461 
 462   result->lbn_low = z->lbnlow;
 463   result->lbn_high = z->lbnhigh;
 464 
 465   result->cyl_low = z->cyllow;
 466   result->cyl_high = z->cylhigh;
 467 
 468   
 469   return 0;
 470 }
